## 1. SPLIT ORDERS INTO PROTECTED VS FLEXIBLE BLOCKS

Protected orders must preserve hard constraints.
Flexible orders may be grouped and moved.

```javascript
function splitOrdersForBalancedSequence(orders) {
  const protectedOrders = [];
  const flexibleOrders = [];

  orders.forEach(order => {
    if (isProtectedOrder(order)) {
      protectedOrders.push(order);
    } else {
      flexibleOrders.push(order);
    }
  });

  return { protectedOrders, flexibleOrders };
}

function isProtectedOrder(order) {
  if (isMTOOrder(order)) return true;

  const status = String(order._n10dStatus || order.inferred_target_label || '').toLowerCase();

  if (status === 'critical') return true;
  if (status === 'urgent') return true;
  if (status === 'monitor') return true;
  if (status.includes('prio replenish')) return true;

  return false;
}

function isFlexibleOrder(order) {
  return !isProtectedOrder(order);
}
```

## 4. SCORE MATERIAL GROUPS BY INTERNAL COHESION + EARLIEST SAFE DATE

```javascript
function scoreMaterialGroup(groupOrders, n10dData) {
  const safeDates = groupOrders.map(order => {
    const window = getBalancedSafeWindow(order, n10dData);
    return window.start;
  });

  const earliestSafeDate = safeDates.sort()[0];

  const totalVolume = groupOrders.reduce((sum, o) => {
    return sum + (parseFloat(o.volume_override || o.volume || o.total_volume_mt || 0) || 0);
  }, 0);

  return {
    earliestSafeDate,
    totalVolume,
    size: groupOrders.length,
  };
}
```

## 5. ORDER FLEXIBLE GROUPS AS BLOCKS INSTEAD OF INDIVIDUAL CHRONO SORT

Instead of sorting every flexible order purely by date, sort material groups as blocks.

```javascript
function buildBalancedFlexibleBlocks(flexibleOrders, n10dData) {
  const grouped = groupFlexibleOrdersByMaterial(flexibleOrders);

  const blocks = Array.from(grouped.entries()).map(([signature, orders]) => {
    const meta = scoreMaterialGroup(orders, n10dData);

    return {
      signature,
      orders,
      earliestSafeDate: meta.earliestSafeDate,
      totalVolume: meta.totalVolume,
      size: meta.size,
    };
  });

  // Sort blocks by earliest safe date first, then by larger groups first
  blocks.sort((a, b) => {
    if (a.earliestSafeDate !== b.earliestSafeDate) {
      return new Date(a.earliestSafeDate) - new Date(b.earliestSafeDate);
    }

    return b.size - a.size;
  });

  return blocks;
}
```

## 6. WITHIN EACH FLEXIBLE BLOCK, ORDER BY LOWEST INTERNAL CHANGEOVER

```javascript
function orderWithinFlexibleBlock(blockOrders, changeoverRules) {
  if (blockOrders.length <= 1) return [...blockOrders];

  const remaining = [...blockOrders];
  const ordered = [];

  // Start with the order having earliest current date
  remaining.sort((a, b) => {
    const da = normalizeDate(a.target_avail_date) || '9999-12-31';
    const db = normalizeDate(b.target_avail_date) || '9999-12-31';
    return new Date(da) - new Date(db);
  });

  ordered.push(remaining.shift());

  while (remaining.length > 0) {
    const last = ordered[ordered.length - 1];

    let bestIndex = 0;
    let bestCost = Infinity;

    remaining.forEach((candidate, idx) => {
      const co = calculateChangeoverBetween(last, candidate, changeoverRules);
      if (co < bestCost) {
        bestCost = co;
        bestIndex = idx;
      }
    });

    ordered.push(remaining.splice(bestIndex, 1)[0]);
  }

  return ordered;
}
```

```javascript
function buildProtectedBlock(protectedOrders) {
  return [...protectedOrders].sort((a, b) => {
    const rankA = getProtectedRank(a);
    const rankB = getProtectedRank(b);

    if (rankA !== rankB) return rankA - rankB;

    const dateA = normalizeDate(a.target_avail_date) || '9999-12-31';
    const dateB = normalizeDate(b.target_avail_date) || '9999-12-31';

    return new Date(dateA) - new Date(dateB);
  });
}

function getProtectedRank(order) {
  if (isMTOOrder(order)) return 0;

  const status = String(order._n10dStatus || order.inferred_target_label || '').toLowerCase();

  if (status === 'critical') return 1;
  if (status === 'urgent' || status.includes('prio replenish')) return 2;
  if (status === 'monitor') return 3;

  return 4;
}
```

```javascript
function buildBalancedSequenceForLine(orders, n10dData, changeoverRules) {
  const { protectedOrders, flexibleOrders } = splitOrdersForBalancedSequence(orders);

  const protectedBlock = buildProtectedBlock(protectedOrders);

  const flexibleBlocks = buildBalancedFlexibleBlocks(flexibleOrders, n10dData);

  const orderedFlexible = flexibleBlocks.flatMap(block =>
```

```javascript
function assignBalancedDatesToSequence(lineOrders, line, n10dData, changeoverRules) {
  const result = [];
  let currentDateCursor = addDays(startOfDay(new Date()), 2);

  lineOrders.forEach((order, index) => {
    const cloned = { ...order };

    if (isProtectedOrder(cloned)) {
      cloned._aiSuggestedDate = normalizeDate(cloned.target_avail_date) || formatISODate(currentDateCursor);
      cloned._aiDateReason = 'Protected order — fixed by hard rule';
      result.push(cloned);

      // Move cursor forward only if needed
      const protectedDate = new Date(cloned._aiSuggestedDate);
      if (protectedDate > currentDateCursor) {
        currentDateCursor = protectedDate;
      }
      return;
    }

    const safeWindow = getBalancedSafeWindow(cloned, n10dData);
    const candidateDate = clampDateToWindow(formatISODate(currentDateCursor), safeWindow);

    cloned._originalAvailDate = normalizeDate(cloned.target_avail_date);
    cloned.target_avail_date = candidateDate;
    cloned._aiSuggestedDate = candidateDate;
    cloned._dateAdjustedByAI = cloned._originalAvailDate !== candidateDate;
    cloned._aiDateReason = 'Date aligned to grouped material block while staying in safe window';

    result.push(cloned);

    // Advance cursor modestly only when accumulated load gets too dense
    currentDateCursor = maybeAdvanceCursorByLoad(result, line, currentDateCursor);
  });

  return result.map((o, i) => ({
    ...o,
    prio: i + 1,
  }));
}

function clampDateToWindow(candidateDate, safeWindow) {
  const candidate = new Date(candidateDate);
  const start = new Date(safeWindow.start);
  const end = new Date(safeWindow.end);

  if (candidate < start) return formatISODate(start);
  if (candidate > end) return formatISODate(end);
  return formatISODate(candidate);
}

function maybeAdvanceCursorByLoad(orderedSoFar, line, currentDateCursor) {
  const currentDay = formatISODate(currentDateCursor);

  const sameDayOrders = orderedSoFar.filter(o =>
    normalizeDate(o._aiSuggestedDate || o.target_avail_date) === currentDay
  );

  const totalHours = sameDayOrders.reduce((sum, o) => {
    const volume = parseFloat(o.volume_override || o.volume || o.total_volume_mt || 0) || 0;
    const runRate = getLineRunRate(normalizeLine(line));
    const prod = runRate > 0 ? volume / runRate : 0;
    const co = parseFloat(o.changeover || 0) || 0;
    return sum + prod + co;
  }, 0);

  if (totalHours >= 14) {
    return addDays(currentDateCursor, 1);
  }

  return currentDateCursor;
}
```
