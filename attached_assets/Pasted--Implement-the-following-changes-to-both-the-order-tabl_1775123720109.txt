## 1. CASCADING COMPLETION DATE — RULES

### CASCADE LOGIC

```
For each order in priority order (top to bottom):

IF order has Start Date AND Start Time set:
  → Use that as the anchor
  → Completion = Start DateTime + Production Time + Changeover
  → Pass completion as the next order's cascaded start

IF order has NO Start Date or NO Start Time:
  → Start = Previous order's Completion DateTime
  → Completion = Start + Production Time + Changeover
  → Pass completion as the next order's cascaded start

IF order has overridden Completion Date:
  → Keep the overridden value
  → Do NOT recalculate
  → Pass overridden completion as the next order's cascaded start
```

### IMPLEMENTATION

```javascript
function recalculateCompletionDates(orders) {
  // Sort by prio
  orders.sort((a, b) => a.prio - b.prio);
  
  let currentEndTime = null;
  
  for (let i = 0; i < orders.length; i++) {
    const order = orders[i];
    
    // Skip Done and Cancelled
    if (order.status === 'Done' || order.status === 'Cancel PO') {
      continue;
    }
    
    const productionTime = parseFloat(order.production_time) || 0;
    const changeover = parseFloat(order.changeover) || 0;
    const totalTime = productionTime + changeover;
    
    // If completion date is overridden, keep it and use it as anchor
    if (order._completionOverridden) {
      const overriddenDateTime = parseDateTime(
        order.completion_date, 
        order.completion_time
      );
      if (overriddenDateTime) {
        currentEndTime = overriddenDateTime;
      }
      continue; // Don't recalculate this order
    }
    
    let startDateTime = null;
    
    // Order has explicit start date and time
    if (order.start_date && order.start_time) {
      startDateTime = parseDateTime(order.start_date, order.start_time);
    }
    // No start date — cascade from previous order's completion
    else if (currentEndTime) {
      startDateTime = new Date(currentEndTime);
      order._cascadedStartDate = formatDateFull(startDateTime);
      order._cascadedStartTime = formatTime(startDateTime);
    }
    
    if (startDateTime) {
      const completionDateTime = new Date(
        startDateTime.getTime() + (totalTime * 60 * 60 * 1000)
      );
      
      order.completion_date = formatDateFull(completionDateTime);
      order.completion_time = formatTime(completionDateTime);
      
      currentEndTime = completionDateTime;
    }
  }
}
```

### APPLY TO BOTH ORDER TABLE AND PREVIEW

```javascript
// Order table — recalculate when anything changes
function handleOrderTableChange(order, field, newValue, line) {
  order[field] = newValue;
  
  if (['start_date', 'start_time', 'volume', 'batch_size', 
       'batches', 'production_time', 'changeover'].includes(field)) {
    const lineOrders = getOrdersForLine(line);
    recalculateCompletionDates(lineOrders);
  }
}

// Preview — recalculate when anything changes
function handlePreviewChange(order, field, newValue, previewOrders) {
  order[field] = newValue;
  
  if (['start_date', 'start_time', 'volume', 'batch_size',
       'batches', 'production_time', 'changeover'].includes(field)) {
    recalculateCompletionDates(previewOrders);
  }
}

```javascript
function applyAutoSequence(previewOrders) {
  previewOrders.forEach((order, index) => {
    order.prio = order.prio;
    
    // Write avail date changes from preview
    if (order._previewAvailDateChanged) {
      // Already updated in order object
      order._previewAvailDateChanged = false;
    }
    
    // Write completion date changes from preview
    if (order._previewCompletionChanged) {
      order._previewCompletionChanged = false;
    }
  });
}

### COMPLETION DATE CELL — PREVIEW

```jsx
function PreviewCompletionDateCell({ order, previewOrders, onChange }) {
  const [showOverrideDialog, setShowOverrideDialog] = useState(false);
  const [pendingDate, setPendingDate] = useState(null);
  
  function handleCompletionClick() {
    // Open date picker
    // When date is selected, show override dialog before applying
  }
  
  function handleDateSelected(newDate) {
    // Check restrictions
    const constraints = getCompletionDateConstraints(order);
    if (constraints.minDate && newDate < constraints.minDate) return;
    if (constraints.maxDate && newDate > constraints.maxDate) return;
    
    setPendingDate(newDate);
    setShowOverrideDialog(true);
  }
  
  function handleOverrideProceed() {
    order.completion_date = formatDateFull(pendingDate);
    order.completion_time = formatTime(pendingDate);
    order._completionOverridden = true;
    order._previewCompletionChanged = true;
    
    setShowOverrideDialog(false);
    setPendingDate(null);
    
    // Recalculate downstream orders
    recalculateCompletionDates(previewOrders);
    onChange();
  }
  
  function handleOverrideCancel() {
    setShowOverrideDialog(false);
    setPendingDate(null);
  }
  
  // Show override indicator if overridden
  const isOverridden = order._completionOverridden;
  
  return (
    <td className={`data-cell editable-cell ${isOverridden ? 'overridden' : ''}`}>
      <div onClick={handleCompletionClick} style={{ cursor: 'pointer' }}>
        {order.completion_date ? (
          <div>
            <span className="text-content">
              {order.completion_date}
            </span>
            {order.completion_time && (
              <div style={{ fontSize: '10px', color: '#6b7280' }}>
                {order.completion_time}
              </div>
            )}
            {isOverridden && (
              <div style={{ fontSize: '10px', color: '#ea580c' }}>
                Overridden ✏
              </div>
            )}
          </div>
        ) : (
          <span className="placeholder-text">Set completion</span>
        )}
      </div>
      
      {showOverrideDialog && (
        <CompletionOverrideDialog
          order={order}
          onCancel={handleOverrideCancel}
          onProceed={handleOverrideProceed}
        />
      )}
    </td>
  );
}
```

## 5. CASCADE EXAMPLE

```
Order 1: Start Date = April 2, 08:00 AM | Prod: 4.19 hrs | CO: 0.17 hrs
  → Completion = April 2, 12:22 PM

Order 2: No start date (cascaded from Order 1)
  → Start = April 2, 12:22 PM | Prod: 2.30 hrs | CO: 0.17 hrs
  → Completion = April 2, 02:50 PM

Order 3: Start Date = April 3, 08:00 AM (resets cascade)
  → Completion = April 3, 12:22 PM

Order 4: No start date (cascaded from Order 3)
  → Start = April 3, 12:22 PM | Prod: 1.05 hrs | CO: 0.17 hrs
  → Completion = April 3, 01:35 PM

Order 5: Completion overridden to April 4, 06:00 PM
  → Keeps April 4, 06:00 PM (no recalculate)
  → Next order cascades from April 4, 06:00 PM

Order 6: No start date (cascaded from Order 5's override)
  → Start = April 4, 06:00 PM | Prod: 0.42 hrs | CO: 0.17 hrs
  → Completion = April 4, 06:35 PM
```
