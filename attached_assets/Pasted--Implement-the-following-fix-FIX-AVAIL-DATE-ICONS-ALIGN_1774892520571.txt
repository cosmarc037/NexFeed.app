```javascript
function getAvailDateIcon(order, n10dData) {
  // Skip orders without prio replenish or safety stocks
  if (!order.avail_date) return null;
```

```javascript
function determineProductStatus(product, dailyColumns, today) {
  const dfl = product.due_for_loading;
  const inventory = product.inventory;
  
  // CORRECT: DFL >= Inventory from Day 0 = Critical
  if (dfl >= inventory) {
    return 'Critical';
  }
  
  // Calculate when cumulative sum exceeds inventory
  let cumulativeSum = dfl;
  let breachDay = null;
  
  for (let i = 0; i < dailyColumns.length; i++) {
    const dayValue = parseFloat(product[dailyColumns[i].key]) || 0;
    cumulativeSum += dayValue;
    
    if (cumulativeSum >= inventory && breachDay === null) {
      breachDay = i;
      break;
    }
  }
  
  if (breachDay === null) {
    return 'Sufficient';  // Never breaches in 10 days
  }
  
  const breachDate = new Date(dailyColumns[breachDay].date);
  const diffMs = breachDate.getTime() - today.getTime();
  const daysUntilBreach = Math.ceil(diffMs / (1000 * 60 * 60 * 24));
  
  if (daysUntilBreach <= 3) {
    return 'Urgent';      // Breaches within 3 days
  }
  
  return 'Monitor';       // Breaches within 4-10 days
}
```
