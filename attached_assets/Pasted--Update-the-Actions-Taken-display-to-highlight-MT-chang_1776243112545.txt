```javascript
function generateFallbackExplanation(entry) {
  if (entry.type === 'combined') {
    const allSameLine = entry.fromLines.every(l => l === entry.toLine);
    const bestLS = entry.lineScores.find(ls => ls.line === entry.toLine);
    
    const allLineDetails = entry.lineScores
      .map(ls => `${getLineShortName(ls.line)}: ${ls.totalMTBefore.toFixed(1)} MT ÷ ${ls.runRate} MT/hr = ${ls.queueTimeBefore.toFixed(2)} hrs`)
      .join('; ');
    
    if (allSameLine) {
      return `${entry.ordersCount} orders of **${entry.product}** already on **${getLineShortName(entry.toLine)}** were combined into a single **${entry.totalVolume} MT** run. Queue times evaluated — ${allLineDetails}. **${getLineShortName(entry.toLine)}** was confirmed as optimal with the lowest queue time of **${entry.bestLineReason.queueTime.toFixed(2)} hrs**. ${getLineShortName(entry.toLine)} MT changed from **${entry.bestLineReason.totalMTBefore.toFixed(1)} MT → ${entry.bestLineReason.totalMTAfter.toFixed(1)} MT** (${entry.bestLineReason.queueTimeAfter.toFixed(2)} hrs). This eliminates **${entry.changeoversSaved} changeover${entry.changeoversSaved > 1 ? 's' : ''}**, saving **${formatTimeShift(entry.timeSaved)}**.`;
    } else {
      const otherLines = [...new Set(entry.fromLines.filter(l => l !== entry.toLine))];
      
      // Show MT changes for source lines
      const sourceChanges = otherLines.map(sourceLine => {
        const sourceScore = entry.lineScores.find(ls => ls.line === sourceLine);
        if (sourceScore) {
          return `${getLineShortName(sourceLine)}: **${(sourceScore.totalMTAfter + entry.totalVolume).toFixed(1)} MT → ${sourceScore.totalMTAfter.toFixed(1)} MT**`;
        }
        return '';
      }).filter(Boolean).join('; ');
      
      return `${entry.ordersCount} orders of **${entry.product}** were combined into **${entry.totalVolume} MT** on **${getLineShortName(entry.toLine)}**, pulling from ${otherLines.map(l => getLineShortName(l)).join(', ')}. Queue times evaluated — ${allLineDetails}. **${getLineShortName(entry.toLine)}** had the lowest queue time of **${entry.bestLineReason.queueTime.toFixed(2)} hrs**. ${getLineShortName(entry.toLine)} MT: **${entry.bestLineReason.totalMTBefore.toFixed(1)} MT → ${entry.bestLineReason.totalMTAfter.toFixed(1)} MT** (${entry.bestLineReason.queueTimeAfter.toFixed(2)} hrs). ${sourceChanges ? 'Source lines: ' + sourceChanges + '.' : ''} This eliminates **${entry.changeoversSaved} changeover${entry.changeoversSaved > 1 ? 's' : ''}**, saving **${formatTimeShift(entry.timeSaved)}**.`;
    }
  }
  
  if (entry.type === 'moved') {
    const fromScore = entry.lineScores.find(ls => ls.line === entry.fromLine);
    const toScore = entry.lineScores.find(ls => ls.line === entry.toLine);
    
    // Calculate original MT on source line (before removal)
    const sourceOriginalMT = (fromScore?.totalMTBefore || 0) + entry.volume;
    const sourceAfterMT = fromScore?.totalMTBefore || 0;
    const destBeforeMT = toScore?.totalMTBefore || 0;
    const destAfterMT = toScore?.totalMTAfter || 0;
    
    const allLineDetails = entry.lineScores
      .map(ls => `${getLineShortName(ls.line)}: ${ls.totalMTBefore.toFixed(1)} MT ÷ ${ls.runRate} MT/hr = ${ls.queueTimeBefore.toFixed(2)} hrs`)
      .join('; ');
    
    return `**${entry.product}** (**${entry.volume} MT**) was moved from **${getLineShortName(entry.fromLine)}** to **${getLineShortName(entry.toLine)}**. Queue times evaluated — ${allLineDetails}. **${getLineShortName(entry.toLine)}** had the lowest queue time of **${toScore?.queueTimeBefore.toFixed(2)} hrs**. ${getLineShortName(entry.fromLine)} decreased from **${sourceOriginalMT.toFixed(1)} MT → ${sourceAfterMT.toFixed(1)} MT**. ${getLineShortName(entry.toLine)} increased from **${destBeforeMT.toFixed(1)} MT → ${destAfterMT.toFixed(1)} MT** (${toScore?.queueTimeAfter.toFixed(2)} hrs).`;
  }
  
  return 'Action details unavailable.';
}
```

## 3. RENDER BOLD TEXT IN EXPLANATIONS

The AI and fallback both use **text** for bold. Parse and render:

```jsx
function ActionExplanation({ text }) {
  // Parse **bold** markers into <strong> elements
  const parts = text.split(/(\*\*[^*]+\*\*)/g);
  
  return (
    <span>
      {parts.map((part, index) => {
        if (part.startsWith('**') && part.endsWith('**')) {
          const boldText = part.slice(2, -2);
          return <strong key={index} className="action-bold">{boldText}</strong>;
        }
        return <span key={index}>{part}</span>;
      })}
    </span>
  );
}
```

## 4. MT CHANGE BADGE COMPONENT

```jsx
function MTChangeBadge({ line, beforeMT, afterMT, direction }) {
  const diff = afterMT - beforeMT;
  const diffFormatted = diff > 0 ? `+${diff.toFixed(1)}` : diff.toFixed(1);
  
  return (
    <div className={`mt-change-badge mt-change-${direction}`}>
      <span className="mt-change-line">{getLineShortName(line)}</span>
      <span className="mt-change-values">
        <span className="mt-change-before">{beforeMT.toFixed(1)} MT</span>
        <span className="mt-change-arrow">→</span>
        <span className="mt-change-after">{afterMT.toFixed(1)} MT</span>
      </span>
      <span className="mt-change-diff">({diffFormatted} MT)</span>
    </div>
  );
}
```
