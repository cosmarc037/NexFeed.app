**Date column parsing:**
- The uploaded Excel file has columns with date-formatted headers like "Mar-18", "26-Feb", "1-Mar", etc.

- The parser must:
  1. Read ALL column headers from the uploaded file.
  2. Attempt to parse each header as a date.
  3. Any header that successfully parses as a date → treat as a daily demand column.
  4. Store both the parsed date object AND the original header text.
  5. Display the original header text in the preview table.
  6. Use the parsed date object for target date calculations.

**Common date header formats to handle:**
- "Mar-18", "Mar-19" (Month abbreviation + day)
- "26-Feb", "1-Mar" (Day + Month abbreviation)
- "3/18", "3/19" (M/DD numeric)
- "2026-03-18" (ISO format)
- Any Excel date serial number that gets parsed as a date string

**Parser pseudocode:**
```
function parseColumns(headers) {
  const fixedColumns = ['Material Code', 'Category', 'Item Description', 
    'Due for loading', 'Inventory', 'Bal. to produce'];
  
  const dateColumns = [];
  
  for (const header of headers) {
    // Skip known fixed columns (case-insensitive)
    if (fixedColumns.some(fc => fc.toLowerCase() === header.toLowerCase())) {
      continue;
    }
    
    // Try to parse as date
    const parsed = tryParseDate(header);
    if (parsed) {
      dateColumns.push({
        originalHeader: header,
        parsedDate: parsed,
        columnIndex: headers.indexOf(header)
      });
    }
  }
  
  // Sort date columns chronologically
  dateColumns.sort((a, b) => a.parsedDate - b.parsedDate);
  
  return { fixedColumns, dateColumns };
}

    const newCumulative = cumulative + dayValue;
    
    if (newCumulative >= inventory) {
      // This day meets/exceeds inventory — red zone starts here
      // Target date = the day BEFORE this
      if (previousDate) {

      } else {

    previousDate = day.date;

  };

## FIX 2: AUTO-SEQUENCE — CORRECT SEQUENCING LOGIC

The auto-sequence must NOT re-sort all orders by effective date. Instead, it must:

1. KEEP dated orders (actual avail dates) in their current positions.
2. ONLY move non-dated orders (prio replenish, safety stocks) to their correct chronological position based on their inferred target dates from Next 10 Days.

---

### THE CORRECT AUTO-SEQUENCE ALGORITHM

```
function autoSequence(orders, next10DaysData) {
  
  // STEP 1: Separate dated and non-dated orders
  const datedOrders = []; // orders with actual avail dates — KEEP positions
  const nonDatedOrders = []; // prio replenish, safety stocks — TO BE INSERTED
  
  for (const order of orders) {
    if (isDateValue(order.avail_date)) {
      datedOrders.push({ ...order, effectiveDate: parseDate(order.avail_date) });
    } else {
      // Look up target date from Next 10 Days
      let effectiveDate = null;
      let stockStatus = 'none';
      let note = null;
      
      if (next10DaysData) {
        const match = next10DaysData.find(
          s => s.material_code === order.material_code_fg
        );
        if (match && match.target_date) {
          effectiveDate = parseDate(match.target_date);
          stockStatus = 'needs_production';
          note = match.note || null;
        } else if (match && match.status === 'stock_sufficient') {
          stockStatus = 'stock_sufficient';
        }
      }
      
      nonDatedOrders.push({ 
        ...order, 
        effectiveDate, 
        stockStatus,
        note
      });
    }
  }
  
  // STEP 2: Sort dated orders by their current position (preserve order)
  // They are already in the order they appear in the interactive table.
  // DO NOT re-sort them.
  
  // STEP 3: Sort non-dated orders into categories
  const targetedNonDated = nonDatedOrders
    .filter(o => o.effectiveDate !== null)
    .sort((a, b) => a.effectiveDate - b.effectiveDate);
  
  const gapFillers = nonDatedOrders
    .filter(o => o.effectiveDate === null && o.stockStatus !== 'stock_sufficient');
  
  const stockSufficient = nonDatedOrders
    .filter(o => o.stockStatus === 'stock_sufficient');
  
  // STEP 4: Build the final sequence by INSERTING non-dated orders
  // among the dated orders based on target dates
  
  const result = [];
  let datedIndex = 0;
  let targetedIndex = 0;
  
  while (datedIndex < datedOrders.length || targetedIndex < targetedNonDated.length) {
    
    // Check if the next targeted non-dated order should go BEFORE the next dated order
    if (targetedIndex < targetedNonDated.length) {
      const nextTargeted = targetedNonDated[targetedIndex];
      
      if (datedIndex < datedOrders.length) {
        const nextDated = datedOrders[datedIndex];
        
        if (nextTargeted.effectiveDate < nextDated.effectiveDate) {
          // Non-dated order's target is BEFORE the next dated order
          // INSERT it before the dated order
          result.push(nextTargeted);
          targetedIndex++;
          continue;
        } else if (nextTargeted.effectiveDate.getTime() === nextDated.effectiveDate.getTime()) {
          // Same date — dated order goes first
          result.push(nextDated);
          datedIndex++;
          continue;
        } else {
          // Dated order goes first
          result.push(nextDated);
          datedIndex++;
          continue;
        }
      } else {
        // No more dated orders, add remaining targeted
        result.push(nextTargeted);
        targetedIndex++;
        continue;
      }
    }
    
    if (datedIndex < datedOrders.length) {
      result.push(datedOrders[datedIndex]);
      datedIndex++;
    }
  }
  
  // STEP 5: Add gap fillers after all dated/targeted orders
  result.push(...gapFillers);
  
  // STEP 6: Add stock sufficient at the very bottom
  result.push(...stockSufficient);
  
  // STEP 7: Re-number prio
  result.forEach((order, index) => {
    order.prio = index + 1;
  });
  
  return result;
}
