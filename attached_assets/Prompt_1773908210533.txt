### DATE HEADER PARSING FUNCTION

The date headers in the Excel file can come in various formats. Handle all common ones:

```javascript
function tryParseExcelDateHeader(headerStr, rawValue) {
  
  // If the raw value is a number, it might be an Excel date serial
  if (typeof rawValue === 'number') {
    // Excel date serial: days since 1900-01-01 (with the 1900 bug)
    const excelEpoch = new Date(1899, 11, 30); // Dec 30, 1899
    const date = new Date(excelEpoch.getTime() + rawValue * 86400000);
    if (!isNaN(date.getTime())) {
      return date;
    }
  }
  
  // Try common string formats
  const str = String(headerStr).trim();
  
  // Format: "Mar-18", "Feb-26", "Jan-05"
  const monthDayMatch = str.match(/^(Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)-(\d{1,2})$/i);
  if (monthDayMatch) {
    const monthNames = ['jan','feb','mar','apr','may','jun','jul','aug','sep','oct','nov','dec'];
    const month = monthNames.indexOf(monthDayMatch[1].toLowerCase());
    const day = parseInt(monthDayMatch[2]);
    // Assume current year or next year
    const year = new Date().getFullYear();
    const date = new Date(year, month, day);
    return date;
  }
  
  // Format: "26-Feb", "1-Mar"
  const dayMonthMatch = str.match(/^(\d{1,2})-(Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)$/i);
  if (dayMonthMatch) {
    const monthNames = ['jan','feb','mar','apr','may','jun','jul','aug','sep','oct','nov','dec'];
    const day = parseInt(dayMonthMatch[1]);
    const month = monthNames.indexOf(dayMonthMatch[2].toLowerCase());
    const year = new Date().getFullYear();
    const date = new Date(year, month, day);
    return date;
  }
  
  // Format: "3/18", "3/19", "12/25"
  const slashMatch = str.match(/^(\d{1,2})\/(\d{1,2})$/);
  if (slashMatch) {
    const month = parseInt(slashMatch[1]) - 1;
    const day = parseInt(slashMatch[2]);
    const year = new Date().getFullYear();
    const date = new Date(year, month, day);
    return date;
  }
  
  // Format: "3/18/2026", "03/18/2026"
  const fullSlashMatch = str.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})$/);
  if (fullSlashMatch) {
    const month = parseInt(fullSlashMatch[1]) - 1;
    const day = parseInt(fullSlashMatch[2]);
    const year = parseInt(fullSlashMatch[3]);
    const date = new Date(year, month, day);
    return date;
  }
  
  // Format: "2026-03-18" (ISO)
  const isoMatch = str.match(/^(\d{4})-(\d{2})-(\d{2})$/);
  if (isoMatch) {
    return new Date(parseInt(isoMatch[1]), parseInt(isoMatch[2]) - 1, parseInt(isoMatch[3]));
  }
  
  // Try native Date.parse as last resort
  const nativeParsed = new Date(str);
  if (!isNaN(nativeParsed.getTime())) {
    return nativeParsed;
  }
  
  // Also try: the raw Excel value might be a Date object already
  if (rawValue instanceof Date) {
    return rawValue;
  }
  
  return null; // Not a date
}
```
