### COMPLETE READINESS CHECK — ALL FIELDS

Here's the full readiness check with corrected validations:

```javascript
function getReadinessStatus(order) {
  const missing = [];
  
  // Production Orders — 10-digit number check
  if (!isValid10Digit(order.production_order_fg1)) {
    missing.push('Production Order FG1');
  }
  if (!isValid10Digit(order.production_order_sfg1)) {
    missing.push('Production Order SFG1');
  }
  if (!isValid10Digit(order.production_order_sfgpmx)) {
    // SFGPMX might be optional for some orders
    // Only flag if the field exists and is expected
    if (order.requires_sfgpmx) {
      missing.push('Production Order SFGPMX');
    }
  }
  
  // HA Info — auto-populated, should never be missing
  // Skip check — it mirrors Number of Batches
  
  // HA Prep — user confirms
  if (!order.ha_prep || order.ha_prep === 'Not confirmed') {
    missing.push('HA Prep');
  }
  
  // SCADA — user sets
  if (!order.scada || String(order.scada).trim() === '') {
    missing.push('SCADA');
  }
  
  // PV — user sets
  if (!order.pv || String(order.pv).trim() === '') {
    missing.push('PV');
  }
  
  // Completion Date — should be set (from cascade or manual)
  if (!order.completion_date) {
    missing.push('Completion Date');
  }
  
  // Material Code (FG)
  if (!order.material_code_fg || String(order.material_code_fg).trim() === '') {
    missing.push('Material Code (FG)');
  }
  
  // Batch Size
  if (!order.batch_size || order.batch_size <= 0) {
    missing.push('Batch Size');
  }
  
  // Return readiness status
  if (missing.length === 0) {
    return { ready: true, icon: 'green_check', missing: [] };
  } else {
    return { ready: false, icon: 'warning', missing: missing };
  }
}

function isValid10Digit(value) {
  if (!value) return false;
  const trimmed = String(value).trim();
  return /^\d{10}$/.test(trimmed);
}
```
