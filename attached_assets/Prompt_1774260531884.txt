### DYNAMIC COLSPAN

When columns are hidden, the colspan of the parent group header must decrease:

```javascript
function getGroupColspan(groupName, visibleColumns) {
  const groupMap = {
    'order_details': ['planned_fg', 'planned_sfg', 'planned_pmx', 'prod_fg1', 'prod_sfg1', 'prod_sfgpmx', 'mat_code_sfg', 'mat_code_fg', 'item_desc'],
    'production_parameters': ['form', 'volume', 'batch_size', 'batches', 'bags', 'prod_time'],
    'mixer_details': ['ha_info', 'ha_prep'],
    'operator': ['status'],
    'scheduling': ['start_date', 'start_time', 'avail_date', 'completion_date'],
    'notes': ['fpr_notes', 'special_remarks'],
    'bagger_details': ['threads', 'sacks', 'markings', 'tags'],
    'completion_tracking': ['end_date', 'history']
  };
  
  const groupCols = groupMap[groupName] || [];
  return groupCols.filter(col => visibleColumns.includes(col)).length;
}
```
