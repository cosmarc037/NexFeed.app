```javascript
function sortNonDatedOrders(nonDatedOrders, next10DaysData) {
  
  // Step 1: Enrich each order with N10D data
  const enrichedOrders = nonDatedOrders.map(order => {
    const stockMatch = next10DaysData.find(
      s => s.material_code === order.material_code_fg
    );
    
    let effectiveDate = null;
    let fulfillmentPct = 100; // default: fully fulfilled (lowest priority)
    let stockStatus = 'none';
    
    if (stockMatch) {
      if (stockMatch.target_date) {
        effectiveDate = new Date(stockMatch.target_date);
        stockStatus = 'needs_production';
      } else if (stockMatch.status === 'stock_sufficient') {
        stockStatus = 'stock_sufficient';
      }
      
      // Calculate fulfillment percentage
      if (stockMatch.inventory > 0) {
        fulfillmentPct = (stockMatch.due_for_loading / stockMatch.inventory) * 100;
      }
    }
    
    // DEBUG — log to verify data is present
    console.log('Order:', order.material_code_fg, 
      'Product:', order.item_description,
      'Target:', effectiveDate,
      'Fulfillment:', fulfillmentPct.toFixed(1) + '%',
      'Status:', stockStatus
    );
    
    return {
      ...order,
      effectiveDate,
      fulfillmentPct,
      stockStatus
    };
  });
  
  // Step 2: Sort
  enrichedOrders.sort((a, b) => {
    
    // Rule 0: Stock sufficient always last
    if (a.stockStatus === 'stock_sufficient' && b.stockStatus !== 'stock_sufficient') return 1;
    if (a.stockStatus !== 'stock_sufficient' && b.stockStatus === 'stock_sufficient') return -1;
    if (a.stockStatus === 'stock_sufficient' && b.stockStatus === 'stock_sufficient') return 0;
    
    // Rule 0b: Orders with date before orders without date
    if (a.effectiveDate && !b.effectiveDate) return -1;
    if (!a.effectiveDate && b.effectiveDate) return 1;
    
    // Both have effective dates
    if (a.effectiveDate && b.effectiveDate) {
      
      // Rule 1: Earlier date first
      const dateCompare = a.effectiveDate.getTime() - b.effectiveDate.getTime();
      if (dateCompare !== 0) return dateCompare;
      
      // Rule 2: SAME DATE — lower fulfillment % first (more urgent)
      // This is the critical tiebreaker
      const pctDiff = a.fulfillmentPct - b.fulfillmentPct;
      if (Math.abs(pctDiff) > 0.01) { // use small epsilon for floating point
        return pctDiff; // lower % comes first
      }
      
      // Rule 3: Same date, same % — higher volume first
      const volA = a.adjusted_volume || a.volume || 0;
      const volB = b.adjusted_volume || b.volume || 0;
      if (volA !== volB) {
        return volB - volA; // higher volume first
      }
      
      // All same — maintain original order
      return 0;
    }
    
    // Both have no dates — maintain original order
    return 0;
  });
  
  // DEBUG — log final sorted order
  console.log('SORTED ORDER:');
  enrichedOrders.forEach((o, i) => {
    console.log(i + 1, o.item_description, 
      'Date:', o.effectiveDate, 
      'Fulfillment:', o.fulfillmentPct.toFixed(1) + '%',
      'Volume:', o.adjusted_volume || o.volume
    );
  });
  
  return enrichedOrders;
}
