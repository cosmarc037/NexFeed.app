```javascript
function calculateInsertionIndex(newOrder, existingOrders) {
  const newDate = parseDate(newOrder.avail_date);
  
  // Filter to active orders only (not Done, not Cancelled)
  const activeOrders = existingOrders.filter(
    o => o.status !== 'Done' && o.status !== 'Cancel PO'
  );
  
  // CASE 1: New order has an actual date
  if (newDate) {
    
    for (let i = 0; i < activeOrders.length; i++) {
      const order = activeOrders[i];
      const orderDate = parseDate(order.avail_date);
      
      // If current order has no date (prio replenish, safety stocks)
      // AND the new order has a date — insert BEFORE non-dated orders
      if (!orderDate) {
        return i; // insert at this index
      }
      
      // If current order has a date AFTER the new order's date
      // Insert BEFORE this order
      if (orderDate > newDate) {
        return i; // insert at this index
      }
      
      // If same date — insert AFTER this order (continue loop)
    }
    
    // New order's date is after all dated orders
    // Find where non-dated orders start
    const lastDatedIndex = activeOrders.findLastIndex(
      o => parseDate(o.avail_date) !== null
    );
    return lastDatedIndex + 1;
  }
  
  // CASE 2: New order is non-dated (prio replenish, safety stocks)
  // Insert at the end of active orders
  return activeOrders.length;
}
```
