function cleanInsightOutput(text) {
  // Fix date formats

    return date.toLocaleDateString('en-US', {

    });

  );

  text = text.replace(/\n\n/g, '</p><p>');
  text = text.replace(/\n/g, '<br>');
  
  // Convert bullets
  text = text.replace(/^• /gm, '<li>');
  
  return text;
}
