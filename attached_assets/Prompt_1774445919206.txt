```javascript
function prepareStockInsightData(next10DaysData, today) {
  
  const products = next10DaysData.map(product => {
    const inventory = product.inventory || 0;
    const dueForLoading = product.due_for_loading || 0;
    const targetDate = product.target_date; // from the Target column in N10D table
    const status = product.status; // "Needs Production" or "Stock Sufficient"
    
    // Buffer calculation
    // Buffer% = (Inventory - Due for Loading) / Inventory × 100
    let bufferPct = 0;
    let bufferStatus = 'deficit';
    
    if (inventory > 0) {
      bufferPct = Math.round(((inventory - dueForLoading) / inventory) * 100);
      
      if (bufferPct > 50) bufferStatus = 'comfortable';
      else if (bufferPct > 20) bufferStatus = 'moderate';
      else if (bufferPct > 0) bufferStatus = 'low';
      else bufferStatus = 'deficit';
    } else {
      bufferPct = -100;
      bufferStatus = 'no_inventory';
    }
    
    // Days until target
    let daysUntilTarget = null;
    let urgencyCategory = 'none';
    
    if (targetDate) {
      const target = new Date(targetDate);
      const diffMs = target.getTime() - today.getTime();
      daysUntilTarget = Math.ceil(diffMs / (1000 * 60 * 60 * 24));
      
      if (daysUntilTarget <= 0) urgencyCategory = 'overdue';
      else if (daysUntilTarget <= 3) urgencyCategory = 'urgent';
      else if (daysUntilTarget <= 7) urgencyCategory = 'upcoming';
      else if (daysUntilTarget <= 10) urgencyCategory = 'extended';
    }
    
    return {
      name: product.item_description,
      dueForLoading: dueForLoading,
      inventory: inventory,
      bufferPct: bufferPct,
      bufferStatus: bufferStatus,
      targetDate: targetDate,
      daysUntilTarget: daysUntilTarget,
      urgencyCategory: urgencyCategory,
      status: status // "Needs Production" or "Stock Sufficient"
    };
  });
  
  // Categorize
  const overdue = products.filter(p => p.urgencyCategory === 'overdue');
  const urgent = products.filter(p => p.urgencyCategory === 'urgent');
  const upcoming = products.filter(p => p.urgencyCategory === 'upcoming');
  const extended = products.filter(p => p.urgencyCategory === 'extended');
  const stockSufficient = products.filter(p => p.status === 'Stock Sufficient');
  const needsProduction = products.filter(p => p.status === 'Needs Production');
  
  // Total deficit
  const totalDeficit = products.reduce((sum, p) => {
    const deficit = p.dueForLoading - p.inventory;
    return deficit > 0 ? sum + deficit : sum;
  }, 0);
  
  return {
    totalProducts: products.length,
    needsProductionCount: needsProduction.length,
    stockSufficientCount: stockSufficient.length,
    totalDeficit: Math.round(totalDeficit * 10) / 10,
    overdue: overdue.sort((a, b) => a.daysUntilTarget - b.daysUntilTarget),
    urgent: urgent.sort((a, b) => a.daysUntilTarget - b.daysUntilTarget),
    upcoming: upcoming.sort((a, b) => a.daysUntilTarget - b.daysUntilTarget),
    extended: extended.sort((a, b) => a.daysUntilTarget - b.daysUntilTarget),
    stockSufficient: stockSufficient
  };
}
```
