```javascript
function canOrderBeDiverted(order, shutdownLines, knowledgeBase) {
  // Step 1: Only check active orders on shutdown lines
  if (!shutdownLines.includes(order.line)) return false;
  if (order.status === 'Done' || order.status === 'Cancel PO') return false;
  
  // Step 2: Find product in Knowledge Base
  // Try multiple matching approaches
  const materialCode = order.material_code_fg;
  
  const product = knowledgeBase.find(p => {
    // Match by material_code_fg
    if (p.material_code_fg === materialCode) return true;
    if (p.fg_material_code === materialCode) return true;
    if (p['FG Material Code'] === materialCode) return true;
    
    // Try string comparison (in case one has leading zeros)
    if (String(p.material_code_fg) === String(materialCode)) return true;
    if (String(p.fg_material_code) === String(materialCode)) return true;
    
    return false;
  });
  
  // DEBUG — log to verify matching
  console.log(
    'Divert check:', 
    order.item_description, 
    'Material:', materialCode, 
    'KB match:', product ? 'FOUND' : 'NOT FOUND'
  );
  
  if (!product) {
    console.log('No KB match for:', materialCode);
    return false;
  }
  
  // Step 3: Check run rates on lines OUTSIDE the shutdown feedmill
  const rateFields = {
    'Line 1': ['l1_rate', 'L1 Rate', 'l1Rate'],
    'Line 2': ['l2_rate', 'L2 Rate', 'l2Rate'],
    'Line 3': ['l3_rate', 'L3 Rate', 'l3Rate'],
    'Line 4': ['l4_rate', 'L4 Rate', 'l4Rate'],
    'Line 5': ['l5_rate', 'L5 Rate', 'l5Rate'],
    'Line 6': ['l6_rate', 'L6 Rate', 'l6Rate'],
    'Line 7': ['l7_rate', 'L7 Rate', 'l7Rate'],
  };
  
  const availableLines = [];
  
  Object.entries(rateFields).forEach(([line, fields]) => {
    // Skip lines that are in the shutdown feedmill
    if (shutdownLines.includes(line)) return;
    
    // Try all possible field names for the rate
    let rate = 0;
    for (const field of fields) {
      if (product[field]) {
        rate = parseFloat(product[field]);
        if (!isNaN(rate) && rate > 0) break;
      }
    }
    
    if (rate > 0) {
      availableLines.push({ line, rate });
    }
  });
  
  // DEBUG — log available lines
  console.log(
    'Available lines for', order.item_description, ':',
    availableLines.length > 0 ? availableLines : 'NONE'
  );
  
  return availableLines.length > 0;
}
