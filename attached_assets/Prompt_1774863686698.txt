### 1. CORRECT UNDERSTANDING

| Field | What it actually means |
|---|---|
| Due for Loading (DFL) | The required amount of MT that needs to be produced/loaded for a product |
| Inventory | The current available stock / threshold |

### PRIORITIZATION RULE

| Condition | Meaning | Priority |
|---|---|---|
| DFL > Inventory | URGENT — required amount exceeds current stock, needs production NOW | ⬆ High |
| DFL < Inventory | Stock sufficient — current stock covers the requirement | ⬇ Low |

### 3. COMPLETION AND AVAIL DATE CALCULATION

Track TWO dates per product based on cumulative sum:

| Column | Definition |
|---|---|
| Completion | Last date BEFORE cumulative sum exceeds inventory — the safety date |
| Avail | Date WHEN cumulative sum exceeds inventory — the actual breach date |

### CALCULATION

```javascript
function calculateCompletionAndAvail(product, dailyColumns, today) {
  const inventory = product.inventory;
  const dfl = product.due_for_loading;
  
  let cumulativeSum = 0;
  let completionDate = null;
  let availDate = null;
  
  // Check if DFL alone already exceeds inventory (Day 0 — critical)
  if (dfl >= inventory) {
    completionDate = null;     // No safe date exists
    availDate = today;         // Already breached, avail is today
    return { completionDate, availDate };
  }
  
  // Start cumulating from DFL as base
  cumulativeSum = dfl;
  
  for (let i = 0; i < dailyColumns.length; i++) {
    const dayValue = parseFloat(product[dailyColumns[i].key]) || 0;
    const previousSum = cumulativeSum;
    cumulativeSum += dayValue;
    
    // Check if cumulative sum exceeds inventory on this day
    if (cumulativeSum >= inventory && !availDate) {
      // Completion = previous day (last safe day)
      if (i > 0) {
        completionDate = dailyColumns[i - 1].date;
      } else {
        completionDate = null; // Exceeded on first day column
      }
      
      // Avail = this day (threshold breached)
      availDate = dailyColumns[i].date;
    }
  }
  
  // If never exceeded inventory within 10 days
  if (!availDate) {
    completionDate = dailyColumns[dailyColumns.length - 1].date;
    availDate = null; // Never breached — show "—"
  }
  
  return { completionDate, availDate };
}
```

### EDGE CASES

| Scenario | Completion | Avail |
|---|---|---|
| DFL already > Inventory (Day 0) | null ("—") | Today's date |
| Exceeds on first day column (Day 1) | null ("—") | Day 1 date |
| Exceeds on Day 5 | Day 4 date | Day 5 date |
| Never exceeds within 10 days | Last day (Day 10) | null ("—") |

```

### 4. STATUS DETERMINATION — 4 LEVELS

```javascript
function determineProductStatus(product, dailyColumns, today) {
  const dfl = product.due_for_loading;
  const inventory = product.inventory;
  
  // CRITICAL: DFL already greater than inventory from Day 0
  if (dfl >= inventory) {
    return 'Critical';
  }
  
  // Calculate which day cumulative sum exceeds inventory
  let cumulativeSum = dfl;
  let breachDay = null;
  
  for (let i = 0; i < dailyColumns.length; i++) {
    const dayValue = parseFloat(product[dailyColumns[i].key]) || 0;
    cumulativeSum += dayValue;
    
    if (cumulativeSum >= inventory && breachDay === null) {
      breachDay = i;
      break;
    }
  }
  
  // No breach found in 10 days
  if (breachDay === null) {
    return 'Sufficient';
  }
  
  // Calculate days from today to breach date
  const breachDate = new Date(dailyColumns[breachDay].date);
  const diffMs = breachDate.getTime() - today.getTime();
  const daysUntilBreach = Math.ceil(diffMs / (1000 * 60 * 60 * 24));
  
  // URGENT: Will breach within 3 days
  if (daysUntilBreach <= 3) {
    return 'Urgent';
  }
  
  // MONITOR: Will breach within 4-10 days
  if (daysUntilBreach <= 10) {
    return 'Monitor';
  }
  
  return 'Sufficient';
}
```

### STATUS RULES

| Status | Condition | Color |
|---|---|---|
| Critical | DFL ≥ Inventory from Day 0 (before any daily values) | Red |
| Urgent | Cumulative sum will exceed inventory within 3 days | Orange/amber |
| Monitor | Cumulative sum will exceed inventory within 4-10 days | Yellow |
| Sufficient | Never exceeds inventory within 10 days | Green |

function getSSLPriority(product) {
  const dfl = product.due_for_loading;
  const inventory = product.inventory;
  
  // Higher DFL-to-Inventory ratio = more urgent = higher priority
  const ratio = inventory > 0 ? dfl / inventory : Infinity;
  
  return {
    ratio: ratio,
    isUrgent: dfl > inventory,
    status: product.status,
    priority: ratio
  };
}

// Sort: Critical first, then Urgent, Monitor, Sufficient

const statusOrder = { 'Critical': 0, 'Urgent': 1, 'Monitor': 2, 'Sufficient': 3 };

products.sort((a, b) => {
  const statusDiff = statusOrder[a.status] - statusOrder[b.status];
  if (statusDiff !== 0) return statusDiff;

### BUFFER CALCULATION

```javascript
function calculateBuffer(dfl, inventory) {
  if (!inventory || inventory === 0) {
    return { buffer: -100, status: 'no_inventory' };
  }
  
  // Positive = stock covers DFL, Negative = deficit
  const buffer = ((inventory - dfl) / inventory) * 100;
  
  return {
    buffer: Math.round(buffer * 10) / 10,
    status: buffer > 20 ? 'sufficient' :
            buffer > 0 ? 'low_buffer' :
            'deficit'
  };
}
