### 2. AI SMART INSIGHT — PER PRODUCT

The AI generates a brief 2-3 sentence insight about the product based on its data. This should be pre-generated or cached, NOT called on every hover.

### PRE-GENERATE INSIGHTS

When the N10D data loads or refreshes, pre-generate a short insight for each product:

```javascript
function generateProductInsight(product, today) {
  const dfl = product.due_for_loading;
  const inventory = product.inventory;
  const status = product.status;
  const completionDate = product.completionDate;
  const availDate = product.availDate;
  const balToProd = product.balanceToProduce;
  const buffer = inventory > 0 ? ((inventory - dfl) / inventory * 100).toFixed(1) : -100;
  const ratio = inventory > 0 ? (dfl / inventory).toFixed(2) : '∞';
  
  // Calculate days until avail
  let daysUntilAvail = null;
  if (availDate) {
    const avail = new Date(availDate);
    daysUntilAvail = Math.ceil((avail - today) / (1000 * 60 * 60 * 24));
  }
  
  // Build insight based on status
  let insight = '';
  
  switch (status) {
    case 'Critical':
      insight = `This product is in Critical status — the required production volume (${dfl} MT) already exceeds current inventory (${inventory} MT). Immediate production is needed to prevent stockout. The deficit stands at ${Math.abs(balToProd).toFixed(1)} MT.`;
      break;
    
    case 'Urgent':
      insight = `This product is in Urgent status. Cumulative demand will exceed inventory by ${formatDate(availDate)}, approximately ${daysUntilAvail} days from now. With a buffer of only ${buffer}%, production should be scheduled immediately to avoid a shortfall.`;
      break;
    
    case 'Monitor':
      insight = `This product is in Monitor status. Demand will exceed inventory by ${formatDate(availDate)}, giving approximately ${daysUntilAvail} days to schedule production. Current buffer of ${buffer}% provides a window, but production should be planned within the coming week.`;
      break;
    
    case 'Sufficient':
      insight = `This product is Stock Sufficient. Current inventory (${inventory} MT) comfortably covers the required production volume (${dfl} MT) throughout the 10-day window. No immediate production action is needed.`;
      break;
  }
  
  return insight;
}
```

### 3. CUMULATIVE SUM ON DAY CELL HOVER

When the user hovers over a specific day cell (not the product name area), show the cumulative sum from DFL up to that day.

### BEHAVIOR

- Hover over **product name / details area** → full tooltip (Order Details + Smart Insight)
- Hover over **specific day cell** → small cumulative sum tooltip on that cell

### DAY CELL TOOLTIP

```

### CALCULATION

```javascript
function getCumulativeSumAtDay(product, dayIndex, dailyColumns) {
  const dfl = product.due_for_loading;
  let cumulative = dfl; // Start with DFL
  
  for (let i = 0; i <= dayIndex; i++) {
    const dayValue = parseFloat(product[dailyColumns[i].key]) || 0;
    cumulative += dayValue;
  }
  
  return cumulative;
}
```

### DAY CELL TOOLTIP CONTENT

```javascript
function getDayCellTooltip(product, dayIndex, dailyColumns) {
  const dfl = product.due_for_loading;
  const inventory = product.inventory;
  const dayValue = parseFloat(product[dailyColumns[dayIndex].key]) || 0;
  const cumulative = getCumulativeSumAtDay(product, dayIndex, dailyColumns);
  const remaining = inventory - cumulative;
  const usagePct = inventory > 0 
    ? Math.min(100, Math.round((cumulative / inventory) * 100)) 
    : 100;
  
  return {
    date: dailyColumns[dayIndex].date,
    dayValue: dayValue,
    cumulative: cumulative,
    inventory: inventory,
    remaining: remaining,
    usagePct: usagePct,
    isBreached: cumulative >= inventory
  };
}
```

### 4. TOOLTIP TRIGGER LOGIC

```jsx
function N10DTableRow({ product, dailyColumns, today }) {
  const [hoverType, setHoverType] = useState(null); // 'product' or 'day'
  const [hoveredDay, setHoveredDay] = useState(null);
  const [tooltipPos, setTooltipPos] = useState({ x: 0, y: 0 });
  const timeoutRef = useRef(null);
  
  function handleProductHover(e) {
    clearTimeout(timeoutRef.current);
    timeoutRef.current = setTimeout(() => {
      setHoverType('product');
      setTooltipPos({ x: e.clientX + 12, y: e.clientY + 12 });
    }, 300);
  }
  
  function handleDayCellHover(e, dayIndex) {
    clearTimeout(timeoutRef.current);
    timeoutRef.current = setTimeout(() => {
      setHoverType('day');
      setHoveredDay(dayIndex);
      setTooltipPos({ x: e.clientX, y: e.clientY - 10 });
    }, 200); // Slightly faster for day cells
  }
  
  function handleMouseLeave() {
    clearTimeout(timeoutRef.current);
    setHoverType(null);
    setHoveredDay(null);
  }
  
  return (
    <>
      <tr onMouseLeave={handleMouseLeave}>
        {/* Product detail cells — trigger product tooltip */}
        <td onMouseEnter={handleProductHover}>{product.material_code}</td>
        <td onMouseEnter={handleProductHover}>{product.category}</td>
        <td onMouseEnter={handleProductHover}>{product.item_description}</td>
        <td onMouseEnter={handleProductHover}>{product.due_for_loading}</td>
        
        {/* Day cells — trigger day cumulative tooltip */}
        {dailyColumns.map((col, index) => (
          <td 
            key={col.key}
            onMouseEnter={(e) => handleDayCellHover(e, index)}
            onMouseMove={(e) => setTooltipPos({ x: e.clientX, y: e.clientY - 10 })}
          >
            {product[col.key]}
          </td>
        ))}
        
        {/* Tracking cells — trigger product tooltip */}
        <td onMouseEnter={handleProductHover}>{product.inventory}</td>
        <td onMouseEnter={handleProductHover}>{product.balanceToProduce}</td>
        <td onMouseEnter={handleProductHover}>{product.completionDate}</td>
        <td onMouseEnter={handleProductHover}>{product.availDate}</td>
        <td onMouseEnter={handleProductHover}>{product.status}</td>
      </tr>
      
      {/* Product tooltip */}
      {hoverType === 'product' && (
        <ProductTooltip product={product} position={tooltipPos} />
      )}
      
      {/* Day cumulative tooltip */}
      {hoverType === 'day' && hoveredDay !== null && (
        <DayCumulativeTooltip 
          product={product} 
          dayIndex={hoveredDay}
          dailyColumns={dailyColumns}
          position={tooltipPos}
        />
      )}
    </>
  );
}
```
