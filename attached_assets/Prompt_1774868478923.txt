function getAutoSequenceTooltipData(order, n10dData, today) {
  // Find matching product in N10D data by material code
  const product = n10dData.find(
    p => p.material_code_fg === order.material_code_fg ||
         String(p.material_code_fg) === String(order.material_code_fg)
  );
  
  if (!product) {
    return null; // No N10D data for this product
  }
  
  const dfl = product.due_for_loading;
  const inventory = product.inventory;
  
  const ratio = inventory > 0 ? (dfl / inventory).toFixed(2) : '∞';
  const buffer = inventory > 0 
    ? (((inventory - dfl) / inventory) * 100).toFixed(1) 
    : '-100.0';
  
  const status = product.status; // Critical, Urgent, Monitor, Sufficient
  const completionDate = product.completionDate;
  const availDate = product.availDate;
  const balToProd = product.balanceToProduce;
  
  const insight = getInsight(product.material_code_fg); // Cached insight
  
  return {
    name: order.item_description,
    dfl: dfl,
    inventory: inventory,
    ratio: ratio,
    buffer: buffer,
    status: status,
    completionDate: completionDate,
    availDate: availDate,
    balToProd: balToProd,
    insight: insight
  };
}

## 2. FIX HOVER TOOLTIP BEING CUT OFF

The tooltip is being clipped by the auto-sequence preview container's overflow. Fix so the tooltip is always fully visible.

### THE PROBLEM

The auto-sequence preview container likely has:
- `overflow: hidden` or `overflow: auto`
- A fixed height that clips content
- The tooltip is positioned inside the container, so it gets cut off at the edges

### THE FIX

Render the tooltip using a **portal** so it's placed at the document body level, outside of any overflow-hidden containers.

```jsx
import { createPortal } from 'react-dom';

function TooltipPortal({ children }) {
  return createPortal(children, document.body);
}

// Usage
function AutoSequenceRow({ order, n10dData, today }) {
  const [showTooltip, setShowTooltip] = useState(false);
  const [tooltipPos, setTooltipPos] = useState({ x: 0, y: 0 });
  const timeoutRef = useRef(null);
  
  function handleMouseEnter(e) {
    timeoutRef.current = setTimeout(() => {
      setShowTooltip(true);
    }, 300);
  }
  
  function handleMouseMove(e) {
    setTooltipPos({ x: e.clientX + 12, y: e.clientY + 12 });
  }
  
  function handleMouseLeave() {
    clearTimeout(timeoutRef.current);
    setShowTooltip(false);
  }
  
  return (
    <>
      <tr
        onMouseEnter={handleMouseEnter}
        onMouseMove={handleMouseMove}
        onMouseLeave={handleMouseLeave}
      >
        {/* ... row cells ... */}
      </tr>
      
      {showTooltip && (
        <TooltipPortal>
          <ProductTooltip
            data={getAutoSequenceTooltipData(order, n10dData, today)}
            position={tooltipPos}
          />
        </TooltipPortal>
      )}
    </>
  );
}
```

### TOOLTIP POSITIONING — VIEWPORT BOUNDARY CHECK

Ensure the tooltip never goes outside the viewport:

```javascript
function getTooltipPosition(mouseX, mouseY, tooltipWidth, tooltipHeight) {
  const viewportWidth = window.innerWidth;
  const viewportHeight = window.innerHeight;
  const padding = 16;
  
  let x = mouseX + 12;
  let y = mouseY + 12;
  
  // If tooltip would go off right edge
  if (x + tooltipWidth + padding > viewportWidth) {
    x = mouseX - tooltipWidth - 12;
  }
  
  // If tooltip would go off bottom edge
  if (y + tooltipHeight + padding > viewportHeight) {
    y = mouseY - tooltipHeight - 12;
  }
  
  // If tooltip would go off left edge
  if (x < padding) {
    x = padding;
  }
  
  // If tooltip would go off top edge
  if (y < padding) {
    y = padding;
  }
  
  return { x, y };
}
```

```jsx
function ProductTooltip({ data, position }) {
  if (!data) return null;
  
  const ratioColor = parseFloat(data.ratio) > 1.0 ? '#fca5a5' :
                     parseFloat(data.ratio) > 0.8 ? '#fde047' : '#86efac';
  
  const bufferColor = parseFloat(data.buffer) < 0 ? '#fca5a5' :
                      parseFloat(data.buffer) < 20 ? '#fde047' : '#86efac';
  
  const statusColors = {
    'Critical': '#fca5a5',
    'Urgent': '#fdba74',
    'Monitor': '#fde047',
    'Sufficient': '#86efac'
  };
  
  const balColor = data.balToProd < 0 ? '#fca5a5' : '#86efac';
  
  return (
    <div className="product-tooltip" style={{
      position: 'fixed',
      left: position.x,
      top: position.y,
      zIndex: 99999
    }}>
      {/* Header — Product Name */}
      <div className="tooltip-header">
        {data.name}
      </div>
      
      {/* Body — Two Columns */}
      <div className="tooltip-body">
        
        {/* Left: Order Details */}
        <div className="tooltip-left">
          <div className="tooltip-grid">
            <span className="tooltip-label">DFL:</span>
            <span className="tooltip-value">{data.dfl} MT</span>
            
            <span className="tooltip-label">Inventory:</span>
            <span className="tooltip-value">{data.inventory} MT</span>
            
            <span className="tooltip-label">Ratio:</span>
            <span className="tooltip-value" style={{ color: ratioColor }}>
              {data.ratio}
            </span>
            
            <span className="tooltip-label">Buffer:</span>
            <span className="tooltip-value" style={{ color: bufferColor }}>
              {data.buffer}%
            </span>
            
            <span className="tooltip-label">Status:</span>
            <span className="tooltip-value" style={{ color: statusColors[data.status] }}>
              {data.status}
            </span>
          </div>
          
          <div className="tooltip-separator" />
          
          <div className="tooltip-grid">
            <span className="tooltip-label">Completion:</span>
            <span className="tooltip-value">{data.completionDate || '—'}</span>
            
            <span className="tooltip-label">Avail:</span>
            <span className="tooltip-value">{data.availDate || '—'}</span>
            
            <span className="tooltip-label">Bal. to Prod:</span>
            <span className="tooltip-value" style={{ color: balColor }}>
              {data.balToProd} MT
            </span>
          </div>
        </div>
        
        {/* Right: Smart Insight */}
        <div className="tooltip-right">
          <div className="tooltip-insight-header">✨ Smart Insight</div>
          <div className="tooltip-insight-text">
            {data.insight}
          </div>
        </div>
        
      </div>
    </div>
  );
}
```
