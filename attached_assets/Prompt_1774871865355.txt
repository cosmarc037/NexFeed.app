The product tooltip is too far from the hovered row. It should appear right next to the mouse cursor — the same way the day cell cumulative tooltip works (see 2nd snip).

### THE FIX

Use mouse position directly, not row centering. Same approach as the day cell tooltip which already works correctly.

```javascript
function N10DTableRow({ product, dailyColumns, today }) {
  const [showTooltip, setShowTooltip] = useState(false);
  const [tooltipPos, setTooltipPos] = useState({ x: 0, y: 0 });
  const timeoutRef = useRef(null);
  
  function handleProductHover(e) {
    clearTimeout(timeoutRef.current);
    timeoutRef.current = setTimeout(() => {
      const pos = clampToViewport(
        e.clientX + 12,
        e.clientY + 12,
        480,  // tooltip width
        220   // tooltip height
      );
      setTooltipPos(pos);
      setShowTooltip(true);
    }, 300);
  }
  
  function handleProductMouseMove(e) {
    if (showTooltip) {
      const pos = clampToViewport(
        e.clientX + 12,
        e.clientY + 12,
        480,
        220
      );
      setTooltipPos(pos);
    }
  }
  
  function handleMouseLeave() {
    clearTimeout(timeoutRef.current);
    setShowTooltip(false);
  }
  
  // ... render
}

### CLAMP TO VIEWPORT — PREVENT CLIPPING

```javascript
function clampToViewport(x, y, tooltipWidth, tooltipHeight) {
  const viewportWidth = window.innerWidth;
  const viewportHeight = window.innerHeight;
  const padding = 8;
  
  // If tooltip goes off right edge, flip to left of cursor
  if (x + tooltipWidth + padding > viewportWidth) {
    x = x - tooltipWidth - 24; // flip to left side of cursor
  }
  
  // If tooltip goes off bottom edge, flip above cursor
  if (y + tooltipHeight + padding > viewportHeight) {
    y = y - tooltipHeight - 24; // flip above cursor
  }
  
  // Ensure not off left edge
  if (x < padding) x = padding;
  
  // Ensure not off top edge
  if (y < padding) y = padding;
  
  return { x, y };
}
```

| Property | Old (wrong) | New (correct) |

| Y position | Above row, centered | Mouse cursor Y + 12px |
| Follows mouse | No | Yes — updates on mouse move |
| Flip behavior | Center vertically | Flips to left/above if near edge |
