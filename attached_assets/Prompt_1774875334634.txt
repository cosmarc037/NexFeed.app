### 2. AI INSIGHT GENERATION — ON N10D

Generate when:
- N10D file is uploaded
- User clicks "Re-apply to Existing Orders"

```javascript
async function generateAndCacheInsights(n10dProducts, allOrders, today) {
  const insightCache = {};
  
  // === PART A: Products WITH N10D data (stock-based insight) ===
  const n10dPrompt = `
Generate a brief 1-2 sentence production insight for each product.
Use the EXACT data provided. Be specific with numbers and dates.
Use full month names (e.g., "April 5, 2026" not "Apr 5").
Maximum 2 sentences per product.
Be actionable — tell the planner what to do.

Products:
${n10dProducts.map(p => `
[${p.material_code_fg}] ${p.item_description}
DFL: ${p.due_for_loading} MT | Inventory: ${p.inventory} MT
Status: ${p.status} | Buffer: ${p.buffer}%
Completion: ${p.completionDate || 'N/A'} | Avail: ${p.availDate || 'N/A'}
Bal to Prod: ${p.balanceToProduce} MT
`).join('\n')}

Output format — one per line:
[material_code]: [1-2 sentence insight]
  `;
  
  const n10dResponse = await callAI(n10dPrompt);
  // Parse and store in cache
  parseInsightResponse(n10dResponse, insightCache);
  
  // === PART B: Dated orders WITHOUT N10D data (deadline-based insight) ===
  const datedOrders = allOrders.filter(order => {
    const hasN10D = n10dProducts.some(
      p => String(p.material_code_fg) === String(order.material_code_fg)
    );
    return !hasN10D && order.avail_date && 
           order.avail_date !== 'prio replenish' && 
           order.avail_date !== 'safety stocks';
  });
  
  if (datedOrders.length > 0) {
    const datedPrompt = `
Generate a brief 1-2 sentence production insight for each order.
These are dated MTO orders with specific deadlines.
Focus on deadline proximity and production urgency.
Use full month names. Maximum 2 sentences.

Orders:
${datedOrders.map(o => `
[${o.material_code_fg}] ${o.item_description}
Volume: ${o.volume} MT | Avail Date: ${formatDateFull(o.avail_date)}
Line: ${o.line} | Status: ${o.status}
Production Time: ${o.production_time} hrs
`).join('\n')}

Output format — one per line:
[material_code]: [1-2 sentence insight]
    `;
    
    const datedResponse = await callAI(datedPrompt);
    parseInsightResponse(datedResponse, insightCache);
  }
  
  // === PART C: Orders with NO N10D and NO avail date ===
  // These get template-based insights (no AI call needed)
  allOrders.forEach(order => {
    if (!insightCache[order.material_code_fg]) {
      insightCache[order.material_code_fg] = getTemplateInsight(order);
    }
  });
  
  // Store in global state
  setInsightCache(insightCache);
}

function parseInsightResponse(response, cache) {
  const lines = response.split('\n').filter(l => l.trim());
  lines.forEach(line => {
    const match = line.match(/^\[?(\d+)\]?:\s*(.+)/);
    if (match) {
      cache[match[1]] = match[2].trim();
    }
  });
}

function getTemplateInsight(order) {
  const volume = order.volume || 0;
  const line = order.line || 'unassigned';
  return `This order is scheduled for ${volume} MT on ${line}. No stock level data available from Next 10 Days.`;
}
```

### 5. AVAIL DATE ICON — ALIGN STATUS WITH N10D

Fix instances where the icon shows a warning but the N10D status is Sufficient.

```javascript
function getAvailDateIcon(order, n10dData) {
  const product = n10dData.find(
    p => String(p.material_code_fg) === String(order.material_code_fg)
  );
  
  if (!product) return null;
  
  const status = product.status;
  
  switch (status) {
    case 'Critical':
      return { icon: '⚠', color: '#dc2626' };    // Red warning
    case 'Urgent':
      return { icon: '⚠', color: '#ea580c' };     // Orange warning
    case 'Monitor':
      return { icon: '⚠', color: '#ca8a04' };     // Yellow warning
    case 'Sufficient':
      return { icon: '✅', color: '#16a34a' };     // Green check
    default:
      return null;
  }
}
```

**Critical rule:** If N10D status is "Sufficient" → show green ✅, NOT a warning ⚠. This fixes the bug where sufficient products incorrectly show warning icons.
