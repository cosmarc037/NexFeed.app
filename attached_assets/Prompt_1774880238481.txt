### MATCHING LOGIC

```javascript
function canOrdersBeCombined(orderA, orderB) {
  // All criteria must match
  const criteria = [
    { field: 'material_code_fg', label: 'Material Code (FG)' },
    { field: 'material_code_sfg', label: 'Material Code (SFG)' },
    { field: 'category', label: 'Category' },
    { field: 'form', label: 'Form' },
    { field: 'batch_size', label: 'Batch Size' },
    { field: 'line', label: 'Line' },
    { field: 'scada', label: 'SCADA' },
  ];
  
  for (const c of criteria) {
    const valA = normalizeValue(orderA[c.field]);
    const valB = normalizeValue(orderB[c.field]);
    
    if (valA !== valB) {
      return {
        canCombine: false,
        mismatch: c.label,
        valueA: valA,
        valueB: valB
      };
    }
  }
  
  return { canCombine: true };
}

function normalizeValue(val) {
  if (val === null || val === undefined || val === '' || val === '-') {
    return ''; // Treat empty, null, dash as same (empty)
  }
  return String(val).trim().toLowerCase();
}
```
