### 3. TEMPLATE LINE GENERATION (INSTANT — NO AI)

```javascript
function generateTemplateLine(product, order, today) {
  const dfl = product?.due_for_loading;
  const inventory = product?.inventory;
  const status = product?.status;
  const buffer = product?.buffer;
  const availDate = product?.availDate;
  const daysUntilBreach = product?.daysUntilBreach;
  
  if (product && status) {
    switch (status) {
      case 'Critical':
        const deficit = (dfl - inventory).toFixed(1);
        return `⚠ CRITICAL — DFL (${dfl} MT) exceeds Inventory (${inventory} MT). Deficit: ${deficit} MT.`;
      
      case 'Urgent':
        return `⚠ URGENT — Stock runs out by ${formatDateFull(availDate)}. Buffer: ${buffer}%. Days remaining: ${daysUntilBreach}.`;
      
      case 'Monitor':
        return `📋 MONITOR — Stock lasts until ${formatDateFull(availDate)}. Buffer: ${buffer}%. Days remaining: ${daysUntilBreach}.`;
      
      case 'Sufficient':
        return `✅ SUFFICIENT — Inventory (${inventory} MT) covers DFL (${dfl} MT). Buffer: ${buffer}%.`;
      
      default:
        return `ℹ Status: ${status}. DFL: ${dfl} MT. Inventory: ${inventory} MT.`;
    }
  }
  
  if (order && order.avail_date && 
      order.avail_date !== 'prio replenish' && 
      order.avail_date !== 'safety stocks') {
    const deadline = parseDate(order.avail_date);
    const daysRemaining = deadline 
      ? Math.ceil((deadline - today) / (1000 * 60 * 60 * 24)) 
      : 'Unknown';
    const prodTime = order.production_time || 'N/A';
    return `📅 DEADLINE — Avail date: ${formatDateFull(order.avail_date)}. Days remaining: ${daysRemaining}. Prod time: ${prodTime} hrs.`;
  }
  
  const volume = order?.volume || 0;
  const line = order?.line || 'unassigned';
  return `ℹ No stock level data available from Next 10 Days. Scheduled for ${volume} MT on ${line}.`;
}
```

### 4. PRE-CALCULATE ENRICHED DATA

```javascript
function enrichProductData(product, dailyColumns) {
  const availDate = product.availDate ? new Date(product.availDate) : null;
  const daysUntilBreach = availDate 
    ? Math.ceil((availDate - new Date()) / (1000 * 60 * 60 * 24)) 
    : null;
  
  let totalDemand = 0;
  let daysWithDemand = 0;
  dailyColumns.forEach(col => {
    const val = parseFloat(product[col.key]) || 0;
    if (val > 0) {
      totalDemand += val;
      daysWithDemand++;
    }
  });
  const avgDailyDemand = daysWithDemand > 0 
    ? (totalDemand / daysWithDemand).toFixed(1) 
    : 0;
  
  const buffer = product.inventory > 0 
    ? (((product.inventory - product.due_for_loading) / product.inventory) * 100).toFixed(1)
    : '-100.0';
  
  return {
    ...product,
    daysUntilBreach: daysUntilBreach,
    avgDailyDemand: avgDailyDemand,
    buffer: buffer
  };
}
```

### 5. GENERATE AND CACHE — TEMPLATE FIRST, THEN AI

```javascript
let insightCache = {};
let insightsLoaded = false;
let isGenerating = false;

function setInsightCache(cache) {
  insightCache = cache;
  insightsLoaded = true;
}

function getSmartInsight(materialCode) {
  const entry = insightCache[materialCode];
  if (!entry) return null;
  return entry.combined;
}

function hasInsights() {
  return insightsLoaded && Object.keys(insightCache).length > 0;
}

async function generateAndCacheInsights(n10dProducts, allOrders, today, dailyColumns) {
  const cache = {};
  
  const enrichedProducts = n10dProducts.map(
    p => enrichProductData(p, dailyColumns)
  );
  
  // === STEP 1: Generate template lines INSTANTLY ===
  enrichedProducts.forEach(product => {
    const templateLine = generateTemplateLine(product, null, today);
    cache[product.material_code_fg] = {
      template: templateLine,
      ai: null,
      combined: templateLine
    };
  });
  
  allOrders.forEach(order => {
    if (!cache[order.material_code_fg]) {
      const templateLine = generateTemplateLine(null, order, today);
      cache[order.material_code_fg] = {
        template: templateLine,
        ai: null,
        combined: templateLine
      };
    }
  });
  
  // Show template lines immediately
  setInsightCache(cache);
  triggerOrderTableRefresh();
  
  // === STEP 2: Generate AI sentences (async) ===
  isGenerating = true;
  triggerOrderTableRefresh();
  
  try {
    // Part A: N10D products — MUST use AI API
    const n10dResponse = await generateN10DInsights(enrichedProducts, today);
    const n10dInsights = parseInsightResponse(n10dResponse);
    
    Object.entries(n10dInsights).forEach(([code, aiText]) => {
      if (cache[code]) {
        cache[code].ai = aiText;
        cache[code].combined = cache[code].template + '\n\n' + aiText;
      }
    });
    
    // Part B: Dated MTO orders — MUST use AI API
    const datedOrders = allOrders.filter(order => {
      const hasN10D = n10dProducts.some(
        p => String(p.material_code_fg) === String(order.material_code_fg)
      );
      return !hasN10D && order.avail_date && 
             order.avail_date !== 'prio replenish' && 
             order.avail_date !== 'safety stocks';
    });
    
    if (datedOrders.length > 0) {
      const datedResponse = await generateDatedOrderInsights(datedOrders, today);
      const datedInsights = parseInsightResponse(datedResponse);
      
      Object.entries(datedInsights).forEach(([code, aiText]) => {
        if (cache[code]) {
          cache[code].ai = aiText;
          cache[code].combined = cache[code].template + '\n\n' + aiText;
        }
      });
    }
    
  } catch (error) {
    console.error('AI insight generation failed:', error);
    // Template lines remain — cells are never empty
  } finally {
    isGenerating = false;
    setInsightCache(cache);
    triggerOrderTableRefresh();
  }
}

function parseInsightResponse(response) {
  const insights = {};
  const lines = response.split('\n').filter(l => l.trim());
  lines.forEach(line => {
    const match = line.match(/^\[?(\d+)\]?:\s*(.+)/);
    if (match) {
      insights[match[1]] = match[2].trim();
    }
  });
  return insights;
}
```

```jsx
function AvailDateCell({ order, n10dData }) {
  const [showTooltip, setShowTooltip] = useState(false);
  const [tooltipPos, setTooltipPos] = useState({ x: 0, y: 0 });
  const timeoutRef = useRef(null);
  
  function handleCellMouseEnter(e) {
    timeoutRef.current = setTimeout(() => {
      const pos = clampToViewport(e.clientX + 12, e.clientY + 12, 280, 120);
      setTooltipPos(pos);
      setShowTooltip(true);
    }, 300);
  }
  
  function handleCellMouseMove(e) {
    if (showTooltip) {
      const pos = clampToViewport(e.clientX + 12, e.clientY + 12, 280, 120);
      setTooltipPos(pos);
    }
  }
  
  function handleCellMouseLeave() {
    clearTimeout(timeoutRef.current);
    setShowTooltip(false);
  }
  
  return (
    <td
      onMouseEnter={handleCellMouseEnter}
      onMouseMove={handleCellMouseMove}
      onMouseLeave={handleCellMouseLeave}
      style={{ cursor: 'pointer' }}
    >
      <span className="avail-date-text">{order.avail_date}</span>
      {order.stockIcon && (
        <span className="avail-date-icon" style={{ cursor: 'pointer' }}>
          {order.stockIcon}

        <TooltipPortal>
          <AvailDateTooltip order={order} n10dData={n10dData} position={tooltipPos} />
```
