### TEMPLATE LINE — UPDATED FORMAT

The template line should read like a natural sentence, not a label.

```javascript
function generateTemplateLine(product, order, today, includeEmoji = false) {
  const dfl = product?.due_for_loading;
  const inventory = product?.inventory;
  const status = product?.status;
  const buffer = product?.buffer;
  const availDate = product?.availDate;
  const completionDate = product?.completionDate;
  const daysUntilBreach = product?.daysUntilBreach;
  
  if (product && status) {
    switch (status) {
      case 'Critical': {
        const deficit = (dfl - inventory).toFixed(1);
        const prefix = includeEmoji ? '⚠ ' : '';
        return `${prefix}This product is on Critical. Current inventory (${inventory} MT) has already fallen below the required demand (${dfl} MT), creating a ${deficit} MT deficit. Immediate production is needed to prevent stockout.`;
      }
      
      case 'Urgent': {
        const prefix = includeEmoji ? '⚠ ' : '';
        return `${prefix}This product is on Urgent. Demand will exceed inventory by ${formatDateFull(availDate)}, with ${formatDateFull(completionDate)} as the last safe production date. The ${buffer}% buffer is thin and will erode quickly — production should be scheduled within 1-2 days.`;
      }
      
      case 'Monitor': {
        const prefix = includeEmoji ? '📋 ' : '';
        return `${prefix}This product is on Monitor. Demand will exceed inventory by ${formatDateFull(availDate)}, with ${formatDateFull(completionDate)} as the last safe production date. The ${buffer}% buffer provides a window — production should be planned within the coming week.`;
      }
      
      case 'Sufficient': {
        const prefix = includeEmoji ? '✅ ' : '';
        return `${prefix}This product is on Sufficient. Current inventory (${inventory} MT) comfortably covers the required demand (${dfl} MT) with a ${buffer}% buffer. No immediate production action is needed.`;
      }
      
      default: {
        return `This product has a status of ${status}. DFL: ${dfl} MT. Inventory: ${inventory} MT.`;
      }
    }
  }
  
  if (order && order.avail_date && 
      order.avail_date !== 'prio replenish' && 
      order.avail_date !== 'safety stocks') {
    const deadline = parseDate(order.avail_date);
    const daysRemaining = deadline 
      ? Math.ceil((deadline - today) / (1000 * 60 * 60 * 24)) 
      : 'Unknown';
    const prodTime = order.production_time || 'N/A';
    const prefix = includeEmoji ? '📅 ' : '';
    return `${prefix}This order has a deadline of ${formatDateFull(order.avail_date)} — ${daysRemaining} days remaining. Estimated production time is ${prodTime} hours.`;
  }
  
  const volume = order?.volume || 0;
  const line = order?.line || 'unassigned';
  return `No stock level data available from Next 10 Days. This order is scheduled for ${volume} MT on ${line}.`;
}
```
