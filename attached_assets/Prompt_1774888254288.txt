### 3. TEMPLATE LINE GENERATION (INSTANT — NO AI)

```javascript
function generateTemplateLine(product, order, today, includeEmoji = false) {
  const dfl = product?.due_for_loading;
  const inventory = product?.inventory;
  const status = product?.status;
  const buffer = product?.buffer;
  const availDate = product?.availDate;
  const completionDate = product?.completionDate;
  const daysUntilBreach = product?.daysUntilBreach;
  
  if (product && status) {
    switch (status) {
      case 'Critical': {
        const deficit = (dfl - inventory).toFixed(1);
        const prefix = includeEmoji ? '⚠ ' : '';
        return `${prefix}This product is on Critical. Current inventory (${inventory} MT) has already fallen below the required demand (${dfl} MT), creating a ${deficit} MT deficit. Immediate production is needed to prevent stockout.`;
      }
      
      case 'Urgent': {
        const prefix = includeEmoji ? '⚠ ' : '';
        return `${prefix}This product is on Urgent. Demand will exceed inventory by ${formatDateFull(availDate)}, with ${formatDateFull(completionDate)} as the last safe production date. The ${buffer}% buffer is thin and will erode quickly — production should be scheduled within 1-2 days.`;
      }
      
      case 'Monitor': {
        const prefix = includeEmoji ? '📋 ' : '';
        return `${prefix}This product is on Monitor. Demand will exceed inventory by ${formatDateFull(availDate)}, with ${formatDateFull(completionDate)} as the last safe production date. The ${buffer}% buffer provides a window — production should be planned within the coming week.`;
      }
      
      case 'Sufficient': {
        const prefix = includeEmoji ? '✅ ' : '';
        return `${prefix}This product is on Sufficient. Current inventory (${inventory} MT) comfortably covers the required demand (${dfl} MT) with a ${buffer}% buffer. No immediate production action is needed.`;
      }
      
      default: {
        return `This product has a status of ${status}. DFL: ${dfl} MT. Inventory: ${inventory} MT.`;
      }
    }
  }
  
  if (order && order.avail_date && 
      order.avail_date !== 'prio replenish' && 
      order.avail_date !== 'safety stocks') {
    const deadline = parseDate(order.avail_date);
    const daysRemaining = deadline 
      ? Math.ceil((deadline - today) / (1000 * 60 * 60 * 24)) 
      : 'Unknown';
    const prodTime = order.production_time || 'N/A';
    const prefix = includeEmoji ? '📅 ' : '';
    return `${prefix}This order has a deadline of ${formatDateFull(order.avail_date)} — ${daysRemaining} days remaining. Estimated production time is ${prodTime} hours.`;
  }
  
  const volume = order?.volume || 0;
  const line = order?.line || 'unassigned';
  return `No stock level data available from Next 10 Days. This order is scheduled for ${volume} MT on ${line}.`;
}
```

### 4. PRE-CALCULATE ENRICHED DATA

```javascript
function enrichProductData(product, dailyColumns) {
  const availDate = product.availDate ? new Date(product.availDate) : null;
  const daysUntilBreach = availDate 
    ? Math.ceil((availDate - new Date()) / (1000 * 60 * 60 * 24)) 
    : null;
  
  let totalDemand = 0;
  let daysWithDemand = 0;
  dailyColumns.forEach(col => {
    const val = parseFloat(product[col.key]) || 0;
    if (val > 0) {
      totalDemand += val;
      daysWithDemand++;
    }
  });
  const avgDailyDemand = daysWithDemand > 0 
    ? (totalDemand / daysWithDemand).toFixed(1) 
    : 0;
  
  const buffer = product.inventory > 0 
    ? (((product.inventory - product.due_for_loading) / product.inventory) * 100).toFixed(1)
    : '-100.0';
  
  return {
    ...product,
    daysUntilBreach: daysUntilBreach,
    avgDailyDemand: avgDailyDemand,
    buffer: buffer
  };
}
```

### 5. AUTO-GENERATE TEMPLATES — NO BUTTON CLICK NEEDED

```javascript
let insightCache = {};
let insightsLoaded = false;
let isGenerating = false;

function setInsightCache(cache) {
  insightCache = cache;
  insightsLoaded = true;
}

function getSmartInsight(materialCode) {
  const entry = insightCache[materialCode];
  if (!entry) return null;
  return entry.combined;
}

function getTemplateOnly(materialCode) {
  const entry = insightCache[materialCode];
  if (!entry) return null;
  return entry.template;
}

function hasInsights() {
  return insightsLoaded && Object.keys(insightCache).length > 0;
}

function autoGenerateTemplates(n10dProducts, allOrders, today, dailyColumns) {
  const cache = {};
  
  const enrichedProducts = n10dProducts.map(
    p => enrichProductData(p, dailyColumns)
  );
  
  // Template lines for N10D products
  enrichedProducts.forEach(product => {
    const templateLine = generateTemplateLine(product, null, today, false);
    cache[product.material_code_fg] = {
      template: templateLine,
      ai: null,
      combined: templateLine
    };
  });
  
  // Template lines for orders without N10D
  allOrders.forEach(order => {
    if (!cache[order.material_code_fg]) {
      const templateLine = generateTemplateLine(null, order, today, false);
      cache[order.material_code_fg] = {
        template: templateLine,
        ai: null,
        combined: templateLine
      };
    }
  });
  
  setInsightCache(cache);
  triggerOrderTableRefresh();
}
```

### 6. SPARKLE BUTTON — AI ENHANCEMENT (ORDER TABLE ONLY)

The ✨ button in the Summary column sub-header triggers AI generation for ALL orders on the current line tab. This is the ONLY way AI text is generated. N10D and Auto-Sequence tooltips NEVER get AI text.

```javascript
async function handleEnhanceWithAI() {
  if (!n10dData || n10dData.length === 0) {
    showNotification('Upload Next 10 Days data first to generate AI insights.');
    return;
  }
  
  isGenerating = true;
  triggerOrderTableRefresh();
  
  try {
    const enrichedProducts = n10dData.map(
      p => enrichProductData(p, dailyColumns)
    );
    
    // Part A: N10D products — MUST use AI API
    const n10dResponse = await generateN10DInsights(enrichedProducts, new Date());
    const n10dInsights = parseInsightResponse(n10dResponse);
    
    Object.entries(n10dInsights).forEach(([code, aiText]) => {
      if (insightCache[code]) {
        insightCache[code].ai = aiText;
        insightCache[code].combined = insightCache[code].template + '\n\n' + aiText;
      }
    });
    
    // Part B: Dated MTO orders — MUST use AI API
    const datedOrders = allOrders.filter(order => {
      const hasN10D = n10dData.some(
        p => String(p.material_code_fg) === String(order.material_code_fg)
      );
      return !hasN10D && order.avail_date && 
             order.avail_date !== 'prio replenish' && 
             order.avail_date !== 'safety stocks';
    });
    
    if (datedOrders.length > 0) {
      const datedResponse = await generateDatedOrderInsights(datedOrders, new Date());
      const datedInsights = parseInsightResponse(datedResponse);
      
      Object.entries(datedInsights).forEach(([code, aiText]) => {
        if (insightCache[code]) {
          insightCache[code].ai = aiText;
          insightCache[code].combined = insightCache[code].template + '\n\n' + aiText;
        }
      });
    }
    
    setInsightCache(insightCache);
  } catch (error) {
    console.error('AI enhancement failed:', error);
    showNotification('Failed to generate AI insights. Try again.');
  } finally {
    isGenerating = false;
    triggerOrderTableRefresh();
  }
}

function parseInsightResponse(response) {
  const insights = {};
  const lines = response.split('\n').filter(l => l.trim());
  lines.forEach(line => {
    const match = line.match(/^\[?(\d+)\]?:\s*(.+)/);
    if (match) {
      insights[match[1]] = match[2].trim();
    }
  });
  return insights;
}
```
