### FIX 4: UPDATED RESPONSE PARSING — HANDLE ALL KEYS

```javascript
function parseOrderInsightResponse(response) {
  const insights = {};
  
  // Split by newlines
  const lines = response.split('\n').filter(l => l.trim());
  
  lines.forEach(line => {
    // Match pattern: [key]: insight text
    // Key can be complex (order ID, material code combo)
    const match = line.match(/^\[?([^\]]+)\]?:\s*(.+)/);
    if (match) {
      const key = match[1].trim();
      const text = match[2].trim();
      insights[key] = text;
    }
  });
  
  return insights;
}
```
