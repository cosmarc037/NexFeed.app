### FIX 1: ASSIGN AVAIL DATE FROM N10D FOR NON-DATED ORDERS

When the auto-sequence processes non-dated orders (prio replenish / safety stocks), it must look up the N10D Avail date and use it as the order's avail date for sorting.

```javascript
function assignAvailDatesFromN10D(orders, n10dData, today) {
  return orders.map(order => {
    const updatedOrder = { ...order };
    
    // Skip dated orders — they already have avail dates
    if (order.avail_date && 
        order.avail_date !== 'prio replenish' && 
        order.avail_date !== 'safety stocks') {
      updatedOrder._sortDate = parseDate(order.avail_date);
      updatedOrder._dateSource = 'dated';
      return updatedOrder;
    }
    
    // Non-dated orders — look up N10D Avail date
    if (!n10dData || n10dData.length === 0) {
      updatedOrder._sortDate = null;
      updatedOrder._dateSource = 'none';
      return updatedOrder;
    }
    
    const product = n10dData.find(
      p => String(p.material_code_fg) === String(order.material_code_fg)
    );
    
    if (!product) {
      updatedOrder._sortDate = null;
      updatedOrder._dateSource = 'none';
      return updatedOrder;
    }
    
    // CORRECT LOGIC:
    // N10D Avail date = date when cumulative DFL exceeds inventory
    // For Critical products: Avail = today (DFL already exceeds inventory)
    // For Urgent: Avail = within 3 days
    // For Monitor: Avail = within 4-10 days
    // For Sufficient: Avail = null (never breaches)
    
    const n10dAvailDate = product.availDate ? parseDate(product.availDate) : null;
    const n10dStatus = product.status;
    
    if (n10dStatus === 'Critical') {
      // Critical = DFL already exceeds Inventory = needs production TODAY
      updatedOrder._sortDate = today;
      updatedOrder._assignedAvailDate = formatDateFull(today);
      updatedOrder._n10dStatus = 'Critical';
    } else if (n10dAvailDate) {
      // Urgent / Monitor = has a breach date
      updatedOrder._sortDate = n10dAvailDate;
      updatedOrder._assignedAvailDate = formatDateFull(n10dAvailDate);
      updatedOrder._n10dStatus = n10dStatus;
    } else {
      // Sufficient = no breach date, no urgency
      updatedOrder._sortDate = null;
      updatedOrder._assignedAvailDate = null;
      updatedOrder._n10dStatus = 'Sufficient';
    }
    
    updatedOrder._dateSource = 'n10d';
    return updatedOrder;
  });
}
```

### FIX 2: SORT ALL ORDERS CHRONOLOGICALLY

After all orders have avail dates (either original or from N10D), sort chronologically:

```javascript
function sortOrdersChronologically(orders) {
  return orders.sort((a, b) => {
    const dateA = a._sortDate;
    const dateB = b._sortDate;
    
    // Orders with dates come before orders without dates
    if (dateA && !dateB) return -1;
    if (!dateA && dateB) return 1;
    if (!dateA && !dateB) return 0;
    
    // Both have dates — sort chronologically (earliest first)
    const diff = dateA.getTime() - dateB.getTime();
    if (diff !== 0) return diff;
    
    // Same date — tiebreaker: higher DFL/Inventory ratio first
    const ratioA = a._n10dRatio || 0;
    const ratioB = b._n10dRatio || 0;
    return ratioB - ratioA;
  });
}
```

### FIX 3: COMPLETE AUTO-SEQUENCE FLOW

```javascript
function runAutoSequence(orders, n10dData) {
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  // Step 1: Assign avail dates from N10D for non-dated orders
  const ordersWithDates = assignAvailDatesFromN10D(orders, n10dData, today);
  
  // Step 2: Add DFL/Inventory ratio for tiebreaking
  ordersWithDates.forEach(order => {
    const product = n10dData?.find(
      p => String(p.material_code_fg) === String(order.material_code_fg)
    );
    if (product && product.inventory > 0) {
      order._n10dRatio = product.due_for_loading / product.inventory;
    } else if (product && product.inventory === 0) {
      order._n10dRatio = Infinity;
    } else {
      order._n10dRatio = 0;
    }
  });
  
  // Step 3: Sort chronologically
  const sorted = sortOrdersChronologically(ordersWithDates);
  
  // Step 4: Assign priorities
  sorted.forEach((order, index) => {
    order.prio = index + 1;
  });
  
  return sorted;
}
```
