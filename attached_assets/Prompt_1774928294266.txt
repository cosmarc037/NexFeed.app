### THE CORRECT LOGIC

```javascript
function determineProductStatus(product, dailyColumns, today) {
  const dfl = parseFloat(product.due_for_loading) || 0;
  const inventory = parseFloat(product.inventory) || 0;
  
  // ========================================
  // RULE: DFL > Inventory = BAD = CRITICAL
  // RULE: DFL < Inventory = GOOD = SUFFICIENT
  // ========================================
  
  // CRITICAL: DFL already greater than or equal to Inventory
  // This means required demand EXCEEDS current stock
  // Example: DFL 24.0, Inventory 0.7 → 24.0 >= 0.7 → CRITICAL
  if (dfl >= inventory) {
    return 'Critical';
  }
  
  // From here, DFL < Inventory (stock covers demand for now)
  // Check when cumulative sum will exceed inventory
  let cumulativeSum = dfl;
  let breachDay = null;
  
  for (let i = 0; i < dailyColumns.length; i++) {
    const dayValue = parseFloat(product[dailyColumns[i].key]) || 0;
    cumulativeSum += dayValue;
    
    if (cumulativeSum >= inventory && breachDay === null) {
      breachDay = i;
      break;
    }
  }
  
  // Never breaches within 10 days = SUFFICIENT
  if (breachDay === null) {
    return 'Sufficient';
  }
  
  // Calculate days until breach
  const breachDate = new Date(dailyColumns[breachDay].date);
  const todayClean = new Date(today);
  todayClean.setHours(0, 0, 0, 0);
  const diffMs = breachDate.getTime() - todayClean.getTime();
  const daysUntilBreach = Math.ceil(diffMs / (1000 * 60 * 60 * 24));
  
  // URGENT: Breaches within 3 days
  if (daysUntilBreach <= 3) {
    return 'Urgent';
  }
  
  // MONITOR: Breaches within 4-10 days
  return 'Monitor';
}
```
