### THE ONE FUNCTION

```javascript
/**
 * THE ONLY STATUS DETERMINATION FUNCTION IN THE ENTIRE APP
 * 
 * RULE:
 * - DFL >= Inventory → CRITICAL (demand exceeds stock)
 * - DFL < Inventory, breaches within 3 days → URGENT
 * - DFL < Inventory, breaches within 4-10 days → MONITOR
 * - DFL < Inventory, never breaches → SUFFICIENT
 * 
 * EXAMPLES:
 * - DFL 24.0, Inventory 0.7 → 24.0 >= 0.7 → CRITICAL
 * - DFL 92.3, Inventory 310.6 → 92.3 < 310.6, never breaches → SUFFICIENT
 * 
 * @param {number} dfl - Due for Loading (required demand)
 * @param {number} inventory - Current available stock
 * @param {Array} dailyColumns - Array of {key, date} for 10-day columns
 * @param {Object} product - Product row from N10D with daily values
 * @param {Date} today - Current date
 * @returns {string} 'Critical' | 'Urgent' | 'Monitor' | 'Sufficient'
 */
function getProductStatus(dfl, inventory, dailyColumns = [], product = null, today = new Date()) {
  // Parse to ensure numbers
  const dflNum = parseFloat(dfl) || 0;
  const invNum = parseFloat(inventory) || 0;
  
  // CRITICAL: DFL >= Inventory
  // The required demand EXCEEDS or EQUALS current stock
  // Production is needed IMMEDIATELY
  if (dflNum >= invNum) {
    return 'Critical';
  }
  
  // From here: DFL < Inventory (stock covers base demand)
  // But check if cumulative daily demand will breach inventory
  
  if (!dailyColumns || dailyColumns.length === 0 || !product) {
    return 'Sufficient';
  }
  
  let cumulativeSum = dflNum;
  let breachDayIndex = null;
  
  for (let i = 0; i < dailyColumns.length; i++) {
    const dayValue = parseFloat(product[dailyColumns[i].key]) || 0;
    cumulativeSum += dayValue;
    
    if (cumulativeSum >= invNum && breachDayIndex === null) {
      breachDayIndex = i;
      break;
    }
  }
  
  // Never breaches within 10 days
  if (breachDayIndex === null) {
    return 'Sufficient';
  }
  
  // Calculate days until breach
  const breachDate = new Date(dailyColumns[breachDayIndex].date);
  const todayClean = new Date(today);
  todayClean.setHours(0, 0, 0, 0);
  breachDate.setHours(0, 0, 0, 0);
  const diffMs = breachDate.getTime() - todayClean.getTime();
  const daysUntilBreach = Math.ceil(diffMs / (1000 * 60 * 60 * 24));
  
  if (daysUntilBreach <= 3) {
    return 'Urgent';
  }
  
  return 'Monitor';
}
```

function runAutoSequence(orders, n10dData, dailyColumns, today) {
  const ordersWithDates = orders.map(order => {
    const updatedOrder = { ...order };
    
    // Dated orders keep their date
    if (order.avail_date && 
        order.avail_date !== 'prio replenish' &&

    }
    
    // Non-dated: get status using THE ONE FUNCTION
    const product = n10dData?.find(
      p => String(p.material_code_fg) === String(order.material_code_fg)
    );
    
    if (!product) {
      updatedOrder._sortDate = null;
      updatedOrder._n10dStatus = null;
      return updatedOrder;
    }
    
    const status = getProductStatus(
      product.due_for_loading,
      product.inventory,
      dailyColumns,
      product,
      today
    );
    
    updatedOrder._n10dStatus = status;
    
    if (status === 'Critical') {
      updatedOrder._sortDate = today;
      updatedOrder._assignedAvailDate = formatDateFull(today);
    } else if (product.availDate) {
      updatedOrder._sortDate = parseDate(product.availDate);
      updatedOrder._assignedAvailDate = formatDateFull(product.availDate);
    } else {
      updatedOrder._sortDate = null;
    }
    
    return updatedOrder;
  });
  
  // Sort chronologically
  ordersWithDates.sort((a, b) => {
    if (a._sortDate && !b._sortDate) return -1;
    if (!a._sortDate && b._sortDate) return 1;
    if (!a._sortDate && !b._sortDate) return 0;
    return a._sortDate.getTime() - b._sortDate.getTime();
  });
  
  return ordersWithDates;
}
