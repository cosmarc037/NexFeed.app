## FIX 1: REPLACE "STOCK SUFFICIENT" WITH ACTUAL N10D AVAIL DATE

In the auto-sequence preview, orders labeled "Stock sufficient" should instead show the actual Avail date from the N10D tab. If no Avail date exists in N10D (never breaches), use the last day of the 10-day window as the date.

### THE PROBLEM

Orders like Elite XP Startex and Gallimax Pigeon Pellet show "Stock sufficient ✅" in the Avail Date column. When "Apply to Schedule" is clicked, this text gets applied instead of an actual date.

### THE FIX

```javascript
function getAvailDateForAutoSequence(order, n10dData, dailyColumns, today) {
  // Dated orders — keep original avail date
  if (order.avail_date && 
      order.avail_date !== 'prio replenish' && 
      order.avail_date !== 'safety stocks') {
    return {
      sortDate: parseDate(order.avail_date),
      displayDate: formatDateFull(order.avail_date),
      source: 'dated'
    };
  }
  
  // Non-dated orders — get date from N10D
  const product = n10dData?.find(
    p => String(p.material_code_fg) === String(order.material_code_fg)
  );
  
  if (!product) return { sortDate: null, displayDate: null, source: 'none' };
  
  const status = getProductStatus(
    product.due_for_loading,
    product.inventory,
    dailyColumns,
    product,
    today
  );
  
  if (status === 'Critical') {
    // Critical — avail is today
    return {
      sortDate: today,
      displayDate: formatDateFull(today),
      source: 'n10d'
    };
  }
  
  if (product.availDate) {
    // Urgent / Monitor — use N10D avail date
    return {
      sortDate: parseDate(product.availDate),
      displayDate: formatDateFull(product.availDate),
      source: 'n10d'
    };
  }
  
  // Sufficient — never breaches, use LAST day of 10-day window as date
  if (dailyColumns && dailyColumns.length > 0) {
    const lastDay = dailyColumns[dailyColumns.length - 1].date;
    return {
      sortDate: parseDate(lastDay),
      displayDate: formatDateFull(lastDay),
      source: 'n10d_last_day'
    };
  }
  
  return { sortDate: null, displayDate: null, source: 'none' };
}
```

## FIX 2: AUTO-UPDATE WHEN N10D IS REFRESHED

When a new N10D file is uploaded (daily update), the avail dates for non-dated orders should automatically update.

```javascript
// When N10D file is uploaded
async function handleN10DUpload(file) {
  const parsedData = parseN10DFile(file);
  setN10dData(parsedData);
  
  // Auto-generate templates
  autoGenerateTemplates(parsedData, allOrders, new Date(), dailyColumns);
  
  // Auto-update avail dates for non-dated orders that were previously assigned N10D dates
  updateNonDatedOrderAvailDates(allOrders, parsedData, dailyColumns, new Date());
}

function updateNonDatedOrderAvailDates(allOrders, n10dData, dailyColumns, today) {
  allOrders.forEach(order => {
    // Only update orders that originally had 'prio replenish' or 'safety stocks'
    // AND were previously assigned an N10D date
    const originalAvail = order._originalAvailDate || order.avail_date;
    
    if (originalAvail === 'prio replenish' || originalAvail === 'safety stocks') {
      // Store original label if not already stored
      if (!order._originalAvailDate) {
        order._originalAvailDate = order.avail_date;
      }
      
      // Get updated date from new N10D data
      const availData = getAvailDateForAutoSequence(
        { ...order, avail_date: originalAvail },
        n10dData,
        dailyColumns,
        today
      );
      
      if (availData.displayDate) {
        order.avail_date = availData.displayDate;
      }
    }
  });
  
  triggerOrderTableRefresh();
}
```
