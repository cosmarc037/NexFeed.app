## 3. STATUS GROUPS — LOCKED VS MOVABLE

### LOCKED STATUSES (cannot be moved, stick in place)

| Status | Behavior |
|---|---|
| Done | Locked at top — already implemented |
| Cancel PO | Locked at bottom — already implemented |
| In Production | Locked — only 1 per line at a time |
| On-going batching | Locked — can coexist with In Production |
| On-going pelleting | Locked — can coexist with In Production |
| On-going bagging | Locked — can coexist with In Production |
| Planned | Locked — stays at its current priority position |

### MOVABLE STATUSES (can be sorted/sequenced)

| Status | Behavior |
|---|---|
| Plotted | Sortable, cannot go above locked orders |
| Hold | Sortable, cannot go above locked orders |
| Cut | Sortable, cannot go above locked orders |
| Uncombine | Sortable, cannot go above locked orders |
| Merge Back | Sortable, cannot go above locked orders |
| Custom (free text) | Sortable, cannot go above locked orders |

### STATUS GROUP FUNCTION

```javascript
function isLockedStatus(status) {
  const lockedStatuses = [
    'Done',
    'Cancel PO',
    'In Production',
    'On-going batching',
    'On-going pelleting',
    'On-going bagging',
    'Planned'
  ];
  
  return lockedStatuses.includes(status);
}

function isMovableStatus(status) {
  return !isLockedStatus(status);
}
```

## 4. AUTO-SEQUENCE — RESPECT LOCKED ORDERS

When running auto-sequence, locked orders stay at their positions. Movable orders are sorted chronologically in the remaining positions.

```javascript
function runAutoSequence(orders, n10dData, dailyColumns, today) {
  // Step 1: Separate locked and movable orders
  const lockedOrders = [];
  const movableOrders = [];
  
  orders.forEach((order, index) => {
    if (isLockedStatus(order.status)) {
      lockedOrders.push({ ...order, _originalPrio: index + 1 });
    } else {
      movableOrders.push({ ...order });
    }
  });
  
  // Step 2: Assign avail dates to movable orders
  const movableWithDates = movableOrders.map(order => {
    const updatedOrder = { ...order };
    
    if (order.avail_date && 
        order.avail_date !== 'prio replenish' && 
        order.avail_date !== 'safety stocks' &&
        !order.avail_date.toLowerCase().includes('stock sufficient')) {
      updatedOrder._sortDate = parseDate(order.avail_date);
      return updatedOrder;
    }
    
    const product = n10dData?.find(
      p => String(p.material_code_fg) === String(order.material_code_fg)
    );
    
    if (product) {
      const status = getProductStatus(
        product.due_for_loading,
        product.inventory,
        dailyColumns,
        product,
        today
      );
      
      updatedOrder._n10dStatus = status;
      
      if (status === 'Critical') {
        updatedOrder._sortDate = new Date(today);
        updatedOrder._displayAvailDate = formatDateFull(today);
      } else if (product.availDate) {
        updatedOrder._sortDate = parseDate(product.availDate);
        updatedOrder._displayAvailDate = formatDateFull(product.availDate);
      } else if (dailyColumns && dailyColumns.length > 0) {
        const lastDay = dailyColumns[dailyColumns.length - 1].date;
        updatedOrder._sortDate = parseDate(lastDay);
        updatedOrder._displayAvailDate = formatDateFull(lastDay);
      }
    }
    
    return updatedOrder;
  });
  
  // Step 3: Sort movable orders chronologically
  movableWithDates.sort((a, b) => {
    const dateA = a._sortDate;
    const dateB = b._sortDate;
    
    if (dateA && !dateB) return -1;
    if (!dateA && dateB) return 1;
    if (!dateA && !dateB) return 0;
    
    const dateDiff = dateA.getTime() - dateB.getTime();
    if (dateDiff !== 0) return dateDiff;
    
    const statusOrder = { 'Critical': 0, 'Urgent': 1, 'Monitor': 2, 'Sufficient': 3 };
    const statusA = statusOrder[a._n10dStatus] ?? 2;
    const statusB = statusOrder[b._n10dStatus] ?? 2;
    if (statusA !== statusB) return statusA - statusB;
    
    const volA = parseFloat(a.volume) || 0;
    const volB = parseFloat(b.volume) || 0;
    return volB - volA;
  });
  
  // Step 4: Merge — locked orders stay at their positions, movable fill the gaps
  const totalSlots = orders.length;
  const result = new Array(totalSlots).fill(null);
  
  // Place locked orders at their original positions
  lockedOrders.forEach(order => {
    const pos = order._originalPrio - 1; // 0-indexed
    if (pos >= 0 && pos < totalSlots) {
      result[pos] = order;
    }
  });
  
  // Fill remaining slots with sorted movable orders
  let movableIndex = 0;
  for (let i = 0; i < totalSlots; i++) {
    if (result[i] === null && movableIndex < movableWithDates.length) {
      result[i] = movableWithDates[movableIndex];
      movableIndex++;
    }
  }
  
  // Step 5: Assign priorities
  result.forEach((order, index) => {
    if (order) {
      order.prio = index + 1;
    }
  });
  
  return result.filter(o => o !== null);
}
```

## 5. DRAG AND DROP — BLOCK MOVING ABOVE LOCKED ORDERS

When a user tries to drag a MOVABLE order above a LOCKED order, block it and show a warning.

```javascript
function canMoveOrderToPosition(order, targetPosition, allOrders) {
  if (!isMovableStatus(order.status)) {
    // Locked orders can't be moved at all
    return {
      allowed: false,
      message: `This order is "${order.status}" and cannot be moved.`
    };
  }
  
  // Check if any locked order is at or above the target position
  const orderAbove = allOrders[targetPosition - 1]; // order at target position
  
  if (orderAbove && isLockedStatus(orderAbove.status) && 
      orderAbove.status !== 'Cancel PO') { // Can go above cancelled
    return {
      allowed: false,
      message: `Cannot place above "${orderAbove.item_description}" which is "${orderAbove.status}". Orders with locked statuses (Done, In Production, On-going, Planned) cannot be overtaken.`
    };
  }
  
  return { allowed: true };
}
```

## 6. IN PRODUCTION — ONLY 1 PER LINE

When a user tries to set an order to "In Production" while another order on the same line is already "In Production", show a warning.

```javascript
function canSetStatus(order, newStatus, allOrders, currentLine) {
  if (newStatus === 'In Production') {
    const existingInProduction = allOrders.find(
      o => o.line === currentLine && 
           o.status === 'In Production' && 
           o.id !== order.id
    );
    
    if (existingInProduction) {
      return {
        allowed: false,
        message: `"${existingInProduction.item_description}" is currently In Production on ${currentLine}. Mark it as Done or On-going before setting another order to In Production.`
      };
    }
  }
  
  return { allowed: true };
}
```
