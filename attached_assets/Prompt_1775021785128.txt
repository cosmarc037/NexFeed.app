## FIX 4: IN PRODUCTION / ON-GOING — ONLY AFTER DONE ORDERS

User can only set an order to "In Production" or "On-going" if ALL orders above it are "Done", "In Production", "On-going", or "Planned" — NOT if a Plotted/Hold/Custom order is above it.

```javascript
function canSetToInProductionOrOngoing(order, newStatus, allOrders) {
  if (newStatus !== 'In Production' && !newStatus.startsWith('On-going')) {
    return { allowed: true };
  }
  
  const orderIndex = allOrders.findIndex(o => o.id === order.id);
  
  // Check all orders above this one
  for (let i = 0; i < orderIndex; i++) {
    const aboveOrder = allOrders[i];
    const aboveStatus = aboveOrder.status;
    
    // Allowed above statuses
    const allowedAbove = [
      'Done', 
      'In Production', 
      'On-going batching', 'On-going pelleting', 'On-going bagging',
      'Planned',
      'Cancel PO'
    ];
    
    if (!allowedAbove.includes(aboveStatus)) {
      return {
        allowed: false,
        message: `Cannot set to "${newStatus}" because "${aboveOrder.item_description}" at Priority ${i + 1} is "${aboveStatus}". All orders above must be Done, In Production, On-going, or Planned before this order can begin production.`
      };
    }
  }
  
  // Additional check: only 1 "In Production" per line
  if (newStatus === 'In Production') {
    const existingInProduction = allOrders.find(
      o => o.line === order.line && 
           o.status === 'In Production' && 
           o.id !== order.id
    );
    
    if (existingInProduction) {
      return {
        allowed: false,
        message: `"${existingInProduction.item_description}" is currently In Production on this line. Mark it as Done or On-going before setting another order to In Production.`
      };
    }
  }
  
  return { allowed: true };
}
```

### AUTO-SEQUENCE LOGIC

```javascript
function runAutoSequence(allOrders, n10dData, dailyColumns, today) {
  // Step 1: Separate into three groups
  const excludedOrders = []; // Done, Cancel, In Prod, On-going — not in preview
  const plannedOrders = [];  // Planned — shown but locked at their prio
  const movableOrders = [];  // Everything else — sorted
  
  allOrders.forEach((order, index) => {
    const prio = index + 1;
    
    if (['Done', 'Cancel PO', 'In Production', 
         'On-going batching', 'On-going pelleting', 'On-going bagging'
        ].includes(order.status)) {
      excludedOrders.push({ ...order, _originalPrio: prio });
    } else if (order.status === 'Planned') {
      plannedOrders.push({ ...order, _originalPrio: prio });
    } else {
      movableOrders.push({ ...order, _originalPrio: prio });
    }
  });
  
  // Step 2: Determine which prio slots are available for the preview
  // Exclude slots taken by Done/Cancel/InProd/OnGoing (not shown)
  // Planned orders keep their prio slot
  
  const excludedPrios = excludedOrders.map(o => o._originalPrio);
  const plannedPrioMap = {};
  plannedOrders.forEach(o => {
    plannedPrioMap[o._originalPrio] = o;
  });
  
  // Step 3: Get available slots (not excluded)
  const totalOrders = allOrders.length;
  const availableSlots = [];
  for (let i = 1; i <= totalOrders; i++) {
    if (!excludedPrios.includes(i)) {
      availableSlots.push(i);
    }
  }
  
  // Step 4: Assign avail dates to movable orders and sort
  const movableWithDates = assignAvailDates(movableOrders, n10dData, dailyColumns, today);
  
  movableWithDates.sort((a, b) => {
    const dateA = a._sortDate;
    const dateB = b._sortDate;
    
    if (dateA && !dateB) return -1;
    if (!dateA && dateB) return 1;
    if (!dateA && !dateB) return 0;
    
    const dateDiff = dateA.getTime() - dateB.getTime();
    if (dateDiff !== 0) return dateDiff;
    
    const statusOrder = { 'Critical': 0, 'Urgent': 1, 'Monitor': 2, 'Sufficient': 3 };
    const statusA = statusOrder[a._n10dStatus] ?? 2;
    const statusB = statusOrder[b._n10dStatus] ?? 2;
    if (statusA !== statusB) return statusA - statusB;
    
    const volA = parseFloat(a.volume) || 0;
    const volB = parseFloat(b.volume) || 0;
    return volB - volA;
  });
  
  // Step 5: Build preview — place Planned at their slots, fill rest with sorted movable
  const previewResult = [];
  let movableIndex = 0;
  
  availableSlots.forEach(slot => {
    if (plannedPrioMap[slot]) {
      // Planned order stays at this slot
      previewResult.push({
        ...plannedPrioMap[slot],
        prio: slot,
        _isLocked: true
      });
    } else {
      // Fill with next sorted movable order
      if (movableIndex < movableWithDates.length) {
        previewResult.push({
          ...movableWithDates[movableIndex],
          prio: slot,
          _isLocked: false
        });
        movableIndex++;
      }
    }
  });
  
  return previewResult;
}
```

function calculatePreviewMetrics(previewOrders, excludedOrders) {
  let withDeadline = 0;

  let stockSufficient = 0;

  };
