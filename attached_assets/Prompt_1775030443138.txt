## 4. PLANNED — BEHAVIOR

### STAYS AT CURRENT PRIO

When set to Planned, order stays at its current priority — does NOT jump to top:

```javascript
function handleStatusChange(order, newStatus, allOrders) {
  // Store original prio when changing to locked
  if (isLockedStatus(newStatus) && !order._originalPrioBeforeLock) {
    order._originalPrioBeforeLock = order.prio;
  }
  
  // When reverting from locked to movable — restore original position
  if (isLockedStatus(order.status) && !isLockedStatus(newStatus)) {
    if (order._originalPrioBeforeLock) {
      const targetPrio = order._originalPrioBeforeLock;
      order._originalPrioBeforeLock = null;
      const maxPrio = allOrders.length;
      const finalPrio = Math.min(targetPrio, maxPrio);
      moveOrderToPrio(order, finalPrio, allOrders);
    }
  }
  
  if (newStatus === 'Planned') {
    order.status = 'Planned';
    order._plannedPrio = order.prio;
    return;
  }
  
  order.status = newStatus;
}

function moveOrderToPrio(order, targetPrio, allOrders) {
  const currentIndex = allOrders.findIndex(o => o.id === order.id);
  if (currentIndex === -1) return;
  allOrders.splice(currentIndex, 1);
  const insertIndex = Math.min(targetPrio - 1, allOrders.length);
  allOrders.splice(insertIndex, 0, order);
  allOrders.forEach((o, i) => { o.prio = i + 1; });
}
```

## 6. IN PRODUCTION / ON-GOING — RESTRICTIONS

```javascript
function canSetToInProductionOrOngoing(order, newStatus, allOrders) {
  if (newStatus !== 'In Production' && !newStatus.startsWith('On-going')) {
    return { allowed: true };
  }
  
  const orderIndex = allOrders.findIndex(o => o.id === order.id);
  
  const allowedAbove = [
    'Done', 'In Production',
    'On-going batching', 'On-going pelleting', 'On-going bagging',
    'Planned', 'Cancel PO'
  ];
  
  const problemOrders = [];
  
  for (let i = 0; i < orderIndex; i++) {
    const aboveOrder = allOrders[i];
    if (!allowedAbove.includes(aboveOrder.status)) {
      problemOrders.push({
        prio: i + 1,
        name: aboveOrder.item_description,
        status: aboveOrder.status
      });
    }
  }
  
  if (problemOrders.length > 0) {
    const statusLabel = newStatus === 'In Production' ? 'In Production' : 'On-going';
    return {
      allowed: false,
      title: `Cannot Set ${statusLabel}`,
      message: `There are orders at earlier priorities that have not been produced or completed yet. Consider producing or completing those orders first before ${newStatus === 'In Production' ? 'starting this one' : 'marking this order as on-going'}.`,
      problemOrders: problemOrders
    };
  }
  
  // In Production — only 1 per line
  if (newStatus === 'In Production') {
    const existingInProduction = allOrders.find(
      o => o.line === order.line &&
           o.status === 'In Production' &&
           o.id !== order.id
    );
    
    if (existingInProduction) {
      return {
        allowed: false,
        title: 'In Production Conflict',
        message: `"${existingInProduction.item_description}" is currently In Production on this line. Consider marking it as Done or On-going before setting another order to In Production.`,
        problemOrders: []
      };
    }
  }
  
  // On-going — can coexist with In Production, no single-order restriction
  
  return { allowed: true };
}
```

### SORTING PRIORITY

```
1. CRITICAL orders (DFL >= Inventory, need production NOW)
   - Among themselves: sorted by DFL/Inventory ratio (highest deficit first)
   - Critical overrides Planned — pushes Planned down
2. PLANNED orders (locked at their position, shifted if Critical takes their spot)
3. All other orders sorted chronologically by avail date
   - Within same date: Urgent > Monitor > Sufficient
   - Within same date and status: higher volume first
```

### COMPLETE AUTO-SEQUENCE FUNCTION

```javascript
function runAutoSequence(allOrders, n10dData, dailyColumns, today) {
  const excludedOrders = [];
  const plannedOrders = [];
  const movableOrders = [];
  
  allOrders.forEach((order) => {
    const realPrio = order.prio;
    
    if (['Done', 'Cancel PO', 'In Production',
         'On-going batching', 'On-going pelleting', 'On-going bagging'
        ].includes(order.status)) {
      excludedOrders.push({ ...order, _realPrio: realPrio });
    } else if (order.status === 'Planned') {
      const lockedPrio = order._plannedPrio || realPrio;
      plannedOrders.push({ ...order, _realPrio: lockedPrio });
    } else {
      movableOrders.push({ ...order, _realPrio: realPrio });
    }
  });
  
  // Excluded prio slots
  const excludedPrios = new Set(excludedOrders.map(o => o._realPrio));
  
  // Available slots (not excluded)
  const totalOrders = allOrders.length;
  const availableSlots = [];
  for (let i = 1; i <= totalOrders; i++) {
    if (!excludedPrios.has(i)) {
      availableSlots.push(i);
    }
  }
  
  // Assign avail dates and N10D data to movable orders
  const movableWithDates = assignAvailDates(movableOrders, n10dData, dailyColumns, today);
  
  movableWithDates.forEach(order => {
    const product = n10dData?.find(
      p => String(p.material_code_fg) === String(order.material_code_fg)
    );
    if (product && product.inventory > 0) {
      order._n10dRatio = product.due_for_loading / product.inventory;
    } else if (product && product.inventory <= 0) {
      order._n10dRatio = Infinity;
    } else {
      order._n10dRatio = 0;
    }
  });
  
  // Sort movable orders
  movableWithDates.sort((a, b) => {
    const statusOrder = { 'Critical': 0, 'Urgent': 1, 'Monitor': 2, 'Sufficient': 3 };
    const statusA = statusOrder[a._n10dStatus] ?? 3;
    const statusB = statusOrder[b._n10dStatus] ?? 3;
    
    const isCriticalA = statusA === 0;
    const isCriticalB = statusB === 0;
    
    // Critical ALWAYS first
    if (isCriticalA && !isCriticalB) return -1;
    if (!isCriticalA && isCriticalB) return 1;
    
    // Both Critical — highest DFL/Inventory ratio first
    if (isCriticalA && isCriticalB) {
      return (b._n10dRatio || 0) - (a._n10dRatio || 0);
    }
    
    // Neither Critical — chronological
    const dateA = a._sortDate;
    const dateB = b._sortDate;
    if (dateA && !dateB) return -1;
    if (!dateA && dateB) return 1;
    if (!dateA && !dateB) return 0;
    
    const dateDiff = dateA.getTime() - dateB.getTime();
    if (dateDiff !== 0) return dateDiff;
    
    // Same date — urgency tiebreaker
    if (statusA !== statusB) return statusA - statusB;
    
    // Same date, same status — higher volume first
    return (parseFloat(b.volume) || 0) - (parseFloat(a.volume) || 0);
  });
  
  // Separate Critical from non-Critical
  const criticalOrders = movableWithDates.filter(o => o._n10dStatus === 'Critical');
  const nonCriticalOrders = movableWithDates.filter(o => o._n10dStatus !== 'Critical');
  
  // Build planned prio map
  const plannedPrioMap = {};
  const bumpedPlanned = [];
  plannedOrders.forEach(o => {
    plannedPrioMap[o._realPrio] = o;
  });
  
  // Build preview
  const previewResult = [];
  let criticalIndex = 0;
  let nonCriticalIndex = 0;
  
  // First: Critical orders take earliest available slots, even bumping Planned
  const criticalSlots = [];
  const remainingSlots = [];
  
  for (const slot of availableSlots) {
    if (criticalIndex < criticalOrders.length) {
      if (plannedPrioMap[slot]) {
        bumpedPlanned.push(plannedPrioMap[slot]);
        delete plannedPrioMap[slot];
      }
      
      previewResult.push({
        ...criticalOrders[criticalIndex],
        prio: slot,
        _isLocked: false,
        _isCritical: true
      });
      criticalIndex++;
    } else {
      remainingSlots.push(slot);
    }
  }
  
  // Find slots for bumped Planned orders
  const remainingPlannedMap = {};
  const allPlanned = [...Object.values(plannedPrioMap), ...bumpedPlanned];
  
  allPlanned.forEach(o => {
    const originalPrio = o._realPrio;
    if (remainingSlots.includes(originalPrio) && !remainingPlannedMap[originalPrio]) {
      remainingPlannedMap[originalPrio] = o;
    } else {
      const nearestSlot = remainingSlots.find(
        s => s >= originalPrio && !remainingPlannedMap[s]
      );
      if (nearestSlot) {
        remainingPlannedMap[nearestSlot] = o;
      } else {
        const anySlot = remainingSlots.find(s => !remainingPlannedMap[s]);
        if (anySlot) {
          remainingPlannedMap[anySlot] = o;
        }
      }
    }
  });
  
  // Fill remaining slots with Planned and non-Critical
  for (const slot of remainingSlots) {
    if (remainingPlannedMap[slot]) {
      previewResult.push({
        ...remainingPlannedMap[slot],
        prio: slot,
        _isLocked: true
      });
    } else {
      if (nonCriticalIndex < nonCriticalOrders.length) {
        previewResult.push({
          ...nonCriticalOrders[nonCriticalIndex],
          prio: slot,
          _isLocked: false
        });
        nonCriticalIndex++;
      }
    }
  }
  
  // Sort by prio for display
  previewResult.sort((a, b) => a.prio - b.prio);
  
  return { previewResult, excludedOrders };
}
```
