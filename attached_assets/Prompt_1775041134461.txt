## FIX 2: VOLUME COUNTING — EXCLUDE DONE AND CANCELLED

The feedmill cards in the Overview are not counting volume properly. The total volume should only include active orders — exclude Done and Cancelled.

### CURRENT (WRONG)

Possibly counting all orders including Done/Cancelled, or not counting some statuses.

### CORRECT

Count volume from active orders only:

```javascript
function calculateLineVolume(orders, line) {
  const activeOrders = orders.filter(order => {
    // Must be on this line
    if (order.line !== line) return false;
    
    // Exclude Done and Cancelled
    if (order.status === 'Done') return false;
    if (order.status === 'Cancel PO') return false;
    
    // Include everything else
    // Plotted, Planned, Hold, Cut, In Production, On-going, 
    // Uncombine, Merge Back, Custom
    return true;
  });
  
  const totalVolume = activeOrders.reduce((sum, order) => {
    return sum + (parseFloat(order.volume) || 0);
  }, 0);
  
  return totalVolume;
}
```

### APPLY TO FEEDMILL CARD CALCULATIONS

```javascript

  const lines = FEEDMILL_LINE_MAP[feedmill];
  let totalVolume = 0;
  let totalRate = 0;
  
  const lineMetrics = lines.map(line => {
    const lineOrders = allOrders.filter(o => o.line === line);
    
    // Active orders only — exclude Done and Cancelled
    const activeOrders = lineOrders.filter(o => 
      o.status !== 'Done' && o.status !== 'Cancel PO'
    );
    
    const lineVolume = activeOrders.reduce((sum, o) => 
      sum + (parseFloat(o.volume) || 0), 0
    );
    
    const lineRate = calculateLineRate(line); // existing rate calculation
    
    totalVolume += lineVolume;
    totalRate += lineRate;
    
    return {
      line: line,
      volume: lineVolume,
      rate: lineRate,

      hasOrders: activeOrders.length > 0
    };
  });
  
  return {
    feedmill: feedmill,
    totalVolume: totalVolume,
    totalRate: totalRate,
    lines: lineMetrics
  };
}
```

```javascript
function calculateCategoryBreakdown(allOrders, feedmill) {

    .filter(o => o.status === 'In Production' || o.status.startsWith('On-going'))
    .reduce((sum, o) => sum + (parseFloat(o.volume) || 0), 0);
  
  const onHold = feedmillOrders
    .filter(o => o.status === 'Hold')
    .reduce((sum, o) => sum + (parseFloat(o.volume) || 0), 0);
  
  const cancelled = feedmillOrders
    .filter(o => o.status === 'Cancel PO')
    .reduce((sum, o) => sum + (parseFloat(o.volume) || 0), 0);

    .filter(o => ['Plotted', 'Planned', 'Cut', 'Uncombine', 'Merge Back'].includes(o.status) || isCustomStatus(o.status))
```
