## FIX: ORDER TABLE — TEXT SELECTABLE + DRAG FROM EMPTY AREAS

Text in order rows should be selectable. Dragging to reorder works from the grip icon AND from empty areas (padding, gaps, cells with no text). Clicking on actual text characters selects text, not drag.

---

### BEHAVIOR SUMMARY

| Mouse position | Cursor | Click action | Click + drag action |
|---|---|---|---|
| Over actual text characters | I-beam (text) | Place cursor / select | Select text |
| Over empty area in row (padding, gaps) | Grab hand | Nothing | Drag to reorder |
| Over ⠿ grip icon | Grab hand | Nothing | Drag to reorder |
| Over placeholder text ("Set start date") | Pointer | Trigger edit/input | Trigger edit/input |
| Over placeholder padding area | Grab hand | Nothing | Drag to reorder |

---

### IMPLEMENTATION

The key is to detect whether the mouse is over actual text content or empty space. Use event target inspection:

```jsx
function OrderRow({ order, index, onDragStart, onDragEnd, onDragOver, onDrop }) {
  const rowRef = useRef(null);
  const [isDragReady, setIsDragReady] = useState(false);
  
  function isOverText(e) {
    const target = e.target;
    
    // Check if target is an input, textarea, select, or button — not draggable
    if (['INPUT', 'TEXTAREA', 'SELECT', 'BUTTON'].includes(target.tagName)) {
      return true;
    }
    
    // Check if target is a placeholder element (Set start date, Add note, etc.)
    if (target.classList.contains('placeholder-text') || 
        target.classList.contains('editable-trigger')) {
      return true;
    }
    
    // Check if the click is over actual text content
    // Use document.caretRangeFromPoint (or caretPositionFromPoint)
    const range = document.caretRangeFromPoint 
      ? document.caretRangeFromPoint(e.clientX, e.clientY)
      : null;
    
    if (range) {
      const node = range.startContainer;
      // If we hit a text node with actual content (not whitespace only)
      if (node.nodeType === Node.TEXT_NODE && node.textContent.trim().length > 0) {
        // Check if the click X is within the text bounds
        const textRange = document.createRange();
        textRange.selectNodeContents(node);
        const rects = textRange.getClientRects();
        
        for (let i = 0; i < rects.length; i++) {
          const rect = rects[i];
          if (e.clientX >= rect.left && e.clientX <= rect.right &&
              e.clientY >= rect.top && e.clientY <= rect.bottom) {
            return true; // Mouse is over actual text
          }
        }
      }
    }
    
    return false; // Mouse is over empty space
  }
  
  function handleMouseDown(e) {
    // Grip icon always triggers drag
    if (e.target.closest('.drag-handle')) {
      setIsDragReady(true);
      return;
    }
    
    // If over text — allow text selection, no drag
    if (isOverText(e)) {
      setIsDragReady(false);
      return;
    }
    
    // Over empty space — prepare for drag
    if (canDragOrder(order)) {
      setIsDragReady(true);
    }
  }
  
  function handleMouseUp() {
    setIsDragReady(false);
  }
  
  function handleDragStart(e) {
    if (!isDragReady || !canDragOrder(order)) {
      e.preventDefault();
      return;
    }
    onDragStart(e, order, index);
  }
  
  function handleDragEnd(e) {
    setIsDragReady(false);
    onDragEnd(e);
  }
  
  return (
    <tr
      ref={rowRef}
      draggable={isDragReady}
      onMouseDown={handleMouseDown}
      onMouseUp={handleMouseUp}
      onDragStart={handleDragStart}
      onDragEnd={handleDragEnd}
      onDragOver={onDragOver}
      onDrop={(e) => onDrop(e, index)}
      className={`order-row ${isDragReady ? 'drag-ready' : ''}`}
    >
      {/* Drag handle */}
      <td className="drag-handle-cell">
        {canDragOrder(order) ? (
          <div className="drag-handle">⠿</div>
        ) : (
          <div className="drag-handle-disabled"></div>
        )}
      </td>
      
      {/* Data cells — text selectable */}
      <td className="data-cell">{order.prio}</td>
      <td className="data-cell">{order.fpr}</td>
      <td className="data-cell">{order.material_code_fg}</td>
      <td className="data-cell">{order.item_description}</td>
      {/* ... other cells ... */}
    </tr>
  );
}
```

### BROWSER COMPATIBILITY — caretRangeFromPoint

```javascript
function isOverText(e) {
  const target = e.target;
  
  if (['INPUT', 'TEXTAREA', 'SELECT', 'BUTTON'].includes(target.tagName)) {
    return true;
  }
  
  if (target.classList.contains('placeholder-text') || 
      target.classList.contains('editable-trigger')) {
    return true;
  }
  
  // Chrome, Safari, Edge
  if (document.caretRangeFromPoint) {
    const range = document.caretRangeFromPoint(e.clientX, e.clientY);
    if (range && range.startContainer.nodeType === Node.TEXT_NODE) {
      const text = range.startContainer.textContent.trim();
      if (text.length > 0) {
        const textRange = document.createRange();
        textRange.selectNodeContents(range.startContainer);
        const rects = textRange.getClientRects();
        for (let i = 0; i < rects.length; i++) {
          if (e.clientX >= rects[i].left && e.clientX <= rects[i].right &&
              e.clientY >= rects[i].top && e.clientY <= rects[i].bottom) {
            return true;
          }
        }
      }
    }
    return false;
  }
  
  // Firefox
  if (document.caretPositionFromPoint) {
    const pos = document.caretPositionFromPoint(e.clientX, e.clientY);
    if (pos && pos.offsetNode && pos.offsetNode.nodeType === Node.TEXT_NODE) {
      const text = pos.offsetNode.textContent.trim();
      if (text.length > 0) {
        return true;
      }
    }
    return false;
  }
  
  // Fallback — check if target has text content
  if (target.closest('.text-content')) {
    return true;
  }
  
  return false;
}
