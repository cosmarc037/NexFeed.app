### COMPLETION DATE CASCADE LOGIC

```javascript
function recalculateCompletionDates(line) {
  const lineOrders = getOrdersForLine(line);
  
  // Sort by current prio
  lineOrders.sort((a, b) => a.prio - b.prio);
  
  // Find the anchor — first order with a set start date and time
  let currentEndTime = null;
  
  for (let i = 0; i < lineOrders.length; i++) {
    const order = lineOrders[i];
    
    // Skip Done and Cancelled — they already have completion dates
    if (order.status === 'Done' || order.status === 'Cancel PO') {
      continue;
    }
    
    // Get production time for this order (hours)
    const productionTime = parseFloat(order.production_time) || 0;
    const changeover = parseFloat(order.changeover) || 0;
    const totalTime = productionTime + changeover;
    
    if (order.start_date && order.start_time) {
      // This order has a set start — use it as anchor
      const startDateTime = parseDateTime(order.start_date, order.start_time);
      
      if (startDateTime) {
        // Calculate completion = start + production time + changeover
        const completionDateTime = new Date(
          startDateTime.getTime() + (totalTime * 60 * 60 * 1000)
        );
        
        order.completion_date = formatDateFull(completionDateTime);
        order.completion_time = formatTime(completionDateTime);
        
        currentEndTime = completionDateTime;
      }
    } else if (currentEndTime) {
      // No start date set — cascade from previous order's completion
      const startDateTime = new Date(currentEndTime);
      
      order._cascadedStartDate = formatDateFull(startDateTime);
      order._cascadedStartTime = formatTime(startDateTime);
      
      const completionDateTime = new Date(
        startDateTime.getTime() + (totalTime * 60 * 60 * 1000)
      );
      
      order.completion_date = formatDateFull(completionDateTime);
      order.completion_time = formatTime(completionDateTime);
      
      currentEndTime = completionDateTime;
    }
    // If no anchor found yet (no order above has start date), skip
  }
}

function parseDateTime(dateStr, timeStr) {
  const date = parseDate(dateStr);
  if (!date) return null;
  
  // Parse time like "08:00 AM", "02:21 PM"
  const timeParts = timeStr.match(/(\d{1,2}):(\d{2})\s*(AM|PM)/i);
  if (!timeParts) return null;
  
  let hours = parseInt(timeParts[1]);
  const minutes = parseInt(timeParts[2]);
  const ampm = timeParts[3].toUpperCase();
  
  if (ampm === 'PM' && hours !== 12) hours += 12;
  if (ampm === 'AM' && hours === 12) hours = 0;
  
  date.setHours(hours, minutes, 0, 0);
  return date;
}

function formatTime(date) {
  if (!date) return '';
  let hours = date.getHours();
  const minutes = date.getMinutes();
  const ampm = hours >= 12 ? 'PM' : 'AM';
  hours = hours % 12;
  if (hours === 0) hours = 12;
  return `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')} ${ampm}`;
}
