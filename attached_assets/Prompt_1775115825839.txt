Implement the following date restriction rules across the entire NexFeed application:

---

## DATE FIELDS — DEFINITIONS

| Field | What it means | Auto or Manual |
|---|---|---|
| Start Date | Expected production start date | Manual — user sets |
| Completion Date | Expected production finish date | Auto-cascaded or manual |
| Avail Date | Customer deadline / availability target | From SAP upload, N10D, or manual |
| End Date | ACTUAL date production completed | Auto-set on Done, can be overridden |

---

## DATE RULES

### CORE RULE

```
Start Date ≤ Completion Date ≤ Avail Date
End Date ≥ Start Date (if start date exists)
End Date — fallback to Avail Date check if Start Date is empty
```

### PER-FIELD CONSTRAINTS

| Field | Min date | Max date | Fallback |
|---|---|---|---|
| Start Date | Today | Avail Date (if real date) | If no avail date: no max |
| Completion Date | Start Date (if set), else Today | Avail Date (if real date) | If no start date: Today. If no avail: no max |
| Avail Date | Today | No upper limit | — |
| End Date | Start Date (if set) | No upper limit | If no Start Date: no min restriction, but warn if below Avail Date |

---

## IMPLEMENTATION

### 1. HELPER FUNCTIONS

```javascript
function getMinSelectableDate() {
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  return today;
}

function isRealDate(availDate) {
  if (!availDate) return false;
  
  const nonDateValues = [
    'prio replenish', 'safety stocks', 'safety stock',
    'stock sufficient'
  ];
  
  if (typeof availDate === 'string' && 
      nonDateValues.includes(availDate.toLowerCase())) return false;
  
  const parsed = parseDate(availDate);
  return parsed !== null && !isNaN(parsed.getTime());
}
```

function getStartDateConstraints(order) {

function getCompletionDateConstraints(order) {
  const today = getMinSelectableDate();
  
  let minDate = today;
  if (order.start_date) {
    const startDate = parseDate(order.start_date);
    if (startDate && startDate > today) {
      minDate = startDate;
    }
  }
  
  let maxDate = null;
  if (order.avail_date && isRealDate(order.avail_date)) {
    maxDate = parseDate(order.avail_date);
  }
  
  return {
    minDate: minDate,
    maxDate: maxDate
  };
}

```javascript
function getEndDateConstraints(order) {
  // If Start Date exists — End Date must be on or after Start Date
  if (order.start_date) {
    const startDate = parseDate(order.start_date);
    if (startDate) {
      return {
        minDate: startDate,
        maxDate: null,
        warnIfBelowAvail: true
      };
    }
  }
  
  // If Start Date is empty — no min restriction
  // But warn if End Date is before Avail Date
  return {
    minDate: null,  // No restriction — user can pick any date
    maxDate: null,
    warnIfBelowAvail: true
  };
}
```

### 3. CALENDAR DATE PICKER — GREY OUT PAST DATES

```jsx
function DatePickerField({ 
  value, 
  minDate, 
  maxDate, 
  onChange, 
  onWarnBelowAvail,
  order,
  field
}) {
  function handleDateSelect(selectedDate) {
    // Check min date
    if (minDate && selectedDate < minDate) {
      return; // Greyed out, shouldn't reach here
    }
    
    // Check max date
    if (maxDate && selectedDate > maxDate) {
      return; // Greyed out, shouldn't reach here
    }
    
    // End Date — check if below Avail Date
    if (field === 'end_date' && onWarnBelowAvail) {
      const availDate = isRealDate(order.avail_date) 
        ? parseDate(order.avail_date) 
        : null;
      
      if (availDate && selectedDate < availDate) {
        // Show warning dialog
        onWarnBelowAvail(selectedDate, availDate);
        return;
      }
    }
    
    onChange(selectedDate);
  }
  
  function isDayDisabled(day) {
    if (minDate && day < minDate) return true;
    if (maxDate && day > maxDate) return true;
    return false;
  }
  
  return (
    <Calendar
      value={value ? parseDate(value) : null}
      onChange={handleDateSelect}
      tileDisabled={({ date }) => isDayDisabled(date)}
      tileClassName={({ date }) => isDayDisabled(date) ? 'date-disabled' : ''}
    />
  );
}
```

### HANDLER

```javascript
function handleEndDateSelection(order, field, allOrders, line) {
  const constraints = getEndDateConstraints(order);
  
  openCalendar({
    currentValue: order.end_date,
    minDate: constraints.minDate,
    maxDate: constraints.maxDate,
    onSelect: (selectedDate) => {
      // Check if below avail date
      const availDate = isRealDate(order.avail_date) 
        ? parseDate(order.avail_date) 
        : null;
      
      if (availDate && selectedDate < availDate) {
        // Show warning dialog
        showEndDateWarningDialog({
          selectedDate: selectedDate,
          availDate: availDate,
          onCancel: () => {
            // Do nothing — calendar stays open or closes
          },
          onProceed: (date) => {
            order.end_date = formatDateFull(date);
            // No cascade needed for end date
          }
        });
      } else {
        // No issue — apply directly
        order.end_date = formatDateFull(selectedDate);
      }
    }
  });
}

### 5. DATE CONFLICT WARNINGS — RED CELL HIGHLIGHT

For any dates that violate the rules (e.g., existing orders with conflicting dates):

```javascript
function checkDateConflicts(order) {
  const warnings = [];
  
  const startDate = order.start_date ? parseDate(order.start_date) : null;
  const completionDate = order.completion_date ? parseDate(order.completion_date) : null;
  const availDate = isRealDate(order.avail_date) ? parseDate(order.avail_date) : null;
  const endDate = order.end_date ? parseDate(order.end_date) : null;
  
  // Start date after completion date
  if (startDate && completionDate && startDate > completionDate) {
    warnings.push({
      field: 'start_date',
      type: 'conflict',
      message: 'Start date is after the completion date.'
    });
  }
  
  // Start date after avail date
  if (startDate && availDate && startDate > availDate) {
    warnings.push({
      field: 'start_date',
      type: 'conflict',
      message: 'Start date is after the availability date.'
    });
  }
  
  // Completion date after avail date
  if (completionDate && availDate && completionDate > availDate) {
    warnings.push({
      field: 'completion_date',
      type: 'conflict',
      message: 'Completion date is after the availability date.'
    });
  }
  
  // End date before start date
  if (endDate && startDate && endDate < startDate) {
    warnings.push({
      field: 'end_date',
      type: 'conflict',
      message: 'End date is before the start date.'
    });
  }
  
  // End date before avail date (soft warning — allowed but flagged)
  if (endDate && availDate && endDate < availDate) {
    warnings.push({
      field: 'end_date',
      type: 'info',
      message: 'End date is before the availability date.'
    });
  }
  
  return warnings;
}
```

### WARNING CELL STYLING

```jsx
function DateCell({ order, field, value }) {
  const warnings = checkDateConflicts(order);
  const fieldWarning = warnings.find(w => w.field === field);
  const hasConflict = fieldWarning && fieldWarning.type === 'conflict';
  const hasInfo = fieldWarning && fieldWarning.type === 'info';
  
  return (
    <td className={`date-cell ${hasConflict ? 'date-conflict' : ''} ${hasInfo ? 'date-info' : ''}`}>
      <span className="text-content">{value}</span>
      {fieldWarning && (
        <span 
          className={`date-warning-icon ${hasInfo ? 'info' : ''}`}
          title={fieldWarning.message}
        >
          ⚠
        </span>
      )}
    </td>
  );
}
```

### 6. EXISTING PAST DATES — KEEP BUT RESTRICT ON EDIT

Orders with dates already in the past display their dates normally. When user opens calendar to change, past dates are greyed out:

```javascript
function handleDateEdit(order, field) {
  const currentValue = order[field];
  const constraints = getConstraintsForField(order, field);
  
  openCalendar({
    currentValue: currentValue,
    minDate: constraints.minDate,
    maxDate: constraints.maxDate,
    onSelect: (newDate) => {
      if (constraints.minDate && newDate < constraints.minDate) {
        return;
      }
      if (constraints.maxDate && newDate > constraints.maxDate) {
        return;
      }
      
      order[field] = formatDateFull(newDate);
      
      // Trigger cascade if needed
      if (field === 'start_date' || field === 'completion_date') {
        recalculateCompletionDates(order.line);
      }
    }
  });
}

function getConstraintsForField(order, field) {
  switch (field) {
    case 'start_date':
      return getStartDateConstraints(order);
    case 'completion_date':
      return getCompletionDateConstraints(order);
    case 'avail_date':
      return getAvailDateConstraints(order);
    case 'end_date':
      return getEndDateConstraints(order);
    default:
      return { minDate: getMinSelectableDate(), maxDate: null };
  }
}
