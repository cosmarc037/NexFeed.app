### IMPLEMENTATION

```jsx
function DateCellWithWarning({ order, field, value }) {
  const [showTooltip, setShowTooltip] = useState(false);
  const [tooltipPos, setTooltipPos] = useState({ x: 0, y: 0 });
  const timeoutRef = useRef(null);
  
  const warnings = checkDateConflicts(order);
  const fieldWarning = warnings.find(w => w.field === field);
  const hasConflict = fieldWarning && fieldWarning.type === 'conflict';
  const hasInfo = fieldWarning && fieldWarning.type === 'info';
  const hasWarning = hasConflict || hasInfo;
  
  function handleCellMouseEnter(e) {
    if (!hasWarning) return;
    
    timeoutRef.current = setTimeout(() => {
      const pos = clampToViewport(e.clientX + 12, e.clientY + 12, 300, 60);
      setTooltipPos(pos);
      setShowTooltip(true);
    }, 300);
  }
  
  function handleCellMouseMove(e) {
    if (!hasWarning || !showTooltip) return;
    
    const pos = clampToViewport(e.clientX + 12, e.clientY + 12, 300, 60);
    setTooltipPos(pos);
  }
  
  function handleCellMouseLeave() {
    clearTimeout(timeoutRef.current);
    setShowTooltip(false);
  }
  
  return (
    <td
      className={`date-cell ${hasConflict ? 'date-conflict' : ''} ${hasInfo ? 'date-info' : ''}`}
      onMouseEnter={handleCellMouseEnter}
      onMouseMove={handleCellMouseMove}
      onMouseLeave={handleCellMouseLeave}
    >
      <div className="date-cell-content">
        <span className="text-content">{value}</span>
        {hasWarning && (
          <span className={`date-warning-icon ${hasInfo ? 'info' : ''}`}>
            ⚠
          </span>
        )}
      </div>
      
      {showTooltip && hasWarning && (
        <TooltipPortal>
          <div
            className="date-warning-tooltip"
            style={{
              position: 'fixed',
              left: tooltipPos.x,
              top: tooltipPos.y,
              zIndex: 99999,
              pointerEvents: 'none'
            }}
          >
            {fieldWarning.message}
          </div>
        </TooltipPortal>
      )}
    </td>
  );
}
```
