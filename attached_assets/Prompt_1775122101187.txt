```jsx
function PreviewRow({ order, index, onDragStart, onDragEnd, onDragOver, onDrop }) {
  const [isDragReady, setIsDragReady] = useState(false);
  
  function isOverText(e) {
    const target = e.target;
    
    if (['INPUT', 'TEXTAREA', 'SELECT', 'BUTTON'].includes(target.tagName)) {
      return true;
    }
    
    if (target.closest('.drag-handle')) return false;
    
    if (document.caretRangeFromPoint) {
      const range = document.caretRangeFromPoint(e.clientX, e.clientY);
      if (range && range.startContainer.nodeType === Node.TEXT_NODE) {
        const text = range.startContainer.textContent.trim();
        if (text.length > 0) {
          const textRange = document.createRange();
          textRange.selectNodeContents(range.startContainer);
          const rects = textRange.getClientRects();
          for (let i = 0; i < rects.length; i++) {
            if (e.clientX >= rects[i].left && e.clientX <= rects[i].right &&
                e.clientY >= rects[i].top && e.clientY <= rects[i].bottom) {
              return true;
            }
          }
        }
      }
    }
    
    if (target.closest('.text-content')) return true;
    
    return false;
  }
  
  function handleMouseDown(e) {
    if (e.target.closest('.drag-handle')) {
      setIsDragReady(true);
      return;
    }
    
    if (isOverText(e)) {
      setIsDragReady(false);
      return;
    }
    
    setIsDragReady(true);
  }
  
  function handleMouseUp() {
    setIsDragReady(false);
  }
  
  return (
    <tr
      draggable={isDragReady}
      onMouseDown={handleMouseDown}
      onMouseUp={handleMouseUp}
      onDragStart={(e) => {
        if (!isDragReady) { e.preventDefault(); return; }
        onDragStart(e, order, index);
      }}
      onDragEnd={onDragEnd}
      onDragOver={onDragOver}
      onDrop={(e) => onDrop(e, index)}
      className={`preview-row ${order._isLocked ? 'preview-row-planned' : ''}`}
    >
      <td className="drag-handle-cell">
        <div className="drag-handle">⠿</div>
      </td>
      
      <td className="data-cell">
        <span className="text-content">
          {order.prio}
          {order._isLocked && <span className="lock-icon">🔒</span>}
        </span>
      </td>
      
      {/* ... other cells ... */}
    </tr>
  );
}
```

### DRAG HANDLER — WITH WARNINGS

```javascript
function handlePreviewDrop(draggedOrder, targetIndex, previewOrders) {
  const targetOrder = previewOrders[targetIndex];
  
  // Warning: Dragging above a Planned order
  if (targetOrder && targetOrder._isLocked && !draggedOrder._isLocked) {
    showDialog({
      title: 'Planned Order Will Be Affected',
      message: `Inserting this order here will push "${targetOrder.item_description}" (Planned at Priority ${targetOrder.prio}) down. Consider adjusting the Planned order first, or proceed if this is intentional.`,
      buttons: [
        { label: 'Cancel', action: 'cancel' },
        { label: 'Proceed Anyway', action: 'proceed', style: 'orange' }
      ],
      onAction: (action) => {
        if (action === 'proceed') {
          executePreviewDrop(draggedOrder, targetIndex, previewOrders);
        }
      }
    });
    return;
  }
  
  // Warning: Moving a Planned order
  if (draggedOrder._isLocked) {
    const newPrio = previewOrders[targetIndex]?.prio || targetIndex + 1;
    showDialog({
      title: 'Moving Planned Order',
      message: `"${draggedOrder.item_description}" is currently Planned at Priority ${draggedOrder.prio}. Moving it to Priority ${newPrio} will change its locked position. Do you want to proceed?`,
      buttons: [
        { label: 'Cancel', action: 'cancel' },
        { label: 'Proceed', action: 'proceed', style: 'orange' }
      ],
      onAction: (action) => {
        if (action === 'proceed') {
          draggedOrder._plannedPrio = newPrio;
          executePreviewDrop(draggedOrder, targetIndex, previewOrders);
        }
      }
    });
    return;
  }
  
  // No warning needed — execute directly
  executePreviewDrop(draggedOrder, targetIndex, previewOrders);
}

function executePreviewDrop(draggedOrder, targetIndex, previewOrders) {
  // Remove from current position
  const currentIndex = previewOrders.findIndex(o => o.id === draggedOrder.id);
  if (currentIndex === -1) return;
  
  previewOrders.splice(currentIndex, 1);
  previewOrders.splice(targetIndex, 0, draggedOrder);
  
  // Reassign prio numbers using available slots
  // (same slot assignment as auto-sequence — skip excluded prios)
  reassignPreviewPrios(previewOrders);
  
  // Mark as manually modified
  draggedOrder._manuallyMoved = true;
  
  triggerPreviewRefresh();
}
```

### HELPER FUNCTIONS

```javascript
function formatProductionHours(hours) {
  if (!hours || hours === 0) return '0.00';
  return parseFloat(hours).toFixed(2);
}

function formatChangeover(hours) {
  if (!hours || hours === 0) return '0:00';
  
  const totalMinutes = Math.round(parseFloat(hours) * 60);
  const h = Math.floor(totalMinutes / 60);
  const m = totalMinutes % 60;
  return `${h}:${m.toString().padStart(2, '0')}`;
}
```
