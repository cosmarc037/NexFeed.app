```jsx
function MultiSelectFilter({ column, allValues, activeValues, onApply, onClear, onClose }) {
  const [selected, setSelected] = useState(new Set(activeValues || []));
  const [searchText, setSearchText] = useState('');
  
  const filteredValues = allValues.filter(v => 
    v.toLowerCase().includes(searchText.toLowerCase())
  );
  
  function toggleValue(val) {
    const next = new Set(selected);
    if (next.has(val)) next.delete(val);
    else next.add(val);
    setSelected(next);
  }
  
  function selectAll() {
    setSelected(new Set(allValues));
  }
  
  function deselectAll() {
    setSelected(new Set());
  }
  
  return (
    <div className="filter-dropdown">
      <div className="filter-dropdown-header">
        <span className="filter-dropdown-title">Filter: {column.label}</span>
        <button className="filter-close-btn" onClick={onClose}>✕</button>
      </div>
      
      {/* Search within options */}
      <div className="filter-search">
        <input
          type="text"
          placeholder="Search..."
          value={searchText}
          onChange={(e) => setSearchText(e.target.value)}
          className="filter-search-input"
        />
      </div>
      
      {/* Select all / Deselect all */}
      <div className="filter-actions">
        <button className="filter-action-btn" onClick={selectAll}>Select all</button>
        <button className="filter-action-btn" onClick={deselectAll}>Deselect all</button>
      </div>
      
      {/* Checkbox list */}
      <div className="filter-options-list">
        {filteredValues.map(val => (
          <label key={val} className="filter-option">
            <input
              type="checkbox"
              checked={selected.has(val)}
              onChange={() => toggleValue(val)}
            />
            <span>{val || '(empty)'}</span>
          </label>
        ))}
      </div>
      
      {/* Apply / Clear */}
      <div className="filter-dropdown-footer">
        <button className="filter-clear-btn" onClick={() => { onClear(); onClose(); }}>
          Clear
        </button>
        <button 
          className="filter-apply-btn"
          onClick={() => { onApply(Array.from(selected)); onClose(); }}
        >
          Apply
        </button>
      </div>
    </div>
  );
}

## 4. FILTER ENGINE

```javascript
function applyFilters(orders, activeFilters) {
  if (Object.keys(activeFilters).length === 0) return orders;
  
  return orders.filter(order => {
    for (const [columnKey, filter] of Object.entries(activeFilters)) {
      const config = COLUMN_FILTER_CONFIG[columnKey];
      if (!config) continue;
      
      const value = order[columnKey];
      
      switch (config.type) {
        case 'multi-select': {
          if (filter.values && filter.values.length > 0) {
            if (!filter.values.includes(value)) return false;
          }
          break;
        }
        
        case 'text-search': {
          if (filter.text && filter.text.trim() !== '') {
            const searchText = filter.text.toLowerCase();
            const cellValue = (value || '').toString().toLowerCase();
            if (!cellValue.includes(searchText)) return false;
          }
          break;
        }
        
        case 'number-range': {
          const numValue = parseFloat(value) || 0;
          if (filter.min !== null && filter.min !== undefined && numValue < filter.min) return false;
          if (filter.max !== null && filter.max !== undefined && numValue > filter.max) return false;
          break;
        }
        
        case 'date-range': {
          if (!value || !isRealDate(value)) {
            if (filter.from || filter.to) return false;
            break;
          }
          const dateValue = parseDate(value);
          if (filter.from) {
            const fromDate = parseDate(filter.from);
            if (fromDate && dateValue < fromDate) return false;
          }
          if (filter.to) {
            const toDate = parseDate(filter.to);
            if (toDate && dateValue > toDate) return false;
          }
          break;
        }
        
        case 'time-range': {
          if (!value) {
            if (filter.from || filter.to) return false;
            break;
          }
          if (filter.from && value < filter.from) return false;
          if (filter.to && value > filter.to) return false;
          break;
        }
        
        case 'toggle': {
          if (filter.value && filter.value !== 'any') {
            if (filter.value === 'checked' && !value) return false;
            if (filter.value === 'unchecked' && value) return false;
            if (filter.value === 'has_history' && (!order.history || order.history.length === 0)) return false;
            if (filter.value === 'none' && order.history && order.history.length > 0) return false;
          }
          break;
        }
      }
    }
    
    return true;
  });
}
```

## 5. FILTER STATE MANAGEMENT

```javascript
function useOrderFilters() {
  // Filters are per line tab
  const [filtersByLine, setFiltersByLine] = useState({});
  
  function getActiveFilters(line) {
    return filtersByLine[line] || {};
  }
  
  function setFilter(line, columnKey, filterData) {
    setFiltersByLine(prev => ({
      ...prev,
      [line]: {
        ...(prev[line] || {}),
        [columnKey]: filterData
      }
    }));
  }
  
  function clearFilter(line, columnKey) {
    setFiltersByLine(prev => {
      const lineFilters = { ...(prev[line] || {}) };
      delete lineFilters[columnKey];
      return { ...prev, [line]: lineFilters };
    });
  }
  
  function clearAllFilters(line) {
    setFiltersByLine(prev => ({
      ...prev,
      [line]: {}
    }));
  }
  
  function hasActiveFilters(line) {
    const filters = filtersByLine[line] || {};
    return Object.keys(filters).length > 0;
  }
  
  return {
    getActiveFilters,
    setFilter,
    clearFilter,
    clearAllFilters,
    hasActiveFilters
  };
}
```

## 8. APPLY FILTERS TO TABLE RENDERING

```jsx
function OrderTable({ allOrders, currentLine, filters }) {
  const lineOrders = allOrders.filter(o => o.line === currentLine);
  const activeFilters = filters.getActiveFilters(currentLine);
  const filteredOrders = applyFilters(lineOrders, activeFilters);
  
  return (
    <table>
      <thead>
        <tr>
          {columns.map(col => (
            <ColumnHeader
              key={col.key}
              column={{
                ...col,
                filterActive: !!activeFilters[col.key]
              }}
              onFilter={(column, e) => openFilterDropdown(column, e)}
              onContextMenu={(e, column) => openContextMenu(e, column)}
            />
          ))}
        </tr>
      </thead>
      <tbody>
        {filteredOrders.map(order => (
          <OrderRow key={order.id} order={order} />
        ))}
      </tbody>
    </table>
  );
}
```
