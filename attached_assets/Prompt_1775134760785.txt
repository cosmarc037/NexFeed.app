## UPDATED FILTER ENGINE

```javascript
function applyFilters(orders, activeFilters) {
  if (Object.keys(activeFilters).length === 0) return orders;
  
  return orders.filter(order => {
    for (const [columnKey, filter] of Object.entries(activeFilters)) {
      
      switch (columnKey) {
        
        case 'item_description': {
          // Two filters: items and categories
          if (filter.categories && filter.categories.length > 0) {
            if (!filter.categories.includes(order.category)) return false;
          }
          if (filter.items && filter.items.length > 0) {
            if (!filter.items.includes(order.item_description)) return false;
          }
          break;
        }
        
        case 'production_time': {
          // Three filters: production hours, changeover, run rate
          const prodTime = parseFloat(order.production_time) || 0;
          const changeover = parseFloat(order.changeover) || 0;
          const runRate = parseFloat(order.run_rate) || 0;
          
          if (filter.prodMin !== null && filter.prodMin !== undefined && prodTime < filter.prodMin) return false;
          if (filter.prodMax !== null && filter.prodMax !== undefined && prodTime > filter.prodMax) return false;
          if (filter.coMin !== null && filter.coMin !== undefined && changeover < filter.coMin) return false;
          if (filter.coMax !== null && filter.coMax !== undefined && changeover > filter.coMax) return false;
          if (filter.rateMin !== null && filter.rateMin !== undefined && runRate < filter.rateMin) return false;
          if (filter.rateMax !== null && filter.rateMax !== undefined && runRate > filter.rateMax) return false;
          break;
        }
        
        case 'ha_info': {
          // Three filters: HA number, SCADA text, PV text
          const ha = parseFloat(order.ha) || 0;
          const scada = (order.scada || '').toString().toLowerCase();
          const pv = (order.pv || '').toString().toLowerCase();
          
          if (filter.haMin !== null && filter.haMin !== undefined && ha < filter.haMin) return false;
          if (filter.haMax !== null && filter.haMax !== undefined && ha > filter.haMax) return false;
          if (filter.scadaText && filter.scadaText.trim() !== '') {
            if (!scada.includes(filter.scadaText.toLowerCase())) return false;
          }
          if (filter.pvText && filter.pvText.trim() !== '') {
            if (!pv.includes(filter.pvText.toLowerCase())) return false;
          }
          break;
        }
        
        case 'status': {
          if (filter.values && filter.values.length > 0) {
            const orderStatus = order.status;
            const isCustom = isCustomStatus(orderStatus);
            
            if (isCustom) {
              if (!filter.values.includes('Custom')) return false;
            } else {
              if (!filter.values.includes(orderStatus)) return false;
            }
          }
          break;
        }
        
        case 'avail_date': {
          // Date range filter
          if (filter.fromDate || filter.toDate) {
            const orderDate = isRealDate(order.avail_date) ? parseDate(order.avail_date) : null;
            if (!orderDate) {
              if (filter.fromDate || filter.toDate) {
                // Has sub-text filter? Check that. Otherwise exclude.
                if (!filter.subTexts || filter.subTexts.length === 0) return false;
              }
            } else {
              if (filter.fromDate) {
                const from = parseDate(filter.fromDate);
                if (from && orderDate < from) return false;
              }
              if (filter.toDate) {
                const to = parseDate(filter.toDate);
                if (to && orderDate > to) return false;
              }
            }
          }
          
          // Sub-text filter
          if (filter.subTexts && filter.subTexts.length > 0) {
            const subText = order._originalAvailDate || order.avail_date;
            let matchesSubText = false;
            
            for (const st of filter.subTexts) {
              if (subText && subText.toLowerCase().includes(st.toLowerCase())) {
                matchesSubText = true;
                break;
              }
            }
            
            if (!matchesSubText) return false;
          }
          break;
        }
        
        case 'completion_date':
        case 'end_date': {
          const dateField = columnKey === 'completion_date' ? 'completion_date' : 'end_date';
          const timeField = columnKey === 'completion_date' ? 'completion_time' : 'end_time';
          
          const orderDate = order[dateField] ? parseDate(order[dateField]) : null;
          const orderTime = order[timeField] || '';
          
          // Date range
          if (filter.fromDate || filter.toDate) {
            if (!orderDate) return false;
            if (filter.fromDate) {
              const from = parseDate(filter.fromDate);
              if (from && orderDate < from) return false;
            }
            if (filter.toDate) {
              const to = parseDate(filter.toDate);
              if (to && orderDate > to) return false;
            }
          }
          
          // Time range
          if (filter.fromTime || filter.toTime) {
            if (!orderTime) return false;
            const timeValue = orderTime.replace(/\s+/g, '').toUpperCase();
            if (filter.fromTime && timeValue < filter.fromTime) return false;
            if (filter.toTime && timeValue > filter.toTime) return false;
          }
          break;
        }
        
        // Default handlers for other columns remain the same
        default: {
          const config = COLUMN_FILTER_CONFIG[columnKey];
          if (!config) continue;
          
          const value = order[columnKey];
          
          switch (config.type) {
            case 'multi-select': {
              if (filter.values && filter.values.length > 0) {
                if (!filter.values.includes(value)) return false;
              }
              break;
            }
            case 'text-search': {
              if (filter.text && filter.text.trim() !== '') {
                const cellValue = (value || '').toString().toLowerCase();
                if (!cellValue.includes(filter.text.toLowerCase())) return false;
              }
              break;
            }
            case 'number-range': {
              const numValue = parseFloat(value) || 0;
              if (filter.min !== null && filter.min !== undefined && numValue < filter.min) return false;
              if (filter.max !== null && filter.max !== undefined && numValue > filter.max) return false;
              break;
            }
            case 'date-range': {
              if (!value || !isRealDate(value)) {
                if (filter.from || filter.to) return false;
                break;
              }
              const dateValue = parseDate(value);
              if (filter.from && dateValue < parseDate(filter.from)) return false;
              if (filter.to && dateValue > parseDate(filter.to)) return false;
              break;
            }
            case 'toggle': {
              if (filter.value && filter.value !== 'any') {
                if (filter.value === 'checked' && !value) return false;
                if (filter.value === 'unchecked' && value) return false;
                if (filter.value === 'has_history' && (!order.history || order.history.length === 0)) return false;
                if (filter.value === 'none' && order.history && order.history.length > 0) return false;
              }
              break;
            }
          }
        }
      }
    }
    
    return true;
  });
}
```
