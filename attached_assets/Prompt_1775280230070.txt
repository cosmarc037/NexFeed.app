### UPDATED VALIDATION FUNCTION

```javascript
function validateKnowledgeBase(data) {
  const errors = [];
  
  data.forEach((row, index) => {
    // Only FG Material Code and FG Item Description are required
    REQUIRED_FIELDS.forEach(f => {
      const value = row[f.key];
      if (!value || value.toString().trim() === '' || value === '-') {
        errors.push({
          rowIndex: index,
          rowId: row.id,
          field: f.key,
          message: `${f.label} is required`
        });
      }
    });
    
    // Numeric validation — only if a value is entered (not required)
    // Batch sizes — optional, but if filled must be positive number
    const BATCH_FIELDS = ['batch_fm1', 'batch_fm2', 'batch_fm3', 'batch_pm'];
    BATCH_FIELDS.forEach(f => {
      const val = row[f];
      if (val && val !== '-' && val.toString().trim() !== '') {
        if (isNaN(parseFloat(val)) || parseFloat(val) <= 0) {
          errors.push({
            rowIndex: index,
            rowId: row.id,
            field: f,
            message: `Must be a positive number`
          });
        }
      }
    });
    
    // Run rates — optional, but if filled must be positive number
    const RUN_RATE_FIELDS = [
      'run_rate_line1', 'run_rate_line2', 'run_rate_line3', 'run_rate_line4',
      'run_rate_line5', 'run_rate_line6', 'run_rate_line7'
    ];
    RUN_RATE_FIELDS.forEach(f => {
      const val = row[f];
      if (val && val !== '-' && val.toString().trim() !== '') {
        if (isNaN(parseFloat(val)) || parseFloat(val) <= 0) {
          errors.push({
            rowIndex: index,
            rowId: row.id,
            field: f,
            message: `Must be a positive number`
          });
        }
      }
    });
    
    // FG Weight — optional, but if filled must be positive number
    const fgWeight = row.fg_weight;
    if (fgWeight && fgWeight !== '-' && fgWeight.toString().trim() !== '') {
      if (isNaN(parseFloat(fgWeight)) || parseFloat(fgWeight) <= 0) {
        errors.push({
          rowIndex: index,
          rowId: row.id,
          field: 'fg_weight',
          message: `FG Weight must be a positive number`
        });
      }
    }
    
    // Form — optional, but if filled must be valid
    const validForms = ['P', 'MP', 'MP-CS', 'M', 'MX', 'C-S', 'C-M', 'C-L'];
    if (row.form && row.form !== '-' && row.form.toString().trim() !== '') {
      if (!validForms.includes(row.form)) {
        errors.push({
          rowIndex: index,
          rowId: row.id,
          field: 'form',
          message: `Form must be one of: ${validForms.join(', ')}`
        });
      }
    }
  });
  
  return errors;
}
```
