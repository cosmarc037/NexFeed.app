```jsx
function HistoryNotesCell({ summary, details }) {
  const [showTooltip, setShowTooltip] = useState(false);

  const timeoutRef = useRef(null);
  
  function handleMouseEnter(e) {
    timeoutRef.current = setTimeout(() => {
      // Position relative to the cell, not cursor
      if (cellRef.current) {
        const rect = cellRef.current.getBoundingClientRect();
        const pos = clampToViewport(
          rect.left,           // Align left edge with cell
          rect.bottom + 4,     // Just below the cell with 4px gap
          400, 
          200
        );
        setTooltipPos(pos);
      }
      setShowTooltip(true);
    }, 300);
  }
  
  function handleMouseLeave() {
    clearTimeout(timeoutRef.current);
    setShowTooltip(false);
  }
  
  const fullText = details ? `${summary}\n\nDetails:\n${details}` : summary;
  const hasDetails = details && details.trim().length > 0;
  
  return (
    <td
      ref={cellRef}
      className="history-notes-cell"
      onMouseEnter={handleMouseEnter}
      onMouseLeave={handleMouseLeave}
    >
      <span className="history-notes-text">{summary}</span>
      
      {showTooltip && (
        <TooltipPortal>
          <div
            className="history-notes-tooltip"
```

### VERIFY generateChangeSummary RETURNS DETAILS

```javascript
function generateChangeSummary() {
  const added = changes.filter(c => c.type === 'add');
  const deleted = changes.filter(c => c.type === 'delete');
  
  // Deduplicate modifications per row+field
  const modMap = {};
  changes.filter(c => c.type === 'modify').forEach(c => {
    const key = `${c.rowId}_${c.field}`;
    modMap[key] = c;
  });
  const modified = Object.values(modMap);
  
  // Group by row
  const modifiedByRow = {};
  modified.forEach(c => {
    if (!modifiedByRow[c.rowId]) {
      modifiedByRow[c.rowId] = {
        materialCode: c.materialCode,
        itemDescription: c.itemDescription,
        changes: []
      };
    }
    modifiedByRow[c.rowId].changes.push(c);
  });
  
  const modifiedRowCount = Object.keys(modifiedByRow).length;
  
  // Summary line
  const parts = [];
  if (modifiedRowCount > 0) parts.push(`${modifiedRowCount} row(s) modified`);
  if (added.length > 0) parts.push(`${added.length} row(s) added`);
  if (deleted.length > 0) parts.push(`${deleted.length} row(s) deleted`);
  const summary = `Manual edit — ${parts.join(', ')}`;
  
  // DETAILS — must build this
  const detailLines = [];
  
  added.forEach(c => {
    detailLines.push(
      `- Added new row: ${c.materialCode || 'New'} - ${c.itemDescription || 'Untitled'}`
    );
  });
  
  deleted.forEach(c => {
    detailLines.push(
      `- Deleted row: ${c.materialCode} - ${c.itemDescription}`
    );
  });
  
  Object.values(modifiedByRow).forEach(row => {
    const fieldChanges = row.changes.map(c => {
      const label = getFieldLabel(c.field);
      return `${label}: ${c.oldValue || '-'} → ${c.newValue || '-'}`;
    }).join(', ');
    detailLines.push(
      `- Modified ${row.materialCode} - ${row.itemDescription}: ${fieldChanges}`
    );
  });
  
  const details = detailLines.join('\n');
  
  return { summary, details };
}
```

When deleting the active entry, the next entry in line becomes active and its snapshot becomes the current KB data:

```javascript
function handleDeleteWithActiveTransfer(entry, index) {
  const isActive = index === 0;
  
  if (isActive && history.length <= 1) {
    // Cannot delete the only entry
    showNotification('warning', 'Cannot delete the only history entry. Upload a new file or make changes first.');
    return;
  }
  
  // Show confirmation dialog
  showDeleteConfirmDialog(entry, () => {
    if (isActive) {
      // Transfer active to next entry
      const nextEntry = history[1]; // Next in line
      
      if (nextEntry && nextEntry.snapshot) {
        // Load the next entry's snapshot as current KB data
        setKbData(JSON.parse(JSON.stringify(nextEntry.snapshot)));
        showNotification('info', 
          `Active version deleted. Reverted to version from ${formatDateTimeFull(nextEntry.timestamp)}.`
        );
      }
    }
    
    // Remove the entry
    removeFromHistory(entry.id);
  });
}
```
