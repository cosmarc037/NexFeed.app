## LOADING STATE ON ACTION BUTTONS — PREVENT DOUBLE CLICK

All action buttons that trigger a process (delete, save, revert, apply, upload, etc.) should show a loading state while the action is being processed. This prevents double-clicking.

---

### REUSABLE LOADING BUTTON COMPONENT

```jsx
function ActionButton({ 
  label, 
  loadingLabel = 'Processing...', 
  onClick, 
  disabled = false, 
  style = 'primary',
  className = ''
}) {
  const [isLoading, setIsLoading] = useState(false);
  
  async function handleClick() {
    if (isLoading || disabled) return;
    
    setIsLoading(true);
    try {
      await onClick();
    } catch (error) {
      console.error('Action failed:', error);
    } finally {
      setIsLoading(false);
    }
  }
  
  const buttonStyles = {
    primary: {
      normal: { background: '#fd5108', color: '#ffffff', border: 'none' },
      loading: { background: '#fb923c', color: '#ffffff', border: 'none' }
    },
    danger: {
      normal: { background: '#dc2626', color: '#ffffff', border: 'none' },
      loading: { background: '#f87171', color: '#ffffff', border: 'none' }
    },
    secondary: {
      normal: { background: '#ffffff', color: '#374151', border: '1px solid #d1d5db' },
      loading: { background: '#f9fafb', color: '#9ca3af', border: '1px solid #e5e7eb' }
    }
  };
  
  const currentStyle = buttonStyles[style] || buttonStyles.primary;
  const appliedStyle = isLoading ? currentStyle.loading : currentStyle.normal;
  
  return (
    <button
      className={`action-btn ${className} ${isLoading ? 'loading' : ''}`}
      onClick={handleClick}
      disabled={isLoading || disabled}
      style={{
        ...appliedStyle,
        borderRadius: '6px',
        padding: '8px 20px',
        fontSize: '13px',
        fontWeight: 600,
        cursor: isLoading || disabled ? 'not-allowed' : 'pointer',
        display: 'inline-flex',
        alignItems: 'center',
        gap: '6px',
        transition: 'background 0.15s',
        opacity: disabled && !isLoading ? 0.5 : 1
      }}
    >
      {isLoading && <LoadingSpinner />}
      {isLoading ? loadingLabel : label}
    </button>
  );
}
```

### LOADING SPINNER

```jsx
function LoadingSpinner() {
  return (
    <svg 
      className="loading-spinner" 
      width="14" 
      height="14" 
      viewBox="0 0 14 14" 
      fill="none"
    >
      <circle 
        cx="7" cy="7" r="5.5" 
        stroke="currentColor" 
        strokeWidth="2" 
        strokeLinecap="round"
        opacity="0.3"
      />
      <path 
        d="M12.5 7a5.5 5.5 0 01-5.5 5.5" 
        stroke="currentColor" 
        strokeWidth="2" 
        strokeLinecap="round"
      />
    </svg>
  );
}
```

```css
.loading-spinner {
  animation: spin 0.8s linear infinite;
}

@keyframes spin {
  from { transform: rotate(0deg); }
  to { transform: rotate(360deg); }
}

.action-btn.loading {
  pointer-events: none;
}
```
