## FIX 1: FIX THE DOWNLOAD BUTTON FOR KNOWLEDGE BASE HISTORY

The download button in the Knowledge Base history table is not clickable or not functioning. Users cannot download the Knowledge Base data from a history entry.

### THE PROBLEM

The download button in the history actions column is either:
1. Not wired to any click handler
2. Being blocked by another element overlapping it (z-index or layout issue)
3. The click handler exists but doesn't trigger the actual file download
4. The button is disabled or has `pointer-events: none` unintentionally

### THE FIX

#### ENSURE THE DOWNLOAD BUTTON HAS A PROPER CLICK HANDLER

```jsx
<td className="history-actions">
  {/* Download button */}
  <button
    title="Download this version"
    className="action-btn action-download"
    onClick={(e) => {
      e.stopPropagation(); // Prevent row click interference
      handleDownloadHistory(entry);
    }}
    style={{ cursor: 'pointer', pointerEvents: 'auto' }}
  >
    <FiDownload size={14} />
  </button>

  {/* Revert button — only show if NOT active */}
  {entry.status !== 'Active' && (
    <button
      title="Revert to this version"
      className="action-btn action-revert"
      onClick={(e) => {
        e.stopPropagation();
        handleRevert(entry);
      }}
    >
      <FiRotateCcw size={14} />
    </button>
  )}

  {/* Delete button */}
  <button
    title="Delete this history"
    className="action-btn action-delete"
    onClick={(e) => {
      e.stopPropagation();
      handleDeleteHistory(entry);
    }}
  >
    <FiTrash2 size={14} />
  </button>
</td>
```

#### IMPLEMENT THE DOWNLOAD HANDLER FOR KNOWLEDGE BASE

This function takes a history entry's snapshot and exports it as an Excel file:

```javascript
import * as XLSX from 'xlsx';

function handleDownloadHistory(entry) {
  if (!entry || !entry.snapshot || entry.snapshot.length === 0) {
    showNotification('error', 'No data available to download for this history entry.');
    return;
  }

  try {
    // Prepare data — remove internal fields (_rowId, _originalIndex)
    const exportData = entry.snapshot.map((row) => {
      const cleanRow = {};
      Object.keys(row).forEach((key) => {
        if (!key.startsWith('_')) {
          cleanRow[key] = row[key];
        }
      });
      return cleanRow;
    });

    // Define column headers in the desired order
    const columnOrder = [
      'fgMaterialCode',
      'fgItemDescription',
      'sfg1MaterialCode',
      'sfgItemDescription',
      'form',
      'batchFm1',
      'batchFm2',
      'batchFm3',
      'batchPmx',
      'fgWeight',
      'thickness',
      'label1',
      'label2',
      'l1Rate',
      'l2Rate',
      'l3Rate',
      'l4Rate',
      'l5Rate',
      'l6Rate',
      'l7Rate',
    ];

    // Map to display-friendly column headers
    const headerMap = {
      fgMaterialCode: 'FG Material Code',
      fgItemDescription: 'FG Item Description',
      sfg1MaterialCode: 'SFG1 Material Code',
      sfgItemDescription: 'SFG Item Description',
      form: 'Form',
      batchFm1: 'Batch FM1',
      batchFm2: 'Batch FM2',
      batchFm3: 'Batch FM3',
      batchPmx: 'Batch PMX',
      fgWeight: 'FG Weight',
      thickness: 'Thickness',
      label1: 'Label 1',
      label2: 'Label 2',
      l1Rate: 'L1 Rate',
      l2Rate: 'L2 Rate',
      l3Rate: 'L3 Rate',
      l4Rate: 'L4 Rate',
      l5Rate: 'L5 Rate',
      l6Rate: 'L6 Rate',
      l7Rate: 'L7 Rate',
    };

    // Build ordered export rows with display headers
    const orderedExportData = exportData.map((row) => {
      const orderedRow = {};
      columnOrder.forEach((key) => {
        const header = headerMap[key] || key;
        orderedRow[header] = row[key] ?? '';
      });
      return orderedRow;
    });

    // Create workbook and worksheet
    const wb = XLSX.utils.book_new();
    const ws = XLSX.utils.json_to_sheet(orderedExportData);

    // Auto-size columns
    const colWidths = {};
    orderedExportData.forEach((row) => {
      Object.keys(row).forEach((key) => {
        const valLength = String(row[key] || '').length;
        const headerLength = key.length;
        const maxLength = Math.max(valLength, headerLength, 10);
        if (!colWidths[key] || colWidths[key] < maxLength) {
          colWidths[key] = maxLength;
        }
      });
    });
    ws['!cols'] = Object.values(colWidths).map((w) => ({ wch: Math.min(w + 2, 40) }));

    XLSX.utils.book_append_sheet(wb, ws, 'Knowledge Base');

    // Generate filename based on history entry
    const dateStr = new Date(entry.timestamp).toISOString().split('T')[0]; // YYYY-MM-DD
    const typeStr = entry.type === 'file_upload' ? 'Upload' : entry.type === 'manual_edit' ? 'Edit' : 'Revert';
    const filename = `Knowledge_Base_${typeStr}_${dateStr}.xlsx`;

    // Trigger download
    XLSX.writeFile(wb, filename);

    showNotification('success', `Downloaded: ${filename}`);
  } catch (error) {
    console.error('Download failed:', error);
    showNotification('error', 'Failed to download. Please try again.');
  }
}
```

#### IMPLEMENT THE N10D DOWNLOAD HANDLER

```javascript
function handleDownloadN10D() {
  if (!n10dData || n10dData.length === 0) {
    showNotification('error', 'No N10D data available to download.');
    return;
  }

  try {
    // Prepare data — remove internal fields
    const exportData = n10dData.map((row) => {
      const cleanRow = {};
      Object.keys(row).forEach((key) => {
        if (!key.startsWith('_')) {
          cleanRow[key] = row[key];
        }
      });
      return cleanRow;
    });

    // Define N10D column order — adjust these to match your actual N10D columns
    const columnOrder = [
      'orderNumber',
      'productionDate',
      'fgMaterialCode',
      'fgItemDescription',
      'sfg1MaterialCode',
      'sfgItemDescription',
      'form',
      'batchFm1',
      'batchFm2',
      'batchFm3',
      'batchPmx',
      'fgWeight',
      'thickness',
      'label1',
      'label2',
      'l1Rate',
      'l2Rate',
      'l3Rate',
      'l4Rate',
      'l5Rate',
      'l6Rate',
      'l7Rate',
      'orderQty',
      'status',
    ];

    // Map to display-friendly column headers
    const headerMap = {
      orderNumber: 'Order Number',
      productionDate: 'Production Date',
      fgMaterialCode: 'FG Material Code',
      fgItemDescription: 'FG Item Description',
      sfg1MaterialCode: 'SFG1 Material Code',
      sfgItemDescription: 'SFG Item Description',
      form: 'Form',
      batchFm1: 'Batch FM1',
      batchFm2: 'Batch FM2',
      batchFm3: 'Batch FM3',
      batchPmx: 'Batch PMX',
      fgWeight: 'FG Weight',
      thickness: 'Thickness',
      label1: 'Label 1',
      label2: 'Label 2',
      l1Rate: 'L1 Rate',
      l2Rate: 'L2 Rate',
      l3Rate: 'L3 Rate',
      l4Rate: 'L4 Rate',
      l5Rate: 'L5 Rate',
      l6Rate: 'L6 Rate',
      l7Rate: 'L7 Rate',
      orderQty: 'Order Qty',
      status: 'Status',
    };

    // Build ordered export rows with display headers
    // Only include columns that actually exist in the data
    const availableColumns = columnOrder.filter((key) =>
      exportData.some((row) => row[key] !== undefined)
    );

    // Also include any columns in the data that are NOT in columnOrder (catch-all)
    const allDataKeys = new Set();
    exportData.forEach((row) => {
      Object.keys(row).forEach((key) => allDataKeys.add(key));
    });
    const extraColumns = [...allDataKeys].filter(
      (key) => !columnOrder.includes(key) && !key.startsWith('_')
    );
    const finalColumnOrder = [...availableColumns, ...extraColumns];

    const orderedExportData = exportData.map((row) => {
      const orderedRow = {};
      finalColumnOrder.forEach((key) => {
        const header = headerMap[key] || key;
        orderedRow[header] = row[key] ?? '';
      });
      return orderedRow;
    });

    // Create workbook and worksheet
    const wb = XLSX.utils.book_new();
    const ws = XLSX.utils.json_to_sheet(orderedExportData);

    // Auto-size columns
    const colWidths = {};
    orderedExportData.forEach((row) => {
      Object.keys(row).forEach((key) => {
        const valLength = String(row[key] || '').length;
        const headerLength = key.length;
        const maxLength = Math.max(valLength, headerLength, 10);
        if (!colWidths[key] || colWidths[key] < maxLength) {
          colWidths[key] = maxLength;
        }
      });
    });
    ws['!cols'] = Object.values(colWidths).map((w) => ({
      wch: Math.min(w + 2, 40),
    }));

    XLSX.utils.book_append_sheet(wb, ws, 'N10D Orders');

    // Generate filename with current date
    const dateStr = new Date().toISOString().split('T')[0];
    const filename = `N10D_Production_Orders_${dateStr}.xlsx`;

    // Trigger download
    XLSX.writeFile(wb, filename);

    showNotification('success', `Downloaded: ${filename}`);
  } catch (error) {
    console.error('N10D download failed:', error);
    showNotification('error', 'Failed to download N10D data. Please try again.');
  }
}
```
