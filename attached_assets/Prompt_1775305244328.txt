When the assistant determines an action should be executed, it renders a confirmation card:

```jsx
function ActionConfirmationCard({ action, params, onConfirm, onCancel }) {
  const [isExecuting, setIsExecuting] = useState(false);
  
  async function handleConfirm() {
    setIsExecuting(true);
    try {
      const result = await action.execute(params);
      onConfirm(result);
    } catch (error) {
      onConfirm({ success: false, error: error.message });
    } finally {
      setIsExecuting(false);
    }
  }
  
  return (
    <div className="chat-action-card">
      <div className="chat-action-header">
        <span className="chat-action-icon">⚡</span>
        <span className="chat-action-title">{action.description}</span>
      </div>
      <div className="chat-action-body">
        <p>{action.confirmMessage(params)}</p>
      </div>
      <div className="chat-action-footer">
        <button 
          className="chat-action-cancel"
          onClick={onCancel}
          disabled={isExecuting}
        >
          Cancel
        </button>
        <button 
          className="chat-action-confirm"
          onClick={handleConfirm}
          disabled={isExecuting}
        >
          {isExecuting ? 'Executing...' : 'Confirm'}
        </button>
      </div>
    </div>
  );
}
```

```jsx
function SmartAssistant() {
  const [isOpen, setIsOpen] = useState(false);
  const [messages, setMessages] = useState([]);
  const [inputValue, setInputValue] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const messagesEndRef = useRef(null);
  
  // Auto-scroll to bottom on new message
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages]);
  
  // Reset messages when chat closes and reopens
  function handleToggle() {
    if (!isOpen) {
      setMessages([{
        id: Date.now(),
        role: 'assistant',
        content: 'Hi! I\'m your production planning assistant. I can help you with order management, line analysis, stock alerts, and more. What would you like to know?',
        type: 'text'
      }]);
    }
    setIsOpen(!isOpen);
  }
  
  async function handleSendMessage(text) {
    if (!text.trim()) return;
    
    const userMessage = {
      id: Date.now(),
      role: 'user',
      content: text,
      type: 'text'
    };
    
    setMessages(prev => [...prev, userMessage]);
    setInputValue('');
    setIsLoading(true);
    
    try {
      const context = buildAssistantContext();
      const systemPrompt = getSystemPrompt(context);
      
      const response = await callAI({
        systemPrompt: systemPrompt,
        messages: messages.filter(m => m.type === 'text').map(m => ({
          role: m.role,
          content: m.content
        })),
        userMessage: text
      });
      
      // Parse response for actions
      const parsedResponse = parseAssistantResponse(response);
      
      // Add assistant message(s)
      parsedResponse.forEach(msg => {
        setMessages(prev => [...prev, {
          id: Date.now() + Math.random(),
          role: 'assistant',
          ...msg
        }]);
      });
      
    } catch (error) {
      setMessages(prev => [...prev, {
        id: Date.now(),
        role: 'assistant',
        content: 'Sorry, I encountered an error. Please try again.',
        type: 'text'
      }]);
    } finally {
      setIsLoading(false);
    }
  }
  
  function handleQuickAction(query) {
    handleSendMessage(query);
  }
  
  function handleActionConfirm(result) {
    if (result.success) {
      setMessages(prev => [...prev, {
        id: Date.now(),
        role: 'assistant',
        content: `✅ ${result.message}`,
        type: 'text'
      }]);
    } else {
      setMessages(prev => [...prev, {
        id: Date.now(),
        role: 'assistant',
        content: `❌ ${result.error || 'Action failed. Please try again.'}`,
        type: 'text'
      }]);
    }
  }
  
  function handleActionCancel() {
    setMessages(prev => [...prev, {
      id: Date.now(),
      role: 'assistant',
      content: 'Action cancelled.',
      type: 'text'
    }]);
  }
  
  return (
    <>
      {/* Floating bubble */}
      <button className="chat-bubble" onClick={handleToggle}>
        {isOpen ? '✕' : '💬'}
      </button>
      
      {/* Chat window */}
      {isOpen && (
        <div className="chat-window">
          {/* Header */}
          <div className="chat-header">
            <div className="chat-header-info">
              <span className="chat-header-title">Smart Assistant</span>
              <span className="chat-header-subtitle">Production Planning AI</span>
            </div>
            <button className="chat-header-close" onClick={handleToggle}>✕</button>
          </div>
          
          {/* Messages */}
          <div className="chat-messages">
            {messages.map(msg => (
              <div key={msg.id} className={`chat-msg chat-msg-${msg.role}`}>
                {msg.type === 'text' && (
                  <div className="chat-msg-bubble">
                    <p>{msg.content}</p>
                  </div>
                )}
                {msg.type === 'action' && (
                  <ActionConfirmationCard
                    action={EXECUTABLE_ACTIONS[msg.actionType]}
                    params={msg.actionParams}
                    onConfirm={handleActionConfirm}
                    onCancel={handleActionCancel}
                  />
                )}
              </div>
            ))}
            {isLoading && (
              <div className="chat-msg chat-msg-assistant">
                <div className="chat-msg-bubble chat-msg-loading">
                  <span className="typing-dots">Thinking</span>
                </div>
              </div>
            )}
            <div ref={messagesEndRef} />
          </div>
          
          {/* Quick actions */}
          <QuickActions onAction={handleQuickAction} />
          
          {/* Input */}
          <div className="chat-input-container">
            <input
              type="text"
              className="chat-input"
              placeholder="Ask me anything about production..."
              value={inputValue}
              onChange={(e) => setInputValue(e.target.value)}
              onKeyDown={(e) => {
                if (e.key === 'Enter' && !e.shiftKey) {
                  e.preventDefault();
                  handleSendMessage(inputValue);
                }
              }}
              disabled={isLoading}
            />
            <button
              className="chat-send-btn"
              onClick={() => handleSendMessage(inputValue)}
              disabled={isLoading || !inputValue.trim()}
            >
              ➤
            </button>
          </div>
        </div>
      )}
    </>
  );
}
```

### 8. RESPONSE PARSER — DETECT ACTIONS IN AI RESPONSE

```javascript
function parseAssistantResponse(response) {
  const results = [];
  
  // Check if AI response contains an action trigger
  const actionPatterns = [
    { pattern: /\[ACTION:COMBINE\](.*?)\[\/ACTION\]/s, type: 'combineOrders' },
    { pattern: /\[ACTION:DIVERT\](.*?)\[\/ACTION\]/s, type: 'divertOrder' },
    { pattern: /\[ACTION:STATUS\](.*?)\[\/ACTION\]/s, type: 'changeStatus' },
    { pattern: /\[ACTION:SEQUENCE\](.*?)\[\/ACTION\]/s, type: 'runAutoSequence' },
    { pattern: /\[ACTION:ADD_ORDER\](.*?)\[\/ACTION\]/s, type: 'addOrder' },
    { pattern: /\[ACTION:SHUTDOWN\](.*?)\[\/ACTION\]/s, type: 'shutdownLine' },
  ];
  
  let textContent = response;
  
  actionPatterns.forEach(({ pattern, type }) => {
    const match = response.match(pattern);
    if (match) {
      try {
        const params = JSON.parse(match[1]);
        results.push({
          type: 'action',
          actionType: type,
          actionParams: params
        });
        textContent = textContent.replace(match[0], '').trim();
      } catch (e) {
        console.error('Failed to parse action params:', e);
      }
    }
  });
  
  if (textContent.trim()) {
    results.unshift({
      type: 'text',
      content: textContent.trim()
    });
  }
  
  if (results.length === 0) {
    results.push({
      type: 'text',
      content: response
    });
  }
  
  return results;
}
```
