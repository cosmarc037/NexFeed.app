## FIX: SEARCH BAR IN "ALL" TAB — APPLY TO ALL LINES, NOT JUST LINE 1

Currently, the search bar in the "All" tab only filters orders on the first line table (Line 1). It should filter across ALL line tables displayed on the page.

---

### THE PROBLEM

The "All" tab shows multiple line tables (Line 1, Line 2, Line 3, etc.). When a user types in the search bar, only Line 1's orders are filtered. Orders in Line 2, Line 3, and other lines remain unfiltered.

This likely happens because:
- The search filter is only applied to the first line's data array
- Each line table manages its own filtered data independently
- The search term is not being passed down to all line table components

---

### THE FIX

The search term must be passed to ALL line tables, and each table must independently filter its own orders based on the search term.

### STEP 1: PASS SEARCH TERM TO ALL LINE TABLES

### STEP 2: EACH LINE TABLE FILTERS ITS OWN DATA

```jsx
function LineTable({ line, searchTerm, filters }) {
  const allLineOrders = getOrdersForLine(line);
  
  // Filter orders based on search term
  const filteredOrders = useMemo(() => {
    if (!searchTerm || !searchTerm.trim()) return allLineOrders;
    
    const term = searchTerm.toLowerCase().trim();
    
    return allLineOrders.filter(order => {
      // Search across all relevant fields
      const searchableFields = [
        order.fpr,
        order.planned_order_fg,
        order.planned_order_sfg,
        order.production_order_fg1,
        order.production_order_sfg1,
        order.material_code_sfg,
        order.material_code_fg,
        order.item_description,
        order.category,
        order.form,
        order.status,
        order.avail_date,
        String(order.volume),
        String(order.prio),
      ];
      
      return searchableFields.some(field => {
        if (!field) return false;
        return String(field).toLowerCase().includes(term);
      });
    });
  }, [allLineOrders, searchTerm]);
  
  // Apply additional filters if any
  const finalOrders = useMemo(() => {
    if (!filters || Object.keys(filters).length === 0) return filteredOrders;
    return applyFilters(filteredOrders, filters);
  }, [filteredOrders, filters]);
  
  // If no orders match search on this line, show a message or hide
  if (finalOrders.length === 0 && searchTerm.trim()) {
    return (
      <div className="line-table-container">
        <div className="line-table-header">
          <span className="line-label">📋 {line}</span>
          <span className="line-order-count">0 orders matching</span>
        </div>
        <div className="line-no-results">
          No orders matching "{searchTerm}" on {line}
        </div>
      </div>
    );
  }
  
  // If no orders at all on this line (regardless of search)
  if (allLineOrders.length === 0) {
    return (
      <div className="line-table-container">
        <div className="line-table-header">
          <span className="line-label">📋 {line}</span>
        </div>
        <div className="line-no-orders">No orders.</div>
      </div>
    );
  }
  
  return (
    <div className="line-table-container">
      <div className="line-table-header">
        <span className="line-label">📋 {line}</span>
        <span className="line-order-count">
          {searchTerm.trim() 
            ? `${finalOrders.length} of ${allLineOrders.length} orders`
            : `${allLineOrders.length} orders`
          }
        </span>
      </div>
      
      <table className="order-table">
        {/* ... table header ... */}
        <tbody>
          {finalOrders.map(order => (
            <OrderRow key={order.id} order={order} />
          ))}
        </tbody>
      </table>
    </div>
  );
}
```
