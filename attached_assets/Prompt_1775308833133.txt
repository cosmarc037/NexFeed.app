### 3. TOUR ENGINE — STEP MANAGEMENT

```jsx
function TourProvider({ children }) {
  const [isTouring, setIsTouring] = useState(false);
  const [currentStepIndex, setCurrentStepIndex] = useState(0);
  const [tourSteps, setTourSteps] = useState([]);
  
  function startTour(type) {
    const currentPage = getCurrentPage(); // 'overview', 'orders', 'n10d', 'knowledgeBase', 'analytics', 'orderHistory', 'configurations'
    
    let steps;
    if (type === 'full') {
      steps = FULL_APP_TOUR_STEPS;
    } else {
      steps = getPageTourSteps(currentPage);
    }
    
    setTourSteps(steps);
    setCurrentStepIndex(0);
    setIsTouring(true);
  }
  
  function nextStep() {
    if (currentStepIndex < tourSteps.length - 1) {
      setCurrentStepIndex(prev => prev + 1);
    } else {
      endTour();
    }
  }
  
  function prevStep() {
    if (currentStepIndex > 0) {
      setCurrentStepIndex(prev => prev - 1);
    }
  }
  
  function endTour() {
    setIsTouring(false);
    setCurrentStepIndex(0);
    setTourSteps([]);
  }
  
  const currentStep = tourSteps[currentStepIndex] || null;
  
  return (
    <TourContext.Provider value={{ startTour, isTouring }}>
      {children}
      {isTouring && currentStep && (
        <TourOverlay
          step={currentStep}
          stepIndex={currentStepIndex}
          totalSteps={tourSteps.length}
          onNext={nextStep}
          onPrev={prevStep}
          onSkip={endTour}
        />
      )}
    </TourContext.Provider>
  );
}
```

### 4. TOUR OVERLAY — HIGHLIGHT AND TOOLTIP

```jsx
function TourOverlay({ step, stepIndex, totalSteps, onNext, onPrev, onSkip }) {
  const [targetRect, setTargetRect] = useState(null);
  
  useEffect(() => {
    // Find the target element
    const element = document.querySelector(step.target);
    if (element) {
      const rect = element.getBoundingClientRect();
      setTargetRect(rect);
      
      // Scroll element into view
      element.scrollIntoView({ behavior: 'smooth', block: 'center' });
      
      // Recalculate after scroll
      setTimeout(() => {
        setTargetRect(element.getBoundingClientRect());
      }, 400);
    } else {
      setTargetRect(null);
    }
  }, [step]);
  
  if (!targetRect) {
    // If target not found, show centered tooltip
    return (
      <div className="tour-overlay-backdrop">
        <TourTooltip
          step={step}
          stepIndex={stepIndex}
          totalSteps={totalSteps}
          onNext={onNext}
          onPrev={onPrev}
          onSkip={onSkip}
          position={{ top: '50%', left: '50%', transform: 'translate(-50%, -50%)' }}
        />
      </div>
    );
  }
  
  const padding = 8;
  const highlightStyle = {
    position: 'fixed',
    top: targetRect.top - padding,
    left: targetRect.left - padding,
    width: targetRect.width + padding * 2,
    height: targetRect.height + padding * 2,
    borderRadius: '8px',
    boxShadow: '0 0 0 9999px rgba(0, 0, 0, 0.5)',
    zIndex: 10001,
    pointerEvents: 'none',
  };
  
  // Calculate tooltip position
  const tooltipPosition = calculateTooltipPosition(targetRect, step.position || 'bottom');
  
  return (
    <>
      {/* Backdrop with cutout */}
      <div className="tour-overlay-backdrop" onClick={onSkip} />
      
      {/* Highlight cutout */}
      <div className="tour-highlight" style={highlightStyle} />
      
      {/* Tooltip */}
      <TourTooltip
        step={step}
        stepIndex={stepIndex}
        totalSteps={totalSteps}
        onNext={onNext}
        onPrev={onPrev}
        onSkip={onSkip}
        position={tooltipPosition}
      />
    </>
  );
}

function calculateTooltipPosition(targetRect, preferredPosition) {
  const tooltipWidth = 340;
  const tooltipHeight = 200;
  const gap = 16;
  const viewportWidth = window.innerWidth;
  const viewportHeight = window.innerHeight;
  
  let top, left;
  
  switch (preferredPosition) {
    case 'bottom':
      top = targetRect.bottom + gap;
      left = targetRect.left + (targetRect.width / 2) - (tooltipWidth / 2);
      break;
    case 'top':
      top = targetRect.top - tooltipHeight - gap;
      left = targetRect.left + (targetRect.width / 2) - (tooltipWidth / 2);
      break;
    case 'right':
      top = targetRect.top + (targetRect.height / 2) - (tooltipHeight / 2);
      left = targetRect.right + gap;
      break;
    case 'left':
      top = targetRect.top + (targetRect.height / 2) - (tooltipHeight / 2);
      left = targetRect.left - tooltipWidth - gap;
      break;
    default:
      top = targetRect.bottom + gap;
      left = targetRect.left;
  }
  
  // Clamp to viewport
  if (left < 16) left = 16;
  if (left + tooltipWidth > viewportWidth - 16) left = viewportWidth - tooltipWidth - 16;
  if (top < 16) top = 16;
  if (top + tooltipHeight > viewportHeight - 16) {
    top = targetRect.top - tooltipHeight - gap;
  }
  
  return { position: 'fixed', top: `${top}px`, left: `${left}px`, zIndex: 10002 };
}
```

### TOUR TOOLTIP COMPONENT

```jsx
function TourTooltip({ step, stepIndex, totalSteps, onNext, onPrev, onSkip, position }) {
  const isFirst = stepIndex === 0;
  const isLast = stepIndex === totalSteps - 1;
  
  return (
    <div className="tour-tooltip" style={position}>
      {/* Header */}
      <div className="tour-tooltip-header">
        <h4 className="tour-tooltip-title">{step.title}</h4>
        <button className="tour-tooltip-skip" onClick={onSkip}>
          Skip Tour
        </button>
      </div>
      
      {/* Body */}
      <div className="tour-tooltip-body">
        <p>{step.description}</p>
      </div>
      
      {/* Footer */}
      <div className="tour-tooltip-footer">
        <span className="tour-tooltip-counter">
          Step {stepIndex + 1} of {totalSteps}
        </span>
        
        <div className="tour-tooltip-actions">
          {!isFirst && (
            <button className="tour-btn-back" onClick={onPrev}>
              Back
            </button>
          )}
          <button className="tour-btn-next" onClick={onNext}>
            {isLast ? 'Finish' : 'Next'}
          </button>
        </div>
      </div>
    </div>
  );
}
```

### 7. NAVIGATION DURING FULL TOUR

For the full app tour, some steps may require navigating to different pages. Handle this by checking if the target element exists and navigating if needed:

```javascript
function navigateToStepPage(step) {
  if (step.page && step.page !== getCurrentPage()) {
    navigateTo(step.page);
    // Wait for page to load before showing tooltip
    return new Promise(resolve => setTimeout(resolve, 500));
  }
  return Promise.resolve();
}
```

Add a `page` property to full tour steps that require navigation:

```javascript
{
  target: '.n10d-upload-btn',
  title: 'Next 10 Days',
  description: '...',
  position: 'bottom',
  page: 'n10d' // Navigate to N10D page for this step
}
```
