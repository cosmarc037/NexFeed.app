## VOLUME OVERRIDE DIALOG — WITH AI IMPACT ANALYSIS

When a user clicks on the Volume (MT) cell to edit it, instead of inline editing, a dialog opens with the current value pre-filled. The user types a new value, and the AI automatically analyzes the impact on following orders before the user confirms.

          }}
          onCancel={() => setShowDialog(false)}

    </>
  );
}

### 2. VOLUME OVERRIDE DIALOG

```jsx
function VolumeOverrideDialog({ order, allOrders, onConfirm, onCancel }) {
  const [newVolume, setNewVolume] = useState(String(order.volume || ''));
  const [impactAnalysis, setImpactAnalysis] = useState(null);
  const [isAnalyzing, setIsAnalyzing] = useState(false);
  const debounceRef = useRef(null);
  
  const currentVolume = parseFloat(order.volume) || 0;
  const parsedNewVolume = parseFloat(newVolume) || 0;
  const volumeDiff = parsedNewVolume - currentVolume;
  const isIncrease = volumeDiff > 0;
  const isDecrease = volumeDiff < 0;
  const isChanged = parsedNewVolume !== currentVolume && parsedNewVolume > 0;
  
  // Auto-generate impact analysis when new volume changes (debounced)
  useEffect(() => {
    if (debounceRef.current) {
      clearTimeout(debounceRef.current);
    }
    
    if (!isChanged || parsedNewVolume <= 0) {
      setImpactAnalysis(null);
      return;
    }
    
    debounceRef.current = setTimeout(() => {
      generateImpactAnalysis();
    }, 600); // 600ms debounce
    
    return () => clearTimeout(debounceRef.current);
  }, [newVolume]);
  
  async function generateImpactAnalysis() {
    setIsAnalyzing(true);
    
    try {
      const context = buildImpactContext(order, parsedNewVolume, allOrders);
      const prompt = buildImpactPrompt(context);
      const response = await callAI(prompt);
      
      setImpactAnalysis(response);
    } catch (error) {
      console.error('Impact analysis failed:', error);
      setImpactAnalysis('Unable to generate impact analysis. You can still proceed with the change.');
    } finally {
      setIsAnalyzing(false);
    }
  }
  
  function handleConfirm() {
    if (parsedNewVolume > 0) {
      onConfirm(parsedNewVolume);
    }
  }
  
  return (
    <div className="dialog-overlay">
      <div className="volume-dialog">
        {/* Header */}
        <div className="volume-dialog-header">
          <span>📦 Override Volume</span>
          <button className="dialog-close" onClick={onCancel}>✕</button>
        </div>
        
        {/* Order info */}
        <div className="volume-dialog-order-info">
          <div className="volume-dialog-order-name">
            {order.item_description}
          </div>
          <div className="volume-dialog-order-details">
            FPR: {order.fpr} · Prio {order.prio} · {order.line}
          </div>
        </div>
        
        {/* Volume input */}
        <div className="volume-dialog-input-section">
          <div className="volume-input-row">
            <div className="volume-input-group">
              <label>Current Volume</label>
              <div className="volume-current">
                {currentVolume} MT
              </div>
            </div>
            
            <div className="volume-input-arrow">→</div>
            
            <div className="volume-input-group">
              <label>New Volume</label>
              <div className="volume-input-wrapper">
                <input
                  type="number"
                  className="volume-input"
                  value={newVolume}
                  onChange={(e) => setNewVolume(e.target.value)}
                  min="0"
                  step="1"
                  autoFocus
                />
                <span className="volume-input-unit">MT</span>
              </div>
            </div>
          </div>
          
          {/* Volume change indicator */}
          {isChanged && (
            <div className={`volume-diff ${isIncrease ? 'volume-diff-increase' : 'volume-diff-decrease'}`}>
              {isIncrease ? '▲' : '▼'} {Math.abs(volumeDiff).toFixed(1)} MT 
              ({isIncrease ? '+' : ''}{((volumeDiff / currentVolume) * 100).toFixed(1)}%)
            </div>
          )}
        </div>
        
        {/* Impact Analysis */}
        <div className="volume-dialog-impact">
          <div className="volume-impact-header">
            <span>✨ Impact Analysis</span>
            {isAnalyzing && (
              <span className="volume-impact-loading">
                <span className="refresh-spinning">🔄</span> Analyzing...
              </span>
            )}
          </div>
          
          <div className="volume-impact-body">
            {!isChanged && (
              <div className="volume-impact-empty">
                Enter a new volume to see the impact analysis.
              </div>
            )}
            
            {isChanged && isAnalyzing && !impactAnalysis && (
              <div className="volume-impact-loading-text">
                Calculating impact on following orders...
              </div>
            )}
            
            {isChanged && isAnalyzing && impactAnalysis && (
              <div className="volume-impact-text" style={{ opacity: 0.5 }}>
                {impactAnalysis}
              </div>
            )}
            
            {isChanged && !isAnalyzing && impactAnalysis && (
              <div className="volume-impact-text">
                {impactAnalysis}
              </div>
            )}
          </div>
        </div>
        
        {/* Footer */}
        <div className="volume-dialog-footer">
          <button className="volume-btn-cancel" onClick={onCancel}>
            Cancel
          </button>
          <button 
            className="volume-btn-confirm"
            onClick={handleConfirm}
            disabled={!isChanged || parsedNewVolume <= 0 || isAnalyzing}
          >
            Apply Change
          </button>
        </div>
      </div>
    </div>
  );
}

### 3. BUILD IMPACT CONTEXT

```javascript
function buildImpactContext(order, newVolume, allOrders) {
  const currentVolume = parseFloat(order.volume) || 0;
  const volumeDiff = newVolume - currentVolume;
  const isIncrease = volumeDiff > 0;
  
  // Get run rate for this line from Knowledge Base
  const runRate = getRunRateForLine(order.line, order.material_code_fg);
  
  // Calculate time difference
  const currentProductionTime = runRate > 0 ? currentVolume / runRate : 0;
  const newProductionTime = runRate > 0 ? newVolume / runRate : 0;
  const timeDiffHours = newProductionTime - currentProductionTime;
  
  // Get following orders on the same line (orders below this one)
  const orderIndex = allOrders.findIndex(o => o.id === order.id);
  const followingOrders = allOrders.slice(orderIndex + 1).filter(o => 
    o.line === order.line && 
    o.status !== 'Done' && 
    o.status !== 'Cancel PO'
  );
  
  // Get N10D data for following orders
  const n10dData = getN10DProducts();
  const followingWithN10D = followingOrders.map(o => {
    const product = n10dData?.find(
      p => String(p.material_code_fg) === String(o.material_code_fg)
    );
    
    const n10dStatus = product ? getProductStatus(
      product.due_for_loading,
      product.inventory,
      getDailyColumns(),
      product,
      new Date()
    ) : null;
    
    return {
      prio: o.prio,
      name: o.item_description,
      fpr: o.fpr,
      volume: o.volume,
      status: o.status,
      availDate: o.avail_date,
      completionDate: o.completion_date,
      n10dStatus: n10dStatus,
      dfl: product?.due_for_loading || null,
      inventory: product?.inventory || null
    };
  });
  
  // Get N10D data for the current order
  const currentProduct = n10dData?.find(
    p => String(p.material_code_fg) === String(order.material_code_fg)
  );
  
  const currentN10DStatus = currentProduct ? getProductStatus(
    currentProduct.due_for_loading,
    currentProduct.inventory,
    getDailyColumns(),
    currentProduct,
    new Date()
  ) : null;
  
  return {
    order: {
      prio: order.prio,
      name: order.item_description,
      fpr: order.fpr,
      line: order.line,
      currentVolume: currentVolume,
      newVolume: newVolume,
      volumeDiff: volumeDiff,
      isIncrease: isIncrease,
      availDate: order.avail_date,
      completionDate: order.completion_date,
      startDate: order.start_date,
      startTime: order.start_time,
      n10dStatus: currentN10DStatus,
      dfl: currentProduct?.due_for_loading || null,
      inventory: currentProduct?.inventory || null
    },
    production: {
      runRate: runRate,
      currentProductionTime: currentProductionTime.toFixed(2),
      newProductionTime: newProductionTime.toFixed(2),
      timeDiffHours: timeDiffHours.toFixed(2),
      timeDiffMinutes: (timeDiffHours * 60).toFixed(0)
    },
    followingOrders: followingWithN10D,
    followingCount: followingWithN10D.length
  };
}
```

function buildImpactPrompt(context) {
  const { order, production, followingOrders, followingCount } = context;
  
  return `
You are a production planning advisor for a feed manufacturing plant.
A planner is changing the volume of an order. Analyze the impact on following orders.

ORDER BEING CHANGED:
- Product: ${order.name}
- FPR: ${order.fpr} | Priority: ${order.prio} | Line: ${order.line}
- Current Volume: ${order.currentVolume} MT → New Volume: ${order.newVolume} MT (${order.isIncrease ? '+' : ''}${order.volumeDiff.toFixed(1)} MT)
- N10D Status: ${order.n10dStatus || 'N/A'}
- DFL: ${order.dfl || 'N/A'} MT | Inventory: ${order.inventory || 'N/A'} MT
- Current Avail Date: ${order.availDate || 'N/A'}
- Current Completion: ${order.completionDate || 'N/A'}

PRODUCTION TIME IMPACT:
- Run Rate: ${production.runRate} MT/hr
- Current Production Time: ${production.currentProductionTime} hrs
- New Production Time: ${production.newProductionTime} hrs
- Time Difference: ${order.isIncrease ? '+' : ''}${production.timeDiffHours} hrs (${order.isIncrease ? '+' : ''}${production.timeDiffMinutes} min)

FOLLOWING ORDERS ON ${order.line} (${followingCount} orders after this one):
${followingOrders.length > 0 
  ? followingOrders.map(o => `
  Priority ${o.prio}: ${o.name}
  Volume: ${o.volume} MT | Status: ${o.status}
  Avail Date: ${o.availDate || 'N/A'} | Completion: ${o.completionDate || 'N/A'}
  N10D Status: ${o.n10dStatus || 'N/A'} | DFL: ${o.dfl || 'N/A'} MT | Inv: ${o.inventory || 'N/A'} MT
  `).join('\n')
  : 'No following orders on this line.'}

RULES:
- Analyze the time impact: how much will following orders be pushed forward or backward
- Identify which following orders (if any) will miss their availability dates due to this change
- For orders with N10D data, assess if the delay will affect stock levels (e.g., "Delaying this Sufficient product by 2 hours won't affect stock" or "Delaying this Urgent product could cause stockout")
- If volume is DECREASED, mention that time is freed up and following orders can complete earlier
- If volume is INCREASED, identify the specific orders at risk
- Be specific — use product names, times, and dates
- Use full month names
- Do NOT use markdown formatting
- Keep it concise — 3-6 sentences total

FORMAT:
[1-2 sentences about the direct time impact of the volume change]
[1-2 sentences about impact on following orders — which ones are affected and how]
[1-2 sentences about stock/N10D impact — whether the delay matters for stock levels]
  `;
}

### 6. APPLY VOLUME CHANGE AFTER CONFIRM

```javascript
function handleVolumeChange(order, newVolume) {
  // Update the order volume
  setAllOrders(prev => prev.map(o => {
    if (o.id === order.id) {
      const updatedOrder = { ...o, volume: newVolume };
      
      // Recalculate dependent fields
      const batchSize = parseFloat(o.batch_size) || 1;
      const newBatches = Math.ceil(newVolume / batchSize);
      updatedOrder.batches = newBatches;
      
      // Recalculate production time using run rate
      const runRate = getRunRateForLine(o.line, o.material_code_fg);
      if (runRate > 0) {
        const newProdTime = newVolume / runRate;
        updatedOrder.production_time = newProdTime.toFixed(2);
      }
      
      // Recalculate bags if applicable
      const fgWeight = parseFloat(o.fg_weight) || 0;
      if (fgWeight > 0) {
        updatedOrder.bags = Math.ceil((newVolume * 1000) / fgWeight);
      }
      
      return updatedOrder;
    }
    return o;
  }));
  
  // Cascade completion dates for following orders
  recalculateCompletionDates();
  
  showNotification('success', `Volume updated: ${order.item_description} → ${newVolume} MT`);
}
```
