```javascript
function migrateLocalStorageKeys() {
  const oldKeys = [

    }
  });
```

## 5. RE-APPLY MASTER DATA — EXCLUDE DONE, CANCELLED, IN PRODUCTION, AND ON-GOING

When re-applying master data to existing orders, do NOT update orders that are Done, Cancelled, In Production, or On-going. Only update orders that are not currently being processed or completed.

### STATUS RULES

| Status | Apply master data changes? |
|---|---|
| Plotted | ✅ Yes |
| Planned | ✅ Yes |
| Hold | ✅ Yes |
| Cut | ✅ Yes |
| Uncombine | ✅ Yes |
| Merge Back | ✅ Yes |
| Custom status | ✅ Yes |
| In Production | ❌ No — mid-production |
| On-going batching | ❌ No — mid-production |
| On-going pelleting | ❌ No — mid-production |
| On-going bagging | ❌ No — mid-production |
| Done | ❌ No — completed |
| Cancel PO | ❌ No — cancelled |

### FUNCTION

```javascript
function isOrderUpdatable(order) {
  const nonUpdatableStatuses = [
    'Done',
    'Cancel PO',
    'In Production',
    'On-going batching',
    'On-going pelleting',
    'On-going bagging'
  ];
  
  return !nonUpdatableStatuses.includes(order.status);
}

function reapplyMasterDataToOrders(masterData, allOrders) {
  const updatedOrders = allOrders.map(order => {
    // Skip non-updatable orders
    if (!isOrderUpdatable(order)) {
      return order;
    }
    
    // Find matching product in master data
    const product = masterData.find(
      p => String(p.fgMaterialCode) === String(order.material_code_fg)
    );
    
    if (!product) return order;
    
    // Apply master data updates
    const updatedOrder = { ...order };
    
    if (product.fgItemDescription) updatedOrder.item_description = product.fgItemDescription;
    if (product.sfg1MaterialCode) updatedOrder.material_code_sfg = product.sfg1MaterialCode;
    if (product.sfgItemDescription) updatedOrder.sfg_item_description = product.sfgItemDescription;
    if (product.form) updatedOrder.form = product.form;
    if (product.fgWeight) updatedOrder.fg_weight = product.fgWeight;
    if (product.thread) updatedOrder.thread = product.thread;
    if (product.sacksCode) updatedOrder.sacks_code = product.sacksCode;
    if (product.sacksDesc) updatedOrder.sacks_desc = product.sacksDesc;
    if (product.tagsCode) updatedOrder.tags_code = product.tagsCode;
    if (product.tagsDesc) updatedOrder.tags_desc = product.tagsDesc;
    if (product.label1) updatedOrder.label_1 = product.label1;
    if (product.label2) updatedOrder.label_2 = product.label2;
    
    // Update batch size based on line
    const lineBatchMap = {
      'Line 1': product.batchFm1,
      'Line 2': product.batchFm1,
      'Line 3': product.batchFm2,
      'Line 4': product.batchFm2,
      'Line 5': product.batchPmx,
      'Line 6': product.batchFm3,
      'Line 7': product.batchFm3,
    };
    
    const newBatchSize = lineBatchMap[order.line];
    if (newBatchSize) {
      updatedOrder.batch_size = parseFloat(newBatchSize);
      if (updatedOrder.volume && updatedOrder.batch_size) {
        updatedOrder.batches = Math.ceil(parseFloat(updatedOrder.volume) / updatedOrder.batch_size);
      }
    }
    
    // Update run rate based on line
    const lineRateMap = {
      'Line 1': product.l1Rate,
      'Line 2': product.l2Rate,
      'Line 3': product.l3Rate,
      'Line 4': product.l4Rate,
      'Line 5': product.l5Rate,
      'Line 6': product.l6Rate,
      'Line 7': product.l7Rate,
    };
    
    const newRunRate = lineRateMap[order.line];
    if (newRunRate) {
      updatedOrder.run_rate = parseFloat(newRunRate);
      if (updatedOrder.volume && updatedOrder.run_rate > 0) {
        updatedOrder.production_time = (parseFloat(updatedOrder.volume) / updatedOrder.run_rate).toFixed(2);
      }
    }
    
    return updatedOrder;
  });
  
  return updatedOrders;
}
```

```javascript
  const updatedOrders = reapplyMasterDataToOrders(masterData, allOrders);
  setAllOrders(updatedOrders);
  
  recalculateCompletionDates();
  
  let message = `Master data applied to ${updatableOrders.length} active order(s).`;
  if (skippedOrders.length > 0) {
    const doneCount = skippedOrders.filter(o => o.status === 'Done').length;
    const cancelledCount = skippedOrders.filter(o => o.status === 'Cancel PO').length;
    const inProgressCount = skippedOrders.filter(o => 
      o.status === 'In Production' || o.status.startsWith('On-going')
    ).length;
    
    const parts = [];
    if (doneCount > 0) parts.push(`${doneCount} completed`);
    if (cancelledCount > 0) parts.push(`${cancelledCount} cancelled`);
    if (inProgressCount > 0) parts.push(`${inProgressCount} in progress`);
    
    message += ` ${parts.join(', ')} order(s) were not updated.`;
  }
  
  showNotification('success', message);
}
```
