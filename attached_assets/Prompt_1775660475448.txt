useEffect(() => {
  if (activePage === 'overview' || activePage === 'analytics') {
    setDashboardExpanded(true);
  }
}, [activePage]);

// Auto-expand Orders if any feedmill is active
useEffect(() => {
  if (activePage === 'orders') {
    setOrdersExpanded(true);
  }
}, [activePage]);

// Auto-expand Configurations if any config page is active
useEffect(() => {
  if (['orderHistory', 'changeoverRules', 'masterData', 'futureDispatches'].includes(activePage)) {
    setConfigExpanded(true);
  }
}, [activePage]);

## 4. UPDATE PAGE DETECTION

```javascript
function getCurrentPage() {
  const path = window.location.pathname;
  const hash = window.location.hash;
  
  if (path.includes('/overview') || hash.includes('overview')) return 'overview';
  if (path.includes('/analytics') || hash.includes('analytics')) return 'analytics';
  if (path.includes('/orders') || hash.includes('orders')) return 'orders';
  if (path.includes('/order-history') || hash.includes('order-history') || hash.includes('orderHistory')) return 'orderHistory';
  if (path.includes('/changeover-rules') || hash.includes('changeover-rules') || hash.includes('changeoverRules')) return 'changeoverRules';
  if (path.includes('/master-data') || hash.includes('master-data') || hash.includes('masterData')) return 'masterData';
  if (path.includes('/future-dispatches') || hash.includes('future-dispatches') || hash.includes('futureDispatches')) return 'futureDispatches';
  
  // Fallback — check active sidebar
  const activeSidebar = document.querySelector('.sidebar-child.active');
  if (activeSidebar) {
    const text = activeSidebar.textContent.toLowerCase();
    if (text.includes('overview')) return 'overview';
    if (text.includes('analytics')) return 'analytics';
    if (text.includes('order history')) return 'orderHistory';
    if (text.includes('changeover')) return 'changeoverRules';
    if (text.includes('master data')) return 'masterData';
    if (text.includes('future dispatches')) return 'futureDispatches';
  }
  
  return 'overview';
}
```
