## THE EXACT FORMULA

For each line, process orders in priority order (top to bottom):

```
FIRST ORDER:
  startTime = user-set start date + start time (e.g., April 08, 2026 08:00 AM)
  productionHours = volume / runRate
  completionTime = startTime + productionHours (in hours, converted to milliseconds)

EVERY SUBSEQUENT ORDER:
  startTime = previousOrder.completionTime + previousOrder.changeover (in hours, converted to milliseconds)
  productionHours = volume / runRate
  completionTime = startTime + productionHours (in hours, converted to milliseconds)
```

**Key:**
- `productionHours` = `volume / runRate` (e.g., 24 / 19.40 = 1.237 hours)
- `changeover` = the PREVIOUS order's total changeover (`_changeoverTotal` or `changeover` field)
- All time math must be done in milliseconds to avoid floating point errors

```javascript
function recalculateCompletionDates(allOrders) {
  // Group orders by line
  const ordersByLine = {};
  allOrders.forEach(order => {
    if (!ordersByLine[order.line]) {
      ordersByLine[order.line] = [];
    }
    ordersByLine[order.line].push(order);
  });
  
  Object.keys(ordersByLine).forEach(line => {
    const lineOrders = ordersByLine[line].sort((a, b) => a.prio - b.prio);
    
    let previousCompletionMs = null;
    let previousChangeoverHrs = 0;
    
    lineOrders.forEach((order, index) => {
      // Skip Done and Cancelled
      if (order.status === 'Done' || order.status === 'Cancel PO') {
        if (order._calculatedCompletionTime) {
          previousCompletionMs = order._calculatedCompletionTime.getTime();
          previousChangeoverHrs = order._changeoverTotal ?? parseFloat(order.changeover) ?? 0;
        }
        return;
      }
      
      // Get production hours
      const volume = parseFloat(order.volume) || 0;
      const runRate = parseFloat(order.run_rate) || 0;
      const productionHours = runRate > 0 ? volume / runRate : 0;
      
      // Convert hours to milliseconds
      const HOUR_MS = 60 * 60 * 1000;
      const productionMs = productionHours * HOUR_MS;
      
      let startMs;
      
      if (index === 0 || previousCompletionMs === null) {
        // FIRST order — use user-set start date/time
        const startDateTime = getOrderStartDateTime(order);
        if (!startDateTime) {
          // No start time set — skip this order and all subsequent
          previousCompletionMs = null;
          previousChangeoverHrs = 0;
          return;
        }
        startMs = startDateTime.getTime();
      } else {
        // SUBSEQUENT orders — start = previous completion + previous changeover
        const changeoverMs = previousChangeoverHrs * HOUR_MS;
        startMs = previousCompletionMs + changeoverMs;
      }
      
      // Calculate completion
      const completionMs = startMs + productionMs;
      
      // Store as Date objects
      const startDate = new Date(startMs);
      const completionDate = new Date(completionMs);
      
      order._calculatedStartTime = startDate;
      order._calculatedCompletionTime = completionDate;
      
      // Format for display
      order._displayStartDate = formatDateForDisplay(startDate);
      order._displayStartTime = formatTimeForDisplay(startDate);
      order._displayCompletionDate = formatDateForDisplay(completionDate);
      order._displayCompletionTime = formatTimeForDisplay(completionDate);
      order.completion_date = `${order._displayCompletionDate} ${order._displayCompletionTime}`;
      
      // Track for next order
      previousCompletionMs = completionMs;
      previousChangeoverHrs = order._changeoverTotal ?? parseFloat(order.changeover) ?? 0;
    });
  });
}

function getOrderStartDateTime(order) {
  // Try to parse the user-set start date and time
  if (order.start_date && order.start_time) {
    return parseDateAndTime(order.start_date, order.start_time);
  }
  if (order.start_date) {
    return parseDateAndTime(order.start_date, '08:00 AM');
  }
  return null;
}

function parseDateAndTime(dateStr, timeStr) {
  try {
    // Combine date and time strings into a Date object
    const combined = `${dateStr} ${timeStr}`;
    const date = new Date(combined);
    if (!isNaN(date.getTime())) return date;
    
    // Try alternative parsing
    // Handle formats like "April 08, 2026" + "08:00 AM"
    const dateClean = dateStr.replace(',', '');
    const alt = new Date(`${dateClean} ${timeStr}`);
    if (!isNaN(alt.getTime())) return alt;
    
    return null;
  } catch (e) {
    return null;
  }
}

function formatDateForDisplay(date) {
  if (!date) return '';
  return date.toLocaleDateString('en-US', {
    year: 'numeric',
    month: 'long',
    day: '2-digit'
  });
}

function formatTimeForDisplay(date) {
  if (!date) return '';
  return date.toLocaleTimeString('en-US', {
    hour: '2-digit',
    minute: '2-digit',
    hour12: true
  });
}
