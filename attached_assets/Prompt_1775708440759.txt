## FIX 1: PER-LINE SHUTDOWN — ONLY AFFECT THE SHUTDOWN LINE, NOT BOTH LINES

Currently, when Line 1 is shut down, both Line 1 and Line 2 orders show yellow highlights and divert hints in the All tab. Only Line 1 orders should be affected.

### THE PROBLEM

The shutdown marking logic is likely applying to all lines in the feedmill instead of just the specific shutdown line.

### THE FIX

When marking divertible orders, check if the ORDER's specific line is shutdown, not the feedmill:

```javascript
function markDivertibleOrders(allOrders, lineShutdowns, masterData) {
  const shutdownLines = Object.entries(lineShutdowns)
    .filter(([line, info]) => info.isShutdown)
    .map(([line]) => line);
  
  allOrders.forEach(order => {
    // Reset markers
    order._isDivertible = false;
    order._canDivertOutsideFeedmill = false;
    order._canDivertWithinFeedmill = false;
    order._divertNote = null;
    
    // Skip Done and Cancelled
    if (order.status === 'Done' || order.status === 'Cancel PO') return;
    
    // Only mark orders on SHUTDOWN lines — not all lines in the feedmill
    if (!shutdownLines.includes(order.line)) return;
    
    const orderFeedmill = LINE_TO_FEEDMILL_MAP[order.line];
    const feedmillLines = FEEDMILL_LINE_MAP[orderFeedmill];
    
    // Check if ENTIRE feedmill is shutdown or just this line
    const allFeedmillLinesShutdown = feedmillLines.every(l => shutdownLines.includes(l));
    
    // Get partner lines in same feedmill (not shutdown)
    const partnerLines = feedmillLines.filter(
      l => l !== order.line && !shutdownLines.includes(l)
    );
    
    // Get lines outside feedmill (not shutdown)
    const outsideLines = Object.keys(LINE_TO_FEEDMILL_MAP).filter(
      l => !feedmillLines.includes(l) && !shutdownLines.includes(l)
    );
    
    // Check run rates from master data for outside lines
    const md = masterData.find(
      m => String(m.fgMaterialCode) === String(order.material_code_fg)
    );
    
    const lineRateMap = md ? {
      'Line 1': parseFloat(md.l1Rate) || 0,
      'Line 2': parseFloat(md.l2Rate) || 0,
      'Line 3': parseFloat(md.l3Rate) || 0,
      'Line 4': parseFloat(md.l4Rate) || 0,
      'Line 5': parseFloat(md.l5Rate) || 0,
      'Line 6': parseFloat(md.l6Rate) || 0,
      'Line 7': parseFloat(md.l7Rate) || 0,
    } : {};
    
    // Check if can divert outside feedmill (has run rate)
    const divertibleOutsideLines = outsideLines.filter(l => lineRateMap[l] > 0);
    const canDivertOutside = divertibleOutsideLines.length > 0;
    
    // Can always divert to partner line within same feedmill
    // (Line 1 and 2 are generally the same, so they can switch)
    const canDivertWithin = partnerLines.length > 0;
    
    if (allFeedmillLinesShutdown) {
      // FEEDMILL SHUTDOWN — only yellow highlight for outside diversion
      if (canDivertOutside) {
        order._isDivertible = true;
        order._canDivertOutsideFeedmill = true;
        order._highlightType = 'yellow';
        order._divertNote = 'Feedmill shutdown — right-click to divert';
      }
    } else {
      // SINGLE LINE SHUTDOWN — differentiate yellow vs white
      if (canDivertOutside) {
        // Can divert to outside feedmill — YELLOW highlight
        order._isDivertible = true;
        order._canDivertOutsideFeedmill = true;
        order._canDivertWithinFeedmill = canDivertWithin;
        order._highlightType = 'yellow';
        order._divertNote = 'Line shutdown — right-click to divert';
      } else if (canDivertWithin) {
        // Can ONLY divert within feedmill (partner line) — WHITE background
        order._isDivertible = true;
        order._canDivertOutsideFeedmill = false;
        order._canDivertWithinFeedmill = true;
        order._highlightType = 'white';
        order._divertNote = 'Line shutdown — right-click to divert';
      } else {
        // Cannot divert anywhere
        order._isDivertible = false;
        order._highlightType = null;
        order._divertNote = 'Cannot be diverted — awaiting line resume';
      }
    }
    
    // Store available lines for the divert dialog
    order._availableDivertLines = {
      partnerLines: partnerLines,
      outsideLines: divertibleOutsideLines,
      allLines: [...partnerLines, ...divertibleOutsideLines]
    };
  });
}
```

function buildSummaryContext(type, data, scope) {
  const context = { /* ... existing context ... */ };
  
  // Add shutdown info with detailed order data
  const shutdownLines = [];
  Object.entries(lineShutdowns).forEach(([line, info]) => {
    if (info.isShutdown) {
      const feedmill = LINE_TO_FEEDMILL_MAP[line];
      const feedmillLines = FEEDMILL_LINE_MAP[feedmill];
      const partnerLines = feedmillLines.filter(
        l => l !== line && !lineShutdowns[l]?.isShutdown
      );
      
      const lineOrders = data.filter(o => 
        o.line === line && 
        o.status !== 'Done' && 
        o.status !== 'Cancel PO'
      );
      
      const masterData = getMasterData();
      
      const ordersWithDivertInfo = lineOrders.map(o => {
        const md = masterData.find(m => String(m.fgMaterialCode) === String(o.material_code_fg));
        const lineRateMap = md ? {
          'Line 1': parseFloat(md.l1Rate) || 0,
          'Line 2': parseFloat(md.l2Rate) || 0,
          'Line 3': parseFloat(md.l3Rate) || 0,
          'Line 4': parseFloat(md.l4Rate) || 0,
          'Line 5': parseFloat(md.l5Rate) || 0,
          'Line 6': parseFloat(md.l6Rate) || 0,
          'Line 7': parseFloat(md.l7Rate) || 0,
        } : {};
        
        const outsideLines = Object.keys(LINE_TO_FEEDMILL_MAP).filter(
          l => !feedmillLines.includes(l) && 
               !lineShutdowns[l]?.isShutdown && 
               lineRateMap[l] > 0
        );
        
        const n10dProduct = getN10DProducts()?.find(
          p => String(p.material_code_fg) === String(o.material_code_fg)
        );
        const n10dStatus = n10dProduct ? getProductStatus(
          n10dProduct.due_for_loading, n10dProduct.inventory,
          getDailyColumns(), n10dProduct, new Date()
        ) : null;
        
        return {
          name: o.item_description,
          volume: o.volume,
          form: o.form,
          prio: o.prio,
          n10dStatus,
          possibleLines: [...partnerLines, ...outsideLines]
        };
      });
      
      shutdownLines.push({
        line,
        feedmill,
        reason: info.reason,
        since: info.since,
        orderCount: lineOrders.length,
        totalVolume: lineOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0),
        divertibleOutsideCount: ordersWithDivertInfo.filter(o => 
          o.possibleLines.some(l => !feedmillLines.includes(l))
        ).length,
        divertibleWithinCount: ordersWithDivertInfo.filter(o => 
          o.possibleLines.some(l => partnerLines.includes(l))
        ).length,
        orders: ordersWithDivertInfo
      });
    }
  });
  
  context.shutdownLines = shutdownLines;
  
  return context;
}
