function markDivertibleOrders(allOrders, lineShutdowns, masterData) {
  const shutdownLines = Object.entries(lineShutdowns)
    .filter(([line, info]) => info.isShutdown)
    .map(([line]) => line);
  
  allOrders.forEach(order => {
    // Reset ALL markers
    order._isDivertible = false;
    order._canDivertOutsideFeedmill = false;
    order._canDivertWithinFeedmill = false;
    order._divertNote = null;
    order._highlightType = null;
    order._availableDivertLines = { partnerLines: [], outsideLines: [], allLines: [] };
    
    // Skip Done and Cancelled
    if (order.status === 'Done' || order.status === 'Cancel PO') return;
    
    // Only process orders on SHUTDOWN lines
    if (!shutdownLines.includes(order.line)) return;
    
    const orderFeedmill = LINE_TO_FEEDMILL_MAP[order.line];
    const feedmillLines = FEEDMILL_LINE_MAP[orderFeedmill];
    
    // Check if ENTIRE feedmill is shutdown
    const allFeedmillLinesShutdown = feedmillLines.every(l => shutdownLines.includes(l));
    
    // Partner lines in same feedmill (not shutdown)
    const partnerLines = feedmillLines.filter(
      l => l !== order.line && !shutdownLines.includes(l)
    );
    
    // Get master data for this product
    const md = masterData.find(
      m => String(m.fgMaterialCode) === String(order.material_code_fg)
    );
    
    const lineRateMap = md ? {
      'Line 1': parseFloat(md.l1Rate) || 0,
      'Line 2': parseFloat(md.l2Rate) || 0,
      'Line 3': parseFloat(md.l3Rate) || 0,
      'Line 4': parseFloat(md.l4Rate) || 0,
      'Line 5': parseFloat(md.l5Rate) || 0,
      'Line 6': parseFloat(md.l6Rate) || 0,
      'Line 7': parseFloat(md.l7Rate) || 0,
    } : {};
    
    // Outside feedmill lines with run rate > 0
    const outsideLines = Object.keys(LINE_TO_FEEDMILL_MAP).filter(
      l => !feedmillLines.includes(l) && 
           !shutdownLines.includes(l) && 
           lineRateMap[l] > 0
    );
    
    const canDivertOutside = outsideLines.length > 0;
    const canDivertWithin = partnerLines.length > 0;
    
    // Store available lines
    order._availableDivertLines = {
      partnerLines: partnerLines,
      outsideLines: outsideLines,
      allLines: [...partnerLines, ...outsideLines]
    };
    
    if (allFeedmillLinesShutdown) {
      // FEEDMILL SHUTDOWN — yellow only if can go outside
      if (canDivertOutside) {
        order._isDivertible = true;
        order._canDivertOutsideFeedmill = true;
        order._canDivertWithinFeedmill = false;
        order._highlightType = 'yellow';
      }
      // No partner lines available (all shutdown), can't go outside = stuck
    } else {
      // SINGLE LINE SHUTDOWN
      if (canDivertOutside) {
        // CAN go outside feedmill → YELLOW
        order._isDivertible = true;
        order._canDivertOutsideFeedmill = true;
        order._canDivertWithinFeedmill = canDivertWithin;
        order._highlightType = 'yellow';
      } else if (canDivertWithin) {
        // Can ONLY go to partner line → WHITE with note
        order._isDivertible = true;
        order._canDivertOutsideFeedmill = false;
        order._canDivertWithinFeedmill = true;
        order._highlightType = 'white';
      } else {
        // Cannot go anywhere
        order._isDivertible = false;
        order._canDivertOutsideFeedmill = false;
        order._canDivertWithinFeedmill = false;
        order._highlightType = null;
      }
    }
  });
}
