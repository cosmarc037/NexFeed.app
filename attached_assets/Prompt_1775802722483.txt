### ALSO UPDATE THE DISPLAY COMPONENT — RENDER AS CARDS

```jsx
function ShutdownAnalysisDisplay({ analysis }) {
  if (!analysis) return null;
  
  const lines = analysis.split('\n').filter(l => l.trim());
  
  return (
    <div className="shutdown-analysis-content">
      {lines.map((line, index) => {
        const trimmed = line.trim();
        
        // Bullet point — render as compact card
        if (trimmed.startsWith('•') || trimmed.startsWith('-')) {
          const text = trimmed.replace(/^[•\-]\s*/, '');
          
          // Try to parse product name (before the — dash)
          const dashSplit = text.split('—');
          if (dashSplit.length >= 2) {
            const productName = dashSplit[0].trim();
            const details = dashSplit.slice(1).join('—').trim();
            
            // Extract N10D status for color coding
            let statusClass = '';
            if (details.toLowerCase().includes('critical')) statusClass = 'shutdown-order-critical';
            else if (details.toLowerCase().includes('urgent')) statusClass = 'shutdown-order-urgent';
            else if (details.toLowerCase().includes('monitor')) statusClass = 'shutdown-order-monitor';
            else if (details.toLowerCase().includes('sufficient')) statusClass = 'shutdown-order-sufficient';
            
            return (
              <div key={index} className={`shutdown-order-card ${statusClass}`}>
                <div className="shutdown-order-name">{productName}</div>
                <div className="shutdown-order-details">{details}</div>
              </div>
            );
          }
          
          // Fallback — simple bullet
          return (
            <div key={index} className="shutdown-analysis-bullet">
              <span className="shutdown-bullet-dot">•</span>
              <span className="shutdown-bullet-text">{text}</span>
            </div>
          );
        }
        
        // Sub-headers (e.g., "Affected orders:", "Capacity note:", "Total impact:")
        if (trimmed.endsWith(':') && trimmed.length < 60) {
          return (
            <div key={index} className="shutdown-analysis-subheader">
              {trimmed}
            </div>
          );
        }
        
        // Capacity note or total impact — highlight box
        if (trimmed.toLowerCase().startsWith('capacity note') || 
            trimmed.toLowerCase().startsWith('partner line capacity') ||
            trimmed.toLowerCase().startsWith('total impact')) {
          return (
            <div key={index} className="shutdown-analysis-note">
              {trimmed}
            </div>
          );
        }
        
        // Regular paragraph
        if (trimmed.length > 0) {
          return (
            <p key={index} className="shutdown-analysis-paragraph">
              {trimmed}
            </p>
          );
        }
        
        return null;
      })}
    </div>
  );
}
```
