function handleShutdownConfirm(shutdownData) {
  const { target, targetType, reason, notes, timestamp } = shutdownData;
  
  if (targetType === 'line') {
    // This triggers the useEffect which saves to localStorage

          reason,
          notes,
          since: timestamp,
          feedmill: target,
          isFeedmillShutdown: true
        };
      });
      return updated;
    });
    
    setFeedmillShutdowns(prev => ({

      [target]: { isShutdown: true, reason, notes, since: timestamp }
    }));

  markDivertibleOrders(allOrders, lineShutdowns, masterData);
  
  // Recalculate
  recalculateAllChangeovers(allOrders, changeoverRules);
  recalculateCompletionDates(allOrders);
