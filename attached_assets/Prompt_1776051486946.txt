```javascript
function buildCombineAnalysisContext(group, lineOrders, masterData, n10dData, changeoverRules) {
  const line = group.line;
  
  // All current orders on this line, sorted by prio
  const currentLineOrders = lineOrders
    .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
    .sort((a, b) => a.prio - b.prio);
  
  // The orders being combined
  const combiningOrders = group.orders;
  
  // Combined order details
  const combinedVolume = combiningOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0);
  const runRate = parseFloat(combiningOrders[0].run_rate) || 0;
  const combinedProductionHours = runRate > 0 ? combinedVolume / runRate : 0;
  const combinedForm = combiningOrders[0].form;
  
  // Determine avail date for combined order
  const datedOrders = combiningOrders.filter(o => {
    const avail = (o.avail_date || '').toLowerCase();
    return avail && !avail.includes('prio') && !avail.includes('safety') && avail !== '' && avail !== '—';
  });
  
  const nonDatedOrders = combiningOrders.filter(o => {
    const avail = (o.avail_date || '').toLowerCase();
    return !avail || avail.includes('prio') || avail.includes('safety') || avail === '' || avail === '—';
  });
  
  let combinedAvailDate = null;
  let availDateType = 'none';
  
  if (datedOrders.length > 0) {
    // Use earliest dated avail date
    const sorted = datedOrders.sort((a, b) => new Date(a.avail_date) - new Date(b.avail_date));
    combinedAvailDate = sorted[0].avail_date;
    availDateType = 'dated';
  } else if (nonDatedOrders.length > 0) {
    // Use the first non-dated avail type
    combinedAvailDate = nonDatedOrders[0].avail_date;
    availDateType = 'priority_based';
  }
  
  // Orders that will be REMOVED after combining (all except the one that stays)
  const orderPrios = combiningOrders.map(o => o.prio).sort((a, b) => a - b);
  const removedPrios = orderPrios.slice(1); // All except the earliest prio
  
  // N10D / Future Dispatches data for the combined product
  const n10dProduct = n10dData?.find(
    p => String(p.material_code_fg) === String(combiningOrders[0].material_code_fg)
  );
  const n10dStatus = n10dProduct ? getProductStatus(
    n10dProduct.due_for_loading, n10dProduct.inventory,
    getDailyColumns(), n10dProduct, new Date()
  ) : null;
  
  // Base changeover for combined order
  const baseChangeover = getBaseChangeover(combinedForm, changeoverRules.baseChangeover);
  
  // Build line context — each order with key details
  const lineContext = currentLineOrders.map(o => {
    const oN10d = n10dData?.find(p => String(p.material_code_fg) === String(o.material_code_fg));
    const oStatus = oN10d ? getProductStatus(
      oN10d.due_for_loading, oN10d.inventory,
      getDailyColumns(), oN10d, new Date()
    ) : null;
    
    const vol = parseFloat(o.volume) || 0;
    const rate = parseFloat(o.run_rate) || 0;
    const prodHrs = rate > 0 ? (vol / rate).toFixed(2) : '0.00';
    
    return {
      prio: o.prio,
      name: o.item_description,
      volume: o.volume,
      form: o.form,
      prodHours: prodHrs,
      availDate: o.avail_date || '—',
      n10dStatus: oStatus || 'N/A',
      completionDate: o.completion_date || '—',
      changeover: (o._changeoverTotal ?? parseFloat(o.changeover) ?? 0).toFixed(2),
      isBeingCombined: combiningOrders.some(co => co.id === o.id),
      color: o.color || '—',
      diameter: o.diameter || '—',
      category: o.category || '—'
    };
  });
  
  return {
    line,
    combiningOrders: combiningOrders.map(o => ({
      prio: o.prio,
      name: o.item_description,
      volume: o.volume,
      availDate: o.avail_date || '—',
      form: o.form
    })),
    combinedVolume,
    combinedProductionHours: combinedProductionHours.toFixed(2),
    combinedForm,
    combinedAvailDate,
    availDateType,
    removedPrios,
    n10dStatus,
    n10dDfl: n10dProduct?.due_for_loading || null,
    n10dInventory: n10dProduct?.inventory || null,
    baseChangeover,
    lineContext,
    totalOrdersOnLine: currentLineOrders.length
  };
}

## 5. RESPONSE PARSER

```javascript
function parseCombineAnalysisResponse(response) {
  let cleaned = response;
  
  // Remove markdown
  cleaned = cleaned.replace(/#{1,6}\s*/g, '');
  cleaned = cleaned.replace(/\*\*/g, '');
  cleaned = cleaned.replace(/\*([^*]+)\*/g, '$1');
  cleaned = cleaned.replace(/_([^_]+)_/g, '$1');
  
  // Parse three sections
  const result = {
    insertionPosition: '',
    schedulingImpact: '',
    alerts: ''
  };
  
  // Try to split by section labels
  const insertionMatch = cleaned.match(/Insertion Position:\s*([\s\S]*?)(?=Scheduling Impact:|$)/i);
  const impactMatch = cleaned.match(/Scheduling Impact:\s*([\s\S]*?)(?=Alerts?:|$)/i);
  const alertsMatch = cleaned.match(/Alerts?:\s*([\s\S]*?)$/i);
  
  if (insertionMatch) result.insertionPosition = insertionMatch[1].trim();
  if (impactMatch) result.schedulingImpact = impactMatch[1].trim();
  if (alertsMatch) result.alerts = alertsMatch[1].trim();
  
  // Fallback — if parsing fails, put everything in insertion
  if (!result.insertionPosition && !result.schedulingImpact && !result.alerts) {
    result.insertionPosition = cleaned.trim();
    result.schedulingImpact = 'See above.';
    result.alerts = 'Please review manually.';
  }
  
  return result;
}

```javascript
function handleApprove(group, analysis) {
  // Parse the AI's recommended position from the analysis
  const recommendedPrio = parseRecommendedPrio(analysis?.insertionPosition, group);
  
  // Combine the orders
  const combiningOrders = group.orders;
  const combinedVolume = combiningOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0);
  
  // Determine avail date — earliest dated, or first non-dated
  const datedOrders = combiningOrders.filter(o => {
    const avail = (o.avail_date || '').toLowerCase();
    return avail && !avail.includes('prio') && !avail.includes('safety') && avail !== '' && avail !== '—';
  }).sort((a, b) => new Date(a.avail_date) - new Date(b.avail_date));
  
  const combinedAvailDate = datedOrders.length > 0
    ? datedOrders[0].avail_date
    : combiningOrders[0].avail_date;
  
  // Create combined order based on the first order
  const baseOrder = { ...combiningOrders[0] };
  baseOrder.volume = combinedVolume;
  baseOrder.avail_date = combinedAvailDate;
  baseOrder.prio = recommendedPrio;
  baseOrder.batches = Math.ceil(combinedVolume / (parseFloat(baseOrder.batch_size) || 1));
  baseOrder.bags = baseOrder.batches * (parseFloat(baseOrder.fg_weight) || 1) * 20;
  
  const runRate = parseFloat(baseOrder.run_rate) || 0;
  baseOrder.production_time = runRate > 0 ? (combinedVolume / runRate).toFixed(2) : '0.00';
  
  // Mark as combined
  baseOrder._isCombined = true;
  baseOrder._combinedFrom = combiningOrders.map(o => o.id);
  
  // Remove all original orders
  const orderIdsToRemove = combiningOrders.map(o => o.id);
  
  // Insert combined order and remove originals
  let updatedOrders = allOrders.filter(o => !orderIdsToRemove.includes(o.id));
  updatedOrders.push(baseOrder);
  
  // Re-sort and re-assign prios
  const lineOrders = updatedOrders
    .filter(o => o.line === group.line)
    .sort((a, b) => a.prio - b.prio);
  
  lineOrders.forEach((o, idx) => {
    o.prio = idx + 1;
  });
  
  // Recalculate everything
  recalculateAllChangeovers(updatedOrders, getDefaultChangeoverRules());
  recalculateCompletionDates(updatedOrders);
  
  setAllOrders(updatedOrders);
  showNotification('success', `Orders combined: ${baseOrder.item_description} (${combinedVolume} MT) at Priority ${recommendedPrio}.`);
}

function parseRecommendedPrio(insertionText, group) {
  if (!insertionText) {
    // Fallback — use earliest avail date logic
    return getEarliestAvailDatePrio(group);
  }
  
  // Try to extract priority number from AI text
  // Look for patterns like "Priority 5", "Prio 5", "position 5", "priority number 5"
  const patterns = [
    /[Pp]riority\s*(\d+)/,
    /[Pp]rio\s*(\d+)/,
    /position\s*(\d+)/i,
    /insert(?:ed)?\s+at\s+(?:priority\s+)?(\d+)/i,
    /(?:recommend|suggest)\w*\s+(?:priority\s+)?(\d+)/i
  ];
  
  for (const pattern of patterns) {
    const match = insertionText.match(pattern);
    if (match) {
      return parseInt(match[1], 10);
    }
  }
  
  // Fallback
  return getEarliestAvailDatePrio(group);
}

function getEarliestAvailDatePrio(group) {
  const datedOrders = group.orders.filter(o => {
    const avail = (o.avail_date || '').toLowerCase();
    return avail && !avail.includes('prio') && !avail.includes('safety') && avail !== '' && avail !== '—';
  });
  
  if (datedOrders.length > 0) {
    const earliest = datedOrders.sort((a, b) => new Date(a.avail_date) - new Date(b.avail_date))[0];
    return earliest.prio;
  }
  
  // Use lowest prio number among combined orders
  return Math.min(...group.orders.map(o => o.prio));
}
