```javascript
function buildCombineAnalysisContext(group, lineOrders, masterData, n10dData, changeoverRules) {
  // ... existing context building ...
  
  // Calculate alerts — downstream orders that would be pushed past avail date

function calculateCombineAlerts(group, lineOrders) {
  const alerts = [];
  const line = group.line;
  
  const currentLineOrders = lineOrders
    .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
    .sort((a, b) => a.prio - b.prio);
  
  const combiningOrders = group.orders;
  const combinedVolume = combiningOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0);
  const runRate = parseFloat(combiningOrders[0].run_rate) || 0;
  const combinedProdHours = runRate > 0 ? combinedVolume / runRate : 0;
  
  // Separate production time
  const separateProdHours = combiningOrders.reduce((s, o) => {
    const v = parseFloat(o.volume) || 0;
    const r = parseFloat(o.run_rate) || 0;
    return s + (r > 0 ? v / r : 0);
  }, 0);
  
  // Time difference from combining
  const timeDifference = combinedProdHours - separateProdHours;
  
  // Check downstream orders for avail date violations
  currentLineOrders.forEach(o => {
    if (combiningOrders.some(co => co.id === o.id)) return;
    
    const availDate = o.avail_date;
    if (!availDate) return;
    
    const availLower = availDate.toLowerCase();
    if (availLower.includes('prio') || availLower.includes('safety') || availLower === '' || availLower === '—') return;
    
    // This is a dated order — check if combining would push it past avail date
    const currentCompletion = o._calculatedCompletionTime;
    if (!currentCompletion) return;
    
    const availDateObj = new Date(availDate);
    if (isNaN(availDateObj.getTime())) return;
    
    // If already past avail date, or would be pushed past
    if (currentCompletion > availDateObj) {
      const delayMs = currentCompletion.getTime() - availDateObj.getTime();
      const delayDays = Math.floor(delayMs / (1000 * 60 * 60 * 24));
      
      alerts.push({
        orderName: o.item_description,
        fpr: o.fpr,
        prio: o.prio,
        availDate: availDate,
        completionDate: o.completion_date,
        delayDays: delayDays
      });
    }
  });
  
  // Volume ceiling check
  if (combinedVolume > 200) {
    alerts.push({
      type: 'volume_ceiling',
      message: `The combined total of ${combinedVolume} MT exceeds the recommended ceiling of 200 MT. Large combined orders may impact scheduling flexibility and downstream production.`
    });
  }
  
  return alerts;
}
```

## FIX 4: INSERTION POSITION LOGIC — EARLIEST AVAIL DATE

Update the AI prompt rules to be very explicit about the insertion position for dated orders. Also update the fallback logic:

### UPDATED APPROVE HANDLER

```javascript
function handleApprove(group, analysis) {
  const combiningOrders = group.orders;
  
  // Determine the correct insertion prio
  let insertionPrio;
  
  // First check: does the AI recommend a position?
  const aiPrio = parseRecommendedPrio(analysis?.insertionPosition, group);
  
  // Second check: enforce dated avail date rule
  const datedOrders = combiningOrders.filter(o => {
    const avail = (o.avail_date || '').toLowerCase();
    return avail && !avail.includes('prio') && !avail.includes('safety') && avail !== '' && avail !== '—';
  });
  
  if (datedOrders.length > 0) {
    // DATED — use the prio of the order with the earliest avail date
    const earliest = datedOrders.sort((a, b) => new Date(a.avail_date) - new Date(b.avail_date))[0];
    insertionPrio = earliest.prio;
  } else {
    // NON-DATED — use AI recommendation
    insertionPrio = aiPrio || Math.min(...combiningOrders.map(o => o.prio));
  }
  
  // Determine avail date
  const combinedAvailDate = datedOrders.length > 0
    ? datedOrders.sort((a, b) => new Date(a.avail_date) - new Date(b.avail_date))[0].avail_date
    : combiningOrders[0].avail_date;
  
  // Create combined order
  const combinedVolume = combiningOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0);
  const baseOrder = { ...combiningOrders.find(o => o.prio === insertionPrio) || combiningOrders[0] };
  
  baseOrder.volume = combinedVolume;
  baseOrder.avail_date = combinedAvailDate;
  baseOrder.prio = insertionPrio;
  baseOrder.batches = Math.ceil(combinedVolume / (parseFloat(baseOrder.batch_size) || 1));
  baseOrder.bags = baseOrder.batches * (parseFloat(baseOrder.fg_weight) || 1) * 20;
  
  const runRate = parseFloat(baseOrder.run_rate) || 0;
  baseOrder.production_time = runRate > 0 ? (combinedVolume / runRate).toFixed(2) : '0.00';
  baseOrder._isCombined = true;
  baseOrder._combinedFrom = combiningOrders.map(o => o.id);
  
  // Remove all original orders and insert combined
  const orderIdsToRemove = combiningOrders.map(o => o.id);
  let updatedOrders = allOrders.filter(o => !orderIdsToRemove.includes(o.id));
  updatedOrders.push(baseOrder);
  
  // Re-sort line orders and reassign prios
  const lineOrders = updatedOrders
    .filter(o => o.line === group.line)
    .sort((a, b) => {
      if (a.id === baseOrder.id) return insertionPrio - 0.5 - b.prio;
      if (b.id === baseOrder.id) return a.prio - (insertionPrio - 0.5);
      return a.prio - b.prio;
    });
  
  lineOrders.forEach((o, idx) => {
    o.prio = idx + 1;
  });
  
  // Recalculate
  recalculateAllChangeovers(updatedOrders, getDefaultChangeoverRules());
  recalculateCompletionDates(updatedOrders);
  
  setAllOrders(updatedOrders);
  showNotification('success', `Orders combined: ${baseOrder.item_description} (${combinedVolume} MT) at Priority ${insertionPrio}.`);
}
