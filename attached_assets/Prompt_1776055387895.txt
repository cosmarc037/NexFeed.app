```jsx
function CombineGroup({ group, lineOrders, masterData, n10dData, changeoverRules }) {
  const [insertionAnalysis, setInsertionAnalysis] = useState(null);
  const [schedulingAnalysis, setSchedulingAnalysis] = useState(null);
  const [alertsAnalysis, setAlertsAnalysis] = useState(null);
  const [isAnalyzing, setIsAnalyzing] = useState(true);
  
  useEffect(() => {
    generateAllAnalyses();
  }, []);
  
  async function generateAllAnalyses() {
    setIsAnalyzing(true);
    try {
      const context = buildCombineAnalysisContext(group, lineOrders, masterData, n10dData, changeoverRules);
      const prompt = buildCombineAnalysisPrompt(context);
      const response = await callAI(prompt);
      const parsed = parseCombineAnalysisResponse(response);
      
      setInsertionAnalysis(parsed.insertionPosition);
      setSchedulingAnalysis(parsed.schedulingImpact);
      setAlertsAnalysis(parsed.alerts);
    } catch (error) {
      console.error('Combine analysis failed:', error);
      setInsertionAnalysis('Unable to determine. Review manually.');
      setSchedulingAnalysis('Analysis unavailable.');
      setAlertsAnalysis('Please verify scheduling impact before approving.');
    } finally {
      setIsAnalyzing(false);
    }
  }
  
  async function refreshSection(section) {
    const context = buildCombineAnalysisContext(group, lineOrders, masterData, n10dData, changeoverRules);
    const prompt = buildCombineAnalysisPrompt(context);
    
    if (section === 'insertion') setInsertionAnalysis(null);
    if (section === 'scheduling') setSchedulingAnalysis(null);
    if (section === 'alerts') setAlertsAnalysis(null);
    
    try {
      const response = await callAI(prompt);
      const parsed = parseCombineAnalysisResponse(response);
      
      if (section === 'insertion') setInsertionAnalysis(parsed.insertionPosition);
      if (section === 'scheduling') setSchedulingAnalysis(parsed.schedulingImpact);
      if (section === 'alerts') setAlertsAnalysis(parsed.alerts);
    } catch (error) {
      console.error(`Refresh ${section} failed:`, error);
    }
  }
  
  return (
    <div className="combine-group">
      {/* Order cards */}
      {group.orders.map(order => (
        <CombineOrderCard key={order.id} order={order} />
      ))}
      
      {/* Combined volume label */}
      <div className="combine-volume-label">
        Combined Volume: {group.combinedVolume} MT
      </div>
      
      {/* Card 1: Combined Production Summary + Changeover Savings */}
      <div className="combine-card">
        <div className="combine-card-header">
          <span>📊 Combined Production Summary</span>
        </div>
        <div className="combine-card-body">
          <div className="combine-card-row">
            <span>Combined Volume</span>
            <span className="combine-card-value">{group.combinedVolume} MT</span>
          </div>
          <div className="combine-card-row">
            <span>Batch Size</span>
            <span className="combine-card-value">{group.batchSize} MT</span>
          </div>
          <div className="combine-card-row">
            <span>Number of Batches</span>
            <span className="combine-card-value">{group.combinedBatches}</span>
          </div>
          <div className="combine-card-row">
            <span>Total Bags</span>
            <span className="combine-card-value">{group.combinedBags}</span>
          </div>
          <div className="combine-card-row">
            <span>Production Time (separate)</span>
            <span className="combine-card-value">{group.separateProductionTime}</span>
          </div>
          <div className="combine-card-row">
            <span>Production Time (combined)</span>
            <span className="combine-card-value">{group.combinedProductionTime}</span>
          </div>
        </div>
        
        {/* Changeover Savings — inside same card */}
        <div className="combine-card-divider"></div>
        <div className="combine-card-header combine-card-header-sub">
          <span>⏱ Changeover Savings</span>
        </div>
        <div className="combine-card-body">
          <div className="combine-card-row">
            <span>Changeovers eliminated</span>
            <span className="combine-card-value">{group.savingsCalculation}</span>
          </div>
          <div className="combine-card-row">
            <span>Total time saved</span>
            <span className="combine-card-value combine-card-value-success">✅ {group.totalSaved} saved</span>
          </div>
        </div>
      </div>

      <div className="combine-card">
```

### ALERTS DISPLAY — HANDLE MULTIPLE ALERTS

```jsx
function CombineAlertsDisplay({ alerts }) {
  if (!alerts) return <span>No alerts.</span>;
  
  // Split by lines that start with ⚠ or bullet-like patterns
  const alertLines = alerts.split('\n').filter(l => l.trim());
  
  if (alertLines.length === 0) {
    return <span className="combine-alerts-none">No alerts — all orders remain on schedule.</span>;
  }
  
  // Check if single "no alerts" message
  const noAlerts = alertLines.length === 1 && 
    alertLines[0].toLowerCase().includes('no alerts');
  
  if (noAlerts) {
    return <span className="combine-alerts-none">{alertLines[0].replace(/^[⚠\s]+/, '').trim()}</span>;
  }
  
  return (
    <div className="combine-alerts-list">
      {alertLines.map((line, index) => {
        const cleaned = line.replace(/^[⚠•\-\s]+/, '').trim();
        if (!cleaned) return null;
        
        return (
          <div key={index} className="combine-alert-item">
            <span className="combine-alert-item-icon">⚠</span>
            <span className="combine-alert-item-text">{cleaned}</span>
          </div>
        );
      })}
    </div>
  );
}
```
