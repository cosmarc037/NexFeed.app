## FIX: INSERTION POSITION — DETERMINISTIC POSITION + AI EXPLANATION

The insertion position is always deterministic:
- Dated orders → position of earliest avail date
- Non-dated orders → position of lowest prio number

The AI must ALWAYS state this exact position. It only adds context and explanation around it.

### CALCULATE THE POSITION BEFORE CALLING AI

```javascript
function getInsertionPrio(combiningOrders) {
  // Check for dated orders
  const datedOrders = combiningOrders.filter(o => {
    const avail = o.target_avail_date;
    return avail && /^\d{4}-\d{2}-\d{2}/.test(avail) && !isNaN(Date.parse(avail));
  });
  
  if (datedOrders.length > 0) {
    // Dated → position of earliest avail date
    const earliest = datedOrders.sort((a, b) =>
      new Date(a.target_avail_date) - new Date(b.target_avail_date)
    )[0];
    return {
      prio: earliest.prio,
      reason: 'dated',
      availDate: earliest.target_avail_date,
      orderName: earliest.item_description
    };
  }
  
  // Non-dated → lowest prio number
  const lowest = combiningOrders.sort((a, b) => a.prio - b.prio)[0];
  return {
    prio: lowest.prio,
    reason: 'lowest_prio',
    availDate: null,
    orderName: lowest.item_description
  };
}
```

```javascript
function handleApprove(group, analysis) {
  const combiningOrders = group.orders;
  
  // Use the SAME deterministic logic as the AI prompt
  const insertion = getInsertionPrio(combiningOrders);
  const insertionPrio = insertion.prio;
  
  // Determine avail date
  const datedOrders = combiningOrders.filter(o => {
    const avail = o.target_avail_date;
    return avail && /^\d{4}-\d{2}-\d{2}/.test(avail) && !isNaN(Date.parse(avail));
  });
  
  const combinedAvailDate = datedOrders.length > 0
    ? datedOrders.sort((a, b) => new Date(a.target_avail_date) - new Date(b.target_avail_date))[0].target_avail_date
    : combiningOrders[0].target_avail_date;
  
  // Create combined order
  const combinedVolume = combiningOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0);
  const baseOrder = { ...combiningOrders.find(o => o.prio === insertionPrio) || combiningOrders[0] };
  
  baseOrder.volume = combinedVolume;
  baseOrder.target_avail_date = combinedAvailDate;
  baseOrder.prio = insertionPrio;
  baseOrder.batches = Math.ceil(combinedVolume / (parseFloat(baseOrder.batch_size) || 1));
  baseOrder.bags = baseOrder.batches * (parseFloat(baseOrder.fg_weight) || 1) * 20;
  
  const runRate = parseFloat(baseOrder.run_rate) || 0;
  baseOrder.production_time = runRate > 0 ? (combinedVolume / runRate).toFixed(2) : '0.00';
  baseOrder._isCombined = true;
  baseOrder._combinedFrom = combiningOrders.map(o => o.id);
  
  // Remove originals and insert combined
  const orderIdsToRemove = combiningOrders.map(o => o.id);
  let updatedOrders = allOrders.filter(o => !orderIdsToRemove.includes(o.id));
  updatedOrders.push(baseOrder);
  
  // Re-sort and reassign prios
  const lineOrders = updatedOrders
    .filter(o => o.line === group.line)
    .sort((a, b) => {
      if (a.id === baseOrder.id) return insertionPrio - 0.5 - b.prio;
      if (b.id === baseOrder.id) return a.prio - (insertionPrio - 0.5);
      return a.prio - b.prio;
    });
  
  lineOrders.forEach((o, idx) => {
    o.prio = idx + 1;
  });
  
  // Recalculate
  recalculateAllChangeovers(updatedOrders, getDefaultChangeoverRules());
  recalculateCompletionDates(updatedOrders);
  
  setAllOrders(updatedOrders);
  showNotification('success', `Orders combined: ${baseOrder.item_description} (${combinedVolume} MT) at Priority ${insertionPrio}.`);
}
```

function parseRecommendedPrio(insertionText, group) {
  // The AI should always match the deterministic prio
  // This is just a validation check
  const insertion = getInsertionPrio(group.orders);
  
  if (insertionText) {
    const patterns = [
      /^(\d+)\./,
      /[Pp]riority\s*(\d+)/,
      /[Pp]rio\s*(\d+)/,
    ];
    
    for (const pattern of patterns) {
      const match = insertionText.match(pattern);
      if (match) {
        const aiPrio = parseInt(match[1], 10);
        if (aiPrio !== insertion.prio) {
          console.warn(`AI suggested Prio ${aiPrio} but deterministic logic says Prio ${insertion.prio}. Using deterministic.`);
        }
      }
    }
  }
  
  // Always return the deterministic result
  return insertion.prio;
}
