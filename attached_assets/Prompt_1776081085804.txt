### SAFETY — CLEAN RESPONSE IF AI STILL STARTS WITH JUST A NUMBER

```javascript
function cleanInsertionResponse(response, expectedPrio) {
  let cleaned = response.trim();
  
  // Remove markdown
  cleaned = cleaned.replace(/\*\*/g, '');
  cleaned = cleaned.replace(/#{1,6}\s*/g, '');
  
  // Remove "Insertion Position:" label if AI includes it
  cleaned = cleaned.replace(/^Insertion Position:\s*/i, '');
  
  // If starts with just a number like "11." or "2.", prepend "Priority"
  const startsWithNumber = /^\d+\.\s/.test(cleaned);
  if (startsWithNumber && !cleaned.toLowerCase().startsWith('priority')) {
    cleaned = 'Priority ' + cleaned;
  }
  
  // Remove "correct" phrasing
  cleaned = cleaned.replace(/is correct because/gi, 'takes this position because');
  cleaned = cleaned.replace(/is correct since/gi, 'is placed here since');
  cleaned = cleaned.replace(/is correct as/gi, 'is placed here as');
  cleaned = cleaned.replace(/correct position/gi, 'position');
  
  return cleaned;
}
```
