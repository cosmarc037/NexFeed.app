### CLEAN AND PARSE AI NOTE INTO INDIVIDUAL SENTENCES

```javascript
function cleanSchedulingNote(response) {
  let cleaned = response.trim();
  
  // Remove markdown
  cleaned = cleaned.replace(/\*\*/g, '');
  cleaned = cleaned.replace(/#{1,6}\s*/g, '');
  cleaned = cleaned.replace(/\*([^*]+)\*/g, '$1');
  cleaned = cleaned.replace(/_([^_]+)_/g, '$1');
  
  // Remove any labels the AI might add
  cleaned = cleaned.replace(/^(INTERSPERSED ORDERS|DOWNSTREAM ORDERS|ANALYSIS|SUMMARY|NOTE)[:\s]*/gim, '');
  cleaned = cleaned.replace(/^\d+\.\s*/gm, '');
  
  // Split into individual sentences/lines
  // First try splitting by newlines (preferred — AI was told to put each on its own line)
  let sentences = cleaned.split('\n')
    .map(s => s.trim())
    .filter(s => s.length > 0);
  
  // If only 1 line (AI wrote a paragraph), split by sentence endings
  if (sentences.length <= 1 && cleaned.length > 100) {
    sentences = cleaned
      .split(/(?<=[.!?])\s+/)
      .map(s => s.trim())
      .filter(s => s.length > 0);
  }
  
  // Limit to 5 sentences max
  if (sentences.length > 5) {
    sentences = sentences.slice(0, 5);
  }
  
  return sentences;
}
```
