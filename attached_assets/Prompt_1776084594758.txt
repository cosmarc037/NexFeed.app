## HELPER — AVAIL DATE DISPLAY

```javascript
function getAvailDisplay(targetAvailDate) {
  if (!targetAvailDate) return '—';
  
  const raw = targetAvailDate.toLowerCase().trim();
  
  if (raw.includes('prio')) return 'prio replenish';
  if (raw.includes('safety')) return 'safety stocks';
  if (raw === 'stock_sufficient') return 'stock sufficient';
  if (raw.includes('sched')) return 'for sched';
  
  if (/^\d{4}-\d{2}-\d{2}/.test(targetAvailDate)) {
    return new Date(targetAvailDate).toLocaleDateString('en-US', {
      month: 'short', day: 'numeric', year: 'numeric'
    });
  }
  
  return targetAvailDate;
}

function getAvailClass(targetAvailDate) {
  if (!targetAvailDate) return '';
  
  const raw = targetAvailDate.toLowerCase().trim();
  
  if (raw.includes('prio')) return 'combine-order-avail-prio';
  if (raw.includes('safety')) return 'combine-order-avail-safety';
  if (raw === 'stock_sufficient') return 'combine-order-avail-stock';
  if (/^\d{4}-\d{2}-\d{2}/.test(targetAvailDate)) return 'combine-order-avail-dated';
  
  return '';
}
```
