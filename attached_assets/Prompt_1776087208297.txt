function buildSchedulingImpactData(group, lineOrders) {
  const combiningOrders = group.orders;
  const combiningIds = combiningOrders.map(o => o.id);
  const line = group.line;
  
  const currentLineOrders = lineOrders
    .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
    .sort((a, b) => a.prio - b.prio);
  
  // Pass 1 — Before (current schedule)
  const beforeStart = {};
  let runningTime = 0;
  
  currentLineOrders.forEach(order => {
    beforeStart[order.id] = runningTime;
    const vol = parseFloat(order.volume) || 0;
    const rate = parseFloat(order.run_rate) || 0;
    const prodHrs = rate > 0 ? vol / rate : 0;
    const co = order._changeoverTotal ?? parseFloat(order.changeover) ?? 0;
    runningTime += prodHrs + co;
  });
  
  // Pass 2 — After (combined schedule)
  const afterStart = {};
  runningTime = 0;
  let combinedDone = false;
  
  const combinedVolume = combiningOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0);
  const combinedRate = parseFloat(combiningOrders[0].run_rate) || 0;
  const combinedProdHrs = combinedRate > 0 ? combinedVolume / combinedRate : 0;
  const combinedCO = combiningOrders[0]._changeoverTotal ?? parseFloat(combiningOrders[0].changeover) ?? 0;
  
  // Determine lead order
  const datedOrders = combiningOrders.filter(o => {
    const avail = o.target_avail_date;
    return avail && /^\d{4}-\d{2}-\d{2}/.test(avail) && !isNaN(Date.parse(avail));
  });
  
  let leadOrderId;
  if (datedOrders.length > 0) {
    const earliest = datedOrders.sort((a, b) =>
      new Date(a.target_avail_date) - new Date(b.target_avail_date)
    )[0];
    leadOrderId = earliest.id;
  } else {
    leadOrderId = combiningOrders.sort((a, b) => a.prio - b.prio)[0].id;
  }
  
  // Store production times for completion calculation
  const orderProdTimes = {};
  
  currentLineOrders.forEach(order => {
    const vol = parseFloat(order.volume) || 0;
    const rate = parseFloat(order.run_rate) || 0;
    const prodHrs = rate > 0 ? vol / rate : 0;
    const co = order._changeoverTotal ?? parseFloat(order.changeover) ?? 0;
    
    orderProdTimes[order.id] = { prodHrs, co };
    
    if (combiningIds.includes(order.id)) {
      if (order.id === leadOrderId && !combinedDone) {
        afterStart[order.id] = runningTime;
        runningTime += combinedProdHrs + combinedCO;
        combinedDone = true;
      }
      return;
    }
    
    afterStart[order.id] = runningTime;
    runningTime += prodHrs + co;
  });
  
  // Get lead order's start date for projecting real completion times
  const leadOrder = currentLineOrders.find(o => o.id === leadOrderId);
  const leadStartDate = leadOrder?.start_date 
    ? new Date(leadOrder.start_date) 
    : new Date();
  
  // Calculate shifts and deadline misses
  const earlierOrders = [];
  const laterOrders = [];
  const deadlineMisses = [];
  
  currentLineOrders.forEach(order => {
    if (combiningIds.includes(order.id)) return;
    if (beforeStart[order.id] === undefined || afterStart[order.id] === undefined) return;
    
    const shiftHrs = beforeStart[order.id] - afterStart[order.id];
    // Positive = earlier, Negative = later
    
    const prodTime = orderProdTimes[order.id]?.prodHrs || 0;
    
    // Check deadline miss — project new completion time
    const targetAvail = order.target_avail_date;
    const isDated = targetAvail && /^\d{4}-\d{2}-\d{2}/.test(targetAvail) && !isNaN(Date.parse(targetAvail));
    
    let deadlineMiss = null;
    
    if (isDated) {
      const availDateObj = new Date(targetAvail);
      const newStartHrs = afterStart[order.id];
      const newCompletionMs = leadStartDate.getTime() + ((newStartHrs + prodTime) * 60 * 60 * 1000);
      const newCompletionDate = new Date(newCompletionMs);
      
      if (newCompletionDate > availDateObj) {
        const delayMs = newCompletionDate.getTime() - availDateObj.getTime();
        const delayDays = Math.ceil(delayMs / (1000 * 60 * 60 * 24));
        
        const availDisplay = availDateObj.toLocaleDateString('en-US', {
          year: 'numeric', month: 'long', day: 'numeric'
        });
        const completionDisplay = newCompletionDate.toLocaleDateString('en-US', {
          year: 'numeric', month: 'long', day: 'numeric'
        });
        
        deadlineMiss = {
          prio: order.prio,
          name: order.item_description,
          availDate: availDisplay,
          newCompletion: completionDisplay,
          delayDays: delayDays
        };
        
        deadlineMisses.push(deadlineMiss);
      }
    }
    
    // Time shift entries
    if (Math.abs(shiftHrs) < 0.01) return;
    
    const entry = {
      prio: order.prio,
      name: order.item_description,
      shiftHrs: Math.abs(shiftHrs),
      shiftFormatted: formatTimeShift(Math.abs(shiftHrs)),
      hasDeadlineMiss: deadlineMiss !== null
    };
    
    if (shiftHrs > 0) {
      earlierOrders.push(entry);
    } else {
      laterOrders.push(entry);
    }
  });
  
  // Sort
  laterOrders.sort((a, b) => b.shiftHrs - a.shiftHrs);
  earlierOrders.sort((a, b) => b.shiftHrs - a.shiftHrs);
  deadlineMisses.sort((a, b) => b.delayDays - a.delayDays);
  
  return { earlierOrders, laterOrders, deadlineMisses };
}

function buildSchedulingImpactNotePrompt(context, impactData) {
  const hasEarlier = impactData.earlierOrders.length > 0;
  const hasLater = impactData.laterOrders.length > 0;
  const hasDeadlineMisses = impactData.deadlineMisses.length > 0;
  
  if (!hasEarlier && !hasLater && !hasDeadlineMisses) {
    return null;
  }
  
  return `
You are a production scheduling advisor. Provide a brief analysis of the scheduling impact from combining orders on ${context.line}.

Orders starting later:
${hasLater
  ? impactData.laterOrders.map(o => `- Prio ${o.prio}: ${o.name} — ${o.shiftFormatted} later`).join('\n')
  : 'None'}

Orders starting earlier:
${hasEarlier
  ? impactData.earlierOrders.map(o => `- Prio ${o.prio}: ${o.name} — ${o.shiftFormatted} earlier`).join('\n')
  : 'None'}

Orders at risk of missing avail date:
${hasDeadlineMisses
  ? impactData.deadlineMisses.map(o => `- Prio ${o.prio}: ${o.name} — new completion ${o.newCompletion} exceeds avail date ${o.availDate} by ~${o.delayDays} days`).join('\n')
  : 'None'}

Product context:
${[...impactData.laterOrders, ...impactData.earlierOrders].map(o => {
  const lineOrder = context.lineContext.find(lo => lo.prio === o.prio);
  return lineOrder ? `- ${o.name}: N10D status ${lineOrder.n10dStatus}, Avail: ${lineOrder.availDate}` : '';
}).filter(Boolean).join('\n')}

RESPONSE RULES:
- Write 3-5 sentences analyzing the significance of the impact.
- Each sentence on its OWN LINE.
- If there are NO delayed orders and NO deadline misses, be POSITIVE — mention the time savings benefit.
- If there ARE delayed orders or deadline misses, highlight the concern and which orders are most at risk.
- If orders start earlier but still miss their avail date, note that the time savings alone don't resolve the deadline issue.
- Do NOT list individual orders again — the bullet lists above already show them.
- Do NOT use markdown formatting.
- Do NOT mention delays if the calculated data shows zero delayed orders and zero deadline misses.
- Keep each sentence concise.
  `;
}

function SchedulingImpactCard({ group, lineOrders, context }) {

  useEffect(() => {
    generateImpact();
  }, [selectedOrderKey]);
  
  async function generateImpact() {
    setIsLoading(true);
    setImpactData(null);
    setAiNote(null);
    
    const data = buildSchedulingImpactData(group, lineOrders);
    setImpactData(data);
    
    const hasAny = data.earlierOrders.length > 0 || 
                   data.laterOrders.length > 0 || 
                   data.deadlineMisses.length > 0;
    
    if (hasAny) {
      try {
        const prompt = buildSchedulingImpactNotePrompt(context, data);
