## 2. RESPONSE PARSER — STRUCTURED SECTIONS

```javascript
function parseSchedulingImpactResponse(response) {
  let cleaned = response.trim();
  
  // Remove markdown
  cleaned = cleaned.replace(/\*\*/g, '');
  cleaned = cleaned.replace(/#{1,6}\s*/g, '');
  cleaned = cleaned.replace(/\*([^*]+)\*/g, '$1');
  cleaned = cleaned.replace(/_([^_]+)_/g, '$1');
  
  const result = {
    earlierSection: [],
    laterSection: [],
    deadlineSection: [],
    summary: []
  };
  
  // Split into sections
  const earlierMatch = cleaned.match(/Orders starting earlier:\s*([\s\S]*?)(?=Orders starting later:|$)/i);
  const laterMatch = cleaned.match(/Orders starting later:\s*([\s\S]*?)(?=Orders at risk|$)/i);
  const deadlineMatch = cleaned.match(/Orders at risk of missing avail date:\s*([\s\S]*?)(?=\n\n[A-Z]|\n[A-Z][a-z]|$)/i);
  
  // Extract earlier lines
  if (earlierMatch) {
    const lines = earlierMatch[1].split('\n').map(l => l.trim()).filter(l => l.length > 0);
    const sectionEnd = lines.findIndex(l => 
      l.toLowerCase().startsWith('orders starting later') ||
      l.toLowerCase().startsWith('orders at risk')
    );
    const sectionLines = sectionEnd >= 0 ? lines.slice(0, sectionEnd) : lines;
    result.earlierSection = sectionLines.filter(l => 
      l.startsWith('✅') || l.toLowerCase().includes('no orders starting earlier')
    );
  }
  
  // Extract later lines
  if (laterMatch) {
    const lines = laterMatch[1].split('\n').map(l => l.trim()).filter(l => l.length > 0);
    const sectionEnd = lines.findIndex(l => 
      l.toLowerCase().startsWith('orders at risk')
    );
    const sectionLines = sectionEnd >= 0 ? lines.slice(0, sectionEnd) : lines;
    result.laterSection = sectionLines.filter(l => 
      l.startsWith('⚠') || l.toLowerCase().includes('no orders starting later')
    );
  }
  
  // Extract deadline lines
  if (deadlineMatch) {
    const lines = deadlineMatch[1].split('\n').map(l => l.trim()).filter(l => l.length > 0);
    // Deadline section ends when we hit the summary (non-icon lines after all ⛔ lines)
    const deadlineLines = [];
    const summaryLines = [];
    let passedDeadlines = false;
    
    lines.forEach(l => {
      if (l.startsWith('⛔') || l.toLowerCase().includes('no orders at risk')) {
        deadlineLines.push(l);
      } else if (deadlineLines.length > 0 || passedDeadlines) {
        passedDeadlines = true;
        summaryLines.push(l);
      } else {
        deadlineLines.push(l);
      }
    });
    
    result.deadlineSection = deadlineLines;
    result.summary = summaryLines.filter(l => l.length > 0);
  }
  
  // If summary not found in deadline section, try to find it at the end
  if (result.summary.length === 0) {
    const allSections = [
      ...(result.earlierSection || []),
      ...(result.laterSection || []),
      ...(result.deadlineSection || [])
    ];
    const allSectionText = allSections.join('\n');
    const remainingText = cleaned.replace(allSectionText, '').trim();
    
    // Find lines after the last section that aren't part of any section
    const allLines = cleaned.split('\n').map(l => l.trim()).filter(l => l.length > 0);
    const lastIconIndex = allLines.reduce((max, l, i) => {
      if (l.startsWith('✅') || l.startsWith('⚠') || l.startsWith('⛔') || 
          l.toLowerCase().includes('no orders')) {
        return i;
      }
      return max;
    }, -1);
    
    if (lastIconIndex >= 0 && lastIconIndex < allLines.length - 1) {
      result.summary = allLines.slice(lastIconIndex + 1).filter(l => 
        !l.toLowerCase().startsWith('orders starting') &&
        !l.toLowerCase().startsWith('orders at risk') &&
        l.length > 0
      );
    }
  }
  
  // Limit summary to 5 sentences
  if (result.summary.length > 5) {
    result.summary = result.summary.slice(0, 5);
  }
  
  return result;
}
```

```jsx
function SchedulingImpactCard({ group, lineOrders, context }) {
  const [calculatedImpact, setCalculatedImpact] = useState(null);
  const [aiResponse, setAiResponse] = useState(null);
  const [isLoading, setIsLoading] = useState(true);
  const [useFallback, setUseFallback] = useState(false);
  
  const selectedOrderKey = group.orders.map(o => o.id).sort().join('|');
  
  useEffect(() => {
    generateImpact();
  }, [selectedOrderKey]);
  
  async function generateImpact() {
    setIsLoading(true);
    setAiResponse(null);
    setUseFallback(false);
    
    // Step 1: Calculate (instant)
    const calculated = buildSchedulingImpactData(group, lineOrders);
    setCalculatedImpact(calculated);
    
    const hasAny = calculated.earlierOrders.length > 0 ||
                   calculated.laterOrders.length > 0 ||
                   calculated.deadlineMisses.length > 0;
    
    if (!hasAny) {
      setIsLoading(false);
      return;
    }
    
    // Step 2: AI analysis
    try {
      const prompt = buildSchedulingImpactPrompt(context, calculated);
      const response = await callAI(prompt);
      const parsed = parseSchedulingImpactResponse(response);
      
      // Validate — check if AI output roughly aligns with calculated
      const aiHasContent = parsed.earlierSection.length > 0 ||
                           parsed.laterSection.length > 0 ||
                           parsed.deadlineSection.length > 0;
      
      if (aiHasContent) {
        setAiResponse(parsed);
      } else {
        console.warn('AI response had no content, using fallback');
        setUseFallback(true);
      }
    } catch (error) {
      console.error('Scheduling impact AI failed, using fallback:', error);
      setUseFallback(true);
    }
    
    setIsLoading(false);
  }
  
  async function handleRefresh() {
    setAiResponse(null);
    setUseFallback(false);
    await generateImpact();
  }
  
  const hasNoImpact = calculatedImpact &&
    calculatedImpact.earlierOrders.length === 0 &&
    calculatedImpact.laterOrders.length === 0 &&
    calculatedImpact.deadlineMisses.length === 0;
  
  return (
    <div className="combine-card">
      <div className="combine-card-header">
        <span>📊 Scheduling Impact</span>
        <button
          className="combine-card-refresh"
          onClick={handleRefresh}
          title="Refresh analysis"
        >
          ↻ Refresh
        </button>
      </div>
      
      <div className="combine-card-body combine-card-body-text">
        {isLoading && (
          <span className="combine-card-loading">Analyzing scheduling impact...</span>
        )}
        
        {!isLoading && hasNoImpact && (
          <span className="scheduling-no-impact">
            No scheduling impact — all orders remain on their current schedule.
          </span>
        )}
        
        {/* AI-generated output */}
        {!isLoading && aiResponse && !useFallback && (
          <SchedulingImpactAIDisplay response={aiResponse} />
        )}
        
        {/* Calculated fallback */}
        {!isLoading && useFallback && calculatedImpact && (
          <SchedulingImpactFallback impact={calculatedImpact} />
        )}
      </div>
    </div>
  );
}
```

```jsx
function SchedulingImpactLine({ line, type }) {
  // Strip leading icon if present — we'll add our own styling
  let text = line.trim();
  let icon = '';
  
  if (text.startsWith('✅')) {
    icon = '✅';
    text = text.replace(/^✅\s*/, '');
  } else if (text.startsWith('⚠')) {
    icon = '⚠';
    text = text.replace(/^⚠\s*/, '');
  } else if (text.startsWith('⛔')) {
    icon = '⛔';
    text = text.replace(/^⛔\s*/, '');
  } else {
    // No icon — assign default based on type
    if (type === 'earlier') icon = '✅';
    else if (type === 'later') icon = '⚠';
    else if (type === 'deadline') icon = '⛔';
  }
  
  // Check for "no orders" message
  const isNoOrders = text.toLowerCase().includes('no orders');
  
  if (isNoOrders) {
    return (
      <div className="scheduling-impact-item scheduling-impact-item-none">
        <span className="scheduling-impact-text-none">{text}</span>
      </div>
    );
  }
  
  const itemClass = type === 'earlier' ? 'scheduling-impact-item-faster' :
                    type === 'later' ? 'scheduling-impact-item-delayed' :
                    'scheduling-impact-item-deadline';
  
  return (
    <div className={`scheduling-impact-item ${itemClass}`}>
      <span className="scheduling-impact-bullet">{icon}</span>
      <span className="scheduling-impact-text">{text}</span>
    </div>
  );
}
```
