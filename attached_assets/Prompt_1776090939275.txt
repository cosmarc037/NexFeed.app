function buildSchedulingImpactPrompt(context, calculatedImpact) {

ORDER CONTEXT (N10D status, avail dates):

ONLY include sections that have actual data. If a section has no orders, DO NOT include that section at all. Skip it entirely — no label, no "No orders..." message.

For sections that DO have data, use the section label on its own line, then the grouped/individual items below.

GROUPING RULES — APPLY TO ALL THREE SECTIONS:

   - For later: ⚠ Prio [range] ([count] orders) — all start ~[time] later. Notable: [2-3 important ones with N10D status].
   - For deadline: ⛔ Prio [range] ([count] orders) — all exceed avail date by ~[X] days. Notable: [2-3 important ones].

2. If orders have DIFFERENT time shifts, group by similar time shifts.

4. Always mention N10D status for Critical and Urgent orders.

TIME FORMAT:

IMPORTANT:

function buildSchedulingImpactData(group, lineOrders) {
  // ... existing code until the shift calculation ...
  
  currentLineOrders.forEach(order => {
    if (combiningIds.includes(order.id)) return;
    if (beforeStart[order.id] === undefined || afterStart[order.id] === undefined) return;
    
    const shiftHrs = beforeStart[order.id] - afterStart[order.id];
    const prodTime = orderProdTimes[order.id]?.prodHrs || 0;
    
    // Get N10D status
    const n10dProduct = context?.n10dData?.find(
      p => String(p.material_code_fg) === String(order.material_code_fg)
    );
    const n10dStatus = n10dProduct ? getProductStatus(
      n10dProduct.due_for_loading, n10dProduct.inventory,
      getDailyColumns(), n10dProduct, new Date()
    ) : null;
    
    // Deadline miss check
    const targetAvail = order.target_avail_date;
    const isDated = targetAvail && /^\d{4}-\d{2}-\d{2}/.test(targetAvail) && !isNaN(Date.parse(targetAvail));
    
    let deadlineMiss = null;
    
    if (isDated) {
      const availDateObj = new Date(targetAvail);
      const newStartHrs = afterStart[order.id];
      const newCompletionMs = leadStartDate.getTime() + ((newStartHrs + prodTime) * 60 * 60 * 1000);
      const newCompletionDate = new Date(newCompletionMs);
      
      if (newCompletionDate > availDateObj) {
        const delayMs = newCompletionDate.getTime() - availDateObj.getTime();
        const delayDays = Math.ceil(delayMs / (1000 * 60 * 60 * 24));
        
        deadlineMiss = {
          prio: order.prio,
          name: order.item_description,
          availDate: availDateObj.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' }),
          newCompletion: newCompletionDate.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' }),
          delayDays: delayDays,
          n10dStatus: n10dStatus || 'N/A'
        };
        
        deadlineMisses.push(deadlineMiss);
      }
    }
    
    if (Math.abs(shiftHrs) < 0.01) return;
    
    const entry = {
      prio: order.prio,
      name: order.item_description,
      shiftHrs: Math.abs(shiftHrs),
      shiftFormatted: formatTimeShift(Math.abs(shiftHrs)),
      n10dStatus: n10dStatus || 'N/A',
      hasDeadlineMiss: deadlineMiss !== null
    };
    
    if (shiftHrs > 0) {
      earlierOrders.push(entry);
    } else {
      laterOrders.push(entry);
    }
  });
  
  // ... rest of sorting and return ...
}

## 4. UPDATE RESPONSE PARSER — HANDLE MISSING SECTIONS

```javascript
function parseSchedulingImpactResponse(response) {
  let cleaned = response.trim();
  
  // Remove markdown
  cleaned = cleaned.replace(/\*\*/g, '');
  cleaned = cleaned.replace(/#{1,6}\s*/g, '');
  cleaned = cleaned.replace(/\*([^*]+)\*/g, '$1');
  cleaned = cleaned.replace(/_([^_]+)_/g, '$1');
  
  const result = {
    earlierSection: [],
    laterSection: [],
    deadlineSection: [],
    summary: []
  };
  
  // Split into lines
  const allLines = cleaned.split('\n').map(l => l.trim()).filter(l => l.length > 0);
  
  let currentSection = null;
  const sectionLines = {
    earlier: [],
    later: [],
    deadline: [],
    summary: []
  };
  
  allLines.forEach(line => {
    const lower = line.toLowerCase();
    
    // Detect section headers
    if (lower.startsWith('orders starting earlier')) {
      currentSection = 'earlier';
      return;
    }
    if (lower.startsWith('orders starting later')) {
      currentSection = 'later';
      return;
    }
    if (lower.startsWith('orders at risk')) {
      currentSection = 'deadline';
      return;
    }
    
    // Lines with icons go to their section
    if (line.startsWith('✅')) {
      currentSection = currentSection || 'earlier';
      sectionLines.earlier.push(line);
      return;
    }
    if (line.startsWith('⚠')) {
      currentSection = currentSection || 'later';
      sectionLines.later.push(line);
      return;
    }
    if (line.startsWith('⛔')) {
      currentSection = currentSection || 'deadline';
      sectionLines.deadline.push(line);
      return;
    }
    
    // Non-icon lines
    if (currentSection === 'earlier' || currentSection === 'later' || currentSection === 'deadline') {
      // Check if this looks like a summary sentence (no icon, not a "no orders" message)
      if (!lower.includes('no orders') && line.length > 30) {
        // Might be start of summary
        currentSection = 'summary';
        sectionLines.summary.push(line);
      }
    } else if (currentSection === 'summary') {
      sectionLines.summary.push(line);
    } else {
      // Unclassified — likely summary
      sectionLines.summary.push(line);
    }
  });
  
  result.earlierSection = sectionLines.earlier;
  result.laterSection = sectionLines.later;
  result.deadlineSection = sectionLines.deadline;
  result.summary = sectionLines.summary.slice(0, 5);
  
  return result;
}
