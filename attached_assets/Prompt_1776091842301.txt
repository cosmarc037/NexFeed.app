## 2. UPDATE RESPONSE PARSER — USE ---SUMMARY--- MARKER

```javascript
function parseSchedulingImpactResponse(response) {
  let cleaned = response.trim();
  
  // Remove markdown
  cleaned = cleaned.replace(/\*\*/g, '');
  cleaned = cleaned.replace(/#{1,6}\s*/g, '');
  cleaned = cleaned.replace(/\*([^*]+)\*/g, '$1');
  cleaned = cleaned.replace(/_([^_]+)_/g, '$1');
  
  const result = {
    earlierSection: [],
    laterSection: [],
    deadlineSection: [],
    summary: []
  };
  
  // Split by ---SUMMARY--- marker
  let bulletsPart = cleaned;
  let summaryPart = '';
  
  const summaryMarkerIndex = cleaned.indexOf('---SUMMARY---');
  if (summaryMarkerIndex >= 0) {
    bulletsPart = cleaned.substring(0, summaryMarkerIndex).trim();
    summaryPart = cleaned.substring(summaryMarkerIndex + '---SUMMARY---'.length).trim();
  } else {
    // Fallback: try other markers
    const altMarkers = ['SUMMARY:', 'Summary:', '---', '==='];
    for (const marker of altMarkers) {
      const idx = cleaned.lastIndexOf(marker);
      if (idx >= 0 && idx > cleaned.length * 0.4) {
        // Only use if it's in the latter half of the response
        bulletsPart = cleaned.substring(0, idx).trim();
        summaryPart = cleaned.substring(idx + marker.length).trim();
        break;
      }
    }
  }
  
  // If still no summary found, try to extract non-icon lines at the end
  if (!summaryPart) {
    const allLines = cleaned.split('\n').map(l => l.trim()).filter(l => l.length > 0);
    
    // Find the last line that has an icon
    let lastIconIndex = -1;
    allLines.forEach((l, i) => {
      if (l.startsWith('✅') || l.startsWith('⚠') || l.startsWith('⛔')) {
        lastIconIndex = i;
      }
    });
    
    // Find last section header
    let lastHeaderIndex = -1;
    allLines.forEach((l, i) => {
      if (l.toLowerCase().startsWith('orders starting') || l.toLowerCase().startsWith('orders at risk')) {
        lastHeaderIndex = i;
      }
    });
    
    const lastContentIndex = Math.max(lastIconIndex, lastHeaderIndex);
    
    if (lastContentIndex >= 0 && lastContentIndex < allLines.length - 1) {
      bulletsPart = allLines.slice(0, lastContentIndex + 1).join('\n');
      summaryPart = allLines.slice(lastContentIndex + 1).join('\n');
    }
  }
  
  // Parse bullets part into sections
  const bulletLines = bulletsPart.split('\n').map(l => l.trim()).filter(l => l.length > 0);
  let currentSection = null;
  
  bulletLines.forEach(line => {
    const lower = line.toLowerCase();
    
    if (lower.startsWith('orders starting earlier')) {
      currentSection = 'earlier';
      return;
    }
    if (lower.startsWith('orders starting later')) {
      currentSection = 'later';
      return;
    }
    if (lower.startsWith('orders at risk')) {
      currentSection = 'deadline';
      return;
    }
    
    if (line.startsWith('✅')) {
      result.earlierSection.push(line);
      currentSection = 'earlier';
      return;
    }
    if (line.startsWith('⚠')) {
      result.laterSection.push(line);
      currentSection = 'later';
      return;
    }
    if (line.startsWith('⛔')) {
      result.deadlineSection.push(line);
      currentSection = 'deadline';
      return;
    }
  });
  
  // Parse summary
  if (summaryPart) {
    result.summary = summaryPart
      .split('\n')
      .map(l => l.trim())
      .filter(l => l.length > 0 && !l.startsWith('---') && !l.startsWith('==='))
      .slice(0, 5);
  }
  
  return result;
}
```
