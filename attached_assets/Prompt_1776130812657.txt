## 2. HELPER — PRE-GROUP ORDERS BY SHIFT

```javascript
function groupOrdersByShift(orders) {
  if (!orders || orders.length === 0) return [];
  
  // Group by similar shift (round to 2 decimal places for matching)
  const groups = {};
  
  orders.forEach(order => {
    const key = order.shiftHrs.toFixed(2);
    if (!groups[key]) {
      groups[key] = {
        shiftHrs: order.shiftHrs,
        shiftFormatted: order.shiftFormatted,
        orders: []
      };
    }
    groups[key].orders.push(order);
  });
  
  // Convert to array and calculate ranges
  return Object.values(groups).map(group => {
    const sortedOrders = group.orders.sort((a, b) => a.prio - b.prio);
    const prios = sortedOrders.map(o => o.prio);
    
    let prioRange;
    if (prios.length === 1) {
      prioRange = String(prios[0]);
    } else {
      prioRange = `${prios[0]}–${prios[prios.length - 1]}`;
    }
    
    return {
      shiftHrs: group.shiftHrs,
      shiftFormatted: group.shiftFormatted,
      prioRange,
      count: sortedOrders.length,
      orders: sortedOrders
    };
  }).sort((a, b) => b.shiftHrs - a.shiftHrs);
}

function groupOrdersByDelay(deadlineMisses) {
  if (!deadlineMisses || deadlineMisses.length === 0) return [];
  
  const groups = {};
  
  deadlineMisses.forEach(order => {
    const key = order.delayDays;
    if (!groups[key]) {
      groups[key] = {
        delayDays: order.delayDays,
        orders: []
      };
    }
    groups[key].orders.push(order);
  });
  
  return Object.values(groups).map(group => {
    const sortedOrders = group.orders.sort((a, b) => a.prio - b.prio);
    const prios = sortedOrders.map(o => o.prio);
    
    let prioRange;
    if (prios.length === 1) {
      prioRange = String(prios[0]);
    } else {
      prioRange = `${prios[0]}–${prios[prios.length - 1]}`;
    }
    
    return {
      delayDays: group.delayDays,
      prioRange,
      count: sortedOrders.length,
      orders: sortedOrders
    };
  }).sort((a, b) => b.delayDays - a.delayDays);
}

## 3. VALIDATION LAYER — AUTO-GROUP IF AI FAILS TO GROUP

After parsing the AI response, check if the output has too many individual lines. If so, auto-group using the calculated data as fallback.

```javascript
function validateAndFixGrouping(parsedResponse, calculatedImpact) {
  const maxBullets = 6;
  
  const totalBullets = parsedResponse.earlierSection.length + 
                       parsedResponse.laterSection.length + 
                       parsedResponse.deadlineSection.length;
  
  if (totalBullets <= maxBullets) {
    // AI output is properly grouped
    return parsedResponse;
  }
  
  // AI failed to group — auto-generate grouped output from calculated data
  console.warn(`AI output has ${totalBullets} bullets (max ${maxBullets}). Auto-grouping from calculated data.`);
  
  const result = {
    earlierSection: [],
    laterSection: [],
    deadlineSection: [],
    summary: parsedResponse.summary // Keep AI summary
  };
  
  // Auto-group earlier
  if (calculatedImpact.earlierOrders.length > 0) {
    const groups = groupOrdersByShift(calculatedImpact.earlierOrders);
    result.earlierSection = groups.map(g => {
      const notables = pickNotableOrders(g.orders);
      if (g.count === 1) {
        return `✅ Prio ${g.prioRange}: ${g.orders[0].name} — starts ${g.shiftFormatted} earlier. ${g.orders[0].n10dStatus !== 'N/A' ? g.orders[0].n10dStatus + '.' : ''}`;
      }
      return `✅ Prio ${g.prioRange} (${g.count} orders) — all start ~${g.shiftFormatted} earlier.${notables ? ' Notable: ' + notables + '.' : ''}`;
    });
  }
  
  // Auto-group later
  if (calculatedImpact.laterOrders.length > 0) {
    const groups = groupOrdersByShift(calculatedImpact.laterOrders);
    result.laterSection = groups.map(g => {
      const notables = pickNotableOrders(g.orders);
      if (g.count === 1) {
        return `⚠ Prio ${g.prioRange}: ${g.orders[0].name} — starts ${g.shiftFormatted} later. ${g.orders[0].n10dStatus !== 'N/A' ? g.orders[0].n10dStatus + '.' : ''}`;
      }
      return `⚠ Prio ${g.prioRange} (${g.count} orders) — all start ~${g.shiftFormatted} later.${notables ? ' Notable: ' + notables + '.' : ''}`;
    });
  }
  
  // Auto-group deadlines
  if (calculatedImpact.deadlineMisses.length > 0) {
    const groups = groupOrdersByDelay(calculatedImpact.deadlineMisses);
    result.deadlineSection = groups.map(g => {
      const notables = pickNotableOrders(g.orders);
      if (g.count === 1) {
        return `⛔ Prio ${g.prioRange}: ${g.orders[0].name} — new completion exceeds avail date ${g.orders[0].availDate} by ~${g.delayDays} days. ${g.orders[0].n10dStatus !== 'N/A' ? g.orders[0].n10dStatus + '.' : ''}`;
      }
      return `⛔ Prio ${g.prioRange} (${g.count} orders) — all exceed avail date by ~${g.delayDays} days.${notables ? ' Notable: ' + notables + '.' : ''}`;
    });
  }
  
  return result;
}

function pickNotableOrders(orders) {
  // Pick Critical first, then Urgent, then others — max 2-3
  const critical = orders.filter(o => o.n10dStatus === 'Critical');
  const urgent = orders.filter(o => o.n10dStatus === 'Urgent');
  const monitor = orders.filter(o => o.n10dStatus === 'Monitor');
  
  const notable = [...critical, ...urgent, ...monitor].slice(0, 3);
  
  if (notable.length === 0) {
    // Pick first 2 unique product names
    const seen = new Set();
    const unique = [];
    for (const o of orders) {
      if (!seen.has(o.name)) {
        seen.add(o.name);
        unique.push(o);
      }
      if (unique.length >= 2) break;
    }
    return unique.map(o => `${o.name} (Prio ${o.prio})`).join(', ');
  }
  
  return notable.map(o => `${o.name} (Prio ${o.prio}) is ${o.n10dStatus}`).join(', ');
}

## 4. INTEGRATE VALIDATION INTO THE FLOW

```javascript
async function generateImpact() {
  setIsLoading(true);
  setAiResponse(null);
  setUseFallback(false);
  
  const calculated = buildSchedulingImpactData(group, lineOrders);
  setCalculatedImpact(calculated);
  
  const hasAny = calculated.earlierOrders.length > 0 ||
                 calculated.laterOrders.length > 0 ||
                 calculated.deadlineMisses.length > 0;
  
  if (!hasAny) {
    setIsLoading(false);
    return;
  }
  
  try {
    const prompt = buildSchedulingImpactPrompt(context, calculated);
    const response = await callAI(prompt);
    const parsed = parseSchedulingImpactResponse(response);
    
    // Validate and fix grouping if AI listed individual orders
    const validated = validateAndFixGrouping(parsed, calculated);
    
    const aiHasContent = validated.earlierSection.length > 0 ||
                         validated.laterSection.length > 0 ||
                         validated.deadlineSection.length > 0;
    
    if (aiHasContent) {
      setAiResponse(validated);
    } else {
      setUseFallback(true);
    }
  } catch (error) {
    console.error('Scheduling impact AI failed:', error);
    setUseFallback(true);
  }
  
  setIsLoading(false);
}
