function calculateDeadlineMisses(group, lineOrders, calculatedShifts) {
  const combiningOrders = group.orders;
  const combiningIds = combiningOrders.map(o => o.id);
  const line = group.line;
  
  const currentLineOrders = lineOrders
    .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
    .sort((a, b) => a.prio - b.prio);
  
  // Determine lead order start date
  const datedOrders = combiningOrders.filter(o => {
    const avail = o.target_avail_date;
    return avail && /^\d{4}-\d{2}-\d{2}/.test(avail) && !isNaN(Date.parse(avail));
  });
  
  let leadOrder;
  if (datedOrders.length > 0) {
    leadOrder = datedOrders.sort((a, b) =>
      new Date(a.target_avail_date) - new Date(b.target_avail_date)
    )[0];
  } else {
    leadOrder = combiningOrders.sort((a, b) => a.prio - b.prio)[0];
  }
  
  const leadStartDate = leadOrder?.start_date
    ? new Date(leadOrder.start_date)
    : new Date();
  
  // Combined production time
  const combinedVolume = combiningOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0);
  const combinedRate = parseFloat(combiningOrders[0].run_rate) || 0;
  const combinedProdHrs = combinedRate > 0 ? combinedVolume / combinedRate : 0;
  const combinedCO = combiningOrders[0]._changeoverTotal ?? parseFloat(combiningOrders[0].changeover) ?? 0;
  
  // Simulate the "after" schedule to get accurate completion times
  let runningTime = 0;
  let combinedDone = false;
  const afterCompletion = {};
  
  currentLineOrders.forEach(order => {
    const vol = parseFloat(order.volume) || 0;
    const rate = parseFloat(order.run_rate) || 0;
    const prodHrs = rate > 0 ? vol / rate : 0;
    const co = order._changeoverTotal ?? parseFloat(order.changeover) ?? 0;
    
    if (combiningIds.includes(order.id)) {
      if (order.id === leadOrder.id && !combinedDone) {
        // Lead order — inject combined block
        const startTime = runningTime;
        runningTime += combinedProdHrs + combinedCO;
        afterCompletion[order.id] = {
          startHrs: startTime,
          endHrs: startTime + combinedProdHrs
        };
        combinedDone = true;
      }
      return; // Skip other combine members
    }
    
    // Non-combine order
    const startTime = runningTime;
    runningTime += prodHrs + co;
    afterCompletion[order.id] = {
      startHrs: startTime,
      endHrs: startTime + prodHrs
    };
  });
  
  // Check each non-combine order for deadline miss
  const deadlineMisses = [];
  
  currentLineOrders.forEach(order => {
    if (combiningIds.includes(order.id)) return;
    if (!afterCompletion[order.id]) return;
    
    const targetAvail = order.target_avail_date;
    const isDated = targetAvail && /^\d{4}-\d{2}-\d{2}/.test(targetAvail) && !isNaN(Date.parse(targetAvail));
    
    if (!isDated) return;
    
    const availDateObj = new Date(targetAvail);
    const newCompletionMs = leadStartDate.getTime() + (afterCompletion[order.id].endHrs * 60 * 60 * 1000);
    const newCompletionDate = new Date(newCompletionMs);
    
    if (newCompletionDate > availDateObj) {
      const delayMs = newCompletionDate.getTime() - availDateObj.getTime();
      const delayHrs = delayMs / (1000 * 60 * 60);
      const delayDays = Math.ceil(delayMs / (1000 * 60 * 60 * 24));
      
      // Get N10D status
      const n10dProduct = group._n10dData?.find(
        p => String(p.material_code_fg) === String(order.material_code_fg)
      );
      const n10dStatus = n10dProduct ? getProductStatus(
        n10dProduct.due_for_loading, n10dProduct.inventory,
        getDailyColumns(), n10dProduct, new Date()
      ) : null;
      
      deadlineMisses.push({
        prio: order.prio,
        name: order.item_description,
        fpr: order.fpr,
        availDate: availDateObj.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' }),
        availDateISO: targetAvail,
        newCompletion: newCompletionDate.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' }),
        newCompletionTime: newCompletionDate.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' }),
        newCompletionFull: `${newCompletionDate.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' })} ${newCompletionDate.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' })}`,
        delayHrs: delayHrs,
        delayFormatted: formatTimeShift(delayHrs),
        delayDays: delayDays,
        n10dStatus: n10dStatus || 'N/A'
      });
    }
  });
  
  return deadlineMisses.sort((a, b) => b.delayHrs - a.delayHrs);
}

function validateAndFixGrouping(parsedResponse, calculatedImpact) {
  const maxBullets = 6;
  
  let result = { ...parsedResponse };
  
  // CRITICAL: If deadline misses exist but AI didn't include them, force-add
  if (calculatedImpact.deadlineMisses.length > 0 && result.deadlineSection.length === 0) {
    console.warn('AI missed deadline section. Force-adding from calculated data.');
    result.deadlineSection = calculatedImpact.deadlineMisses.map(o => 
      `⛔ Prio ${o.prio}: ${o.name} — new completion ${o.newCompletionFull} exceeds avail date ${o.availDate} by ~${o.delayFormatted}. ${o.n10dStatus !== 'N/A' ? o.n10dStatus + '.' : ''}`
    );
  }
  
  // CRITICAL: If deadline misses exist but summary says "safe to proceed", fix summary
  if (calculatedImpact.deadlineMisses.length > 0 && result.summary.length > 0) {
    const summaryText = result.summary.join(' ').toLowerCase();
    if (summaryText.includes('safe to proceed') || 
        summaryText.includes('no risks') || 
        summaryText.includes('no delays') ||
        summaryText.includes('no orders at risk')) {
      console.warn('AI summary contradicts deadline data. Replacing summary.');
      
      const atRiskNames = calculatedImpact.deadlineMisses
        .slice(0, 3)
        .map(o => `${o.name} (Prio ${o.prio})`)
        .join(', ');
      
      result.summary = [
        `The combination will push ${calculatedImpact.deadlineMisses.length} order(s) past their avail date.`,
        `The most affected: ${atRiskNames}.`,
        `${calculatedImpact.deadlineMisses[0].name} will be delayed by approximately ${calculatedImpact.deadlineMisses[0].delayFormatted} past its avail date of ${calculatedImpact.deadlineMisses[0].availDate}.`,
        `Proceed with caution — review the at-risk orders before approving this combination.`
      ];
    }
  }
  
  // Check total bullet count for earlier/later sections (deadline is always individual)
  const earlierLaterBullets = result.earlierSection.length + result.laterSection.length;
  
  if (earlierLaterBullets > maxBullets) {
    // Auto-group earlier and later
    if (calculatedImpact.earlierOrders.length > 0) {
      const groups = groupOrdersByShift(calculatedImpact.earlierOrders);
      result.earlierSection = groups.map(g => {
        const notables = pickNotableOrders(g.orders);
        if (g.count === 1) {
          return `✅ Prio ${g.prioRange}: ${g.orders[0].name} — starts ${g.shiftFormatted} earlier. ${g.orders[0].n10dStatus !== 'N/A' ? g.orders[0].n10dStatus + '.' : ''}`;
        }
        return `✅ Prio ${g.prioRange} (${g.count} orders) — all start ~${g.shiftFormatted} earlier.${notables ? ' Notable: ' + notables + '.' : ''}`;
      });
    }
    
    if (calculatedImpact.laterOrders.length > 0) {
      const groups = groupOrdersByShift(calculatedImpact.laterOrders);
      result.laterSection = groups.map(g => {
        const notables = pickNotableOrders(g.orders);
        if (g.count === 1) {
          return `⚠ Prio ${g.prioRange}: ${g.orders[0].name} — starts ${g.shiftFormatted} later. ${g.orders[0].n10dStatus !== 'N/A' ? g.orders[0].n10dStatus + '.' : ''}`;
        }
        return `⚠ Prio ${g.prioRange} (${g.count} orders) — all start ~${g.shiftFormatted} later.${notables ? ' Notable: ' + notables + '.' : ''}`;
      });
    }
  }
  
  return result;
}
