```javascript
function calculateDeadlineMisses(group, lineOrders, n10dData) {
  const combiningOrders = group.orders;
  const combiningIds = combiningOrders.map(o => o.id);
  const line = group.line;
  
  const currentLineOrders = lineOrders
    .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
    .sort((a, b) => a.prio - b.prio);
  
  // Determine lead order
  const datedCombining = combiningOrders.filter(o => {
    const avail = o.target_avail_date;
    return avail && /^\d{4}-\d{2}-\d{2}/.test(avail) && !isNaN(Date.parse(avail));
  });
  
  let leadOrder;
  if (datedCombining.length > 0) {
    leadOrder = datedCombining.sort((a, b) =>
      new Date(a.target_avail_date) - new Date(b.target_avail_date)
    )[0];
  } else {
    leadOrder = combiningOrders.sort((a, b) => a.prio - b.prio)[0];
  }
  
  const leadStartDate = leadOrder?.start_date
    ? new Date(leadOrder.start_date)
    : new Date();
  
  // Combined production
  const combinedVolume = combiningOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0);
  const combinedRate = parseFloat(combiningOrders[0].run_rate) || 0;
  const combinedProdHrs = combinedRate > 0 ? combinedVolume / combinedRate : 0;
  const combinedCO = combiningOrders[0]._changeoverTotal ?? parseFloat(combiningOrders[0].changeover) ?? 0;
  
  // Simulate "before" schedule
  let runningTimeBefore = 0;
  const beforeCompletion = {};
  
  currentLineOrders.forEach(order => {
    const vol = parseFloat(order.volume) || 0;
    const rate = parseFloat(order.run_rate) || 0;
    const prodHrs = rate > 0 ? vol / rate : 0;
    const co = order._changeoverTotal ?? parseFloat(order.changeover) ?? 0;
    
    beforeCompletion[order.id] = {
      startHrs: runningTimeBefore,
      endHrs: runningTimeBefore + prodHrs
    };
    runningTimeBefore += prodHrs + co;
  });
  
  // Simulate "after" schedule
  let runningTimeAfter = 0;
  let combinedDone = false;
  const afterCompletion = {};
  
  currentLineOrders.forEach(order => {
    const vol = parseFloat(order.volume) || 0;
    const rate = parseFloat(order.run_rate) || 0;
    const prodHrs = rate > 0 ? vol / rate : 0;
    const co = order._changeoverTotal ?? parseFloat(order.changeover) ?? 0;
    
    if (combiningIds.includes(order.id)) {
      if (order.id === leadOrder.id && !combinedDone) {
        const startTime = runningTimeAfter;
        runningTimeAfter += combinedProdHrs + combinedCO;
        afterCompletion[order.id] = {
          startHrs: startTime,
          endHrs: startTime + combinedProdHrs
        };
        combinedDone = true;
      }
      return;
    }
    
    const startTime = runningTimeAfter;
    runningTimeAfter += prodHrs + co;
    afterCompletion[order.id] = {
      startHrs: startTime,
      endHrs: startTime + prodHrs
    };
  });
  
  const deadlineMisses = [];
  
  currentLineOrders.forEach(order => {
    if (combiningIds.includes(order.id)) return;
    if (!afterCompletion[order.id]) return;
    
    const targetAvail = order.target_avail_date;
    const isDated = targetAvail && /^\d{4}-\d{2}-\d{2}/.test(targetAvail) && !isNaN(Date.parse(targetAvail));
    
    // Get N10D status
    const n10dProduct = n10dData?.find(
      p => String(p.material_code_fg) === String(order.material_code_fg)
    );
    const n10dStatus = n10dProduct ? getProductStatus(
      n10dProduct.due_for_loading, n10dProduct.inventory,
      getDailyColumns(), n10dProduct, new Date()
    ) : null;
    
    // Calculate time shift
    const beforeEnd = beforeCompletion[order.id]?.endHrs || 0;
    const afterEnd = afterCompletion[order.id]?.endHrs || 0;
    const shiftHrs = afterEnd - beforeEnd; // Positive = delayed
    
    // TYPE 1: Dated order — check if new completion exceeds avail date
    if (isDated) {
      const availDateObj = new Date(targetAvail);
      const newCompletionMs = leadStartDate.getTime() + (afterCompletion[order.id].endHrs * 60 * 60 * 1000);
      const newCompletionDate = new Date(newCompletionMs);
      
      if (newCompletionDate > availDateObj) {
        const delayMs = newCompletionDate.getTime() - availDateObj.getTime();
        const delayHrs = delayMs / (1000 * 60 * 60);
        const delayDays = Math.ceil(delayMs / (1000 * 60 * 60 * 24));
        
        deadlineMisses.push({
          type: 'dated',
          prio: order.prio,
          name: order.item_description,
          fpr: order.fpr,
          availDate: availDateObj.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' }),
          availDateISO: targetAvail,
          newCompletion: newCompletionDate.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' }),
          newCompletionTime: newCompletionDate.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' }),
          newCompletionFull: `${newCompletionDate.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' })} ${newCompletionDate.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' })}`,
          delayHrs: delayHrs,
          delayFormatted: formatTimeShift(delayHrs),
          delayDays: delayDays,
          n10dStatus: n10dStatus || 'N/A',
          shiftHrs: shiftHrs,
          shiftFormatted: formatTimeShift(Math.abs(shiftHrs))
        });
      }
    }
    
    // TYPE 2: Non-dated order — check if N10D is Critical or Urgent AND order is being delayed
    if (!isDated && shiftHrs > 0.01) {
      const isCritical = n10dStatus === 'Critical';
      const isUrgent = n10dStatus === 'Urgent';
      
      if (isCritical || isUrgent) {
        deadlineMisses.push({
          type: 'n10d_risk',
          prio: order.prio,
          name: order.item_description,
          fpr: order.fpr,
          availDate: null,
          availDateISO: null,
          newCompletion: null,
          newCompletionFull: null,
          delayHrs: shiftHrs,
          delayFormatted: formatTimeShift(shiftHrs),
          delayDays: null,
          n10dStatus: n10dStatus,
          shiftHrs: shiftHrs,
          shiftFormatted: formatTimeShift(shiftHrs)
        });
      }
    }
  });
  
  return deadlineMisses.sort((a, b) => {
    // Critical first, then Urgent, then by delay
    const statusOrder = { 'Critical': 0, 'Urgent': 1 };
    const aOrder = statusOrder[a.n10dStatus] ?? 2;
    const bOrder = statusOrder[b.n10dStatus] ?? 2;
    if (aOrder !== bOrder) return aOrder - bOrder;
    return (b.delayHrs || 0) - (a.delayHrs || 0);
  });
}
```

```javascript
function validateAndFixGrouping(parsedResponse, calculatedImpact) {
  let result = { ...parsedResponse };
  
  // FIX: Remove literal "[N10D status]", "[N10D: —]" etc.
  function cleanN10DPlaceholders(lines) {
    return lines.map(line => {
      let cleaned = line;
      cleaned = cleaned.replace(/\[N10D status\]/gi, '');
      cleaned = cleaned.replace(/\[N10D: —\]/gi, '');
      cleaned = cleaned.replace(/\[N10D: N\/A\]/gi, '');
      cleaned = cleaned.replace(/\[N10D:.*?\]/gi, '');
      cleaned = cleaned.replace(/N10D: —/gi, '');
      cleaned = cleaned.replace(/\.\s*\./g, '.'); // Clean double periods
      cleaned = cleaned.replace(/\s+/g, ' ').trim();
      return cleaned;
    });
  }
  
  result.earlierSection = cleanN10DPlaceholders(result.earlierSection);
  result.laterSection = cleanN10DPlaceholders(result.laterSection);
  result.deadlineSection = cleanN10DPlaceholders(result.deadlineSection);
  result.summary = cleanN10DPlaceholders(result.summary);
  
  // FIX: Remove non-dated orders that are NOT Critical/Urgent from deadline section
  // These should not have been flagged
  result.deadlineSection = result.deadlineSection.filter(line => {
    const lower = line.toLowerCase();
    // Keep if it mentions an actual avail date (dated order)
    if (lower.includes('exceeds avail date') && !lower.includes('no date') && !lower.includes('n/a')) {
      return true;
    }
    // Keep if it mentions Critical or Urgent stock
    if (lower.includes('critical') || lower.includes('urgent')) {
      return true;
    }
    // Keep if it mentions stock depletion
    if (lower.includes('stock depletion') || lower.includes('stockout')) {
      return true;
    }
    // Remove — likely a non-dated, non-critical order that shouldn't be here
    return false;
  });
  
  // FIX: Force-add deadline section if calculated data has misses but AI missed them
  if (calculatedImpact.deadlineMisses.length > 0 && result.deadlineSection.length === 0) {
    console.warn('AI missed deadline section. Force-adding from calculated data.');
    result.deadlineSection = calculatedImpact.deadlineMisses.map(o => {
      if (o.type === 'dated') {
        return `⛔ Prio ${o.prio}: ${o.name} — new completion ${o.newCompletionFull} exceeds avail date ${o.availDate} by ~${o.delayFormatted}. ${o.n10dStatus !== 'N/A' ? o.n10dStatus + '.' : ''}`;

        return `⛔ Prio ${o.prio}: ${o.name} — ${o.n10dStatus} stock status. Delay of ~${o.delayFormatted} may accelerate stock depletion.`;
      }
      return '';
    }).filter(Boolean);
  }
  
  // FIX: Summary contradiction check
  if (calculatedImpact.deadlineMisses.length > 0 && result.summary.length > 0) {
    const summaryText = result.summary.join(' ').toLowerCase();
    if (summaryText.includes('safe to proceed') ||
        summaryText.includes('no risks') ||
        summaryText.includes('no delays') ||
        summaryText.includes('no orders at risk')) {
      console.warn('AI summary contradicts deadline data. Replacing.');
      
      const datedMisses = calculatedImpact.deadlineMisses.filter(d => d.type === 'dated');
      const n10dMisses = calculatedImpact.deadlineMisses.filter(d => d.type === 'n10d_risk');
      
      const summaryLines = [];
      
      if (datedMisses.length > 0) {
        const names = datedMisses.slice(0, 2).map(o => `${o.name} (Prio ${o.prio})`).join(' and ');
        summaryLines.push(`The combination will push ${datedMisses.length} order(s) past their avail date, including ${names}.`);
        summaryLines.push(`${datedMisses[0].name} will be delayed by approximately ${datedMisses[0].delayFormatted} past its avail date of ${datedMisses[0].availDate}.`);
      }
      
      if (n10dMisses.length > 0) {
        const names = n10dMisses.slice(0, 2).map(o => `${o.name} (Prio ${o.prio}, ${o.n10dStatus})`).join(' and ');
        summaryLines.push(`Additionally, ${n10dMisses.length} order(s) with critical stock levels will be delayed, including ${names}, which may accelerate stock depletion.`);
```
