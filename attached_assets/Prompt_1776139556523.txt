## 1. STORE ORIGINAL ORDER BEFORE SEQUENCING

When auto-sequence is triggered, save a snapshot of the original order before any changes:

```javascript
function handleAutoSequence(line) {
  // Save snapshot of current orders BEFORE sequencing
  const currentLineOrders = allOrders
    .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
    .sort((a, b) => a.prio - b.prio);
  
  const originalSnapshot = currentLineOrders.map(o => ({
    ...o,
    _originalPrio: o.prio
  }));
  
  // Store in state
  setOriginalOrderSnapshot(originalSnapshot);
  
  // Run auto-sequence algorithm
  const sequencedOrders = runAutoSequence(currentLineOrders, changeoverRules, masterData, n10dData);
  
  // Calculate movement indicators
  const sequencedWithMovement = sequencedOrders.map((order, newIndex) => {
    const newPrio = newIndex + 1;
    const originalIndex = originalSnapshot.findIndex(o => o.id === order.id);
    const originalPrio = originalIndex >= 0 ? originalIndex + 1 : newPrio;
    
    let movement = 'same';
    let movementDelta = originalPrio - newPrio;
    
    if (movementDelta > 0) movement = 'up';
    else if (movementDelta < 0) movement = 'down';
    
    return {
      ...order,
      prio: newPrio,
      _originalPrio: originalPrio,
      _movement: movement,
      _movementDelta: Math.abs(movementDelta)
    };
  });
  
  setSequencedOrders(sequencedWithMovement);
  setShowAutoSequencePreview(true);
}
```

## 3. MOVEMENT INDICATOR COMPONENT

```jsx
function MovementIndicator({ movement, delta }) {
  if (movement === 'same' || delta === 0) return null;
  
  if (movement === 'up') {
    return (
      <span className="as-movement as-movement-up" title={`Moved up ${delta} position${delta > 1 ? 's' : ''}`}>
        ↑{delta}
      </span>
    );
  }
  
  if (movement === 'down') {
    return (
      <span className="as-movement as-movement-down" title={`Moved down ${delta} position${delta > 1 ? 's' : ''}`}>
        ↓{delta}
      </span>
    );
  }
  
  return null;
}
```
