## FIX 2: MOVEMENT INDICATOR — ICON ONLY, NO HIGHLIGHT

```jsx
function MovementIndicator({ movement, delta }) {
  if (movement === 'same' || delta === 0) return null;
  
  if (movement === 'up') {
    return (
      <span 
        className="as-movement-icon as-movement-icon-up" 
        title={`Moved up ${delta} position${delta > 1 ? 's' : ''}`}
      >
        ▲
      </span>
    );
  }
  
  if (movement === 'down') {
    return (
      <span 
        className="as-movement-icon as-movement-icon-down" 
        title={`Moved down ${delta} position${delta > 1 ? 's' : ''}`}
      >
        ▼
      </span>
    );
  }
  
  return null;
}
```
