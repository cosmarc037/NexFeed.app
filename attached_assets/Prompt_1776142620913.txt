```javascript
function handleAutoSequence(line) {
  // Get current line orders BEFORE any sequencing
  const currentLineOrders = allOrders
    .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
    .sort((a, b) => a.prio - b.prio);
  
  // DEEP COPY — critical! Must not be a reference to the same objects
  const originalSnapshot = currentLineOrders.map(o => ({
    ...JSON.parse(JSON.stringify(o)),
    _originalPrio: o.prio
  }));
  
  // Store the snapshot BEFORE running auto-sequence
  setOriginalOrderSnapshot(originalSnapshot);
  
  // NOW run auto-sequence on a SEPARATE copy
  const ordersToSequence = currentLineOrders.map(o => ({ ...o }));
  const sequencedOrders = runAutoSequence(ordersToSequence, changeoverRules, masterData, n10dData);
  
  // Calculate movement by comparing against originalSnapshot
  const sequencedWithMovement = sequencedOrders.map((order, newIndex) => {
    const newPrio = newIndex + 1;
    
    // Find this order's ORIGINAL position in the snapshot
    const originalIndex = originalSnapshot.findIndex(o => o.id === order.id);
    const originalPrio = originalIndex >= 0 ? originalIndex + 1 : newPrio;
    
    let movement = 'same';
    let movementDelta = originalPrio - newPrio;
    
    if (movementDelta > 0) movement = 'up';
    else if (movementDelta < 0) movement = 'down';
    
    return {
      ...order,
      prio: newPrio,
      _originalPrio: originalPrio,
      _movement: movement,
      _movementDelta: Math.abs(movementDelta)
    };
  });
  
  setSequencedOrders(sequencedWithMovement);
  setShowAutoSequencePreview(true);
}
```
