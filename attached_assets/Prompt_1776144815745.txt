## 2. CHANGEOVER CALCULATION IN PREVIEW — USE CHANGEOVER RULES

When the auto-sequence preview generates the sequenced orders, apply the full changeover calculation (base + additional) instead of just showing 0.10:

```javascript
function calculatePreviewChangeovers(sequencedOrders, changeoverRules) {
  if (!changeoverRules) {
    changeoverRules = getDefaultChangeoverRules();
  }
  
  // Handle old format
  if (Array.isArray(changeoverRules)) {
    changeoverRules = {
      baseChangeover: { p: 0.17, mp: 0.17, c: 0.33, others: 0.00 },
      additionalRules: changeoverRules
    };
  }
  
  sequencedOrders.forEach((order, index) => {
    // Base changeover from form
    const baseChangeover = getBaseChangeover(order.form, changeoverRules.baseChangeover);
    
    // Get following order for additional changeover calculation
    let followingOrder = null;
    for (let j = index + 1; j < sequencedOrders.length; j++) {
      if (sequencedOrders[j].status !== 'Done' && sequencedOrders[j].status !== 'Cancel PO') {
        followingOrder = sequencedOrders[j];
        break;
      }
    }
    
    // Calculate additional changeover
    const additional = followingOrder 
      ? calculateAdditionalChangeover(order, followingOrder, changeoverRules.additionalRules)
      : { total: 0, breakdown: [] };
    
    const totalChangeover = baseChangeover + additional.total;
    
    // Update order
    order._changeoverTotal = totalChangeover;
    order._changeoverBase = baseChangeover;
    order._changeoverBaseForm = order.form || '—';
    order._changeoverAdditional = additional.total;
    order._changeoverBreakdown = additional.breakdown;
    order.changeover = baseChangeover;
  });
  
  return sequencedOrders;
}
```
