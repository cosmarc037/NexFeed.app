```javascript
// Feedmill run rates (MT per hour)
const FEEDMILL_RUN_RATES = {
  'Feedmill 1': 40,
  'Feedmill 2': 20,
  'Feedmill 3': 20,
  'PMX': 20
};

const MAX_COMBINE_MT = 180;

function getFeedmillFromLine(line) {
  // e.g., "Line 1" in Feedmill 1 context
  // This should map based on your app's structure
  // Assuming line objects have a feedmill property
  if (line.includes('Feedmill 1') || line.startsWith('FM1')) return 'Feedmill 1';
  if (line.includes('Feedmill 2') || line.startsWith('FM2')) return 'Feedmill 2';
  if (line.includes('Feedmill 3') || line.startsWith('FM3')) return 'Feedmill 3';
  if (line.includes('PMX')) return 'PMX';
  return 'Feedmill 1'; // default
}

function getFeedmillRunRate(feedmill) {
  return FEEDMILL_RUN_RATES[feedmill] || 20;
}

function getLineShortName(line) {
  // Convert "Line 1" to "L1", "Line 2" to "L2", etc.
  const match = line.match(/Line\s*(\d+)/i);
  if (match) return `L${match[1]}`;
  return line;
}

function canProduceOnLine(order, line, masterData) {
  // Check if product exists in master data for this line
  // AND has a run rate
  const entry = masterData.find(m =>
    String(m.material_code_fg) === String(order.material_code_fg) &&
    m.line === line
  );
  return entry && (parseFloat(entry.run_rate) || 0) > 0;
}

function calculateLineQueueTime(lineOrders, feedmill) {
  const totalMT = lineOrders.reduce((sum, o) => sum + (parseFloat(o.volume) || 0), 0);
  const runRate = getFeedmillRunRate(feedmill);
  return runRate > 0 ? totalMT / runRate : Infinity;
}
```

```javascript
function plantLevelCombineAndPlace(allOrders, masterData, n10dData, allLines) {
  // Deep copy all orders
  let workingOrders = allOrders
    .filter(o => o.status !== 'Done' && o.status !== 'Cancel PO')
    .map(o => ({ ...JSON.parse(JSON.stringify(o)), _originalLine: o.line, _movedFromLine: null }));
  
  // Track orders per line
  const lineOrdersMap = {};
  allLines.forEach(line => {
    lineOrdersMap[line] = workingOrders
      .filter(o => o.line === line)
      .sort((a, b) => a.prio - b.prio);
  });
  
  // Step 1: Sort dated orders first
  const datedOrders = workingOrders.filter(o => {
    const avail = o.target_avail_date;
    return avail && /^\d{4}-\d{2}-\d{2}/.test(avail) && !isNaN(Date.parse(avail));
  }).sort((a, b) => new Date(a.target_avail_date) - new Date(b.target_avail_date));
  
  const nonDatedOrders = workingOrders.filter(o => {
    const avail = o.target_avail_date;
    return !avail || !/^\d{4}-\d{2}-\d{2}/.test(avail) || isNaN(Date.parse(avail));
  });
  
  const sortedOrders = [...datedOrders, ...nonDatedOrders];
  
  // Step 2: Group by Material Code + Formula Version (ignore line)
  const matchGroups = {};
  sortedOrders.forEach(order => {
    const key = `${order.material_code_fg}||${order.formula_version || '—'}`;
    if (!matchGroups[key]) {
      matchGroups[key] = [];
    }
    matchGroups[key].push(order);
  });
  
  // Separate matched (2+ orders) and non-matched (1 order)
  const matchedGroups = {};
  const nonMatchedOrders = [];
  
  Object.entries(matchGroups).forEach(([key, orders]) => {
    if (orders.length >= 2) {
      matchedGroups[key] = orders;
    } else {
      nonMatchedOrders.push(orders[0]);
    }
  });
  
  // Track combined orders and placements
  const combinedOrders = [];
  const placementLog = [];
  const processedOrderIds = new Set();
  
  // Step 3: Process matched groups
  Object.entries(matchedGroups).forEach(([key, orders]) => {
    // Find eligible lines for this product
    const eligibleLines = allLines.filter(line =>
      canProduceOnLine(orders[0], line, masterData)
    );
    
    if (eligibleLines.length === 0) {
      // No eligible lines — leave orders where they are
      orders.forEach(o => processedOrderIds.add(o.id));
      return;
    }
    
    // Sort orders by volume descending for greedy combine
    const sortedByVolume = [...orders].sort((a, b) =>
      (parseFloat(b.volume) || 0) - (parseFloat(a.volume) || 0)
    );
    
    // Greedy combine up to 180 MT
    const combineGroups = [];
    const remaining = [...sortedByVolume];
    
    while (remaining.length > 0) {
      const group = [remaining.shift()];
      let groupVolume = parseFloat(group[0].volume) || 0;
      
      // Try to add more orders to this group
      for (let i = remaining.length - 1; i >= 0; i--) {
        const candidateVolume = parseFloat(remaining[i].volume) || 0;
        if (groupVolume + candidateVolume <= MAX_COMBINE_MT) {
          groupVolume += candidateVolume;
          group.push(remaining.splice(i, 1)[0]);
        }
      }
      
      combineGroups.push({ orders: group, totalVolume: groupVolume });
    }
    
    // Place each combine group on the best line
    combineGroups.forEach(combineGroup => {
      // Calculate queue time per eligible line
      const lineScores = eligibleLines.map(line => {
        const feedmill = getFeedmillFromLine(line);
        const currentLineOrders = lineOrdersMap[line] || [];
        const queueTime = calculateLineQueueTime(currentLineOrders, feedmill);
        const lineNumber = parseInt(line.match(/\d+/)?.[0] || '99');
        
        return { line, feedmill, queueTime, lineNumber };
      });
      
      // Sort: lowest queue time, then lowest line number
      lineScores.sort((a, b) => {
        if (Math.abs(a.queueTime - b.queueTime) > 0.01) return a.queueTime - b.queueTime;
        return a.lineNumber - b.lineNumber;
      });
      
      const bestLine = lineScores[0].line;
      
      if (combineGroup.orders.length === 1) {
        // Single order — just place on best line
        const order = combineGroup.orders[0];
        if (order.line !== bestLine) {
          order._movedFromLine = order.line;
          order.line = bestLine;
        }
        processedOrderIds.add(order.id);
        
        // Update line orders map
        if (order._movedFromLine) {
          lineOrdersMap[order._movedFromLine] = lineOrdersMap[order._movedFromLine]
            .filter(o => o.id !== order.id);
        }
        if (!lineOrdersMap[bestLine].some(o => o.id === order.id)) {
          lineOrdersMap[bestLine].push(order);
        }
        
        placementLog.push({
          type: 'placed',
          order: order.item_description,
          fromLine: order._movedFromLine,
          toLine: bestLine,
          volume: order.volume
        });
      } else {
        // Multiple orders — combine
        const baseOrder = { ...combineGroup.orders[0] };
        const combinedVolume = combineGroup.totalVolume;
        
        // Determine avail date — earliest dated
        const datedInGroup = combineGroup.orders.filter(o => {
          const avail = o.target_avail_date;
          return avail && /^\d{4}-\d{2}-\d{2}/.test(avail) && !isNaN(Date.parse(avail));
        }).sort((a, b) => new Date(a.target_avail_date) - new Date(b.target_avail_date));
        
        if (datedInGroup.length > 0) {
          baseOrder.target_avail_date = datedInGroup[0].target_avail_date;
        }
        
        baseOrder.volume = combinedVolume;
        baseOrder.line = bestLine;
        baseOrder._isCombined = true;
        baseOrder._combinedFrom = combineGroup.orders.map(o => ({
          id: o.id,
          line: o._originalLine,
          volume: o.volume,
          fpr: o.fpr
        }));
        baseOrder._movedFromLine = baseOrder._originalLine !== bestLine ? baseOrder._originalLine : null;
        baseOrder._combinedFromLines = [...new Set(combineGroup.orders.map(o => o._originalLine))];
        
        // Recalculate batches, bags, production time
        const batchSize = parseFloat(baseOrder.batch_size) || 1;
        baseOrder.batches = Math.ceil(combinedVolume / batchSize);
        const fgWeight = parseFloat(baseOrder.fg_weight) || 1;
        baseOrder.bags = baseOrder.batches * fgWeight * 20;
        const runRate = parseFloat(baseOrder.run_rate) || 0;
        baseOrder.production_time = runRate > 0 ? (combinedVolume / runRate).toFixed(2) : '0.00';
        
        // Remove original orders from their lines
        combineGroup.orders.forEach(o => {
          processedOrderIds.add(o.id);
          const origLine = o._originalLine;
          if (lineOrdersMap[origLine]) {
            lineOrdersMap[origLine] = lineOrdersMap[origLine].filter(lo => lo.id !== o.id);
          }
        });
        
        // Add combined order to best line
        lineOrdersMap[bestLine].push(baseOrder);
        combinedOrders.push(baseOrder);
        
        placementLog.push({
          type: 'combined',
          product: baseOrder.item_description,
          ordersCount: combineGroup.orders.length,
          fromLines: [...new Set(combineGroup.orders.map(o => o._originalLine))],
          toLine: bestLine,
          totalVolume: combinedVolume
        });
      }
    });
  });
  
  // Step 4: Process non-matched orders
  nonMatchedOrders.forEach(order => {
    if (processedOrderIds.has(order.id)) return;
    
    // Check if product has master data on any line
    const eligibleLines = allLines.filter(line =>
      canProduceOnLine(order, line, masterData)
    );
    
    if (eligibleLines.length === 0) {
      // No master data — leave where it is
      processedOrderIds.add(order.id);
      return;
    }
    
    // Find best line by queue time
    const lineScores = eligibleLines.map(line => {
      const feedmill = getFeedmillFromLine(line);
      const currentLineOrders = lineOrdersMap[line] || [];
      const queueTime = calculateLineQueueTime(currentLineOrders, feedmill);
      const lineNumber = parseInt(line.match(/\d+/)?.[0] || '99');
      
      return { line, queueTime, lineNumber };
    });
    
    lineScores.sort((a, b) => {
      if (Math.abs(a.queueTime - b.queueTime) > 0.01) return a.queueTime - b.queueTime;
      return a.lineNumber - b.lineNumber;
    });
    
    const bestLine = lineScores[0].line;
    
    if (order.line !== bestLine) {
      order._movedFromLine = order.line;
      
      // Remove from original line
      lineOrdersMap[order.line] = lineOrdersMap[order.line].filter(o => o.id !== order.id);
      
      // Add to best line
      order.line = bestLine;
      lineOrdersMap[bestLine].push(order);
      
      placementLog.push({
        type: 'moved',
        order: order.item_description,
        fromLine: order._movedFromLine,
        toLine: bestLine,
        volume: order.volume
      });
    }
    
    processedOrderIds.add(order.id);
  });
  
  return { lineOrdersMap, combinedOrders, placementLog };
}
```

```javascript
function handlePlantLevelAutoSequence() {
  // Get all lines for all feedmills
  const allLines = getAllLines(); // Returns array of all line identifiers
  
  // Get all active orders
  const allActiveOrders = allOrders.filter(
    o => o.status !== 'Done' && o.status !== 'Cancel PO'
  );
  
  // Save original snapshot — deep copy per line
  const originalSnapshot = {};
  allLines.forEach(line => {
    originalSnapshot[line] = allActiveOrders
      .filter(o => o.line === line)
      .sort((a, b) => a.prio - b.prio)
      .map(o => ({ ...JSON.parse(JSON.stringify(o)) }));
  });
  
  setOriginalPlantSnapshot(originalSnapshot);
  
  // Phase 1: Combine and place across lines
  const { lineOrdersMap, combinedOrders, placementLog } = plantLevelCombineAndPlace(
    allActiveOrders, masterData, n10dData, allLines
  );
  
  // Phase 2: Auto-sequence each line independently
  const sequencedResults = {};
  const changeoverRules = getDefaultChangeoverRules();
  
  allLines.forEach(line => {
    const lineOrders = lineOrdersMap[line] || [];
    
    if (lineOrders.length === 0) {
      sequencedResults[line] = [];
      return;
    }
    
    // Run per-line auto-sequence
    const sequenced = runAutoSequence(lineOrders, changeoverRules, masterData, n10dData);
    
    // Calculate changeovers
    const withChangeovers = calculatePreviewChangeovers(sequenced, changeoverRules);
    
    // Calculate movement indicators (compared to original)
    const originalLineOrders = originalSnapshot[line] || [];
    
    const withMovement = withChangeovers.map((order, newIndex) => {
      const newPrio = newIndex + 1;
      
      // Check if order was originally on this line
      const originalIndex = originalLineOrders.findIndex(o => o.id === order.id);
      let originalPrio = -1;
      let movement = 'same';
      let movementDelta = 0;
      
      if (originalIndex >= 0) {
        originalPrio = originalIndex + 1;
        movementDelta = originalPrio - newPrio;
        if (movementDelta > 0) movement = 'up';
        else if (movementDelta < 0) movement = 'down';
      } else {
        // Order came from another line
        movement = 'new_to_line';
      }
      
      return {
        ...order,
        prio: newPrio,
        _originalPrio: originalPrio,
        _movement: movement,
        _movementDelta: Math.abs(movementDelta)
      };
    });
    
    sequencedResults[line] = withMovement;
  });
  
  // Calculate summary stats
  const summaryStats = calculatePlantSummary(originalSnapshot, sequencedResults, combinedOrders, placementLog, allLines);
  
  setPlantSequenceResults(sequencedResults);
  setPlantSummaryStats(summaryStats);
  setPlantPlacementLog(placementLog);
  setShowPlantAutoSequencePreview(true);
}
```

```javascript
function calculatePlantSummary(originalSnapshot, sequencedResults, combinedOrders, placementLog, allLines) {
  const totalOrdersBefore = Object.values(originalSnapshot).reduce((sum, orders) => sum + orders.length, 0);
  const totalOrdersAfter = Object.values(sequencedResults).reduce((sum, orders) => sum + orders.length, 0);
  
  const ordersCombined = combinedOrders.length;
  const ordersMovedBetweenLines = placementLog.filter(p => p.type === 'moved' || p.type === 'placed').length;
  const ordersCombinedAndMoved = placementLog.filter(p => p.type === 'combined').length;
  
  const perLineSummary = allLines.map(line => {
    const feedmill = getFeedmillFromLine(line);
    const beforeOrders = originalSnapshot[line] || [];
    const afterOrders = sequencedResults[line] || [];
    
    const beforeMT = beforeOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0);
    const afterMT = afterOrders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0);
    
    const beforeQueueTime = calculateLineQueueTime(beforeOrders, feedmill);
    const afterQueueTime = calculateLineQueueTime(afterOrders, feedmill);
    
    const newOrders = afterOrders.filter(o => o._movement === 'new_to_line').length;
    const removedOrders = beforeOrders.filter(bo =>
      !afterOrders.some(ao => ao.id === bo.id)
    ).length;
    
    return {
      line,
      feedmill,
      beforeCount: beforeOrders.length,
      afterCount: afterOrders.length,
      beforeMT: beforeMT.toFixed(1),
      afterMT: afterMT.toFixed(1),
      beforeQueueHrs: beforeQueueTime.toFixed(2),
      afterQueueHrs: afterQueueTime.toFixed(2),
      queueDiff: (afterQueueTime - beforeQueueTime).toFixed(2),
      newOrders,
      removedOrders
    };
  }).filter(ls => ls.beforeCount > 0 || ls.afterCount > 0);
  
  return {
    totalOrdersBefore,
    totalOrdersAfter,
    ordersCombined,
    ordersMovedBetweenLines,
    ordersCombinedAndMoved,
    perLineSummary
  };
}
```

## 9. APPLY TO SCHEDULE — ALL LINES

```javascript
function handlePlantApply(sequencedResults) {
  let updatedOrders = [...allOrders];
  
  Object.entries(sequencedResults).forEach(([line, lineOrders]) => {
    // Remove all existing active orders for this line
    updatedOrders = updatedOrders.filter(o =>
      o.line !== line || o.status === 'Done' || o.status === 'Cancel PO'
    );
    
    // Add sequenced orders
    lineOrders.forEach((order, index) => {
      order.prio = index + 1;
      updatedOrders.push(order);
    });
  });
  
  // Recalculate changeovers and completion dates for all lines
  const allActiveLines = [...new Set(Object.keys(sequencedResults))];
  allActiveLines.forEach(line => {
    const lineOrders = updatedOrders
      .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
      .sort((a, b) => a.prio - b.prio);
    
    calculatePreviewChangeovers(lineOrders, getDefaultChangeoverRules());
  });
  
  recalculateCompletionDates(updatedOrders);
  setAllOrders(updatedOrders);
  setShowPlantAutoSequencePreview(false);
  
  showNotification('success', `Plant-level auto-sequence applied across ${allActiveLines.length} lines.`);
}
```
