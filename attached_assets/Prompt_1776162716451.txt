## FIX: APPLY TO SCHEDULE — PLANT LEVEL

The issue is that clicking "Apply to Schedule" does nothing. The handler needs to:
1. Remove all original active orders from all affected lines
2. Insert the sequenced orders (including combined orders) into each line
3. Reassign prio numbers
4. Recalculate changeovers and completion dates
5. Close the preview
6. Show success notification

```javascript
function handlePlantApply(sequencedResults) {
  // Get all affected lines
  const affectedLines = Object.keys(sequencedResults);
  
  // Start with a copy of all current orders
  let updatedOrders = allOrders.map(o => ({ ...o }));
  
  // For each affected line, remove existing active orders and replace with sequenced
  affectedLines.forEach(line => {
    const sequencedLineOrders = sequencedResults[line] || [];
    
    // Remove all existing active orders for this line
    updatedOrders = updatedOrders.filter(o => {
      if (o.line !== line) return true; // Keep orders from other lines
      if (o.status === 'Done' || o.status === 'Cancel PO') return true; // Keep Done/Cancelled
      return false; // Remove active orders — they'll be replaced
    });
    
    // Also remove orders that were MOVED from this line to another line
    // These orders now exist in a different line's sequenced results
    const allSequencedIds = new Set();
    Object.values(sequencedResults).forEach(lineOrders => {
      lineOrders.forEach(o => {
        allSequencedIds.add(o.id);
        // Also track combined sub-order IDs
        if (o._combinedFrom) {
          o._combinedFrom.forEach(sub => allSequencedIds.add(sub.id));
        }
      });
    });
    
    // Remove any remaining originals that are now part of combined orders
    updatedOrders = updatedOrders.filter(o => {
      if (o.status === 'Done' || o.status === 'Cancel PO') return true;
      // If this order's ID is consumed by a combined order, remove it
      if (allSequencedIds.has(o.id)) {
        // Check if it exists as a top-level sequenced order
        const isTopLevel = Object.values(sequencedResults).some(lineOrders =>
          lineOrders.some(so => so.id === o.id)
        );
        if (!isTopLevel) return false; // It's a sub-order of a combined order — remove
      }
      return true;
    });
  });
  
  // Remove ALL original active orders that were consumed by combinations
  // across any line (some originals may have been on non-affected lines)
  const allCombinedSubIds = new Set();
  Object.values(sequencedResults).forEach(lineOrders => {
    lineOrders.forEach(o => {
      if (o._combinedFrom) {
        o._combinedFrom.forEach(sub => allCombinedSubIds.add(sub.id));
      }
    });
  });
  
  updatedOrders = updatedOrders.filter(o => {
    if (o.status === 'Done' || o.status === 'Cancel PO') return true;
    if (allCombinedSubIds.has(o.id)) return false; // Consumed by a combination
    return true;
  });
  
  // Now insert sequenced orders for each line
  affectedLines.forEach(line => {
    const sequencedLineOrders = sequencedResults[line] || [];
    
    sequencedLineOrders.forEach((order, index) => {
      // Clean up internal tracking properties
      const cleanOrder = { ...order };
      cleanOrder.prio = index + 1;
      cleanOrder.line = line;
      
      // Keep _isCombined and _combinedFrom for reference
      // But remove movement tracking
      delete cleanOrder._movement;
      delete cleanOrder._movementDelta;
      delete cleanOrder._originalPrio;
      delete cleanOrder._movedFromLine;
      delete cleanOrder._originalLine;
      delete cleanOrder._combinedFromLines;
      
      // Check if this order already exists in updatedOrders (by ID)
      const existingIndex = updatedOrders.findIndex(uo => uo.id === cleanOrder.id);
      if (existingIndex >= 0) {
        // Update existing
        updatedOrders[existingIndex] = cleanOrder;
      } else {
        // Add new (combined order)
        updatedOrders.push(cleanOrder);
      }
    });
  });
  
  // Recalculate changeovers for all affected lines
  const changeoverRules = getDefaultChangeoverRules();
  
  affectedLines.forEach(line => {
    const lineOrders = updatedOrders
      .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
      .sort((a, b) => a.prio - b.prio);
    
    calculatePreviewChangeovers(lineOrders, changeoverRules);
  });
  
  // Recalculate completion dates
  recalculateCompletionDates(updatedOrders);
  
  // Update state
  setAllOrders(updatedOrders);
  
  // Close preview
  setShowPlantAutoSequencePreview(false);
  
  // Show notification
  const totalOrders = Object.values(sequencedResults).reduce((sum, lo) => sum + lo.length, 0);
  const combinedCount = Object.values(sequencedResults).reduce((sum, lo) =>
    sum + lo.filter(o => o._isCombined).length, 0
  );
  
  showNotification(
    'success',
    `Plant-level auto-sequence applied: ${totalOrders} orders across ${affectedLines.length} lines.${combinedCount > 0 ? ` ${combinedCount} orders combined.` : ''}`
  );
}
```
