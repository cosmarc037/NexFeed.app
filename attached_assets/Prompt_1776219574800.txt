```jsx
function AllFeedmillsView({ 
  allOrders, 
  masterData, 
  n10dData, 
  changeoverRules,
  onOrderUpdate,
  onAutoSequenceAll
}) {
  return (
    <div className="all-feedmills-view">
      {/* Header with Auto-Sequence All button */}
      <div className="all-feedmills-header">
        <h2 className="all-feedmills-title">All Feedmills</h2>
        <button 
          className="btn-auto-sequence-all"
          onClick={onAutoSequenceAll}
        >
          ✨ Auto-Sequence (All)
        </button>
      </div>
      
      {/* All line tables */}
      <div className="all-feedmills-tables">
        {ALL_LINES_ORDERED.map(({ feedmill, line }) => {
          const lineOrders = allOrders
            .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
            .sort((a, b) => a.prio - b.prio);
          
          return (
            <div key={`${feedmill}-${line}`} className="all-feedmills-line-section">
              {/* Line header */}
              <div className="all-feedmills-line-header">
                <span className="all-feedmills-line-label">
                  {feedmill} — {line}
                </span>
                <span className="all-feedmills-line-count">
                  {lineOrders.length} order{lineOrders.length !== 1 ? 's' : ''}
                </span>
              </div>
              
              {/* Reuse the SAME order table component used in individual feedmill/line views */}
              <OrderTable
                orders={lineOrders}
                line={line}
                feedmill={feedmill}
                masterData={masterData}
                n10dData={n10dData}
                changeoverRules={changeoverRules}
                onOrderUpdate={onOrderUpdate}
                showAutoSequenceButton={false}
              />
            </div>
          );
        })}
      </div>
    </div>
  );
}
```

## 4. DISABLE AUTO-SEQUENCE IN [ALL] TABS PER FEEDMILL

```jsx
function FeedmillView({ feedmill, activeLine, onLineChange, allOrders, ... }) {
  const lines = FEEDMILL_LINES[feedmill] || [];
  const tabs = ['All', ...lines];
  
  return (
    <div className="feedmill-view">
      {/* Line tabs */}
      <div className="feedmill-line-tabs">
        {tabs.map(tab => (
          <button
            key={tab}
            className={`feedmill-line-tab ${activeLine === tab ? 'active' : ''}`}
            onClick={() => onLineChange(tab)}
          >
            {tab}
          </button>
        ))}
      </div>
      
      {/* Content */}
      {activeLine === 'All' ? (
        /* All lines in this feedmill — auto-sequence DISABLED */
        <div className="feedmill-all-lines">
          {lines.map(line => {
            const lineOrders = allOrders
              .filter(o => o.line === line && o.status !== 'Done' && o.status !== 'Cancel PO')
              .sort((a, b) => a.prio - b.prio);
            
            return (
              <div key={line} className="feedmill-line-section">
                <div className="feedmill-line-section-header">
                  <span>{line}</span>
                  <span className="feedmill-line-section-count">
                    {lineOrders.length} orders
                  </span>
                </div>
                <OrderTable
                  orders={lineOrders}
                  line={line}
                  feedmill={feedmill}
                  showAutoSequenceButton={false}
                />
              </div>
            );
          })}
        </div>
      ) : (
        /* Individual line — auto-sequence ENABLED */
        <div className="feedmill-single-line">
          <OrderTable
            orders={allOrders.filter(o => o.line === activeLine && o.status !== 'Done' && o.status !== 'Cancel PO').sort((a, b) => a.prio - b.prio)}
            line={activeLine}
            feedmill={feedmill}
            showAutoSequenceButton={true}
            onAutoSequence={() => handleLineAutoSequence(activeLine)}
          />
        </div>
      )}
    </div>
  );
}
```

## 5. ORDER TABLE — ACCEPT showAutoSequenceButton PROP

```jsx
function OrderTable({ orders, line, feedmill, showAutoSequenceButton = true, onAutoSequence, ... }) {
  return (
    <div className="order-table-container">
      {/* Table header with optional auto-sequence button */}
      <div className="order-table-header">
        {showAutoSequenceButton && (
          <button 
            className="btn-auto-sequence"
            onClick={onAutoSequence}
          >
            ✨ Auto-Sequence
          </button>
        )}
      </div>
      
      {/* The actual order table — SAME for all views */}
      <table className="order-table">
        <thead>
          <tr>
            <th></th>
            <th>Prio</th>
            <th>FPR</th>
            <th>Planned Order</th>
            <th>Material Code (FG)</th>
            <th>Item Description</th>
            <th>Form</th>
            <th>Volume (MT)</th>
            <th>Batch Size</th>
            <th>Batches</th>
            <th>Production Time</th>
            <th>Start Date</th>
            <th>Start Time</th>
            <th>Avail Date</th>
            <th>Estimated Completion Date</th>
          </tr>
        </thead>
        <tbody>
          {orders.map((order, index) => (
            <OrderRow key={order.id} order={order} index={index} />
          ))}
          {orders.length === 0 && (
            <tr>
              <td colSpan="15" className="order-table-empty">
                No orders on this line.
              </td>
            </tr>
          )}
        </tbody>
      </table>
    </div>
  );
}
```
