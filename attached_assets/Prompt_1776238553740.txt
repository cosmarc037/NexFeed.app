function canProduceOnLine(order, line, masterData) {
  if (!masterData || !Array.isArray(masterData)) return false;
  
  const materialCode = String(order.material_code_fg || '').trim();
  if (!materialCode) return false;
  
  // Normalize line name for comparison
  const normalizedLine = normalizeLine(line);
  
  const entries = masterData.filter(m => {
    const masterMaterial = String(m.material_code_fg || '').trim();
    const masterLine = normalizeLine(String(m.line || ''));
    return masterMaterial === materialCode && masterLine === normalizedLine;
  });
  
  const hasRunRate = entries.some(m => (parseFloat(m.run_rate) || 0) > 0);
  
  // Debug log for troubleshooting
  if (!hasRunRate && entries.length === 0) {
    // Check if product exists in master data at all
    const allEntriesForProduct = masterData.filter(m => 
      String(m.material_code_fg || '').trim() === materialCode
    );
    
    if (allEntriesForProduct.length > 0) {
      console.log(`canProduceOnLine: ${materialCode} NOT found for "${line}" (normalized: "${normalizedLine}")`);
      console.log(`  Available lines in master data:`, allEntriesForProduct.map(m => `"${m.line}"`).join(', '));
    }
  }
  
  return hasRunRate;
}

// Normalize line names to handle different formats
function normalizeLine(line) {
  if (!line) return '';
  
  let normalized = String(line).trim();
  
  // Handle various formats:
  // "Line 1", "line 1", "L1", "l1", "Line1", "LINE 1"
  
  // If it's already "Line X" format, standardize spacing and case
  const lineMatch = normalized.match(/^line\s*(\d+)$/i);
  if (lineMatch) {
    return `Line ${lineMatch[1]}`;
  }
  
  // If it's "LX" format, convert to "Line X"
  const shortMatch = normalized.match(/^l(\d+)$/i);
  if (shortMatch) {
    return `Line ${shortMatch[1]}`;
  }
  
  return normalized;
}

function generateFallbackExplanation(entry) {
  const allLineDetails = entry.lineScores
    .map(ls => `${getLineShortName(ls.line)}: ${ls.totalMTBefore.toFixed(1)} MT ÷ ${ls.runRate} MT/hr = ${ls.queueTimeBefore.toFixed(2)} hrs`)

    }
