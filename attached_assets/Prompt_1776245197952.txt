```javascript
function generateFallbackExplanation(entry) {
  if (entry.type === 'combined') {
    const allSameLine = entry.fromLines.every(l => l === entry.toLine);
    
    const allLineDetails = entry.lineScores
      .map(ls => `${getLineShortName(ls.line)}: ${ls.totalMTBefore.toFixed(1)} MT ÷ ${ls.runRate} MT/hr = ${ls.queueTimeBefore.toFixed(2)} hrs`)
      .join(', ');
    
    if (allSameLine) {
      return `${entry.ordersCount} orders of ${entry.product} already on ${getLineShortName(entry.toLine)} were combined into a single ${entry.totalVolume} MT run. Queue times evaluated: ${allLineDetails}. ${getLineShortName(entry.toLine)} was confirmed as optimal with the lowest queue time of ${entry.bestLineReason.queueTime.toFixed(2)} hrs. After combining, ${getLineShortName(entry.toLine)} increased from ${entry.bestLineReason.totalMTBefore.toFixed(1)} MT to ${entry.bestLineReason.totalMTAfter.toFixed(1)} MT (${entry.bestLineReason.queueTimeAfter.toFixed(2)} hrs). This saved ${entry.changeoversSaved} changeover${entry.changeoversSaved > 1 ? 's' : ''} and ${formatTimeShift(entry.timeSaved)}.`;
    } else {
      const otherLines = [...new Set(entry.fromLines.filter(l => l !== entry.toLine))];
      return `${entry.ordersCount} orders of ${entry.product} were combined into ${entry.totalVolume} MT on ${getLineShortName(entry.toLine)}, pulling from ${otherLines.map(l => getLineShortName(l)).join(', ')}. Queue times evaluated: ${allLineDetails}. ${getLineShortName(entry.toLine)} had the lowest queue time of ${entry.bestLineReason.queueTime.toFixed(2)} hrs. After combining, ${getLineShortName(entry.toLine)} increased from ${entry.bestLineReason.totalMTBefore.toFixed(1)} MT to ${entry.bestLineReason.totalMTAfter.toFixed(1)} MT (${entry.bestLineReason.queueTimeAfter.toFixed(2)} hrs). This saved ${entry.changeoversSaved} changeover${entry.changeoversSaved > 1 ? 's' : ''} and ${formatTimeShift(entry.timeSaved)}.`;
    }
  }
  
  if (entry.type === 'moved') {
    const fromScore = entry.lineScores.find(ls => ls.line === entry.fromLine);
    const toScore = entry.lineScores.find(ls => ls.line === entry.toLine);
    
    const allLineDetails = entry.lineScores
      .map(ls => `${getLineShortName(ls.line)}: ${ls.totalMTBefore.toFixed(1)} MT ÷ ${ls.runRate} MT/hr = ${ls.queueTimeBefore.toFixed(2)} hrs`)
      .join(', ');
    
    return `${entry.product} (${entry.volume} MT) was moved from ${getLineShortName(entry.fromLine)} to ${getLineShortName(entry.toLine)}. Queue times evaluated: ${allLineDetails}. ${getLineShortName(entry.toLine)} had the lowest queue time of ${toScore?.queueTimeBefore.toFixed(2)} hrs compared to ${getLineShortName(entry.fromLine)} at ${fromScore?.queueTimeBefore.toFixed(2)} hrs. After placement, ${getLineShortName(entry.toLine)} increased from ${toScore?.totalMTBefore.toFixed(1)} MT to ${toScore?.totalMTAfter.toFixed(1)} MT.`;
  }
  
  return 'Action details unavailable.';
}
```

## 3. UPDATED DISPLAY — BADGES AT BOTTOM, INLINE WITH SUB-ORDERS

```jsx
function PlantActionsTaken({ placementLog, aiExplanations, isLoadingAI, useFallback, onRefreshInsights }) {
  if (placementLog.length === 0) {
    return (
      <div className="plant-actions-section">
        <div className="plant-actions-header">
          <h3 className="plant-summary-title">📋 Actions Taken</h3>
        </div>
        <div className="plant-actions-empty">No actions taken — all orders remain on their original lines.</div>
      </div>
    );
  }
  
  return (
    <div className="plant-actions-section">
      <div className="plant-actions-header">
        <h3 className="plant-summary-title">📋 Actions Taken</h3>
        <button className="plant-actions-refresh" onClick={onRefreshInsights}>
          ↻ Refresh
        </button>
      </div>
      
      {isLoadingAI && (
        <div className="plant-actions-loading">Analyzing actions...</div>
      )}
      
      {!isLoadingAI && placementLog.map((entry, index) => {
        const explanation = (!useFallback && aiExplanations && aiExplanations[index])
          ? aiExplanations[index]
          : generateFallbackExplanation(entry);
        
        const isAI = !useFallback && aiExplanations && aiExplanations[index];
        
        // Calculate MT change for destination
        const toScore = entry.lineScores.find(ls => ls.line === entry.toLine);
        
        return (
          <div key={index} className="plant-action-card">
            {/* Header */}
            <div className="plant-action-header">
              {entry.type === 'combined' ? (
                <>
                  <span className="plant-action-icon">🔗</span>
                  <span className="plant-action-title">
                    Combined <strong>{entry.ordersCount}</strong> orders of <strong>{entry.product}</strong> ({entry.totalVolume} MT)
                    {entry.fromLines.every(l => l === entry.toLine) 
                      ? ` in place on ${entry.toLine}`
                      : ` from ${[...new Set(entry.fromLines)].map(l => getLineShortName(l)).join(', ')} → ${entry.toLine}`
                    }
                  </span>
                </>
              ) : (
                <>
                  <span className="plant-action-icon">↗</span>
                  <span className="plant-action-title">
                    Moved <strong>{entry.product}</strong> ({entry.volume} MT) {getLineShortName(entry.fromLine)} → {entry.toLine}
                  </span>
                </>
              )}
              {isAI && <span className="plant-action-ai-tag">AI</span>}
            </div>
            
            {/* Explanation — plain text, no bold */}
            <div className="plant-action-explanation">
              {explanation}
            </div>
            
            {/* Bottom badges — MT change + sub-orders all in one row */}
            <div className="plant-action-badges">
              {/* MT change badge for destination */}
              {toScore && (
                <span className="plant-action-mt-badge plant-action-mt-increase">
                  {getLineShortName(entry.toLine)} {toScore.totalMTBefore.toFixed(1)} MT → {toScore.totalMTAfter.toFixed(1)} MT
                  <span className="plant-action-mt-diff">
                    (+{(toScore.totalMTAfter - toScore.totalMTBefore).toFixed(1)} MT)
                  </span>
                </span>
              )}
              
              {/* Sub-order chips for combined */}
              {entry.type === 'combined' && entry.individualVolumes && (
                entry.individualVolumes.map((iv, subIdx) => (
                  <span key={subIdx} className="plant-action-sub-chip">
                    {getLineShortName(iv.fromLine)} · {iv.volume} MT
                  </span>
                ))
              )}
            </div>
          </div>
        );
      })}
    </div>
  );
}
```
