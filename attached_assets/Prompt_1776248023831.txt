## 1. FILTER ACTIONS PER LINE

```javascript
function getActionsForLine(placementLog, line) {
  return placementLog.filter(entry => {
    // Show if this line is the destination
    if (entry.toLine === line) return true;
    
    // Show if this line is the source (order moved away)
    if (entry.type === 'moved' && entry.fromLine === line) return true;
    
    // Show if any sub-order in a combined group came from this line
    if (entry.type === 'combined' && entry.fromLines && entry.fromLines.includes(line)) return true;
    
    return false;
  });
}
```

```jsx
function LineActions({ line, actions, placementLog, aiExplanations, isLoadingAI, useFallback }) {
  const [isExpanded, setIsExpanded] = useState(true);
  
  return (
    <div className="line-actions-section">
      <div className="line-actions-header" onClick={() => setIsExpanded(!isExpanded)}>
        <span className="line-actions-title">
          📋 Actions on {line}
          <span className="line-actions-count">{actions.length}</span>
        </span>
        <span className="line-actions-toggle">
          {isExpanded ? '▲ Collapse' : '▼ Expand'}
        </span>
      </div>
      
      {isExpanded && (
        <div className="line-actions-body">
          {isLoadingAI && (
            <div className="plant-actions-loading">Analyzing actions...</div>
          )}
          
          {!isLoadingAI && actions.map((entry, actionIdx) => {
            // Find the global index of this action in the full placementLog
            const globalIndex = placementLog.findIndex(p => p === entry);
            
            const explanation = (!useFallback && aiExplanations && aiExplanations[globalIndex])
              ? aiExplanations[globalIndex]
              : generateFallbackExplanation(entry);
            
            const isAI = !useFallback && aiExplanations && aiExplanations[globalIndex];
            
            // Determine the relationship to this line
            const isDestination = entry.toLine === line;
            const isSource = (entry.type === 'moved' && entry.fromLine === line) ||
                             (entry.type === 'combined' && entry.fromLines?.includes(line) && entry.toLine !== line);
            
            const toScore = entry.lineScores?.find(ls => ls.line === entry.toLine);
            
            return (
              <div key={actionIdx} className={`plant-action-card ${isSource && !isDestination ? 'plant-action-card-source' : ''}`}>
                {/* Relationship tag */}
                <div className="line-action-relationship">
                  {isDestination && isSource && (
                    <span className="line-action-tag line-action-tag-both">Combined in place</span>
                  )}
                  {isDestination && !isSource && (
                    <span className="line-action-tag line-action-tag-incoming">Incoming</span>
                  )}
                  {isSource && !isDestination && (
                    <span className="line-action-tag line-action-tag-outgoing">Outgoing</span>
                  )}
                </div>
                
                {/* Header */}
                <div className="plant-action-header">
                  {entry.type === 'combined' ? (
                    <>
                      <span className="plant-action-icon">🔗</span>
                      <span className="plant-action-title">
                        Combined <strong>{entry.ordersCount}</strong> orders of <strong>{entry.product}</strong> ({entry.totalVolume} MT)
                        {entry.fromLines.every(l => l === entry.toLine)
                          ? ` in place on ${entry.toLine}`
                          : ` from ${[...new Set(entry.fromLines)].map(l => getLineShortName(l)).join(', ')} → ${entry.toLine}`
                        }
                      </span>
                    </>
                  ) : (
                    <>
                      <span className="plant-action-icon">↗</span>
                      <span className="plant-action-title">
                        Moved <strong>{entry.product}</strong> ({entry.volume} MT) {getLineShortName(entry.fromLine)} → {entry.toLine}
                      </span>
                    </>
                  )}
                  {isAI && <span className="plant-action-ai-tag">AI</span>}
                </div>
                
                {/* Explanation */}
                <div className="plant-action-explanation">
                  {explanation}
                </div>
                
                {/* Bottom badges */}
                <div className="plant-action-badges">
                  {toScore && (
                    <span className="plant-action-mt-badge plant-action-mt-increase">
                      {getLineShortName(entry.toLine)} {toScore.totalMTBefore.toFixed(1)} MT → {toScore.totalMTAfter.toFixed(1)} MT
                      <span className="plant-action-mt-diff">
                        (+{(toScore.totalMTAfter - toScore.totalMTBefore).toFixed(1)} MT)
                      </span>
                    </span>
                  )}
                  
                  {entry.type === 'combined' && entry.individualVolumes && (
                    entry.individualVolumes.map((iv, subIdx) => (
                      <span key={subIdx} className="plant-action-sub-chip">
                        {getLineShortName(iv.fromLine)} · {iv.volume} MT
                      </span>
                    ))
                  )}
                </div>
              </div>
            );
          })}
        </div>
      )}
    </div>
  );
}
```
