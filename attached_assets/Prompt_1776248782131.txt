```jsx
function PlantLineTab({ 
  line, 
  originalOrders, 
  sequencedOrders, 
  placementLog,
  aiExplanations,
  isLoadingAI,
  useFallback
}) {
  const movedCount = sequencedOrders.filter(o => o._movement !== 'same').length;
  const newFromOtherLines = sequencedOrders.filter(o => o._movement === 'new_to_line').length;
  const combinedCount = sequencedOrders.filter(o => o._isCombined).length;
  
  // Get actions relevant to this line
  const lineActions = getActionsForLine(placementLog, line);
  
  return (
    <div className="plant-line-tab">
      {/* Line stats */}
      <div className="auto-sequence-stats" style={{ padding: '12px 24px' }}>
        <StatBadge label="Before" value={originalOrders.length} />
        <StatBadge label="After" value={sequencedOrders.length} />
        {newFromOtherLines > 0 && (
          <StatBadge label="From Other Lines" value={newFromOtherLines} color="blue" />
        )}
        <StatBadge label="Orders Moved" value={movedCount} color="orange" />
        {combinedCount > 0 && (
          <StatBadge label="Combined Groups" value={combinedCount} color="green" />
        )}
      </div>
      
      {/* Before and After comparison tables */}
      <div className="auto-sequence-comparison">
        <div className="auto-sequence-comparison-container">
          <div className="auto-sequence-comparison-scroll">
            {/* ... existing Before and After tables ... */}
          </div>
        </div>
      </div>
      
      {/* Actions for this line — below tables, no container, no collapse */}
      {lineActions.length > 0 && (
        <div className="line-actions-plain">
          <h3 className="plant-summary-title">📋 Actions on {line}</h3>
          
          {isLoadingAI && (
            <div className="plant-actions-loading">Analyzing actions...</div>
          )}
          
          {!isLoadingAI && lineActions.map((entry, actionIdx) => {
            const globalIndex = placementLog.findIndex(p => p === entry);
            
            const explanation = (!useFallback && aiExplanations && aiExplanations[globalIndex])
              ? aiExplanations[globalIndex]
              : generateFallbackExplanation(entry);
            
            const isAI = !useFallback && aiExplanations && aiExplanations[globalIndex];
            
            const toScore = entry.lineScores?.find(ls => ls.line === entry.toLine);
            
            return (
              <div key={actionIdx} className="plant-action-card">
                {/* Header */}
                <div className="plant-action-header">
                  {entry.type === 'combined' ? (
                    <>
                      <span className="plant-action-icon">🔗</span>
                      <span className="plant-action-title">
                        Combined <strong>{entry.ordersCount}</strong> orders of <strong>{entry.product}</strong> ({entry.totalVolume} MT)
                        {entry.fromLines.every(l => l === entry.toLine)
                          ? ` in place on ${entry.toLine}`
                          : ` from ${[...new Set(entry.fromLines)].map(l => getLineShortName(l)).join(', ')} → ${entry.toLine}`
                        }
                      </span>
                    </>
                  ) : (
                    <>
                      <span className="plant-action-icon">↗</span>
                      <span className="plant-action-title">
                        Moved <strong>{entry.product}</strong> ({entry.volume} MT) {getLineShortName(entry.fromLine)} → {entry.toLine}
                      </span>
                    </>
                  )}
                  {isAI && <span className="plant-action-ai-tag">AI</span>}
                </div>
                
                {/* Explanation */}
                <div className="plant-action-explanation">
                  {explanation}
                </div>
                
                {/* Bottom badges */}
                <div className="plant-action-badges">
                  {toScore && (
                    <span className="plant-action-mt-badge plant-action-mt-increase">
                      {getLineShortName(entry.toLine)} {toScore.totalMTBefore.toFixed(1)} MT → {toScore.totalMTAfter.toFixed(1)} MT
                      <span className="plant-action-mt-diff">
                        (+{(toScore.totalMTAfter - toScore.totalMTBefore).toFixed(1)} MT)
                      </span>
                    </span>
                  )}
                  
                  {entry.type === 'combined' && entry.individualVolumes && (
                    entry.individualVolumes.map((iv, subIdx) => (
                      <span key={subIdx} className="plant-action-sub-chip">
                        {getLineShortName(iv.fromLine)} · {iv.volume} MT
                      </span>
                    ))
                  )}
                </div>
              </div>
            );
          })}
        </div>
      )}
    </div>
  );
}
```
