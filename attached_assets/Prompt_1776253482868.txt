```javascript
function handlePlantApply(sequencedResults) {
  setIsApplying(true);
  
  // Use setTimeout to allow UI to update with loading state
  setTimeout(() => {
    try {
      // Step 1: Collect all order IDs that are consumed by combinations
      const allCombinedSubIds = new Set();
      Object.values(sequencedResults).forEach(lineOrders => {
        lineOrders.forEach(order => {
          if (order._isCombined && order._combinedFrom) {
            order._combinedFrom.forEach(sub => allCombinedSubIds.add(sub.id));
          }
        });
      });
      
      // Step 2: Collect all sequenced order IDs (top-level)
      const allSequencedIds = new Set();
      Object.values(sequencedResults).forEach(lineOrders => {
        lineOrders.forEach(order => {
          allSequencedIds.add(order.id);
        });
      });
      
      // Step 3: Start with orders NOT affected by the sequencing
      // Keep Done, Cancel PO, and orders on lines NOT in sequencedResults
      const affectedLines = new Set(Object.keys(sequencedResults));
      
      let updatedOrders = allOrders.filter(order => {
        // Always keep Done and Cancel PO
        if (order.status === 'Done' || order.status === 'Cancel PO') return true;
        
        // If order is on an affected line, remove it (will be replaced)
        if (affectedLines.has(normalizeLine(order.line))) return false;
        
        // If order was consumed by a combination (could be from non-affected line)
        if (allCombinedSubIds.has(order.id)) return false;
        
        // If order was moved to an affected line, remove from here (it's in sequencedResults)
        if (allSequencedIds.has(order.id)) return false;
        
        // Keep everything else
        return true;
      });
      
      // Step 4: Add all sequenced orders from each line
      Object.entries(sequencedResults).forEach(([line, lineOrders]) => {
        lineOrders.forEach((order, index) => {
          // Create a clean order for storage
          const cleanOrder = { ...order };
          cleanOrder.prio = index + 1;
          cleanOrder.line = line;
          
          // Remove internal tracking properties
          delete cleanOrder._movement;
          delete cleanOrder._movementDelta;
          delete cleanOrder._originalPrio;
          delete cleanOrder._movedFromLine;
          delete cleanOrder._originalLine;
          delete cleanOrder._combinedFromLines;
          
          // Keep _isCombined and _combinedFrom for reference
          
          updatedOrders.push(cleanOrder);
        });
      });
      
      // Step 5: Recalculate changeovers for all affected lines
      const changeoverRules = getDefaultChangeoverRules();
      
      Object.keys(sequencedResults).forEach(line => {
        const lineOrders = updatedOrders
          .filter(o => normalizeLine(o.line) === line && o.status !== 'Done' && o.status !== 'Cancel PO')
          .sort((a, b) => a.prio - b.prio);
        
        if (lineOrders.length > 0) {
          calculatePreviewChangeovers(lineOrders, changeoverRules);
        }
      });
      
      // Step 6: Recalculate completion dates
      recalculateCompletionDates(updatedOrders);
      
      // Step 7: Update state
      setAllOrders(updatedOrders);
      
      // Step 8: Close preview
      setShowPlantAutoSequencePreview(false);
      setIsApplying(false);
      
      // Step 9: Show notification
      const totalOrders = Object.values(sequencedResults).reduce((sum, lo) => sum + lo.length, 0);
      const combinedCount = Object.values(sequencedResults).reduce((sum, lo) =>
        sum + lo.filter(o => o._isCombined).length, 0
      );
      const affectedLineCount = Object.keys(sequencedResults).filter(line =>
        (sequencedResults[line] || []).length > 0
      ).length;
      
      showNotification(
        'success',
        `Plant-level auto-sequence applied: ${totalOrders} orders across ${affectedLineCount} lines.${combinedCount > 0 ? ` ${combinedCount} combined groups created.` : ''}`
      );
      
    } catch (error) {
      console.error('Error applying plant-level auto-sequence:', error);
      setIsApplying(false);
      showNotification('error', 'Failed to apply auto-sequence. Please try again.');
    }
  }, 100);
}
```
