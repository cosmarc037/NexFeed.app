## 3. SEQUENCE-ONLY ALGORITHM (NO COMBINE)

```javascript
function sequenceOnlyAcrossLines(allOrders, masterData, n10dData, allLines) {
  let workingOrders = allOrders
    .filter(o => o.status !== 'Done' && o.status !== 'Cancel PO')
    .map(o => ({
      ...JSON.parse(JSON.stringify(o)),
      _originalLine: normalizeLine(o.feedmill_line || o.line),
      _movedFromLine: null
    }));
  
  // Track MT per line
  const lineTotalMT = {};
  allLines.forEach(line => {
    const lineOrders = workingOrders.filter(o => normalizeLine(o.feedmill_line || o.line) === line);
    lineTotalMT[line] = lineOrders.reduce((sum, o) => sum + (parseFloat(o.volume) || 0), 0);
  });
  
  const lineOrdersMap = {};
  allLines.forEach(line => {
    lineOrdersMap[line] = workingOrders
      .filter(o => normalizeLine(o.feedmill_line || o.line) === line)
      .sort((a, b) => (a.priority_seq || 0) - (b.priority_seq || 0));
  });
  
  const placementLog = [];
  
  // Sort: dated first, then by avail date
  const datedOrders = workingOrders.filter(o => {
    const avail = o.target_avail_date;
    return avail && /^\d{4}-\d{2}-\d{2}/.test(avail) && !isNaN(Date.parse(avail));
  }).sort((a, b) => new Date(a.target_avail_date) - new Date(b.target_avail_date));
  
  const nonDatedOrders = workingOrders.filter(o => {
    const avail = o.target_avail_date;
    return !avail || !/^\d{4}-\d{2}-\d{2}/.test(avail) || isNaN(Date.parse(avail));
  });
  
  const sortedOrders = [...datedOrders, ...nonDatedOrders];
  
  // For each order, evaluate if it should be on a different line
  sortedOrders.forEach(order => {
    const eligibleLines = allLines.filter(line =>
      canProduceOnLine(order, line, masterData)
    );
    
    if (eligibleLines.length === 0) return;
    
    const orderVolume = parseFloat(order.volume) || 0;
    const origLine = normalizeLine(order.feedmill_line || order.line);
    
    // Remove from current line
    lineTotalMT[origLine] = (lineTotalMT[origLine] || 0) - orderVolume;
    lineOrdersMap[origLine] = (lineOrdersMap[origLine] || []).filter(o => o.id !== order.id);
    
    // Calculate queue times
    const lineScores = eligibleLines.map(line => {
      const currentTotalMT = lineTotalMT[line] || 0;
      const runRate = getLineRunRate(line);
      const queueTime = runRate > 0 ? currentTotalMT / runRate : Infinity;
      const lineNumber = parseInt(line.match(/\d+/)?.[0] || '99');
      
      return { line, runRate, totalMTBefore: currentTotalMT, queueTimeBefore: queueTime, lineNumber };
    });
    
    lineScores.sort((a, b) => {
      const diff = a.queueTimeBefore - b.queueTimeBefore;
      if (Math.abs(diff) > 0.001) return diff;
      return a.lineNumber - b.lineNumber;
    });
    
    const bestLine = lineScores[0].line;
    
    // Place on best line
    if (origLine !== bestLine) {
      order._movedFromLine = origLine;
      order.line = bestLine;
      order.feedmill_line = bestLine;
      
      const toScore = lineScores.find(ls => ls.line === bestLine);
      
      placementLog.push({
        type: 'moved',
        product: order.item_description,
        materialCode: order.material_code_fg,
        fromLine: origLine,
        toLine: bestLine,
        volume: orderVolume,
        fpr: order.fpr,
        lineScores: lineScores.map(ls => ({
          line: ls.line,
          runRate: ls.runRate,
          totalMTBefore: ls.totalMTBefore,
          queueTimeBefore: ls.queueTimeBefore,
          totalMTAfter: ls.line === bestLine ? (ls.totalMTBefore + orderVolume) : ls.totalMTBefore,
          queueTimeAfter: ls.line === bestLine ? ((ls.totalMTBefore + orderVolume) / ls.runRate) : ls.queueTimeBefore
        })),
        bestLineReason: {
          line: bestLine,
          runRate: toScore.runRate,
          queueTime: toScore.queueTimeBefore,
          totalMTBefore: toScore.totalMTBefore,
          totalMTAfter: toScore.totalMTBefore + orderVolume,
          queueTimeAfter: (toScore.totalMTBefore + orderVolume) / toScore.runRate
        },
        fromLineReason: {
          line: origLine,
          queueTime: lineScores.find(ls => ls.line === origLine)?.queueTimeBefore || 0,
          runRate: getLineRunRate(origLine)
        }
      });
    }
    
    lineTotalMT[bestLine] = (lineTotalMT[bestLine] || 0) + orderVolume;
    if (!lineOrdersMap[bestLine]) lineOrdersMap[bestLine] = [];
    lineOrdersMap[bestLine].push(order);
  });
  
  return { lineOrdersMap, placementLog };
}
```

## 4. COMBINE-ONLY ALGORITHM (RUN ON TOP OF SEQUENCED ORDERS)

```javascript
function plantLevelCombineOnly(allOrders, masterData, n10dData, allLines) {
  // This is the same as plantLevelCombineAndPlace but ONLY does the combine step
  // It does NOT move orders between lines — just combines matching orders on the same line
  // or across lines if queue time is better
  
  let workingOrders = allOrders.map(o => ({
    ...JSON.parse(JSON.stringify(o)),
    _originalLine: normalizeLine(o.line || o.feedmill_line),
    _movedFromLine: null
  }));
  
  const lineTotalMT = {};
  allLines.forEach(line => {
    const lineOrders = workingOrders.filter(o => normalizeLine(o.line || o.feedmill_line) === line);
    lineTotalMT[line] = lineOrders.reduce((sum, o) => sum + (parseFloat(o.volume) || 0), 0);
  });
  
  const lineOrdersMap = {};
  allLines.forEach(line => {
    lineOrdersMap[line] = workingOrders.filter(o => normalizeLine(o.line || o.feedmill_line) === line);
  });
  
  // Group by Material Code + Formula Version
  const matchGroups = {};
  workingOrders.forEach(order => {
    const key = `${String(order.material_code_fg || '').trim()}||${order.formula_version || '—'}`;
    if (!matchGroups[key]) matchGroups[key] = [];
    matchGroups[key].push(order);
  });
  
  const combinedOrders = [];
  const placementLog = [];
  const processedIds = new Set();
  
  Object.entries(matchGroups).forEach(([key, orders]) => {
    if (orders.length < 2) return;
    
    const eligibleLines = allLines.filter(line =>
      canProduceOnLine(orders[0], line, masterData)
    );
    
    if (eligibleLines.length === 0) return;
    
    // Greedy combine up to 180 MT
    const sorted = [...orders].sort((a, b) => (parseFloat(b.volume) || 0) - (parseFloat(a.volume) || 0));
    const combineGroups = [];
    const remaining = [...sorted];
    
    while (remaining.length > 0) {
      const group = [remaining.shift()];
      let vol = parseFloat(group[0].volume) || 0;
      
      for (let i = remaining.length - 1; i >= 0; i--) {
        const cv = parseFloat(remaining[i].volume) || 0;
        if (vol + cv <= MAX_COMBINE_MT) {
          vol += cv;
          group.push(remaining.splice(i, 1)[0]);
        }
      }
      
      if (group.length > 1) {
        combineGroups.push({ orders: group, totalVolume: vol });
      }
    }
    
    // Place each combine group
    combineGroups.forEach(combineGroup => {
      // Remove from lines
      combineGroup.orders.forEach(o => {
        const orig = normalizeLine(o.line || o.feedmill_line);
        lineTotalMT[orig] = (lineTotalMT[orig] || 0) - (parseFloat(o.volume) || 0);
        lineOrdersMap[orig] = (lineOrdersMap[orig] || []).filter(lo => lo.id !== o.id);
        processedIds.add(o.id);
      });
      
      // Evaluate lines
      const lineScores = eligibleLines.map(line => {
        const mt = lineTotalMT[line] || 0;
        const rate = getLineRunRate(line);
        return { line, runRate: rate, totalMTBefore: mt, queueTimeBefore: rate > 0 ? mt / rate : Infinity, lineNumber: parseInt(line.match(/\d+/)?.[0] || '99') };
      });
      lineScores.sort((a, b) => {
        const d = a.queueTimeBefore - b.queueTimeBefore;
        if (Math.abs(d) > 0.001) return d;
        return a.lineNumber - b.lineNumber;
      });
      
      const bestLine = lineScores[0].line;
      
      // Create combined order
      const baseOrder = { ...combineGroup.orders[0] };
      baseOrder.volume = combineGroup.totalVolume;
      baseOrder.line = bestLine;
      baseOrder.feedmill_line = bestLine;
      baseOrder._isCombined = true;
      baseOrder._combinedFrom = combineGroup.orders.map(o => ({
        id: o.id, line: o._originalLine, fpr: o.fpr, volume: o.volume,
        item_description: o.item_description, form: o.form,
        material_code_fg: o.material_code_fg, planned_order_fg: o.planned_order_fg,
        planned_order_sfg: o.planned_order_sfg, batch_size: o.batch_size,
        batches: o.batches, production_time: o.production_time,
        target_avail_date: o.target_avail_date, category: o.category
      }));
      baseOrder._combinedFromLines = [...new Set(combineGroup.orders.map(o => o._originalLine))];
      
      const batchSize = parseFloat(baseOrder.batch_size) || 1;
      baseOrder.batches = Math.ceil(combineGroup.totalVolume / batchSize);
      
      lineTotalMT[bestLine] = (lineTotalMT[bestLine] || 0) + combineGroup.totalVolume;
      lineOrdersMap[bestLine].push(baseOrder);
      combinedOrders.push(baseOrder);
      
      const afterMT = lineTotalMT[bestLine];
      const changeoversSaved = combineGroup.orders.length - 1;
      const baseCO = getBaseChangeover(baseOrder.form, getDefaultChangeoverRules()?.baseChangeover);
      
      placementLog.push({
        type: 'combined',
        product: baseOrder.item_description,
        materialCode: baseOrder.material_code_fg,
        ordersCount: combineGroup.orders.length,
        fromLines: [...new Set(combineGroup.orders.map(o => o._originalLine))],
        toLine: bestLine,
        totalVolume: combineGroup.totalVolume,
        individualVolumes: combineGroup.orders.map(o => ({ name: o.item_description, volume: parseFloat(o.volume) || 0, fromLine: o._originalLine, fpr: o.fpr })),
        lineScores: lineScores.map(ls => ({ ...ls, totalMTAfter: lineTotalMT[ls.line] || 0, queueTimeAfter: (lineTotalMT[ls.line] || 0) / ls.runRate })),
        bestLineReason: { line: bestLine, runRate: getLineRunRate(bestLine), queueTime: lineScores[0].queueTimeBefore, totalMTBefore: lineScores[0].totalMTBefore, totalMTAfter: afterMT, queueTimeAfter: afterMT / getLineRunRate(bestLine) },
        changeoversSaved, baseChangeover: baseCO, timeSaved: changeoversSaved * baseCO
      });
    });
  });
  
  return { lineOrdersMap, combinedOrders, placementLog };
}
```

```jsx
      await handlePlantApply(sequencedResults);
      setShowUploadAutoSequence(false);
    }}
    onUploadOnly={() => {
      setShowUploadAutoSequence(false);
      toast.success('Orders uploaded without sequencing.');
    }}
    onCancel={() => {
      setShowUploadAutoSequence(false);
    }}
  />
)}
```
