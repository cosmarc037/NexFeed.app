## 2. UPDATE ProductionInsights — SUPPORT "ALL" SCOPE

The existing ProductionInsights component likely filters by feedmill. Add support for scope="all":

```jsx
function ProductionInsights({ orders, n10dData, masterData, scope, scopeLabel }) {
  const [summary, setSummary] = useState('');
  const [isLoading, setIsLoading] = useState(true);
  const [isCollapsed, setIsCollapsed] = useState(false);
  
  useEffect(() => {
    generateInsights();
  }, [orders]);
  
  async function generateInsights() {
    setIsLoading(true);
    
    const totalOrders = orders.length;
    const totalMT = orders.reduce((sum, o) => sum + (parseFloat(o.volume) || 0), 0);
    
    // Count per line
    const perLine = {};
    orders.forEach(o => {
      const line = normalizeLine(o.feedmill_line || o.line || 'Unassigned');
      if (!perLine[line]) perLine[line] = { count: 0, mt: 0 };
      perLine[line].count++;
      perLine[line].mt += parseFloat(o.volume) || 0;
    });
    
    // Urgent / critical orders
    const urgentOrders = orders.filter(o => {
      const n10d = n10dData.find(n => String(n.material_code) === String(o.material_code_fg));
      return n10d && (n10d.status === 'Critical' || n10d.status === 'Urgent');
    });
    
    // In production
    const inProduction = orders.filter(o => o.status === 'In Production' || o.status === 'On-going');
    
    // Build prompt based on scope
    let scopeContext = '';
    if (scope === 'all') {
      const lineBreakdown = Object.entries(perLine)
        .map(([line, data]) => `${line}: ${data.count} orders, ${data.mt.toFixed(1)} MT`)
        .join('\n');
      
      scopeContext = `
This is a plant-wide overview across ALL feedmills and lines.

Line Breakdown:
${lineBreakdown}

Total: ${totalOrders} orders, ${totalMT.toFixed(1)} MT
In Production: ${inProduction.length}
Urgent/Critical: ${urgentOrders.length}
${urgentOrders.map(o => `- ${o.item_description} (${o.volume} MT) on ${o.feedmill_line}`).join('\n')}
      `;
    } else {
      scopeContext = `
This is for ${scopeLabel}.
Total: ${totalOrders} orders, ${totalMT.toFixed(1)} MT
In Production: ${inProduction.length}
Urgent/Critical: ${urgentOrders.length}
      `;
    }
    
    try {
      const prompt = `
You are a production scheduling advisor. Provide a brief smart summary and 3 recommended actions.

${scopeContext}

Write:
1. A 2-3 sentence summary of the current production status.
2. 3 numbered recommended actions.

Keep it concise. No markdown. Use plain text.
      `;
      
      const response = await callAI(prompt);
      setSummary(response);
    } catch {
      // Fallback
      setSummary(
        `${scopeLabel} has a total of ${totalOrders} orders totaling ${totalMT.toFixed(1)} MT across ${Object.keys(perLine).length} lines. ${urgentOrders.length > 0 ? `There are ${urgentOrders.length} urgent orders requiring immediate attention.` : 'No urgent orders at this time.'} ${inProduction.length > 0 ? `${inProduction.length} order${inProduction.length > 1 ? 's are' : ' is'} currently in production.` : 'No production is currently underway.'}`
      );
    }
    
    setIsLoading(false);
  }
  
  // Check for N10D stock-level sequencing
  const stockLevelOrders = orders.filter(o => {
    const n10d = n10dData.find(n => String(n.material_code) === String(o.material_code_fg));
    return n10d && n10d.inferred_target_date;
  });
  
  return (
    <div className="production-insights-card">
      <div className="production-insights-header" onClick={() => setIsCollapsed(!isCollapsed)}>
        <span className="production-insights-title">🍳 Production Insights</span>
        <span className="production-insights-toggle">{isCollapsed ? '▼' : '▲'}</span>
      </div>
      
      {!isCollapsed && (
        <div className="production-insights-body">
          {/* Smart Summary */}
          <div className="production-insights-section">
            <div className="production-insights-section-header">
              <span>✨ Smart Summary</span>
              <button className="production-insights-refresh" onClick={generateInsights}>
                ↻ Refresh
              </button>
            </div>
            {isLoading ? (
              <div className="production-insights-loading">Generating insights...</div>
            ) : (
              <div className="production-insights-text">{summary}</div>
            )}
          </div>
          
          {/* Stock-Level Sequencing */}
          {stockLevelOrders.length > 0 && (
            <div className="production-insights-section">
              <div className="production-insights-stock-label">
                📊 Stock-Level Sequencing Available
              </div>
              <div className="production-insights-stock-text">
                Future Dispatches data shows {stockLevelOrders.length} order{stockLevelOrders.length > 1 ? 's' : ''} with inferred target dates. Run Auto-Sequence to optimize the production schedule based on stock demand forecasts.
              </div>
            </div>
          )}
        </div>
      )}
    </div>
  );
}
```

## 3. UPDATE AlertsAndReminders — SUPPORT "ALL" SCOPE

```jsx
function AlertsAndReminders({ orders, n10dData, masterData, scope, scopeLabel }) {
  const [isCollapsed, setIsCollapsed] = useState(false);
  const [aiSummary, setAiSummary] = useState('');
  const [isLoading, setIsLoading] = useState(true);
  
  // Build alerts from ALL orders
  const alerts = useMemo(() => {
    const result = [];
    
    orders.forEach(order => {
      const n10d = n10dData.find(n => String(n.material_code) === String(order.material_code_fg));
      
      if (!n10d) return;
      
      // Critical — stock already run out
      if (n10d.status === 'Critical') {
        result.push({
          type: 'critical',
          icon: '🔴',
          label: 'Stock Running Low',
          order: order,
          n10d: n10d,
          message: `FPR: ${order.fpr} (${order.item_description}) (Prio ${order.priority_seq}) has inferred target ${n10d.inferred_target_date ? new Date(n10d.inferred_target_date).toLocaleDateString() : '—'} — target has passed! This order has no avail date set. Consider prioritizing it.`,
          line: order.feedmill_line
        });
      }
      
      // Urgent — stock about to run out
      if (n10d.status === 'Urgent') {
        const daysAway = n10d.inferred_target_date 
          ? Math.ceil((new Date(n10d.inferred_target_date) - new Date()) / (1000 * 60 * 60 * 24))
          : null;
        
        result.push({
          type: 'urgent',
          icon: '🟡',
          label: 'Stock Running Low',
          order: order,
          n10d: n10d,
          message: `FPR: ${order.fpr} (${order.item_description}) (Prio ${order.priority_seq}) has inferred target ${n10d.inferred_target_date ? new Date(n10d.inferred_target_date).toLocaleDateString() : '—'} — ${daysAway} day(s) away. This order has no avail date set. Consider prioritizing it.`,
          line: order.feedmill_line
        });
      }
    });
    
    // Sort: critical first, then urgent
    result.sort((a, b) => {
      if (a.type === 'critical' && b.type !== 'critical') return -1;
      if (a.type !== 'critical' && b.type === 'critical') return 1;
      return 0;
    });
    
    return result;
  }, [orders, n10dData]);
  
  useEffect(() => {
    generateAlertSummary();
  }, [alerts]);
  
  async function generateAlertSummary() {
    setIsLoading(true);
    
    if (alerts.length === 0) {
      setAiSummary(`No critical or urgent stock issues detected across ${scopeLabel}.`);
      setIsLoading(false);
      return;
    }
    
    const criticalCount = alerts.filter(a => a.type === 'critical').length;
    const urgentCount = alerts.filter(a => a.type === 'urgent').length;
    
    try {
      const prompt = `
You are a production scheduling advisor. Summarize the current alerts for ${scopeLabel}.

Alerts:
- ${criticalCount} critical (stock already depleted)
- ${urgentCount} urgent (stock about to run out)

Critical items:
${alerts.filter(a => a.type === 'critical').map(a => `- ${a.order.item_description} on ${a.line}`).join('\n')}

Urgent items:
${alerts.filter(a => a.type === 'urgent').map(a => `- ${a.order.item_description} on ${a.line}`).join('\n')}

Write a 2-3 sentence summary and 3 recommended actions. No markdown. Plain text.
      `;
      
      const response = await callAI(prompt);
      setAiSummary(response);
    } catch {
      setAiSummary(
        `There are ${criticalCount} critical and ${urgentCount} urgent stock alerts across ${scopeLabel}. ${criticalCount > 0 ? `Critical items like ${alerts[0].order.item_description} require immediate production attention.` : ''}`
      );
    }
    
    setIsLoading(false);
  }
  
  return (
    <div className="alerts-reminders-card">
      <div className="alerts-reminders-header" onClick={() => setIsCollapsed(!isCollapsed)}>
        <span className="alerts-reminders-title">🔔 Alerts & Reminders</span>
        <span className="alerts-reminders-toggle">{isCollapsed ? '▼' : '▲'}</span>
      </div>
      
      {!isCollapsed && (
        <div className="alerts-reminders-body">
          {/* Smart Summary */}
          <div className="alerts-reminders-section">
            <div className="alerts-reminders-section-header">
              <span>✨ Smart Summary</span>
              <button className="alerts-reminders-refresh" onClick={generateAlertSummary}>
                ↻ Refresh
              </button>
            </div>
            {isLoading ? (
              <div className="alerts-reminders-loading">Analyzing alerts...</div>
            ) : (
              <div className="alerts-reminders-text">{aiSummary}</div>
            )}
          </div>
          
          {/* Alert items */}
          {alerts.length === 0 ? (
            <div className="alerts-reminders-empty">
              No critical or urgent alerts at this time.
            </div>
          ) : (
            <div className="alerts-reminders-list">
              {alerts.map((alert, index) => (
                <div key={index} className={`alert-item alert-item-${alert.type}`}>
                  <div className="alert-item-header">
                    <span className="alert-item-icons">
                      {alert.type === 'critical' ? '🔴 🍳' : '🟡 🍳'}
                    </span>
                    <span className="alert-item-label">{alert.label}</span>
                  </div>
                  <div className="alert-item-message">
                    {alert.message}
                  </div>
                  {/* Show line info for All Feedmills scope */}
                  {scope === 'all' && alert.line && (
                    <span className="alert-item-line-badge">
                      {alert.line}
                    </span>
                  )}
                </div>
              ))}
            </div>
          )}
        </div>
      )}
    </div>
  );
}
```
