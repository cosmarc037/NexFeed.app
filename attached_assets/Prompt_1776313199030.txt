```javascript
function handleLineAutoSequence(targetLine) {
  const allActiveOrders = enrichedOrders.filter(
    o => o.status !== 'Done' && o.status !== 'Cancel PO'
  );
  
  const normalizedTarget = normalizeLine(targetLine);
  
  // Save original snapshot for ALL lines (since orders may be pulled from any line)
  const originalSnapshot = {};
  ALL_LINES_ORDERED.forEach(({ line }) => {
    const normalized = normalizeLine(line);
    originalSnapshot[normalized] = allActiveOrders
      .filter(o => normalizeLine(o.feedmill_line) === normalized)
      .sort((a, b) => (a.priority_seq || 0) - (b.priority_seq || 0))
      .map(o => ({ ...JSON.parse(JSON.stringify(o)) }));
  });
  
  // Step 1: Find orders that CAN be produced on the target line
  const targetLineOrders = originalSnapshot[normalizedTarget] || [];
  const otherLineOrders = [];
  
  ALL_LINES_ORDERED.forEach(({ line }) => {
    const normalized = normalizeLine(line);
    if (normalized === normalizedTarget) return;
    
    const lineOrders = originalSnapshot[normalized] || [];
    lineOrders.forEach(order => {
      if (canProduceOnLine(order, normalizedTarget, masterData)) {
        otherLineOrders.push({
          ...order,
          _sourceLineNormalized: normalized,
          _sourceLine: line
        });
      }
    });
  });
  
  // Step 2: Calculate current queue times
  const lineTotalMT = {};
  ALL_LINES_ORDERED.forEach(({ line }) => {
    const normalized = normalizeLine(line);
    const orders = originalSnapshot[normalized] || [];
    lineTotalMT[normalized] = orders.reduce((sum, o) => sum + (parseFloat(o.volume) || 0), 0);
  });
  
  const targetQueueTime = calculateLineQueueTime(targetLineOrders, normalizedTarget);
  
  // Step 3: Evaluate which orders are worth moving
  // An order is worth moving if moving it REDUCES overall production time
  // i.e., the target line queue time (after adding) is still lower than the source line (after removing)
  const candidateOrders = [];
  
  otherLineOrders.forEach(order => {
    const orderVolume = parseFloat(order.volume) || 0;
    const sourceLine = order._sourceLineNormalized;
    
    // Source line queue after removing this order
    const sourceMTBefore = lineTotalMT[sourceLine] || 0;
    const sourceMTAfter = sourceMTBefore - orderVolume;
    const sourceQueueBefore = sourceMTBefore / getLineRunRate(sourceLine);
    const sourceQueueAfter = sourceMTAfter / getLineRunRate(sourceLine);
    
    // Target line queue after adding this order
    const targetMTBefore = lineTotalMT[normalizedTarget] || 0;
    const targetMTAfter = targetMTBefore + orderVolume;
    const targetQueueBefore = targetMTBefore / getLineRunRate(normalizedTarget);
    const targetQueueAfter = targetMTAfter / getLineRunRate(normalizedTarget);
    
    // Worth moving if target queue AFTER is still lower than source queue BEFORE
    // This means the order finishes EARLIER on the target line
    const isWorthMoving = targetQueueAfter < sourceQueueBefore;
    
    candidateOrders.push({
      ...order,
      _isWorthMoving: isWorthMoving,
      _sourceMTBefore: sourceMTBefore,
      _sourceMTAfter: sourceMTAfter,
      _sourceQueueBefore: sourceQueueBefore,
      _sourceQueueAfter: sourceQueueAfter,
      _targetMTBefore: targetMTBefore,
      _targetMTAfter: targetMTAfter,
      _targetQueueBefore: targetQueueBefore,
      _targetQueueAfter: targetQueueAfter,
      _queueImprovement: sourceQueueBefore - targetQueueAfter
    });
  });
  
  // Sort candidates: best improvement first
  const worthMoving = candidateOrders
    .filter(o => o._isWorthMoving)
    .sort((a, b) => b._queueImprovement - a._queueImprovement);
  
  // Step 4: Greedily select orders to move (recalculating queue time after each)
  const selectedToMove = [];
  let runningTargetMT = lineTotalMT[normalizedTarget] || 0;
  const runningSourceMT = { ...lineTotalMT };
  
  worthMoving.forEach(candidate => {
    const orderVolume = parseFloat(candidate.volume) || 0;
    const sourceLine = candidate._sourceLineNormalized;
    
    // Recalculate with running totals
    const sourceQueueAfter = ((runningSourceMT[sourceLine] || 0) - orderVolume) / getLineRunRate(sourceLine);
    const targetQueueAfter = (runningTargetMT + orderVolume) / getLineRunRate(normalizedTarget);
    
    // Still worth moving?
    const sourceQueueBefore = (runningSourceMT[sourceLine] || 0) / getLineRunRate(sourceLine);
    
    if (targetQueueAfter < sourceQueueBefore) {
      selectedToMove.push({
        ...candidate,
        _finalSourceQueueBefore: sourceQueueBefore,
        _finalSourceQueueAfter: sourceQueueAfter,
        _finalTargetQueueBefore: runningTargetMT / getLineRunRate(normalizedTarget),
        _finalTargetQueueAfter: targetQueueAfter
      });
      
      runningTargetMT += orderVolume;
      runningSourceMT[sourceLine] = (runningSourceMT[sourceLine] || 0) - orderVolume;
    }
  });
  
  // Step 5: Also find combinable orders on the target line + moved orders
  const allTargetOrders = [
    ...targetLineOrders,
    ...selectedToMove
  ];
  
  // Group by Material Code + Formula Version for combine candidates
  const combineGroups = {};
  allTargetOrders.forEach(order => {
    const key = `${String(order.material_code_fg || '').trim()}||${order.formula_version || '—'}`;
    if (!combineGroups[key]) combineGroups[key] = [];
    combineGroups[key].push(order);
  });
  
  const combineCandidates = Object.entries(combineGroups)
    .filter(([_, orders]) => orders.length >= 2)
    .map(([key, orders]) => ({
      key,
      orders,
      totalVolume: orders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0),
      canCombine: orders.reduce((s, o) => s + (parseFloat(o.volume) || 0), 0) <= MAX_COMBINE_MT
    }));
  
  // Store results for preview
  setLineAutoSequenceData({
    targetLine: normalizedTarget,
    targetLineLabel: targetLine,
    originalSnapshot,
    targetLineOrders,
    candidateOrders: worthMoving,
    selectedToMove,
    combineCandidates,
    lineTotalMT: { ...lineTotalMT }
  });
  
  setShowLineAutoSequencePreview(true);
}
```

```jsx
              disabled={isApplying || !sequencedResult}
            >
              {isApplying ? (

                  <span className="btn-apply-spinner"></span>
                  Applying...
                </>
              ) : (
                'Apply to Schedule'
              )}
            </button>
          </div>
        </div>
      </div>
    </div>
  );
}
```

## 3. APPLY HANDLER FOR PER-LINE

```javascript
async function handleLineApply({ targetLine, sequencedOrders, affectedSourceLines }) {
  try {
    const mutationPayloads = [];
    let prioCounter = 1;
    
    for (const order of sequencedOrders) {
      if (order._isCombined && order._combinedFrom && order._combinedFrom.length > 1) {
        const leadId = order.id;
        const childIds = order._combinedFrom.map(sub => sub.id);
        
        mutationPayloads.push({
          id: leadId,
          data: {
            feedmill_line: targetLine,
            priority_seq: prioCounter,
            volume_override: parseFloat(order.volume) || 0,
            original_order_ids: childIds,
            parent_id: null,
          }
        });
        
        const children = order._combinedFrom.filter(sub => sub.id !== leadId);
        for (const child of children) {
          mutationPayloads.push({
            id: child.id,
            data: {
              parent_id: leadId,
              feedmill_line: targetLine,
              priority_seq: null,
            }
          });
        }
        
        prioCounter++;
      } else {
        mutationPayloads.push({
          id: order.id,
          data: {
            feedmill_line: targetLine,
            priority_seq: prioCounter,
            parent_id: null,
            volume_override: null,
            original_order_ids: null,
          }
        });
        prioCounter++;
      }
    }
    
    // Execute in batches
    const BATCH_SIZE = 10;
    for (let i = 0; i < mutationPayloads.length; i += BATCH_SIZE) {
      const batch = mutationPayloads.slice(i, i + BATCH_SIZE);
      await Promise.all(
        batch.map(p => updateOrderMutation.mutateAsync({ id: p.id, data: p.data }))
      );
    }
    
    // Cascade schedule for target line
    await queryClient.invalidateQueries({ queryKey: ["orders"] });
    await new Promise(resolve => setTimeout(resolve, 500));
    
    const topLevel = sequencedOrders.map((o, i) => ({
      ...o,
      id: o.id,
      priority_seq: i + 1,
      feedmill_line: targetLine
    }));
    await cascadeSchedule(topLevel, 0);
    
    await queryClient.invalidateQueries({ queryKey: ["orders"] });
    
    setShowLineAutoSequencePreview(false);
    
    toast.success(
      `Auto-sequence applied to ${targetLine}: ${sequencedOrders.length} orders.`
    );
    
  } catch (error) {
    console.error('Error applying line auto-sequence:', error);
    toast.error('Failed to apply. Please try again.');
  }
}
```
