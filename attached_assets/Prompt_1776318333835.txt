async function applySequencedResultsToDB(sequencedResults, options = {}) {
  const { label = 'Auto-sequence' } = options;
  
  // Separate pending (new) orders from existing orders
  const pendingPayloads = [];
  const existingPayloads = [];
  
  // Track combined groups separately — children need lead's REAL DB ID
  const combinedGroups = []; // { leadOrder, childOrders, line, prio }
  
  for (const [line, lineOrders] of Object.entries(sequencedResults)) {
    if (!lineOrders || lineOrders.length === 0) continue;
    
    let prioCounter = 1;
    
    for (const order of lineOrders) {
      
      if (order._isCombined && order._combinedFrom && order._combinedFrom.length > 1) {
        // Store combined group for later processing
        combinedGroups.push({
          leadOrder: order,
          allSubOrders: order._combinedFrom,
          line,
          prio: prioCounter
        });
        prioCounter++;
        
      } else {
        // Regular order
        const updateData = {
          feedmill_line: line,
          priority_seq: prioCounter,
        };
        
        // Clear any previous combine state
        updateData.parent_id = null;
        updateData.volume_override = null;
        updateData.original_order_ids = null;
        
        const isPending = String(order.id).startsWith('__pending_');
        
        if (isPending) {
          pendingPayloads.push({ order, data: updateData });
        } else {
          existingPayloads.push({ id: order.id, data: updateData });
        }
        
        prioCounter++;
      }
    }
  }
  
  console.log(`=== ${label} ===`);
  console.log(`  Regular: ${existingPayloads.length} existing, ${pendingPayloads.length} pending`);
  console.log(`  Combined groups: ${combinedGroups.length}`);
  
  // Step 1: Update regular existing orders
  if (existingPayloads.length > 0) {
    const BATCH_SIZE = 10;
    for (let i = 0; i < existingPayloads.length; i += BATCH_SIZE) {
      const batch = existingPayloads.slice(i, i + BATCH_SIZE);
      await Promise.all(
        batch.map(p => {
          console.log(`  [Update] Order ${p.id}:`, JSON.stringify(p.data));
          return updateOrderMutation.mutateAsync({ id: p.id, data: p.data });
        })
      );
    }
  }
  
  // Step 2: Create pending orders (if any from upload flow)
  // This returns a map of __pending_id → real DB id
  const pendingIdMap = {};
  if (pendingPayloads.length > 0) {
    for (const pp of pendingPayloads) {
      const created = await createOrderMutation.mutateAsync({
        data: {
          ...pp.order,
          ...pp.data,
          // Remove internal fields
          _isCombined: undefined,
          _combinedFrom: undefined,
          _movement: undefined,
          _movementDelta: undefined,
          _originalPrio: undefined,
          _movedFromLine: undefined,
          _originalLine: undefined,
          _combinedFromLines: undefined,
        }
      });
      pendingIdMap[pp.order.id] = created.id;
      console.log(`  [Create] ${pp.order.id} → ${created.id}`);
    }
  }
  
  // Step 3: Process combined groups — this is where the fix is critical
  for (const group of combinedGroups) {
    const { leadOrder, allSubOrders, line, prio } = group;
    
    // Resolve lead ID (might be pending)
    const leadId = String(leadOrder.id).startsWith('__pending_')
      ? pendingIdMap[leadOrder.id]
      : leadOrder.id;
    
    if (!leadId) {
      console.error(`  [Error] Could not resolve lead ID for: ${leadOrder.id}`);
      continue;
    }
    
    // Resolve ALL sub-order IDs (some might be pending)
    const resolvedSubIds = allSubOrders.map(sub => {
      const subId = String(sub.id).startsWith('__pending_')
        ? pendingIdMap[sub.id]
        : sub.id;
      return subId;
    }).filter(Boolean);
    
    // Get children (all sub-orders except the lead)
    const childSubOrders = allSubOrders.filter(sub => {
      const resolvedId = String(sub.id).startsWith('__pending_')
        ? pendingIdMap[sub.id]
        : sub.id;
      // Compare as strings to avoid type mismatch
      return String(resolvedId) !== String(leadId);
    });
    
    const combinedVolume = parseFloat(leadOrder.volume) || 0;
    
    console.log(`  [Combined] Lead: ${leadId}, Children: ${childSubOrders.length}, Volume: ${combinedVolume}`);
    console.log(`    All sub IDs: ${resolvedSubIds.join(', ')}`);
    
    // Update LEAD order
    await updateOrderMutation.mutateAsync({
      id: leadId,
      data: {
        feedmill_line: line,
        priority_seq: prio,
        volume_override: combinedVolume,
        original_order_ids: resolvedSubIds,
        parent_id: null,
      }
    });
    console.log(`    [Lead updated] ${leadId}: volume_override=${combinedVolume}, original_order_ids=[${resolvedSubIds.join(',')}]`);
    
    // Update EACH child order
    for (const child of childSubOrders) {
      const childId = String(child.id).startsWith('__pending_')
        ? pendingIdMap[child.id]
        : child.id;
      
      if (!childId) {
        console.error(`    [Error] Could not resolve child ID: ${child.id}`);
        continue;
      }
      
      await updateOrderMutation.mutateAsync({
        id: childId,
        data: {
          parent_id: leadId,
          feedmill_line: line,
          priority_seq: null,
        }
      });
      console.log(`    [Child updated] ${childId}: parent_id=${leadId}`);
    }
  }
  
  // Step 4: Refetch
  await queryClient.invalidateQueries({ queryKey: ["orders"] });
  await new Promise(resolve => setTimeout(resolve, 500));
  
  // Step 5: Cascade per line
  for (const [line, lineOrders] of Object.entries(sequencedResults)) {
    if (!lineOrders || lineOrders.length === 0) continue;
    
    // Only top-level orders for cascade (not children)
    const topLevel = lineOrders.map((o, i) => {
      const resolvedId = String(o.id).startsWith('__pending_')
        ? pendingIdMap[o.id] || o.id
        : o.id;
      return {
        ...o,
        id: resolvedId,
        priority_seq: i + 1,
        feedmill_line: line,
      };
    });
    
    console.log(`  [Cascade] ${line}: ${topLevel.length} orders`);
    await cascadeSchedule(topLevel, 0);
  }
  
  // Step 6: Final refetch
  await queryClient.invalidateQueries({ queryKey: ["orders"] });
  
  // Return stats
  const totalOrders = Object.values(sequencedResults).reduce((s, lo) => s + lo.length, 0);
  const combinedCount = combinedGroups.length;
  const linesUsed = Object.keys(sequencedResults).filter(l =>
    (sequencedResults[l] || []).length > 0
  ).length;
  
  return { totalOrders, combinedCount, linesUsed };
}
