async function handleOrderUpload(file) {
  try {
    // Step 1: Parse the file
    const parsedOrders = await parseOrderFile(file);
    
    if (parsedOrders.length === 0) {
      toast.error('No orders found in the uploaded file.');
      return;
    }
    
    // Step 2: Check if Future Dispatches is uploaded
    const n10dData = queryClient.getQueryData(["n10d"]) || [];
    const hasN10D = n10dData.length > 0;
    
    // Step 3: Check if all lines were blank BEFORE upload
    const existingOrders = queryClient.getQueryData(["orders"]) || [];
    const existingActive = existingOrders.filter(
      o => o.status !== 'Done' && o.status !== 'Cancel PO'
    );
    const allLinesWereBlank = existingActive.length === 0;
    
    // Step 4: Show loading overlay
    setUploadState('saving');
    
    // Step 5: Save ALL orders to DB first — they get real IDs
    const BATCH_SIZE = 10;
    for (let i = 0; i < parsedOrders.length; i += BATCH_SIZE) {
      const batch = parsedOrders.slice(i, i + BATCH_SIZE);
      await Promise.all(
        batch.map(order => createOrderMutation.mutateAsync({ data: order }))
      );
    }
    
    console.log(`Saved ${parsedOrders.length} orders to DB`);
    
    // Step 6: Refetch to get real DB orders with IDs
    await queryClient.invalidateQueries({ queryKey: ["orders"] });
    await new Promise(resolve => setTimeout(resolve, 500));
    
    // Step 7: Now trigger auto-sequence on real DB orders
    setUploadState('sequencing');
    
    setUploadContext({
      allLinesWereBlank,
      hasExistingOrders: !allLinesWereBlank,
      hasN10D,
      uploadedOrderCount: parsedOrders.length
    });
    
    setUploadState(null);
    setShowUploadAutoSequence(true);
    
  } catch (error) {
    console.error('Upload error:', error);
    setUploadState(null);
    toast.error(`Upload failed: ${error.message}`);
  }
}

          movement = 'new_to_line';

          ...order,
          prio: newPrio,
          _originalPrio: originalIndex >= 0 ? originalIndex + 1 : -1,
          _movement: movement,
          _movementDelta: Math.abs(movementDelta)
