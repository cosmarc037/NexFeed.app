async function handleOrderUpload(file) {
  try {
    // Step 1: Parse the file — get raw order data
    const parsedOrders = await parseOrderFile(file);
    
    if (parsedOrders.length === 0) {
      toast.error('No orders found in the uploaded file.');
      return;
    }
    
    // Step 2: Assign temporary IDs so the algorithm can track them
    const tempOrders = parsedOrders.map((order, index) => ({
      ...order,
      id: `__upload_${Date.now()}_${index}`,
      _isNewUpload: true,
    }));
    
    // Step 3: Enrich with master data and N10D
    const enriched = enrichOrders(tempOrders, masterData, n10dData);
    
    // Step 4: Check context
    const existingOrders = (queryClient.getQueryData(["orders"]) || []);
    const existingActive = existingOrders.filter(
      o => o.status !== 'Done' && o.status !== 'Cancel PO' && 
           o.status !== 'In Production' && o.status !== 'On-going' &&
           o.status !== 'Planned'
    );
    const hasN10D = (queryClient.getQueryData(["n10d"]) || []).length > 0;
    const allLinesWereBlank = existingActive.length === 0;
    
    // Step 5: Get existing enriched orders that can be moved/combined
    const existingEnriched = enrichOrders(
      existingOrders.filter(o => 
        o.status !== 'Done' && o.status !== 'Cancel PO' &&
        o.status !== 'In Production' && o.status !== 'On-going' &&
        o.status !== 'Planned'
      ),
      masterData, 
      n10dData
    );
    
    // Step 6: Combine all orders for the algorithm
    const allOrdersForAlgorithm = [...existingEnriched, ...enriched];
    
    setUploadContext({
      allLinesWereBlank,
      hasN10D,
      uploadedOrderCount: parsedOrders.length,
      newOrders: enriched,
      existingOrders: existingEnriched,
      allOrders: allOrdersForAlgorithm

async function handleUploadApplySequence(sequencedResults) {
  setIsApplying(true);
  
  try {
    // Track temp ID → real DB ID for combined order references
    const tempIdToRealId = {};
    
    // Step 1: Separate new uploads from existing DB orders
    // and separate combined from regular
    
    for (const [line, lineOrders] of Object.entries(sequencedResults)) {
      if (!lineOrders || lineOrders.length === 0) continue;
      
      let prioCounter = 1;
      
      for (const order of lineOrders) {
        
        if (order._isCombined && order._combinedFrom && order._combinedFrom.length > 1) {
          // === COMBINED ORDER ===
          // Step A: Create or update ALL sub-orders first to get real IDs
          const resolvedSubIds = [];
          
          for (const sub of order._combinedFrom) {
            const isNew = String(sub.id).startsWith('__upload_');
            
            if (isNew) {
              // Create new order in DB
              const created = await createOrderMutation.mutateAsync({
                data: {
                  ...cleanOrderForDB(sub),
                  feedmill_line: line,
                  priority_seq: null, // Children don't get prio
                }
              });
              tempIdToRealId[sub.id] = created.id;
              resolvedSubIds.push(created.id);
              console.log(`  [Create sub] ${sub.id} → ${created.id}`);
            } else {
              // Existing order — just track its ID
              resolvedSubIds.push(sub.id);
            }
          }
          
          // Step B: Determine lead — resolve its real ID
          const leadTempId = order.id;
          const isLeadNew = String(leadTempId).startsWith('__upload_');
          const leadRealId = isLeadNew ? tempIdToRealId[leadTempId] : leadTempId;
          
          if (!leadRealId) {
            console.error(`  [Error] Lead ID not resolved: ${leadTempId}`);
            continue;
          }
          
          const combinedVolume = parseFloat(order.volume) || 0;
          
          // Step C: Update lead order with combined state
          await updateOrderMutation.mutateAsync({
            id: leadRealId,
            data: {
              feedmill_line: line,
              priority_seq: prioCounter,
              volume_override: combinedVolume,
              original_order_ids: resolvedSubIds,
              parent_id: null,
            }
          });
          console.log(`  [Lead] ${leadRealId}: volume_override=${combinedVolume}, subs=[${resolvedSubIds.join(',')}]`);
          
          // Step D: Update children with parent_id
          for (const sub of order._combinedFrom) {
            const subRealId = String(sub.id).startsWith('__upload_') 
              ? tempIdToRealId[sub.id] 
              : sub.id;
            
            if (!subRealId || String(subRealId) === String(leadRealId)) continue;
            
            await updateOrderMutation.mutateAsync({
              id: subRealId,
              data: {
                parent_id: leadRealId,
                feedmill_line: line,
                priority_seq: null,
              }
            });
            console.log(`  [Child] ${subRealId}: parent_id=${leadRealId}`);
          }
          
          prioCounter++;
          
        } else {
          // === REGULAR ORDER ===
          const isNew = String(order.id).startsWith('__upload_');
          
          if (isNew) {
            // Create new order
            const created = await createOrderMutation.mutateAsync({
              data: {
                ...cleanOrderForDB(order),
                feedmill_line: line,
                priority_seq: prioCounter,
                parent_id: null,
                volume_override: null,
                original_order_ids: null,
              }
            });
            tempIdToRealId[order.id] = created.id;
            console.log(`  [Create] ${order.id} → ${created.id}`);
          } else {
            // Update existing order
            await updateOrderMutation.mutateAsync({
              id: order.id,
              data: {
                feedmill_line: line,
                priority_seq: prioCounter,
                parent_id: null,
                volume_override: null,
                original_order_ids: null,
              }
            });
            console.log(`  [Update] ${order.id}: line=${line}, prio=${prioCounter}`);
          }
          
          prioCounter++;
        }
      }
    }
    
    // Step 2: Refetch
    await queryClient.invalidateQueries({ queryKey: ["orders"] });
    await new Promise(resolve => setTimeout(resolve, 500));
    
    // Step 3: Cascade per line
    for (const [line, lineOrders] of Object.entries(sequencedResults)) {
      if (!lineOrders || lineOrders.length === 0) continue;
      
      // Build cascade with real IDs
      const topLevel = lineOrders.map((o, i) => {
        const realId = String(o.id).startsWith('__upload_')
          ? tempIdToRealId[o.id] || o.id
          : o.id;
        return {
          ...o,
          id: realId,
          priority_seq: i + 1,
          feedmill_line: line,
        };
      });
      
      await cascadeSchedule(topLevel, 0);
    }
    
    // Step 4: Final refetch
    await queryClient.invalidateQueries({ queryKey: ["orders"] });
    
    // Step 5: Close and notify
    setShowUploadAutoSequence(false);
    setIsApplying(false);
    
    const totalOrders = Object.values(sequencedResults).reduce((s, lo) => s + lo.length, 0);
    const combinedCount = Object.values(sequencedResults).reduce((s, lo) =>
      s + lo.filter(o => o._isCombined).length, 0
    );
    const linesUsed = Object.keys(sequencedResults).filter(l =>
      (sequencedResults[l] || []).length > 0
    ).length;
    
    toast.success(
      `Auto-sequence applied: ${totalOrders} orders across ${linesUsed} lines.${combinedCount > 0 ? ` ${combinedCount} combined groups.` : ''}`
    );
    
  } catch (error) {
    console.error('Upload apply error:', error);
    setIsApplying(false);
    toast.error(`Failed to apply: ${error.message}`);
  }
}

// Helper: clean order object for DB creation
function cleanOrderForDB(order) {
  const cleaned = { ...order };
  
  // Remove internal tracking fields
  delete cleaned._isNewUpload;
  delete cleaned._isCombined;
  delete cleaned._combinedFrom;
  delete cleaned._combinedFromLines;
  delete cleaned._movement;
  delete cleaned._movementDelta;
  delete cleaned._originalPrio;
  delete cleaned._movedFromLine;
  delete cleaned._originalLine;
  delete cleaned._sourceLineNormalized;
  delete cleaned._sourceLine;
  delete cleaned._isWorthMoving;
  delete cleaned._queueImprovement;
  delete cleaned._sourceMTBefore;
  delete cleaned._sourceMTAfter;
  delete cleaned._sourceQueueBefore;
  delete cleaned._sourceQueueAfter;
  delete cleaned._targetMTBefore;
  delete cleaned._targetMTAfter;
  delete cleaned._targetQueueBefore;
  delete cleaned._targetQueueAfter;
  delete cleaned._finalSourceQueueBefore;
  delete cleaned._finalSourceQueueAfter;
  delete cleaned._finalTargetQueueBefore;
  delete cleaned._finalTargetQueueAfter;
  delete cleaned.prio;
  
  // Remove temp ID
  if (String(cleaned.id).startsWith('__upload_')) {
    delete cleaned.id;
  }
  
  return cleaned;
}
