## 1. GENERATE PER-ROW INSIGHTS — SINGLE BATCH AI CALL

```javascript
async function generateRowInsights(sequencedOrders, originalOrders, line, placementLog) {
  const rows = sequencedOrders.map((order, index) => {
    const newPrio = index + 1;
    const originalIndex = originalOrders.findIndex(o => o.id === order.id);
    const isNew = originalIndex === -1;
    const originalPrio = originalIndex >= 0 ? originalIndex + 1 : null;
    const isCombined = order._isCombined && order._combinedFrom;
    const movedFromLine = order._movedFromLine || order._originalLine;
    const movedFromDifferentLine = movedFromLine && normalizeLine(movedFromLine) !== normalizeLine(line);
    
    // Find relevant placement log entry
    const logEntry = placementLog.find(entry => {
      if (entry.type === 'combined' && isCombined) {
        return entry.toLine === line && entry.product === order.item_description;
      }
      if (entry.type === 'moved' && movedFromDifferentLine) {
        return entry.toLine === line && String(entry.materialCode) === String(order.material_code_fg);
      }
      return false;
    });
    
    // Build context for this row
    let context = '';
    
    if (isCombined) {
      const subCount = order._combinedFrom.length;
      const fromLines = [...new Set(order._combinedFrom.map(s => s.line))];
      const allSameLine = fromLines.every(l => normalizeLine(l) === normalizeLine(line));
      
      context = `COMBINED: ${subCount} orders combined into ${order.volume} MT.`;
      if (!allSameLine) {
        context += ` Orders pulled from ${fromLines.map(l => getLineShortName(l)).join(', ')}.`;
      } else {
        context += ` Combined in place on ${getLineShortName(line)}.`;
      }
      if (logEntry) {
        context += ` Changeovers saved: ${logEntry.changeoversSaved}. Time saved: ${logEntry.timeSaved.toFixed(2)} hrs.`;
        context += ` Queue time: ${logEntry.lineScores.map(ls => `${getLineShortName(ls.line)}: ${ls.queueTimeBefore.toFixed(2)} hrs`).join(', ')}.`;
      }
    } else if (movedFromDifferentLine) {
      context = `MOVED from ${getLineShortName(movedFromLine)} to ${getLineShortName(line)}.`;
      if (logEntry) {
        context += ` Queue times: ${logEntry.lineScores.map(ls => `${getLineShortName(ls.line)}: ${ls.totalMTBefore.toFixed(1)} MT ÷ ${ls.runRate} MT/hr = ${ls.queueTimeBefore.toFixed(2)} hrs`).join(', ')}.`;
        context += ` ${getLineShortName(line)} had lowest queue time.`;
      }
    } else if (isNew) {
      context = `NEW to this line.`;
    } else if (originalPrio !== newPrio) {
      const direction = originalPrio > newPrio ? 'up' : 'down';
      const delta = Math.abs(originalPrio - newPrio);
      context = `REPOSITIONED: moved ${direction} ${delta} position${delta > 1 ? 's' : ''} (Prio ${originalPrio} → ${newPrio}).`;
    } else {
      context = `UNCHANGED: stayed at Prio ${newPrio}.`;
    }
    
    // Add avail date context
    const availDate = order.target_avail_date;
    if (availDate && /^\d{4}-\d{2}-\d{2}/.test(availDate)) {
      context += ` Has avail date: ${new Date(availDate).toLocaleDateString('en-GB', { day: '2-digit', month: 'short', year: 'numeric' })}.`;
    } else if (order._category === 'B') {
      context += ` N10D status: prio replenish.`;
    } else if (order._category === 'D') {
      context += ` N10D status: stock sufficient.`;
    }
    
    // Add form context if relevant
    if (index > 0) {
      const prevOrder = sequencedOrders[index - 1];
      if (prevOrder.form === order.form) {
        context += ` Same form as previous order (${order.form}) — no form changeover.`;
      } else {
        context += ` Form change from ${prevOrder.form} to ${order.form}.`;
      }
    }
    
    return {
      prio: newPrio,
      product: order.item_description,
      volume: order.volume,
      form: order.form,
      context
    };
  });
  
  // Build AI prompt
  const prompt = `
You are a production scheduling advisor. For each order in the sequence below, write a SHORT 1-2 sentence insight explaining why this order is at this position and any relevant action taken.

LINE: ${line}
LINE RUN RATE: ${getLineRunRate(line)} MT/hr

ORDERS IN SEQUENCE:

${rows.map(row => `
PRIO ${row.prio}: ${row.product} (${row.volume} MT, ${row.form})
Context: ${row.context}
`).join('\n')}

RULES:
- Write 1-2 sentences MAX per order. Be concise.
- No markdown, no bold, no bullet points.
- Use short line names (L1, L2, etc.).
- For UNCHANGED orders, just say why its position makes sense (e.g., "No change needed — same product group as adjacent orders").
- For REPOSITIONED orders, explain what drove the position (avail date, N10D urgency, form grouping).
- For MOVED orders, mention the queue time comparison briefly.
- For COMBINED orders, mention how many orders merged, volume, changeovers saved.
- Reference avail dates and N10D status when relevant.
- Do NOT repeat the product name — the user can see it in the table.

RESPONSE FORMAT:
PRIO 1: [1-2 sentence insight]
PRIO 2: [1-2 sentence insight]
...
  `;
  
  try {
    const response = await callAI(prompt);
    return parseRowInsights(response, sequencedOrders.length);
  } catch (error) {
    console.error('Row insights AI failed:', error);
    return generateFallbackRowInsights(rows);
  }
}

function parseRowInsights(response, orderCount) {
  const insights = {};
  const lines = response.split('\n').filter(l => l.trim());
  
  lines.forEach(line => {
    const match = line.match(/^PRIO\s*(\d+)\s*:\s*(.+)$/i);
    if (match) {
      const prio = parseInt(match[1]);
      const text = match[2].trim();
      insights[prio] = text;
    }
  });
  
  // Fill gaps with empty
  for (let i = 1; i <= orderCount; i++) {
    if (!insights[i]) {
      insights[i] = '';
    }
  }
  
  return insights;
}

function generateFallbackRowInsights(rows) {
  const insights = {};
  
  rows.forEach(row => {
    let text = '';
    
    if (row.context.startsWith('COMBINED')) {
      text = row.context.replace('COMBINED: ', '');
    } else if (row.context.startsWith('MOVED')) {
      text = row.context.replace('MOVED ', 'Moved ');
    } else if (row.context.startsWith('REPOSITIONED')) {
      text = row.context.replace('REPOSITIONED: ', '');
    } else if (row.context.startsWith('UNCHANGED')) {
      text = 'No change in position.';
    } else if (row.context.startsWith('NEW')) {
      text = 'Newly assigned to this line.';
    }
    
    insights[row.prio] = text;
  });
  
  return insights;
}
```
