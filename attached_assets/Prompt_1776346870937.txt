```javascript
async function applySequencedResultsToDB(sequencedResults, options = {}) {
  const { label = 'Auto-sequence' } = options;
  
  const combinedLeadMap = {};
  
  for (const [line, lineOrders] of Object.entries(sequencedResults)) {
    if (!lineOrders || lineOrders.length === 0) continue;
    
    let prioCounter = 1;
    
    for (const order of lineOrders) {
      
      if (order._isCombined && order._combinedFrom && order._combinedFrom.length > 1) {
        // === COMBINED ORDER ===
        // Step 1: Sum ALL child volumes
        const combinedVolume = order._combinedFrom.reduce((sum, sub) => {
          return sum + (parseFloat(sub.volume) || parseFloat(sub.total_volume_mt) || 0);
        }, 0);
        
        const allChildIds = order._combinedFrom.map(sub => sub.id);
        const firstChild = order._combinedFrom[0];
        
        // Step 2: CREATE a brand new lead order
        // Copy product details from the first child
        const newLead = await createOrderMutation.mutateAsync({
          data: {
            fpr: firstChild.fpr,
            feedmill_line: line,
            priority_seq: prioCounter,
            
            // Product details — same as children
            material_code_fg: firstChild.material_code_fg,
            material_code_sfg: firstChild.material_code_sfg,
            item_description: firstChild.item_description,
            form: firstChild.form,
            batch_size: firstChild.batch_size,
            run_rate: firstChild.run_rate,
            category: firstChild.category,
            color: firstChild.color,
            diameter: firstChild.diameter,
            formula_version: firstChild.formula_version,
            changeover: firstChild.changeover,
            
            // Use first child's planned order numbers
            planned_order_fg: firstChild.planned_order_fg,
            planned_order_sfg: firstChild.planned_order_sfg,
            
            // Avail date — earliest from children
            target_avail_date: order.target_avail_date || firstChild.target_avail_date,
            
            // Combined-specific fields
            volume_override: combinedVolume > 0 ? combinedVolume : null,
            original_order_ids: allChildIds,
            parent_id: null,
            
            // Volume fields
            total_volume_mt: combinedVolume > 0 ? combinedVolume : null,
          }
        });
        
        const newLeadId = newLead.id;
        combinedLeadMap[order.id] = newLeadId;
        
        console.log(`  [New Lead Created] ID: ${newLeadId}, ${firstChild.item_description}, ${combinedVolume} MT`);
        console.log(`    Children: ${allChildIds.join(', ')}`);
        
        // Step 3: ALL original orders become children
        for (const sub of order._combinedFrom) {
          await updateOrderMutation.mutateAsync({
            id: sub.id,
            data: {
              parent_id: newLeadId,
              feedmill_line: line,
              priority_seq: null,
            }
          });
          console.log(`    [Child] ${sub.id} (${sub.volume} MT) → parent_id=${newLeadId}`);
        }
        
        prioCounter++;
        
      } else {
        // === REGULAR ORDER (not combined) ===
        await updateOrderMutation.mutateAsync({
          id: order.id,
          data: {
            feedmill_line: line,
            priority_seq: prioCounter,
            parent_id: null,
            volume_override: null,
            original_order_ids: null,
          }
        });
        
        prioCounter++;
      }
    }
  }
  
  // Refetch
  await queryClient.invalidateQueries({ queryKey: ["orders"] });
  await new Promise(resolve => setTimeout(resolve, 500));
  
  // Cascade — use new lead IDs
  for (const [line, lineOrders] of Object.entries(sequencedResults)) {
    if (!lineOrders || lineOrders.length === 0) continue;
    
    const topLevel = lineOrders.map((o, i) => {
      const realId = combinedLeadMap[o.id] || o.id;
      return {
        ...o,
        id: realId,
        priority_seq: i + 1,
        feedmill_line: line,
      };
    });
    
    await cascadeSchedule(topLevel, 0);
  }
  
  // Final refetch
  await queryClient.invalidateQueries({ queryKey: ["orders"] });
  
  const totalOrders = Object.values(sequencedResults).reduce((s, lo) => s + lo.length, 0);
  const combinedCount = Object.keys(combinedLeadMap).length;
  const linesUsed = Object.keys(sequencedResults).filter(l =>
    (sequencedResults[l] || []).length > 0
  ).length;
  
  return { totalOrders, combinedCount, linesUsed };
}
```
