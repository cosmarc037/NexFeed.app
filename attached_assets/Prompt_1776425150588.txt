## 3. UPDATE THE MAIN HANDLER — REPORT PHASES

```javascript
async function handlePlantLevelAutoSequence() {
  setProcessingPhase('scanning');
  setProcessingStats({ processed: 0, total: 0, combined: 0, moved: 0 });
  setShowProcessingOverlay(true);
  
  // Small delay to let UI render
  await new Promise(r => setTimeout(r, 100));
  
  try {
    const allActiveOrders = enrichedOrders.filter(
      o => o.status !== 'Done' && o.status !== 'Cancel PO' &&
           o.status !== 'In Production' && o.status !== 'On-going'
    );
    
    setProcessingStats(prev => ({ ...prev, total: allActiveOrders.length }));
    
    // Phase 1: Scanning
    setProcessingPhase('scanning');
    await new Promise(r => setTimeout(r, 300));
    
    // Save original snapshot
    const snapshot = {};
    const allLines = ALL_LINES_ORDERED.map(l => normalizeLine(l.line));
    allLines.forEach(line => {
      snapshot[line] = allActiveOrders
        .filter(o => normalizeLine(o.feedmill_line) === line)
        .sort((a, b) => (a.priority_seq || 0) - (b.priority_seq || 0))
        .map(o => ({ ...JSON.parse(JSON.stringify(o)) }));
    });
    
    // Phase 2: Combining
    setProcessingPhase('combining');
    await new Promise(r => setTimeout(r, 200));
    
    // Phase 3: Calculating queue times
    // Phase 4: Placing
    // These happen inside plantLevelAutoSequence — use callbacks to report progress
    
    const onProgress = (stats) => {
      setProcessingStats(prev => ({ ...prev, ...stats }));
    };
    
    const onPhaseChange = (phase) => {
      setProcessingPhase(phase);
    };
    
    const { lineOrdersMap, combinedOrders, placementLog } = await runAutoSequenceWithProgress(
      allActiveOrders, masterData, n10dData, allLines, changeoverRules,
      onProgress, onPhaseChange
    );
    
    // Phase 5: N10D
    setProcessingPhase('n10d');
    await new Promise(r => setTimeout(r, 200));
    
    // Phase 6: Sequencing
    setProcessingPhase('sequencing');
    await new Promise(r => setTimeout(r, 200));
    
    // Phase 7: Changeovers
    setProcessingPhase('changeovers');
    await new Promise(r => setTimeout(r, 200));
    
    // Build results with movement indicators
    const results = {};
    allLines.forEach(line => {
      const lineOrders = lineOrdersMap[line] || [];
      if (lineOrders.length === 0) {
        results[line] = [];
        return;
      }
      
      const originalLineOrders = snapshot[line] || [];
      const withMovement = lineOrders.map((order, newIndex) => {
        const newPrio = newIndex + 1;
        const originalIndex = originalLineOrders.findIndex(o => o.id === order.id);
        let movement = 'same';
        let movementDelta = 0;
        
        if (originalIndex >= 0) {
          const originalPrio = originalIndex + 1;
          movementDelta = originalPrio - newPrio;
          if (movementDelta > 0) movement = 'up';
          else if (movementDelta < 0) movement = 'down';
        } else {
          movement = 'new_to_line';
        }
        
        return {
          ...order,
          prio: newPrio,
          _originalPrio: originalIndex >= 0 ? originalIndex + 1 : -1,
          _movement: movement,
          _movementDelta: Math.abs(movementDelta),
        };
      });
      
      results[line] = withMovement;
    });
    
    // Phase 8: AI insights
    setProcessingPhase('insights');
    await new Promise(r => setTimeout(r, 200));
    
    // Phase 9: Done
    setProcessingPhase('done');
    await new Promise(r => setTimeout(r, 500));
    
    // Store results and open preview
    setOriginalPlantSnapshot(snapshot);
    setPlantSequenceResults(results);
    setPlantPlacementLog(placementLog);
    
    const stats = calculatePlantSummary(snapshot, results, combinedOrders, placementLog, allLines);
    setPlantSummaryStats(stats);
    
    setShowProcessingOverlay(false);
    setShowPlantAutoSequencePreview(true);
    
  } catch (error) {
    console.error('Auto-sequence error:', error);
    setShowProcessingOverlay(false);
    toast.error(`Auto-sequence failed: ${error.message}`);
  }
}
```

## 4. WRAP ALGORITHM WITH PROGRESS CALLBACKS

```javascript
async function runAutoSequenceWithProgress(allOrders, masterData, n10dData, allLines, changeoverRules, onProgress, onPhaseChange) {
  // Deep copy
  const activeOrders = allOrders.map(o => ({ ...JSON.parse(JSON.stringify(o)) }));
  
  const lineOrdersMap = {};
  allLines.forEach(line => {
    lineOrdersMap[line] = activeOrders
      .filter(o => normalizeLine(o.feedmill_line) === line)
      .sort((a, b) => (a.priority_seq || 0) - (b.priority_seq || 0))
      .map(o => ({
        ...o,
        _originalLine: normalizeLine(o.feedmill_line),
        _isPlanned: o.status === 'Planned',
        _processed: false,
      }));
  });
  
  const lineTotalMT = {};
  allLines.forEach(line => {
    lineTotalMT[line] = (lineOrdersMap[line] || [])
      .reduce((sum, o) => sum + (parseFloat(o.volume) || parseFloat(o.total_volume_mt) || 0), 0);
  });
  
  const placementLog = [];
  const combinedOrders = [];
  const processedIds = new Set();
  let processedCount = 0;
  let combinedCount = 0;
  let movedCount = 0;
  
  onPhaseChange('combining');
  
  // Process order by order
  for (const line of allLines) {
    const lineOrders = [...(lineOrdersMap[line] || [])];
    
    for (const order of lineOrders) {
      if (order._processed || processedIds.has(order.id)) continue;
      
      const orderVolume = parseFloat(order.volume) || parseFloat(order.total_volume_mt) || 0;
      const orderMaterialCode = String(order.material_code_fg || '').trim();
      const orderFormulaVersion = order.formula_version || '—';
      
      if (!orderMaterialCode) {
        order._processed = true;
        processedIds.add(order.id);
        processedCount++;
        continue;
      }
      
      // Find matches
      const matches = findMatchableOrders(
        order, orderMaterialCode, orderFormulaVersion,
        allLines, lineOrdersMap, processedIds, masterData
      );
      
      if (matches.length > 0) {
        const combineResult = combineOrdersGreedy(order, orderVolume, matches, MAX_COMBINE_MT);
        
        if (combineResult.combinedOrders.length > 0) {
          onPhaseChange('calculating');
          
          const placement = placeCombiledOrder(
            order, combineResult, allLines, lineOrdersMap,
            lineTotalMT, masterData, processedIds
          );
          
          combinedOrders.push(placement.combinedOrder);
          if (placement.logEntry) placementLog.push(placement.logEntry);
          
          combinedCount++;
          processedCount += 1 + combineResult.combinedOrders.length;
          if (placement.combinedOrder._movedFromLine) movedCount++;
          
          onPhaseChange('placing');
          onProgress({ processed: processedCount, combined: combinedCount, moved: movedCount });
          
          // Yield to UI every 5 orders
          if (processedCount % 5 === 0) {
            await new Promise(r => setTimeout(r, 0));
          }
          
          continue;
        }
      }
      
      // No match — evaluate queue time
      const movePlacement = evaluateAndMoveOrder(
        order, orderVolume, line, allLines, lineOrdersMap,
        lineTotalMT, masterData, processedIds
      );
      
      if (movePlacement.logEntry) {
        placementLog.push(movePlacement.logEntry);
        movedCount++;
      }
      
      processedCount++;
      onProgress({ processed: processedCount, combined: combinedCount, moved: movedCount });
      
      if (processedCount % 5 === 0) {
        await new Promise(r => setTimeout(r, 0));
      }
    }
  }
  
  // N10D phase
  onPhaseChange('n10d');
  await new Promise(r => setTimeout(r, 100));
  
  const finalLineOrders = {};
  allLines.forEach(line => {
    const orders = lineOrdersMap[line] || [];
    if (orders.length === 0) {
      finalLineOrders[line] = [];
      return;
    }
    const categorized = preSortOrders(orders, n10dData, masterData);
    finalLineOrders[line] = sortOrdersByCategory(categorized);
  });
  
  // Changeovers phase
  onPhaseChange('changeovers');
  await new Promise(r => setTimeout(r, 100));
  
  allLines.forEach(line => {
    const orders = finalLineOrders[line] || [];
    if (orders.length > 0) {
      calculatePreviewChangeovers(orders, changeoverRules);
    }
  });
  
  return { lineOrdersMap: finalLineOrders, combinedOrders, placementLog };
}
```
