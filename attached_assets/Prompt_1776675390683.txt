## 2. REUSE THE EXACT preSortOrders FUNCTION

This is the same function from the per-line auto-sequence. It:
- Looks up N10D data for each order
- Assigns _category (A/B/C/D/E)
- Assigns _targetDate for non-dated orders
- Assigns _n10dStatus (Critical/Urgent/Sufficient/etc.)
- Colors the avail date accordingly

```javascript
function preSortOrders(orders, n10dData, masterData) {
  return orders.map(order => {
    const enriched = { ...order };
    
    // Check if order has a real avail date
    const availDate = order.target_avail_date;
    const hasRealDate = availDate && /^\d{4}-\d{2}-\d{2}/.test(availDate) && !isNaN(Date.parse(availDate));
    
    if (hasRealDate) {
      enriched._category = 'A';
      enriched._sortDate = new Date(availDate);
      enriched._availDateColor = 'default';
      return enriched;
    }
    
    // Look up N10D data
    const materialCode = String(order.material_code_fg || '').trim();
    const n10dEntry = n10dData.find(n => 
      String(n.material_code || '').trim() === materialCode
    );
    
    if (!n10dEntry) {
      enriched._category = 'E';
      enriched._sortDate = new Date('2099-12-31');
      enriched._availDateDisplay = availDate || '—';
      enriched._availDateColor = 'default';
      return enriched;
    }
    
    // Determine N10D status and target date
    const n10dStatus = n10dEntry.status || n10dEntry.stock_status || '';
    const inferredDate = n10dEntry.inferred_target_date || n10dEntry.target_date;
    
    enriched._n10dStatus = n10dStatus;
    enriched._n10dEntry = n10dEntry;
    
    if (n10dStatus === 'Critical' || n10dStatus === 'Urgent') {
      // Category C: urgent/critical — no specific date but needs attention
      if (inferredDate && !isNaN(Date.parse(inferredDate))) {
        // Has inferred date — use it
        enriched._category = 'B';
        enriched._targetDate = inferredDate;
        enriched._sortDate = new Date(inferredDate);
        enriched._availDateDisplay = new Date(inferredDate).toLocaleDateString('en-US', {
          month: 'short', day: 'numeric', year: 'numeric'
        });
        enriched._availDateColor = n10dStatus === 'Critical' ? 'red' : 'orange';
        
        // Update target_avail_date for display
        enriched.target_avail_date = inferredDate;
      } else {
        enriched._category = 'C';
        enriched._sortDate = new Date('2000-01-01'); // Sort to top
        enriched._availDateDisplay = availDate || 'prio replenish';
        enriched._availDateColor = n10dStatus === 'Critical' ? 'red' : 'orange';
      }
    } else if (n10dStatus === 'Sufficient' || n10dStatus === 'Safe' || n10dStatus === 'OK') {
      enriched._category = 'D';
      
      if (inferredDate && !isNaN(Date.parse(inferredDate))) {
        enriched._targetDate = inferredDate;
        enriched._sortDate = new Date(inferredDate);
        enriched._availDateDisplay = new Date(inferredDate).toLocaleDateString('en-US', {
          month: 'short', day: 'numeric', year: 'numeric'
        });
      } else {
        enriched._sortDate = new Date('2099-12-31');
        enriched._availDateDisplay = availDate || 'safety stocks';
      }
      enriched._availDateColor = 'green';
    } else {
      // Prio replenish or other statuses with inferred date
      if (inferredDate && !isNaN(Date.parse(inferredDate))) {
        enriched._category = 'B';
        enriched._targetDate = inferredDate;
        enriched._sortDate = new Date(inferredDate);
        enriched._availDateDisplay = new Date(inferredDate).toLocaleDateString('en-US', {
          month: 'short', day: 'numeric', year: 'numeric'
        });
        enriched._availDateColor = 'orange';
        enriched.target_avail_date = inferredDate;
      } else {
        enriched._category = 'E';
        enriched._sortDate = new Date('2099-12-31');
        enriched._availDateDisplay = availDate || '—';
        enriched._availDateColor = 'default';
      }
    }
    
    return enriched;
  });
}
```

## 3. REUSE THE EXACT autoSequenceSort FUNCTION

```javascript
function autoSequenceSort(orders, changeoverRules) {
  // Sort priority:
  // 1. Category C (critical/urgent non-dated) — at the TOP
  // 2. Category A (dated) — sorted by avail date ascending
  // 3. Category B (N10D targeted) — sorted by inferred date ascending
  // 4. Category D (stock sufficient) — sorted by inferred date ascending
  // 5. Category E (no data) — at the bottom
  
  const catC = orders.filter(o => o._category === 'C')
    .sort((a, b) => {
      // Critical before Urgent
      if (a._n10dStatus === 'Critical' && b._n10dStatus !== 'Critical') return -1;
      if (a._n10dStatus !== 'Critical' && b._n10dStatus === 'Critical') return 1;
      return 0;
    });
  
  const catA = orders.filter(o => o._category === 'A')
    .sort((a, b) => a._sortDate - b._sortDate);
  
  const catB = orders.filter(o => o._category === 'B')
    .sort((a, b) => a._sortDate - b._sortDate);
  
  const catD = orders.filter(o => o._category === 'D')
    .sort((a, b) => a._sortDate - b._sortDate);
  
  const catE = orders.filter(o => o._category === 'E');
  
  // Combine in priority order
  const sorted = [...catC, ...catA, ...catB, ...catD, ...catE];
  
  // Assign prio
  sorted.forEach((order, index) => {
    order.prio = index + 1;
    order.priority_seq = index + 1;
  });
  
  return sorted;
}
```

```javascript
function buildFinalLineSequence(lineOrders, existingInProductionOrders) {
```
