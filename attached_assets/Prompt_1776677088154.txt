function PlantAutoSequencePreview({ ... }) {
  // ... existing state ...
  
  // Cache insights per line — keyed by line name
  const insightCacheRef = useRef({});
  // Track fingerprints to detect changes
  const fingerprintCacheRef = useRef({});
  
  function getLineFingerprint(line) {
    const orders = sequencedResults?.[line] || [];
    return orders.map(o => `${o.id}_${o.prio}_${o._isCombined ? 'C' : 'R'}_${o._movement || 's'}`).join('|');
  }
  
  function getCachedInsights(line) {
    const currentFingerprint = getLineFingerprint(line);
    const cachedFingerprint = fingerprintCacheRef.current[line];
    
    // Return cached if fingerprint matches
    if (cachedFingerprint === currentFingerprint && insightCacheRef.current[line]) {
      return insightCacheRef.current[line];
    }
    
    return null;
  }
  
  function setCachedInsights(line, insights) {
    const fingerprint = getLineFingerprint(line);
    insightCacheRef.current[line] = insights;
    fingerprintCacheRef.current[line] = fingerprint;
  }
  
  function invalidateLineCache(line) {
    delete insightCacheRef.current[line];
    delete fingerprintCacheRef.current[line];
  }
  
  // When uncombine happens, only invalidate that line
  function handleUncombineSingle(line, combinedOrder) {
    // ... existing uncombine logic ...
    
    // Invalidate ONLY this line's insight cache
    invalidateLineCache(line);
  }
  
  // Also cache AI explanations (Actions Taken)
  const aiExplanationsCacheRef = useRef(null);
  const aiPlacementLogFingerprintRef = useRef(null);
  
  function getPlacementLogFingerprint() {
    return placementLog.map(e => `${e.type}_${e.product}_${e.toLine}`).join('|');
  }
  
  async function generateAIExplanations(log) {
    const fingerprint = getPlacementLogFingerprint();
    
    // Return cached if unchanged
    if (fingerprint === aiPlacementLogFingerprintRef.current && aiExplanationsCacheRef.current) {
      setAiExplanations(aiExplanationsCacheRef.current);
      return;
    }
    
    setIsLoadingAI(true);
    try {
      const prompt = buildActionsPrompt(log);
      const response = await callAI(prompt);
      const parsed = parseActionsResponse(response, log);
      if (Object.keys(parsed).length >= log.length * 0.5) {
        setAiExplanations(parsed);
        aiExplanationsCacheRef.current = parsed;
        aiPlacementLogFingerprintRef.current = fingerprint;
      } else {
        setUseFallback(true);
      }
    } catch {
      setUseFallback(true);
    }
    setIsLoadingAI(false);
  }
  
  return (
    // ... existing JSX ...
    
    {activeTab !== 'summary' && (
      <PlantLineTab
        line={activeTab}
        originalOrders={originalSnapshot[activeTab] || []}
        sequencedOrders={sequencedResults[activeTab] || []}
        placementLog={placementLog}
        aiExplanations={aiExplanations}
        isLoadingAI={isLoadingAI}
        useFallback={useFallback}
        onUncombineSingle={(combinedOrder) => handleUncombineSingle(activeTab, combinedOrder)}
        getCachedInsights={() => getCachedInsights(activeTab)}
        setCachedInsights={(insights) => setCachedInsights(activeTab, insights)}
      />
    )}
  );
}

## 4. UPDATE PlantLineTab — USE PARENT CACHE

```jsx
function PlantLineTab({ 
  line, 
  originalOrders, 
  sequencedOrders, 
  placementLog,
  aiExplanations,
  isLoadingAI,
  useFallback,
  onUncombineSingle,
  getCachedInsights,
  setCachedInsights
}) {
  const [rowInsights, setRowInsights] = useState(null);
  const [isLoadingRowInsights, setIsLoadingRowInsights] = useState(false);
  
  useEffect(() => {
    if (sequencedOrders.length === 0) return;
    
    // Check parent cache first
    const cached = getCachedInsights ? getCachedInsights() : null;
    
    if (cached) {
      // Use cached — no AI call
      setRowInsights(cached);
      setIsLoadingRowInsights(false);
    } else {
      // Generate new
      loadRowInsights();
    }
  }, [line, sequencedOrders]);
  
  async function loadRowInsights() {
    setIsLoadingRowInsights(true);
    const insights = await generateRowInsights(
      sequencedOrders, originalOrders, line, placementLog
    );
    setRowInsights(insights);
    setIsLoadingRowInsights(false);
    
    // Store in parent cache
    if (setCachedInsights) {
      setCachedInsights(insights);
    }
  }
  
  // ... rest of component unchanged ...
  
  return (
    <div className="plant-line-tab">
      {/* ... */}
      <tbody>
        {sequencedOrders.map(order => (
          <PlantAfterRow
            key={order.id}
            order={order}
            onUncombineSingle={onUncombineSingle}
            insight={rowInsights?.[order.prio] || null}
            isLoadingInsight={isLoadingRowInsights}
          />
        ))}
      </tbody>
      {/* ... */}
    </div>
  );
}
