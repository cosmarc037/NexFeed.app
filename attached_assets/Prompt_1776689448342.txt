function autoSequenceSort(orders, changeoverRules) {
  // ALL orders sort chronologically — no exceptions for Planned
  // Sort priority:
  // 1. Category C (critical/urgent non-dated) — at the TOP
  // 2. Category A (dated) — sorted by avail date ascending
  // 3. Category B (N10D targeted) — sorted by inferred date ascending
  // 4. Category D (stock sufficient) — sorted by inferred date ascending
  // 5. Category E (no data) — at the bottom
  
  // NO special handling for Planned — they sort by date like everything else
  
  const catC = orders.filter(o => o._category === 'C')
    .sort((a, b) => {
      if (a._n10dStatus === 'Critical' && b._n10dStatus !== 'Critical') return -1;
      if (a._n10dStatus !== 'Critical' && b._n10dStatus === 'Critical') return 1;
      return 0;
    });
  
  const catA = orders.filter(o => o._category === 'A')
    .sort((a, b) => a._sortDate - b._sortDate);
  
  const catB = orders.filter(o => o._category === 'B')
    .sort((a, b) => a._sortDate - b._sortDate);
  
  const catD = orders.filter(o => o._category === 'D')
    .sort((a, b) => a._sortDate - b._sortDate);
  
  const catE = orders.filter(o => o._category === 'E');
  
  const sorted = [...catC, ...catA, ...catB, ...catD, ...catE];
  
  sorted.forEach((order, index) => {
    order.prio = index + 1;
    order.priority_seq = index + 1;
  });
  
  return sorted;
}

function evaluateAndMoveOrder(order, orderVolume, currentLine, allLines, lineOrdersMap, lineTotalMT, masterData, processedIds) {
  
  // Planned orders: DO NOT move to a different line
  // But they CAN be re-sorted within their current line
  if (order._isPlanned || order.status === 'Planned') {
    // Keep on current line — don't evaluate queue time for other lines
    order._processed = true;
    processedIds.add(order.id);
    
    // Just make sure it stays on its current line
    // The autoSequenceSort will handle its position chronologically
    return { logEntry: null };
  }
  
  // Non-planned: evaluate queue time across all lines
  const eligibleLines = allLines.filter(line =>
    canProduceOnLine(order, line, masterData)
  );
  
  if (eligibleLines.length === 0) {
    order._processed = true;
    processedIds.add(order.id);
    return { logEntry: null };
  }
  
  // Remove from current line
  lineTotalMT[currentLine] = (lineTotalMT[currentLine] || 0) - orderVolume;
  lineOrdersMap[currentLine] = (lineOrdersMap[currentLine] || []).filter(o => o.id !== order.id);
  
  // Find best line by queue time
  const lineScores = eligibleLines.map(line => {
    const currentMT = lineTotalMT[line] || 0;
    const runRate = getLineRunRate(line);
    const queueTime = runRate > 0 ? currentMT / runRate : Infinity;
    const lineNumber = parseInt(line.match(/\d+/)?.[0] || '99');
    return {
      line, runRate, totalMTBefore: currentMT, queueTimeBefore: queueTime,
      totalMTAfter: currentMT + orderVolume,
      queueTimeAfter: runRate > 0 ? (currentMT + orderVolume) / runRate : Infinity,
      lineNumber,
    };
  });
  
  lineScores.sort((a, b) => {
    const diff = a.queueTimeBefore - b.queueTimeBefore;
    if (Math.abs(diff) > 0.001) return diff;
    return a.lineNumber - b.lineNumber;
  });
  
  const bestLine = lineScores[0].line;
  
  order.feedmill_line = bestLine;
  order.line = bestLine;
  order.status = 'Planned';
  order._processed = true;
  processedIds.add(order.id);
  
  lineTotalMT[bestLine] = (lineTotalMT[bestLine] || 0) + orderVolume;
  if (!lineOrdersMap[bestLine]) lineOrdersMap[bestLine] = [];
  lineOrdersMap[bestLine].push(order);
  
  let logEntry = null;
  if (bestLine !== currentLine) {
    order._movedFromLine = currentLine;
    logEntry = {
      type: 'moved',
      product: order.item_description,
      materialCode: order.material_code_fg,
      fromLine: currentLine,
      toLine: bestLine,
      volume: orderVolume,
      fpr: order.fpr,
      lineScores: lineScores.map(ls => ({ ...ls })),
      bestLineReason: { ...lineScores[0] },
      fromLineReason: {
        line: currentLine,
        queueTime: lineScores.find(ls => ls.line === currentLine)?.queueTimeBefore || 0,
        runRate: getLineRunRate(currentLine),
      },
    };
  }
  
  return { logEntry };
}
