function calculatePlantSummary(originalSnapshot, sequencedResults, combinedOrders, placementLog, allLines) {
  let totalBefore = 0;
  let totalAfter = 0;
  let totalMoved = 0;
  let totalRepositioned = 0;
  let totalCombinedGroups = 0;
  let totalNewToLine = 0;
  let totalUnchanged = 0;
  
  allLines.forEach(line => {
    const before = originalSnapshot[line] || [];
    const after = sequencedResults[line] || [];
    
    totalBefore += before.length;
    totalAfter += after.length;
    
    after.forEach(order => {
      if (order._movement === 'new_to_line') totalNewToLine++;
      else if (order._movement === 'up' || order._movement === 'down') totalRepositioned++;
      else totalUnchanged++;
      
      if (order._isCombined) totalCombinedGroups++;
    });
  });
  
  // Count orders that actually moved between lines
  totalMoved = placementLog.filter(e => e.type === 'moved').length;
  
  // Determine if there are actual changes
  const hasChanges = totalMoved > 0 || totalRepositioned > 0 || totalNewToLine > 0 ||
    placementLog.some(e => e.type === 'combined' && !e._existedBefore);
  
  return {
    totalOrdersBefore: totalBefore,
    totalOrdersAfter: totalAfter,
    ordersMovedBetweenLines: totalMoved,
    ordersRepositioned: totalRepositioned,
    ordersCombined: totalCombinedGroups,
    ordersNewToLine: totalNewToLine,
    ordersUnchanged: totalUnchanged,
    hasChanges,
  };
}

// After table header — dynamic based on actual changes
function getAfterLabel(sequencedOrders, originalOrders) {
  const afterCount = sequencedOrders.length;
  const beforeCount = originalOrders.length;
  
  // Count actual changes
  const repositioned = sequencedOrders.filter(o => 
    o._movement === 'up' || o._movement === 'down'
  ).length;
  const newToLine = sequencedOrders.filter(o => o._movement === 'new_to_line').length;
  const combined = sequencedOrders.filter(o => o._isCombined).length;
  const unchanged = sequencedOrders.filter(o => o._movement === 'same').length;
  
  // Check if any combined orders are NEW (not already combined before)
  const newCombined = sequencedOrders.filter(o => 
    o._isCombined && !originalOrders.some(orig => 
      orig.original_order_ids && orig.original_order_ids.length > 0 && orig.id === o.id
    )
  ).length;
  
  // Existing combined (already were combined before auto-sequence)
  const existingCombined = combined - newCombined;
  
  const parts = [];
  
  if (afterCount !== beforeCount) {
    parts.push(`${afterCount} orders`);
  } else {
    parts.push(`${afterCount} orders`);
  }
  
  if (repositioned > 0) parts.push(`${repositioned} repositioned`);
  if (newToLine > 0) parts.push(`${newToLine} from other lines`);
  if (newCombined > 0) parts.push(`${newCombined} newly combined`);
  if (existingCombined > 0 && newCombined === 0 && repositioned === 0 && newToLine === 0) {
    parts.push(`${existingCombined} already combined`);
  }
  
  if (repositioned === 0 && newToLine === 0 && newCombined === 0 && afterCount === beforeCount) {
    parts.push('no changes');
  }
  
  return parts.join(' · ');
}
