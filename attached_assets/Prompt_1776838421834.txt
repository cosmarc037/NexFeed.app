## 3. HANDLE FEEDMILL AUTO-SEQUENCE — SAME AS PER-LINE BUT ACROSS FEEDMILL LINES

```javascript
async function handleFeedmillAutoSequence(feedmill) {
  const feedmillLines = getFeedmillLines(feedmill);
  
  if (feedmillLines.length === 0) {
    toast.error(`No lines found for ${feedmill}`);
    return;
  }
  
  const feedmillShortName = getFeedmillShortName(feedmill);
  
  // Show processing overlay
  setProcessingPhase('scanning');
  setProcessingStats({ processed: 0, total: 0, combined: 0, moved: 0 });
  setShowProcessingOverlay(true);
  setProcessingLabel(`${feedmillShortName} Auto-Sequence`);
  
  await new Promise(r => setTimeout(r, 100));
  
  try {
    // Get active orders ONLY from this feedmill's lines
    const allActiveOrders = enrichedOrders.filter(
      o => o.status !== 'Done' && o.status !== 'Cancel PO' &&
           o.status !== 'In Production' && o.status !== 'On-going'
    );
    
    const feedmillOrders = allActiveOrders.filter(o =>
      feedmillLines.some(line => normalizeLine(o.feedmill_line) === normalizeLine(line))
    );
    
    // Also get orders from OTHER lines that might be candidates to move INTO this feedmill
    const otherLineOrders = allActiveOrders.filter(o =>
      !feedmillLines.some(line => normalizeLine(o.feedmill_line) === normalizeLine(line))
    );
    
    setProcessingStats(prev => ({ ...prev, total: feedmillOrders.length }));
    
    // Build snapshot — only for this feedmill's lines
    const snapshot = {};
    const normalizedFeedmillLines = feedmillLines.map(l => normalizeLine(l));
    
    normalizedFeedmillLines.forEach(line => {
      snapshot[line] = allActiveOrders
        .filter(o => normalizeLine(o.feedmill_line) === line)
        .filter(o => !o.parent_id)
        .sort((a, b) => (a.priority_seq || 0) - (b.priority_seq || 0))
        .map(o => ({ ...JSON.parse(JSON.stringify(o)) }));
    });
    
    // Run auto-sequence — same algorithm but scoped to feedmill lines
    setProcessingPhase('combining');
    await new Promise(r => setTimeout(r, 200));
    
    const onProgress = (stats) => {
      setProcessingStats(prev => ({ ...prev, ...stats }));
    };
    
    const onPhaseChange = (phase) => {
      setProcessingPhase(phase);
    };
    
    // Use the same algorithm but pass only feedmill lines
    const { lineOrdersMap, combinedOrders, placementLog } = await runAutoSequenceWithProgress(
      [...feedmillOrders, ...otherLineOrders],  // All orders for scanning matches
      masterData, n10dData,
      normalizedFeedmillLines,  // Only place on feedmill's lines
      changeoverRules,
      onProgress, onPhaseChange,
      {
        // Scope: only move/place orders onto these lines
        targetLines: normalizedFeedmillLines,
        // But scan ALL lines for combinable matches
        scanAllLines: true,
        sourceOrders: feedmillOrders,
      }
    );
    
    // Build results
    setProcessingPhase('sequencing');
    await new Promise(r => setTimeout(r, 200));
    
    const results = {};
    normalizedFeedmillLines.forEach(line => {
      const lineOrders = lineOrdersMap[line] || [];
      if (lineOrders.length === 0) {
        results[line] = [];
        return;
      }
      
      const originalLineOrders = snapshot[line] || [];
      const withMovement = lineOrders.map((order, newIndex) => {
        const newPrio = newIndex + 1;
        const originalIndex = originalLineOrders.findIndex(o => o.id === order.id);
        let movement = 'same';
        let movementDelta = 0;
        
        if (originalIndex >= 0) {
          const originalPrio = originalIndex + 1;
          movementDelta = originalPrio - newPrio;
          if (movementDelta > 0) movement = 'up';
          else if (movementDelta < 0) movement = 'down';
        } else {
          movement = 'new_to_line';
        }
        
        return {
          ...order,
          prio: newPrio,
          _originalPrio: originalIndex >= 0 ? originalIndex + 1 : -1,
          _movement: movement,
          _movementDelta: Math.abs(movementDelta),
        };
      });
      
      results[line] = withMovement;
    });
    
    setProcessingPhase('done');
    await new Promise(r => setTimeout(r, 500));
    
    // Store and show preview
    setFeedmillSequenceSnapshot(snapshot);
    setFeedmillSequenceResults(results);
    setFeedmillPlacementLog(placementLog);
    setFeedmillCombinedOrders(combinedOrders);
    setCurrentFeedmillForPreview(feedmill);
    
    const stats = calculatePlantSummary(snapshot, results, combinedOrders, placementLog, normalizedFeedmillLines);
    setFeedmillSummaryStats(stats);
    
    setShowProcessingOverlay(false);
    setShowFeedmillAutoSequencePreview(true);
    
  } catch (error) {
    console.error('Feedmill auto-sequence error:', error);
    setShowProcessingOverlay(false);
    toast.error(`Auto-sequence failed: ${error.message}`);
  }
}
```
