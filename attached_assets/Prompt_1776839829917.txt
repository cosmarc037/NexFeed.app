## 1. UPDATE handleFeedmillAutoSequence — SAME AS PER-LINE BUT FOR MULTIPLE LINES

```javascript
async function handleFeedmillAutoSequence(feedmill) {
  const feedmillLines = getFeedmillLines(feedmill);
  
  if (feedmillLines.length === 0) {
    toast.error(`No lines found for ${feedmill}`);
    return;
  }
  
  const feedmillShortName = getFeedmillShortName(feedmill);
  const normalizedFeedmillLines = feedmillLines.map(l => normalizeLine(l));
  
  try {
    // Get all active orders
    const allActiveOrders = enrichedOrders.filter(
      o => o.status !== 'Done' && o.status !== 'Cancel PO' &&
           o.status !== 'In Production' && o.status !== 'On-going' &&
           !o.parent_id
    );
    
    // Step 1: Find candidates from OTHER lines that can move to this feedmill
    const feedmillOrders = allActiveOrders.filter(o =>
      normalizedFeedmillLines.includes(normalizeLine(o.feedmill_line))
    );
    
    const otherLineOrders = allActiveOrders.filter(o =>
      !normalizedFeedmillLines.includes(normalizeLine(o.feedmill_line))
    );
    
    // Step 2: Check which external orders can be produced on any of the feedmill's lines
    const candidates = [];
    
    for (const order of otherLineOrders) {
      // Skip Planned orders — they stay on their line
      if (order.status === 'Planned') continue;
      
      const materialCode = String(order.material_code_fg || '').trim();
      if (!materialCode) continue;
      
      // Check if this order can run on ANY of the feedmill's lines
      const eligibleFeedmillLines = normalizedFeedmillLines.filter(line =>
        canProduceOnLine(order, line, masterData)
      );
      
      if (eligibleFeedmillLines.length === 0) continue;
      
      const orderVolume = parseFloat(order.volume) || parseFloat(order.total_volume_mt) || 0;
      const currentLine = normalizeLine(order.feedmill_line);
      const currentLineRunRate = getLineRunRate(currentLine);
      const currentLineMT = allActiveOrders
        .filter(o => normalizeLine(o.feedmill_line) === currentLine)
        .reduce((sum, o) => sum + (parseFloat(o.volume) || parseFloat(o.total_volume_mt) || 0), 0);
      const currentQueueTime = currentLineRunRate > 0 ? currentLineMT / currentLineRunRate : Infinity;
      
      // Find best feedmill line for this order
      let bestLine = null;
      let bestQueueTime = Infinity;
      
      for (const fmLine of eligibleFeedmillLines) {
        const lineMT = feedmillOrders
          .filter(o => normalizeLine(o.feedmill_line) === fmLine)
          .reduce((sum, o) => sum + (parseFloat(o.volume) || parseFloat(o.total_volume_mt) || 0), 0);
        const lineRunRate = getLineRunRate(fmLine);
        const queueTime = lineRunRate > 0 ? lineMT / lineRunRate : Infinity;
        
        if (queueTime < bestQueueTime) {
          bestQueueTime = queueTime;
          bestLine = fmLine;
        }
      }
      
      // Only suggest if moving improves queue time
      if (bestLine && bestQueueTime < currentQueueTime) {
        candidates.push({
          order,
          volume: orderVolume,
          fromLine: currentLine,
          toBestLine: bestLine,
          currentQueueTime,
          bestQueueTime,
          improvement: currentQueueTime - bestQueueTime,
          eligibleLines: eligibleFeedmillLines,
        });
      }
    }
    
    // Step 3: Also look for combinable matches WITHIN and INTO feedmill
    const combineMatches = [];
    
    for (const fmOrder of feedmillOrders) {
      const materialCode = String(fmOrder.material_code_fg || '').trim();
      const formulaVersion = fmOrder.formula_version || '—';
      
      if (!materialCode) continue;
      
      // Look for matches from OTHER lines
      for (const extOrder of otherLineOrders) {
        if (extOrder.status === 'Planned') continue;
        
        const extMaterial = String(extOrder.material_code_fg || '').trim();
        const extFormula = extOrder.formula_version || '—';
        
        if (extMaterial !== materialCode || extFormula !== formulaVersion) continue;
        
        const fmVolume = parseFloat(fmOrder.volume_override || fmOrder.volume || fmOrder.total_volume_mt) || 0;
        const extVolume = parseFloat(extOrder.volume || extOrder.total_volume_mt) || 0;
        
        if (fmVolume + extVolume > 180) continue;
        
        // Check if external order can run on the feedmill order's line
        const fmLine = normalizeLine(fmOrder.feedmill_line);
        if (!canProduceOnLine(extOrder, fmLine, masterData)) continue;
        
        combineMatches.push({
          feedmillOrder: fmOrder,
          externalOrder: extOrder,
          combinedVolume: fmVolume + extVolume,
          targetLine: fmLine,
        });
      }
      
      // Also look for combinable orders WITHIN the feedmill's own lines
      for (const otherFmOrder of feedmillOrders) {
        if (otherFmOrder.id === fmOrder.id) continue;
        if (otherFmOrder._processed) continue;
        
        const otherMaterial = String(otherFmOrder.material_code_fg || '').trim();
        const otherFormula = otherFmOrder.formula_version || '—';
        
        if (otherMaterial !== materialCode || otherFormula !== formulaVersion) continue;
        
        const fmVolume = parseFloat(fmOrder.volume_override || fmOrder.volume || fmOrder.total_volume_mt) || 0;
        const otherVolume = parseFloat(otherFmOrder.volume_override || otherFmOrder.volume || otherFmOrder.total_volume_mt) || 0;
        
        if (fmVolume + otherVolume > 180) continue;
        
        combineMatches.push({
          feedmillOrder: fmOrder,
          externalOrder: otherFmOrder,
          combinedVolume: fmVolume + otherVolume,
          targetLine: normalizeLine(fmOrder.feedmill_line),
          isWithinFeedmill: true,
        });
      }
    }
    
    // Step 4: Also check if orders can be moved BETWEEN the feedmill's own lines
    const internalMoves = [];
    
    for (const fmOrder of feedmillOrders) {
      if (fmOrder.status === 'Planned') continue;
      
      const currentLine = normalizeLine(fmOrder.feedmill_line);
      const orderVolume = parseFloat(fmOrder.volume) || parseFloat(fmOrder.total_volume_mt) || 0;
      
      // Check other feedmill lines
      for (const otherLine of normalizedFeedmillLines) {
        if (otherLine === currentLine) continue;
        if (!canProduceOnLine(fmOrder, otherLine, masterData)) continue;
        
        const currentLineMT = feedmillOrders
          .filter(o => normalizeLine(o.feedmill_line) === currentLine)
          .reduce((sum, o) => sum + (parseFloat(o.volume) || parseFloat(o.total_volume_mt) || 0), 0);
        const otherLineMT = feedmillOrders
          .filter(o => normalizeLine(o.feedmill_line) === otherLine)
          .reduce((sum, o) => sum + (parseFloat(o.volume) || parseFloat(o.total_volume_mt) || 0), 0);
        
        const currentQT = getLineRunRate(currentLine) > 0 ? currentLineMT / getLineRunRate(currentLine) : Infinity;
        const otherQT = getLineRunRate(otherLine) > 0 ? otherLineMT / getLineRunRate(otherLine) : Infinity;
        
        if (otherQT < currentQT) {
          internalMoves.push({
            order: fmOrder,
            volume: orderVolume,
            fromLine: currentLine,
            toLine: otherLine,
            currentQueueTime: currentQT,
            newQueueTime: otherQT,
            improvement: currentQT - otherQT,
          });
        }
      }
    }
    
    // Step 5: If no candidates at all, show message
    if (candidates.length === 0 && combineMatches.length === 0 && internalMoves.length === 0) {
      toast.info(`No optimization opportunities found for ${feedmillShortName}. Lines are already optimal.`);
      return;
    }
    
    // Step 6: Run the full sequence with the identified opportunities
    // Combine all orders: feedmill orders + candidates to move in
    const allOrdersForSequence = [
      ...feedmillOrders,
      ...candidates.map(c => c.order),
    ];
    
    // Remove duplicates
    const uniqueOrders = [];
    const seenIds = new Set();
    for (const order of allOrdersForSequence) {
      if (!seenIds.has(order.id)) {
        uniqueOrders.push(order);
        seenIds.add(order.id);
      }
    }
    
    // Show processing overlay
    setProcessingPhase('scanning');
    setProcessingStats({ processed: 0, total: uniqueOrders.length, combined: 0, moved: 0 });
    setShowProcessingOverlay(true);
    setProcessingLabel(`${feedmillShortName} Auto-Sequence`);
    
    await new Promise(r => setTimeout(r, 100));
    
    const onProgress = (stats) => {
      setProcessingStats(prev => ({ ...prev, ...stats }));
    };
    
    const onPhaseChange = (phase) => {
      setProcessingPhase(phase);
    };
    
    // Run auto-sequence scoped to feedmill lines
    const { lineOrdersMap, combinedOrders, placementLog } = await runAutoSequenceWithProgress(
      uniqueOrders,
      masterData, n10dData,
      normalizedFeedmillLines,
      changeoverRules,
      onProgress, onPhaseChange
    );
    
    // Build snapshot and results
    const snapshot = {};
    normalizedFeedmillLines.forEach(line => {
      snapshot[line] = allActiveOrders
        .filter(o => normalizeLine(o.feedmill_line) === line && !o.parent_id)
        .sort((a, b) => (a.priority_seq || 0) - (b.priority_seq || 0))
        .map(o => ({ ...JSON.parse(JSON.stringify(o)) }));
    });
    
    const results = {};
    normalizedFeedmillLines.forEach(line => {
      const lineOrders = lineOrdersMap[line] || [];
      if (lineOrders.length === 0) {
        results[line] = [];
        return;
      }
      
      const originalLineOrders = snapshot[line] || [];
      const withMovement = lineOrders.map((order, newIndex) => {
        const newPrio = newIndex + 1;
        const originalIndex = originalLineOrders.findIndex(o => o.id === order.id);
        let movement = 'same';
        let movementDelta = 0;
        
        if (originalIndex >= 0) {
          const originalPrio = originalIndex + 1;
          movementDelta = originalPrio - newPrio;
          if (movementDelta > 0) movement = 'up';
          else if (movementDelta < 0) movement = 'down';
        } else {
          movement = 'new_to_line';
        }
        
        return {
          ...order,
          prio: newPrio,
          _originalPrio: originalIndex >= 0 ? originalIndex + 1 : -1,
          _movement: movement,
          _movementDelta: Math.abs(movementDelta),
        };
      });
      
      results[line] = withMovement;
    });
    
    setProcessingPhase('done');
    await new Promise(r => setTimeout(r, 500));
    
    const stats = calculatePlantSummary(snapshot, results, combinedOrders, placementLog, normalizedFeedmillLines);
    
    setShowProcessingOverlay(false);
    
    // Show preview using the same PlantAutoSequencePreview component
    setFeedmillSequenceSnapshot(snapshot);
    setFeedmillSequenceResults(results);
    setFeedmillSummaryStats(stats);
    setFeedmillPlacementLog(placementLog);
    setFeedmillCombinedOrders(combinedOrders);
    setCurrentFeedmillForPreview(feedmill);
    setShowFeedmillAutoSequencePreview(true);
    
  } catch (error) {
    console.error('Feedmill auto-sequence error:', error);
    setShowProcessingOverlay(false);
    toast.error(`Auto-sequence failed: ${error.message}`);
  }
}
```

### Per-feedmill does THREE things
1. ✅ Finds orders from OTHER lines that can move INTO the feedmill's lines

4. ✅ Combines orders WITHIN the feedmill's own lines
