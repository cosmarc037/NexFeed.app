## 1. HANDLER — FIND CANDIDATES FOR THE FEEDMILL'S LINES

```javascript
async function handleFeedmillAutoSequence(feedmill) {
  const feedmillLines = getFeedmillLines(feedmill);
  const feedmillShortName = getFeedmillShortName(feedmill);
  const normalizedFeedmillLines = feedmillLines.map(l => normalizeLine(l));
  
  // Get all active orders
  const allActiveOrders = enrichedOrders.filter(
    o => o.status !== 'Done' && o.status !== 'Cancel PO' &&
         o.status !== 'In Production' && o.status !== 'On-going' &&
         !o.parent_id
  );
  
  // Orders currently on the feedmill's lines
  const feedmillOrders = allActiveOrders.filter(o =>
    normalizedFeedmillLines.includes(normalizeLine(o.feedmill_line))
  );
  
  // Orders on OTHER lines (outside this feedmill)
  const otherLineOrders = allActiveOrders.filter(o =>
    !normalizedFeedmillLines.includes(normalizeLine(o.feedmill_line))
  );
  
  // Find candidates: orders from other lines that CAN be produced on any feedmill line
  const candidates = [];
  
  for (const order of otherLineOrders) {
    if (order.status === 'Planned') continue;
    
    const materialCode = String(order.material_code_fg || '').trim();
    if (!materialCode) continue;
    
    // Check which feedmill lines can produce this order
    const eligibleLines = normalizedFeedmillLines.filter(line =>
      canProduceOnLine(order, line, masterData)
    );
    
    if (eligibleLines.length === 0) continue;
    
    const orderVolume = parseFloat(order.volume) || parseFloat(order.total_volume_mt) || 0;
    const currentLine = normalizeLine(order.feedmill_line);
    const currentLineRunRate = getLineRunRate(currentLine);
    const currentLineMT = allActiveOrders
      .filter(o => normalizeLine(o.feedmill_line) === currentLine)
      .reduce((sum, o) => sum + (parseFloat(o.volume) || parseFloat(o.total_volume_mt) || 0), 0);
    const currentQueueTime = currentLineRunRate > 0 ? currentLineMT / currentLineRunRate : Infinity;
    
    // Find best feedmill line
    let bestLine = null;
    let bestQueueTime = Infinity;
    
    for (const fmLine of eligibleLines) {
      const lineMT = feedmillOrders
        .filter(o => normalizeLine(o.feedmill_line) === fmLine)
        .reduce((sum, o) => sum + (parseFloat(o.volume) || parseFloat(o.total_volume_mt) || 0), 0);
      const lineRunRate = getLineRunRate(fmLine);
      const queueTime = lineRunRate > 0 ? lineMT / lineRunRate : Infinity;
      
      if (queueTime < bestQueueTime) {
        bestQueueTime = queueTime;
        bestLine = fmLine;
      }
    }
    
    if (!bestLine) continue;
    
    // Calculate improvement
    const improvement = currentQueueTime - bestQueueTime;
    
    candidates.push({
      order,
      volume: orderVolume,
      fromLine: currentLine,
      bestLine,
      eligibleLines,
      currentQueueTime,
      bestQueueTime,
      improvement,
      selected: improvement > 0, // Auto-select if queue improves
    });
  }
  
  // Sort: biggest improvement first
  candidates.sort((a, b) => b.improvement - a.improvement);
  
  if (candidates.length === 0) {
    toast.info(`No orders from other lines can be moved to ${feedmillShortName}. Lines are already optimal.`);
    return;
  }
  
  // Store and show preview
  setFeedmillAutoSequenceData({
    feedmill,
    feedmillShortName,
    feedmillLines: normalizedFeedmillLines,
    currentOrders: feedmillOrders,
    candidates,
  });
  setShowFeedmillAutoSequencePreview(true);
}
```

## 2. FEEDMILL AUTO-SEQUENCE PREVIEW COMPONENT — SAME UI AS PER-LINE

```jsx
function FeedmillAutoSequencePreview({
  feedmillData,
  allOrders,
  masterData,
  n10dData,
  changeoverRules,
  onApplySequence,
  onCancel
}) {
  const { feedmill, feedmillShortName, feedmillLines, currentOrders, candidates: initialCandidates } = feedmillData;
  
  const [candidates, setCandidates] = useState(initialCandidates);
  const [sequencedResults, setSequencedResults] = useState(null);
  const [isCombined, setIsCombined] = useState(false);
  const [isCombining, setIsCombining] = useState(false);
  const [isApplying, setIsApplying] = useState(false);
  
  const selectedCandidates = candidates.filter(c => c.selected);
  
  // Build before snapshot — orders currently on feedmill lines
  const beforeOrders = {};
  feedmillLines.forEach(line => {
    beforeOrders[line] = currentOrders
      .filter(o => normalizeLine(o.feedmill_line) === line)
      .sort((a, b) => (a.priority_seq || 0) - (b.priority_seq || 0));
  });
  
  const totalCurrentOrders = currentOrders.length;
  
  // Build after — current orders + selected candidates, assigned to best lines
  useEffect(() => {
    buildAfterSequence();
  }, [candidates]);
  
  function buildAfterSequence() {
    const afterOrders = {};
    
    // Start with current orders per line
    feedmillLines.forEach(line => {
      afterOrders[line] = currentOrders
        .filter(o => normalizeLine(o.feedmill_line) === line)
        .map(o => ({ ...o, _movement: 'same', _movementDelta: 0 }));
    });
    
    // Add selected candidates to their best lines
    for (const candidate of selectedCandidates) {
      const targetLine = candidate.bestLine;
      if (!afterOrders[targetLine]) afterOrders[targetLine] = [];
      
      afterOrders[targetLine].push({
        ...candidate.order,
        feedmill_line: targetLine,
        line: targetLine,
        _movement: 'new_to_line',
        _movedFromLine: candidate.fromLine,
        _movementDelta: 0,
      });
    }
    
    // Sort each line and assign prio
    feedmillLines.forEach(line => {
      const orders = afterOrders[line] || [];
      
      // Run preSortOrders for N10D dates
      const categorized = preSortOrders(orders, n10dData, masterData);
      const sorted = autoSequenceSort(categorized, changeoverRules);
      
      // Calculate changeovers
      calculatePreviewChangeovers(sorted, changeoverRules);
      
      // Assign prio and movement
      const originalLineOrders = beforeOrders[line] || [];
      sorted.forEach((order, index) => {
        order.prio = index + 1;
        
        if (order._movement !== 'new_to_line') {
          const originalIndex = originalLineOrders.findIndex(o => o.id === order.id);
          if (originalIndex >= 0) {
            const originalPrio = originalIndex + 1;
            const delta = originalPrio - order.prio;
            order._originalPrio = originalPrio;
            order._movement = delta > 0 ? 'up' : delta < 0 ? 'down' : 'same';
            order._movementDelta = Math.abs(delta);
          }
        }
      });
      
      afterOrders[line] = sorted;
    });
    
    setSequencedResults(afterOrders);
  }
  
  function toggleCandidate(index) {
    setCandidates(prev => prev.map((c, i) => 
      i === index ? { ...c, selected: !c.selected } : c
    ));
  }
  
  function selectAll() {
    setCandidates(prev => prev.map(c => ({ ...c, selected: true })));
  }
  
  function deselectAll() {
    setCandidates(prev => prev.map(c => ({ ...c, selected: false })));
  }
  
  async function handleCombineOrders() {
    if (isCombining || !sequencedResults) return;
    setIsCombining(true);
    
    // Run combine logic on the after orders
    const allAfterOrders = [];
    feedmillLines.forEach(line => {
      (sequencedResults[line] || []).forEach(o => {
        allAfterOrders.push({ ...o, line, feedmill_line: line });
      });
    });
    
    const { lineOrdersMap, combinedOrders, placementLog } = plantLevelCombineOnly(
      allAfterOrders, masterData, n10dData, feedmillLines
    );
    
    // Re-sequence each line
    const results = {};
    feedmillLines.forEach(line => {
      const lineOrders = lineOrdersMap[line] || [];
      if (lineOrders.length === 0) {
        results[line] = [];
        return;
      }
      
      const sequenced = runAutoSequence(lineOrders, changeoverRules, masterData, n10dData);
      calculatePreviewChangeovers(sequenced, changeoverRules);
      
      const originalLineOrders = beforeOrders[line] || [];
      sequenced.forEach((order, index) => {
        order.prio = index + 1;
        const originalIndex = originalLineOrders.findIndex(o => o.id === order.id);
        if (originalIndex >= 0) {
          const delta = (originalIndex + 1) - order.prio;
          order._originalPrio = originalIndex + 1;
          order._movement = delta > 0 ? 'up' : delta < 0 ? 'down' : 'same';
          order._movementDelta = Math.abs(delta);
        } else if (!order._movement) {
          order._movement = 'new_to_line';
        }
      });
      
      results[line] = sequenced;
    });
    
    setSequencedResults(results);
    setIsCombined(true);
    setIsCombining(false);
  }
  
  const totalAfterOrders = feedmillLines.reduce((sum, line) => 
    sum + (sequencedResults?.[line]?.length || 0), 0
  );
  const totalFromOtherLines = selectedCandidates.length;
  
  return (
    <div className="auto-sequence-overlay" onClick={e => e.stopPropagation()}>
      <div className="auto-sequence-modal" onClick={e => e.stopPropagation()}>
        {/* Header */}
        <div className="auto-sequence-header">
          <div>
            <h2>✨ Auto-Sequence — {feedmillShortName}</h2>
            <span className="auto-sequence-subtitle">
              Find orders from other lines that can be moved to {feedmill} ({feedmillLines.map(l => getLineShortLabel(l)).join(' & ')})
            </span>
          </div>
          <button className="auto-sequence-close" onClick={onCancel} disabled={isApplying}>✕</button>
        </div>
        
        {/* Stats */}
        <div className="auto-sequence-stats">
          <StatBadge label="Current Orders" value={totalCurrentOrders} />
          <StatBadge label="Candidates Found" value={candidates.length} color="blue" />
          <StatBadge label="Selected to Move" value={selectedCandidates.length} color="red" />
          <StatBadge label="After" value={totalAfterOrders} />
        </div>
        
        {/* Candidate list */}
        <div className="line-as-candidates">
          <div className="line-as-candidates-header">
            <span className="line-as-candidates-icon">📋</span>
            <span className="line-as-candidates-title">
              Orders available to move to {feedmillShortName}
            </span>
            <div className="line-as-candidates-actions">
              <button className="line-as-select-all" onClick={selectAll}>Select All</button>
              <button className="line-as-deselect-all" onClick={deselectAll}>Deselect All</button>
            </div>
          </div>
          
          <div className="line-as-candidates-list">
            {candidates.map((candidate, index) => (
              <label key={candidate.order.id} className="line-as-candidate-row">
                <input
                  type="checkbox"
                  checked={candidate.selected}
                  onChange={() => toggleCandidate(index)}
                />
                <span className={`line-as-from-badge line-as-from-${getLineShortLabel(candidate.fromLine).toLowerCase()}`}>
                  {getLineShortLabel(candidate.fromLine)}
                </span>
                <span className="line-as-candidate-name">
                  {candidate.order.item_description}
                </span>
                <span className="line-as-candidate-target">
                  → {getLineShortLabel(candidate.bestLine)}
                </span>
                <span className="line-as-candidate-volume">
                  {candidate.volume} MT
                </span>
                <span className="line-as-candidate-improvement">
                  ↓ {candidate.improvement.toFixed(2)} hrs
                </span>
              </label>
            ))}
            
            {candidates.length === 0 && (
              <div className="line-as-no-candidates">
                No orders from other lines can be moved to {feedmillShortName}
              </div>
            )}
          </div>
        </div>
        
        {/* Per-line tabs for Before / After */}
        <FeedmillBeforeAfterTabs
          feedmillLines={feedmillLines}
          beforeOrders={beforeOrders}
          afterOrders={sequencedResults}
          allOrders={allOrders}
          changeoverRules={changeoverRules}
        />
        
        {/* Applying overlay */}
        {isApplying && (
          <div className="auto-sequence-applying-overlay">
            <div className="auto-sequence-applying-content">
              <div className="auto-sequence-applying-spinner"></div>
              <span className="auto-sequence-applying-text">Applying to schedule...</span>
            </div>
          </div>
        )}
        
        {/* Footer */}
        <div className="auto-sequence-footer">
          <div className="auto-sequence-footer-left">
            {!isCombined ? (
              <button
                className={`btn-combine-orders ${isCombining ? 'btn-combine-loading' : ''}`}
                onClick={handleCombineOrders}
                disabled={isCombining || isApplying || selectedCandidates.length === 0}
              >
                {isCombining ? (
                  <><span className="btn-apply-spinner"></span> Combining...</>
                ) : (
                  '🔗 Combine Orders'
                )}
              </button>
            ) : (
              <button
                className="btn-uncombine-orders"
                onClick={() => { setIsCombined(false); buildAfterSequence(); }}
                disabled={isApplying}
              >
                ↩ Uncombine Orders
              </button>
            )}
          </div>
          <div className="auto-sequence-footer-right">
            <button className="btn-cancel" onClick={onCancel} disabled={isApplying}>
              Cancel
            </button>
            <button
              className={`btn-apply ${isApplying ? 'btn-apply-loading' : ''}`}
              onClick={async () => {
                if (isApplying) return;
                setIsApplying(true);
                await onApplySequence(sequencedResults);
                setIsApplying(false);
              }}
              disabled={isApplying || selectedCandidates.length === 0}
            >
              {isApplying ? (
                <><span className="btn-apply-spinner"></span> Applying...</>
              ) : (
                'Apply to Schedule'
              )}
            </button>
          </div>
        </div>
      </div>
    </div>
  );
}
```
