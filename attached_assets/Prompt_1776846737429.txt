```jsx
          order._movement = order._movedFromLine ? 'new_to_line' : 'same';
        }
      });
      
      marginResults[line] = sorted;
    });
    
    setSequencedResults(marginResults);
```

## 2. SORTING FUNCTION WITH MARGIN

```javascript
function sortWithMargin(orders) {
  // Sorting criteria:
  // 1. Category C (Critical/Urgent non-dated) — always at the top
  // 2. Category A (dated) — by avail date, THEN by margin within same date
  // 3. Category B (N10D targeted) — by inferred date, THEN by margin within same date
  // 4. Category D (stock sufficient) — by margin (higher first), then by date
  // 5. Category E (no data) — by margin (higher first)
  
  const catC = orders.filter(o => o._category === 'C')
    .sort((a, b) => {
      if (a._n10dStatus === 'Critical' && b._n10dStatus !== 'Critical') return -1;
      if (a._n10dStatus !== 'Critical' && b._n10dStatus === 'Critical') return 1;
      // Same urgency — higher margin first
      return (b._margin || 0) - (a._margin || 0);
    });
  
  const catA = orders.filter(o => o._category === 'A')
    .sort((a, b) => {
      // First by date
      const dateA = a._sortDate || new Date(a.target_avail_date);
      const dateB = b._sortDate || new Date(b.target_avail_date);
      const dateDiff = dateA - dateB;
      
      // If same date — higher margin first
      if (isSameDay(dateA, dateB)) {
        return (b._margin || 0) - (a._margin || 0);
      }
      
      return dateDiff;
    });
  
  const catB = orders.filter(o => o._category === 'B')
    .sort((a, b) => {
      const dateA = a._sortDate || new Date(a._targetDate);
      const dateB = b._sortDate || new Date(b._targetDate);
      const dateDiff = dateA - dateB;
      
      if (isSameDay(dateA, dateB)) {
        return (b._margin || 0) - (a._margin || 0);
      }
      
      return dateDiff;
    });
  
  const catD = orders.filter(o => o._category === 'D')
    .sort((a, b) => {
      // For stock sufficient: margin first, then date
      const marginDiff = (b._margin || 0) - (a._margin || 0);
      if (Math.abs(marginDiff) > 0.01) return marginDiff;
      
      // Same margin — by date if available
      const dateA = a._sortDate || new Date('2099-12-31');
      const dateB = b._sortDate || new Date('2099-12-31');
      return dateA - dateB;
    });
  
  const catE = orders.filter(o => o._category === 'E')
    .sort((a, b) => {
      // No N10D data: margin first, then by avail date if exists
      const marginDiff = (b._margin || 0) - (a._margin || 0);
      if (Math.abs(marginDiff) > 0.01) return marginDiff;
      
      const dateA = a._sortDate || new Date('2099-12-31');
      const dateB = b._sortDate || new Date('2099-12-31');
      return dateA - dateB;
    });
  
  const sorted = [...catC, ...catA, ...catB, ...catD, ...catE];
  
  sorted.forEach((order, index) => {
    order.prio = index + 1;
    order.priority_seq = index + 1;
  });
  
  return sorted;
}

function isSameDay(dateA, dateB) {
  if (!dateA || !dateB) return false;
  return dateA.getFullYear() === dateB.getFullYear() &&
         dateA.getMonth() === dateB.getMonth() &&
         dateA.getDate() === dateB.getDate();
}
```
