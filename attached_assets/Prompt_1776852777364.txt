```javascript
function sortWithMargin(orders, masterData, n10dData) {
  
  // Step 1: Resolve date and margin for EVERY order
  const enriched = orders.map(order => {
    const materialCode = String(order.material_code_fg || '').trim();
    
    // Resolve margin from master data
    const mdEntry = masterData.find(md =>
      String(md.material_code_fg || '').trim() === materialCode
    );
    const margin = mdEntry ? (parseFloat(mdEntry.margin) || 0) : 0;
    const hasMarginData = mdEntry && mdEntry.margin != null && mdEntry.margin !== '';
    
    // Resolve date — try three sources in order
    const resolvedDate = resolveOrderDate(order, n10dData, materialCode);
    
    return {
      ...order,
      _margin: margin,
      _hasMarginData: hasMarginData,
      _resolvedDate: resolvedDate.date,
      _dateSource: resolvedDate.source,
      _dateCategory: resolvedDate.category,
    };
  });
  
  // Step 2: Sort
  enriched.sort((a, b) => {
    // Category priority: Critical/Urgent first
    const catOrder = getCategoryPriority(a._dateCategory) - getCategoryPriority(b._dateCategory);
    if (catOrder !== 0) return catOrder;
    
    // Within same category: sort by date ascending
    if (a._resolvedDate && b._resolvedDate) {
      const dateA = new Date(a._resolvedDate);
      const dateB = new Date(b._resolvedDate);
      
      // Different dates — earlier first
      if (!isSameDay(dateA, dateB)) {
        return dateA - dateB;
      }
      
      // SAME DATE — higher margin first
      if (a._hasMarginData || b._hasMarginData) {
        const marginDiff = (b._margin || 0) - (a._margin || 0);
        if (Math.abs(marginDiff) > 0.01) return marginDiff;
      }
      
      // Same date and same margin — keep original order
      return 0;
    }
    
    // One has date, one doesn't — dated first
    if (a._resolvedDate && !b._resolvedDate) return -1;
    if (!a._resolvedDate && b._resolvedDate) return 1;
    
    // Neither has date — higher margin first
    if (a._hasMarginData || b._hasMarginData) {
      const marginDiff = (b._margin || 0) - (a._margin || 0);
      if (Math.abs(marginDiff) > 0.01) return marginDiff;
    }
    
    return 0;
  });
  
  // Step 3: Reassign prio
  enriched.forEach((order, index) => {
    order.prio = index + 1;
    order.priority_seq = index + 1;
  });
  
  return enriched;
}
```

```javascript
function resolveOrderDate(order, n10dData, materialCode) {
  // Source 1: Actual avail date on the order (hard date from SAP)
  const availDate = order.target_avail_date;
  if (availDate && isValidISODate(availDate)) {
    return {
      date: availDate,
      source: 'avail_date',
      category: 'dated',
    };
  }
  
  // Source 2: N10D lookup — inferred target date
  if (n10dData && n10dData.length > 0 && materialCode) {
    const n10dEntry = n10dData.find(n =>
      String(n.material_code || n.material_code_fg || '').trim() === materialCode
    );
    
    if (n10dEntry) {
      const inferredDate = n10dEntry.inferred_target_date || n10dEntry.target_date;
      const n10dStatus = n10dEntry.status || n10dEntry.stock_status || '';
      
      if (inferredDate && isValidISODate(inferredDate)) {
        // Has a real inferred date
        const isCritical = n10dStatus === 'Critical';
        const isUrgent = n10dStatus === 'Urgent';
        
        return {
          date: inferredDate,
          source: 'n10d',
          category: isCritical ? 'critical' : isUrgent ? 'urgent' : 'n10d_dated',
        };
      }
      
      // N10D entry exists but no date — check status
      if (n10dStatus === 'Critical') {
        return { date: null, source: 'n10d_status', category: 'critical' };
      }
      if (n10dStatus === 'Urgent') {
        return { date: null, source: 'n10d_status', category: 'urgent' };
      }
      if (n10dStatus === 'Sufficient' || n10dStatus === 'Safe' || n10dStatus === 'OK') {
        return { date: null, source: 'n10d_status', category: 'sufficient' };
      }
    }
  }
  
  // Source 3: Fallback — existing _effectiveDate or _targetDate on the order
  const fallbackDate = order._effectiveDate || order._targetDate;
  if (fallbackDate) {
    const dateStr = fallbackDate instanceof Date 
      ? fallbackDate.toISOString().split('T')[0]
      : String(fallbackDate);
    
    if (isValidISODate(dateStr)) {
      return {
        date: dateStr,
        source: 'fallback',
        category: order._category === 'C' ? 'critical' : 
                  order._category === 'B' ? 'n10d_dated' :
                  order._category === 'D' ? 'sufficient' : 'dated',
      };
    }
  }
  
  // No date at all
  return {
    date: null,
    source: 'none',
    category: 'no_data',
  };
}

function isValidISODate(str) {
  if (!str) return false;
  const s = String(str).trim();
  // Must look like a date, not "prio replenish" or "safety stocks"
  if (!/^\d{4}-\d{2}-\d{2}/.test(s)) return false;
  const d = new Date(s);
  return !isNaN(d.getTime());
}

function getCategoryPriority(category) {
  switch (category) {
    case 'critical': return 0;
    case 'urgent': return 1;
    case 'dated': return 2;
    case 'n10d_dated': return 3;
    case 'sufficient': return 4;
    case 'no_data': return 5;
    default: return 5;
  }
}

function isSameDay(dateA, dateB) {
  if (!dateA || !dateB) return false;
  if (isNaN(dateA.getTime()) || isNaN(dateB.getTime())) return false;
  return dateA.getFullYear() === dateB.getFullYear() &&
         dateA.getMonth() === dateB.getMonth() &&
         dateA.getDate() === dateB.getDate();
}
```
