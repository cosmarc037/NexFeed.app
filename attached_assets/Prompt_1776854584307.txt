function sortWithMargin(orders, masterData) {
  // Step 1: Enrich each order with its margin from master data
  const enriched = orders.map(order => {
    const materialCode = String(order.material_code_fg || '').trim();
    const mdEntry = masterData.find(md =>
      String(md.material_code_fg || '').trim() === materialCode
    );
    const margin = mdEntry ? (parseFloat(mdEntry.margin) || 0) : 0;
    const hasMarginData = mdEntry && mdEntry.margin != null && mdEntry.margin !== '';
    
    return {
      ...order,
      _margin: margin,
      _hasMarginData: hasMarginData,
    };
  });
  
  // Step 2: Group orders by their display date
  // Orders are ALREADY sorted correctly (by avail date, N10D, urgency, etc.)
  // We just need to find groups with the SAME date and re-sort within those groups
  
  const groups = [];
  let currentGroup = [];
  let currentDateKey = null;
  
  for (const order of enriched) {
    const dateKey = getOrderDateKey(order);
    
    if (dateKey !== currentDateKey) {
      if (currentGroup.length > 0) {
        groups.push(currentGroup);
      }
      currentGroup = [order];
      currentDateKey = dateKey;
    } else {
      currentGroup.push(order);
    }
  }
  if (currentGroup.length > 0) {
    groups.push(currentGroup);
  }
  
  // Step 3: Within each group, sort by margin descending
  const result = [];
  for (const group of groups) {
    if (group.length > 1) {
      group.sort((a, b) => (b._margin || 0) - (a._margin || 0));
    }
    result.push(...group);
  }
  
  // Step 4: Reassign prio
  result.forEach((order, index) => {
    order.prio = index + 1;
    order.priority_seq = index + 1;
  });
  
  return result;
}

function getOrderDateKey(order) {
  // Get the date that this order is sorted by
  // This could be target_avail_date, _targetDate, or a status string
  
  const availDate = order.target_avail_date;
  
  // If it's a real date, extract YYYY-MM-DD
  if (availDate && /^\d{4}-\d{2}-\d{2}/.test(String(availDate).trim())) {
    return String(availDate).trim().substring(0, 10);
  }
  
  // If it has an inferred target date
  if (order._targetDate && /^\d{4}-\d{2}-\d{2}/.test(String(order._targetDate).trim())) {
    return String(order._targetDate).trim().substring(0, 10);
  }
  
  // If it has _sortDate
  if (order._sortDate && order._sortDate instanceof Date && !isNaN(order._sortDate.getTime())) {
    return order._sortDate.toISOString().substring(0, 10);
  }
  
  // Not a date — use the display string as the key
  // This groups "safety stocks" together, "prio replenish" together, etc.
  return String(availDate || order._availDateDisplay || 'no_date').trim().toLowerCase();
}
