## 2. GROUP CHILD ORDERS TOGETHER IN ORDER HISTORY

When a combined order's children are transferred to history, they should stay grouped (visually indented/connected). In the Order History table, add visual grouping:

```jsx
// In OrderHistoryTable.jsx — render sub-orders grouped with lead

function renderOrderHistoryRow(order, index, allOrders) {
  // Check if this order is a child (has parent_id)
  const isChild = order.parent_id != null;
  const leadOrder = isChild 
    ? allOrders.find(o => o.id === order.parent_id)
    : null;
  
  // If this is a lead order with children, render it and all children together
  if (!isChild && order.original_order_ids && order.original_order_ids.length > 0) {
    const children = allOrders.filter(o => o.parent_id === order.id);
    
    return (
      <React.Fragment key={order.id}>
        {/* Lead row */}
        <tr className="order-history-row order-history-row-lead">
          <td className="order-history-indicator">🔗</td>
          {/* ... rest of columns ... */}
        </tr>
        
        {/* Child rows — grouped together, indented */}
        {children.map((child, childIndex) => (
          <tr key={child.id} className="order-history-row order-history-row-sub">
            <td className="order-history-indicator">
              {childIndex === children.length - 1 ? '└─' : '├─'}
            </td>
            {/* ... rest of columns ... */}
          </tr>
        ))}
      </React.Fragment>
    );
  }
  
  // If this is a child, skip it (already rendered above)
  if (isChild) return null;
  
  // Regular order
  return (
    <tr key={order.id} className="order-history-row">
      <td className="order-history-indicator"></td>
      {/* ... rest of columns ... */}
    </tr>
  );
}
```
