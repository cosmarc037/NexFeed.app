```jsx
// In OrderTable.jsx

function OrderTableRow({ order, onOrderUpdate }) {
  const [editingField, setEditingField] = useState(null);
  const [editValue, setEditValue] = useState('');
  const inputRef = useRef(null);
  
  // Close edit when clicking outside
  useEffect(() => {
    function handleClickOutside(e) {
      if (inputRef.current && !inputRef.current.contains(e.target)) {
        setEditingField(null);
      }
    }
    
    if (editingField) {
      document.addEventListener('mousedown', handleClickOutside);
      return () => document.removeEventListener('mousedown', handleClickOutside);
    }
  }, [editingField]);
  
  async function handleFieldEdit(fieldKey) {
    setEditingField(fieldKey);
    setEditValue(order[fieldKey] || '');
    setTimeout(() => inputRef.current?.focus(), 0);
  }
  
  async function handleFieldChange(e, fieldKey) {
    const newValue = e.target.value;
    setEditValue(newValue);
    
    // Auto-save immediately
    try {
      await updateOrderMutation.mutateAsync({
        id: order.id,
        data: { [fieldKey]: newValue }
      });
      setEditingField(null);
      toast.success(`${fieldKey} updated`);
    } catch (error) {
      toast.error(`Update failed: ${error.message}`);
    }
  }
  
  return (
    <tr className="order-table-row">
      {/* ... other columns ... */}
      
      {/* Planned Order FG */}
      <td className="order-table-cell order-table-editable">
        {editingField === 'planned_order_fg' ? (
          <input
            ref={inputRef}
            type="text"
            value={editValue}
```

```jsx
// In MasterDataFilter.jsx

function MasterDataFilter({ onFilter }) {
  const [searchTerm, setSearchTerm] = useState('');
  const [selectedCategories, setSelectedCategories] = useState([]);
  const [selectedColors, setSelectedColors] = useState([]);
  const [selectedDiameters, setSelectedDiameters] = useState([]);
  const [selectedDescriptions, setSelectedDescriptions] = useState([]);
  
  const uniqueCategories = [...new Set(masterData.map(m => m.category))].filter(Boolean).sort();
  const uniqueColors = [...new Set(masterData.map(m => m.color))].filter(Boolean).sort();
  const uniqueDiameters = [...new Set(masterData.map(m => m.diameter))].filter(Boolean).sort();
  const uniqueDescriptions = [...new Set(masterData.map(m => m.item_description))].filter(Boolean).sort();
  
  function applyFilters() {
    const filtered = masterData.filter(item => {
      // Search term (OR match across multiple fields)
      let matchesSearch = true;
      if (searchTerm) {
        const term = searchTerm.toLowerCase();
        matchesSearch =
          (item.item_description || '').toLowerCase().includes(term) ||
          (item.color || '').toLowerCase().includes(term) ||
          (item.diameter || '').toLowerCase().includes(term) ||
          (item.category || '').toLowerCase().includes(term) ||
          (item.form || '').toLowerCase().includes(term);
      }
      
      // Category filter (AND — all selected must match)
      const matchesCategory = selectedCategories.length === 0 || selectedCategories.includes(item.category);
      
      // Color filter
      const matchesColor = selectedColors.length === 0 || selectedColors.includes(item.color);
      
      // Diameter filter
      const matchesDiameter = selectedDiameters.length === 0 || selectedDiameters.includes(item.diameter);
      
      // Description filter
      const matchesDescription = selectedDescriptions.length === 0 || selectedDescriptions.includes(item.item_description);
      
      return matchesSearch && matchesCategory && matchesColor && matchesDiameter && matchesDescription;
    });
    
    onFilter(filtered);
  }
  
  useEffect(() => {
    applyFilters();
  }, [searchTerm, selectedCategories, selectedColors, selectedDiameters, selectedDescriptions]);
  
  function toggleCategory(category) {
    setSelectedCategories(prev =>
      prev.includes(category) ? prev.filter(c => c !== category) : [...prev, category]
    );
  }
  
  function toggleColor(color) {
    setSelectedColors(prev =>
      prev.includes(color) ? prev.filter(c => c !== color) : [...prev, color]
    );
  }
  
  function toggleDiameter(diameter) {
    setSelectedDiameters(prev =>
      prev.includes(diameter) ? prev.filter(d => d !== diameter) : [...prev, diameter]
    );
  }
  
  function toggleDescription(desc) {
    setSelectedDescriptions(prev =>
      prev.includes(desc) ? prev.filter(d => d !== desc) : [...prev, desc]
    );
  }
  
  return (
    <div className="master-data-filters">
      {/* Search */}
      <div className="filter-group">
        <input
          type="text"
          placeholder="Search by item, color, diameter, category, or form..."
          value={searchTerm}
          onChange={(e) => setSearchTerm(e.target.value)}
          className="filter-search-input"
        />
      </div>
      
      {/* Category Filter */}
      <div className="filter-group">
        <label className="filter-group-label">Category</label>
        <div className="filter-checkboxes">
          {uniqueCategories.map(cat => (
            <label key={cat} className="filter-checkbox">
              <input
                type="checkbox"
                checked={selectedCategories.includes(cat)}
                onChange={() => toggleCategory(cat)}
              />
              <span>{cat}</span>
            </label>
          ))}
        </div>
      </div>
      
      {/* Color Filter */}
      <div className="filter-group">
        <label className="filter-group-label">Color</label>
        <div className="filter-checkboxes">
          {uniqueColors.map(color => (
            <label key={color} className="filter-checkbox">
              <input
                type="checkbox"
                checked={selectedColors.includes(color)}
                onChange={() => toggleColor(color)}
              />
              <span>
                <span
                  className="color-dot"
                  style={{background: getColorHex(color)}}
                ></span>
                {color}
              </span>
            </label>
          ))}
        </div>
      </div>
      
      {/* Diameter Filter */}
      <div className="filter-group">
        <label className="filter-group-label">Diameter</label>
        <div className="filter-checkboxes">
          {uniqueDiameters.map(diameter => (
            <label key={diameter} className="filter-checkbox">
              <input
                type="checkbox"
                checked={selectedDiameters.includes(diameter)}
                onChange={() => toggleDiameter(diameter)}
              />
              <span>{diameter}mm</span>
            </label>
          ))}
        </div>
      </div>
      
      {/* Item Description Filter */}
      <div className="filter-group">
        <label className="filter-group-label">Item Description</label>
        <div className="filter-checkboxes filter-checkboxes-scroll">
          {uniqueDescriptions.map(desc => (
            <label key={desc} className="filter-checkbox">
              <input
                type="checkbox"
                checked={selectedDescriptions.includes(desc)}
                onChange={() => toggleDescription(desc)}
              />
              <span>{desc}</span>
            </label>
          ))}
        </div>
      </div>
    </div>
  );
}
```

```jsx
// In Navbar.jsx — grouped theme switcher

const THEMES = {
  pilmico: { name: 'Pilmico', color: '#0099DD' },
  aboitiz: { name: 'Aboitiz', color: '#4CAF50' },
  default: { name: 'Default', color: '#EA580C' },
};

function ThemeSwitcher() {
  const [theme, setTheme] = useLocalStorage('app-theme', 'default');
  const [isDarkMode, setIsDarkMode] = useLocalStorage('app-dark-mode', false);
  
  function applyTheme(themeName) {
    setTheme(themeName);
    const themeData = THEMES[themeName];
    
    document.documentElement.style.setProperty('--color-primary', themeData.color);
    document.documentElement.style.setProperty('--color-primary-dark', darkenColor(themeData.color, 0.2));
    document.documentElement.style.setProperty('--color-primary-light', lightenColor(themeData.color, 0.2));
  }
  
  function toggleDarkMode() {
    const newValue = !isDarkMode;
    setIsDarkMode(newValue);
    
    if (newValue) {
      document.documentElement.classList.add('dark-mode');
    } else {
      document.documentElement.classList.remove('dark-mode');
    }
  }
  
  // Apply saved theme on mount
  useEffect(() => {
    applyTheme(theme);
    if (isDarkMode) {
      document.documentElement.classList.add('dark-mode');
    }
  }, []);
  
  return (
    <div className="theme-switcher-container">
      {/* Dark Mode Toggle */}
      <div className="theme-switcher-dark-mode">
        <span className="theme-label">Dark Mode</span>
        <button
          className={`dark-mode-toggle ${isDarkMode ? 'active' : ''}`}
          onClick={toggleDarkMode}
          title="Toggle dark mode"
        >
          <span className="toggle-track"></span>
          <span className="toggle-thumb"></span>
        </button>
      </div>
      
      {/* Divider */}
      <div className="theme-switcher-divider"></div>
      
      {/* Themes */}
      <div className="theme-switcher-themes">
        <span className="theme-label">Themes</span>
        <div className="theme-circles">
          {Object.entries(THEMES).map(([key, themeData]) => (
            <button
              key={key}
              className={`theme-circle ${theme === key ? 'active' : ''}`}
              onClick={() => applyTheme(key)}
              style={{background: themeData.color}}
              title={themeData.name}
            />
          ))}
        </div>
      </div>
    </div>
  );
}
```
