```jsx
// In OrderTableFilters.jsx — add Color and Diameter checkboxes
// Keep existing Category and Item Description filters as-is

function OrderTableFilters({ orders, onFilter }) {
  const [searchTerm, setSearchTerm] = useState('');
  const [selectedCategories, setSelectedCategories] = useState([]);
  const [selectedDescriptions, setSelectedDescriptions] = useState([]);
  const [selectedColors, setSelectedColors] = useState([]);
  const [selectedDiameters, setSelectedDiameters] = useState([]);
  
  // Extract unique values from current orders
  const uniqueCategories = useMemo(() => 
    [...new Set(orders.map(o => o.category))].filter(Boolean).sort(),
    [orders]
  );
  
  const uniqueColors = useMemo(() =>
    [...new Set(orders.map(o => o.color))].filter(Boolean).sort(),
    [orders]
  );
  
  const uniqueDiameters = useMemo(() =>
    [...new Set(orders.map(o => o.diameter))].filter(Boolean).sort(),
    [orders]
  );
  
  const uniqueDescriptions = useMemo(() =>
    [...new Set(orders.map(o => o.item_description))].filter(Boolean).sort(),
    [orders]
  );
  
  // Apply filters (case-insensitive OR match)
  useEffect(() => {
    const filtered = orders.filter(order => {
      // Search term — OR match across multiple fields (case-insensitive)
      let matchesSearch = true;
      if (searchTerm.trim()) {
        const term = searchTerm.toLowerCase().trim();
        matchesSearch =
          (order.item_description || '').toLowerCase().includes(term) ||
          (order.color || '').toLowerCase().includes(term) ||
          String(order.diameter || '').toLowerCase().includes(term) ||
          (order.category || '').toLowerCase().includes(term) ||
          (order.form || '').toLowerCase().includes(term) ||
          (order.material_code_fg || '').toLowerCase().includes(term) ||
          (order.fpr || '').toLowerCase().includes(term);
      }
      
      // Category filter — OR match
      const matchesCategory = selectedCategories.length === 0 || 
                              selectedCategories.includes(order.category);
      
      // Description filter — OR match
      const matchesDescription = selectedDescriptions.length === 0 || 
                                 selectedDescriptions.includes(order.item_description);
      
      // Color filter — OR match
      const matchesColor = selectedColors.length === 0 || 
                          selectedColors.includes(order.color);
      
      // Diameter filter — OR match
      const matchesDiameter = selectedDiameters.length === 0 || 
                             selectedDiameters.includes(order.diameter);
      
      return matchesSearch && matchesCategory && matchesDescription && 
             matchesColor && matchesDiameter;
    });
    
    onFilter(filtered);
  }, [searchTerm, selectedCategories, selectedDescriptions, selectedColors, selectedDiameters, orders]);
  
  return (
    <div className="order-filters">
      {/* Search input — existing */}
      <input
        type="text"
        placeholder="Search by item, color, diameter, category, form, material code..."
        value={searchTerm}

          options: uniqueCategories,

          onChange: setSelectedCategories

          options: uniqueDescriptions, 
          selected: selectedDescriptions,

function FilterDropdown({ categories, descriptions, colors, diameters }) {
  const [isOpen, setIsOpen] = useState(false);

function FilterSection({ label, options, selected, onChange, scrollable, showColorDot, suffix }) {
  function toggle(option) {
    if (selected.includes(option)) {
      onChange(selected.filter(o => o !== option));
    } else {
      onChange([...selected, option]);
    }
  }
  
  return (
    <div className="filter-section">
      <div className="filter-section-header">
        <span>{label}</span>
        {selected.length > 0 && (
          <button className="filter-clear" onClick={() => onChange([])}>Clear</button>
        )}
      </div>
      <div className={`filter-checkboxes ${scrollable ? 'scrollable' : ''}`}>
        {options.map(option => (
          <label key={option} className="filter-checkbox">
            <input
              type="checkbox"
              checked={selected.includes(option)}
              onChange={() => toggle(option)}
            />
            {showColorDot && (
              <span className="color-dot" style={{background: getColorHex(option)}}></span>
            )}
            <span>{option}{suffix}</span>
          </label>
        ))}
      </div>
    </div>
  );
}

function getColorHex(color) {
  const colorMap = {
    'Brown': '#8B4513',
    'Yellow': '#FFC107',
    'White': '#F5F5F5',
    'Red': '#DC2626',
    'Blue': '#2563EB',
    'Green': '#16A34A',
  };
  return colorMap[color] || '#9ca3af';
}
```

## 4. PER-USER THEME PERSISTENCE

```javascript
// Custom hook that saves to the DB per user (not localStorage)

function useUserPreference(key, defaultValue) {
  const { user } = useAuth();
  const [value, setValue] = useState(defaultValue);
  
  // Load from user preferences on mount
  useEffect(() => {
    if (!user) return;
    
    async function loadPref() {
      try {
        const pref = await UserPreference.filter({ 
          user_id: user.id, 
          key: key 
        });
        if (pref && pref.length > 0) {
          setValue(JSON.parse(pref[0].value));
        }
      } catch (error) {
        console.error('Load pref failed:', error);
      }
    }
    
    loadPref();
  }, [user, key]);
  
  // Save on change
  const updateValue = async (newValue) => {
    setValue(newValue);
    
    if (!user) return;
    
    try {
      const existing = await UserPreference.filter({ 
        user_id: user.id, 
        key: key 
      });
      
      if (existing && existing.length > 0) {
        await UserPreference.update(existing[0].id, {
          value: JSON.stringify(newValue)
        });
      } else {
        await UserPreference.create({
          user_id: user.id,
          key: key,
          value: JSON.stringify(newValue)
        });
      }
    } catch (error) {
      console.error('Save pref failed:', error);
    }
  };
  
  return [value, updateValue];
}
```

```javascript
function useUserPreference(key, defaultValue) {
  const { user } = useAuth();
  const storageKey = user ? `user_${user.id}_${key}` : key;
  
  const [value, setValue] = useState(() => {
    try {
      const stored = localStorage.getItem(storageKey);
      return stored ? JSON.parse(stored) : defaultValue;
    } catch {
      return defaultValue;
    }
  });
  
  useEffect(() => {
    localStorage.setItem(storageKey, JSON.stringify(value));
  }, [storageKey, value]);
  
  return [value, setValue];
}
```
