function preSortOrders(orders, n10dData, masterData) {
  return orders.map(order => {
    const materialCode = String(order.material_code_fg || '').trim();
    
    // Skip if already has a hard target date from SAP
    if (order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date))) {
      return {
        ...order,
        _category: 'A',
        _sortDate: new Date(order.target_avail_date),
        _dateSource: 'hard_date',
        date_source: 'hard_date',
      };
    }
    
    // Look up N10D data
    if (n10dData && n10dData.length > 0 && materialCode) {
      const n10dEntry = n10dData.find(n =>
        String(n.material_code || n.material_code_fg || '').trim() === materialCode
      );
      
      if (n10dEntry) {
        const inferredDate = n10dEntry.inferred_target_date || n10dEntry.target_date;
        const status = String(n10dEntry.status || n10dEntry.stock_status || '').trim();
        
        // Check if order already has a date from a PREVIOUS N10D upload
        const hasExistingN10DDate = order.date_source === 'n10d' && order.inferred_target_date;
        
        if (inferredDate && /^\d{4}-\d{2}-\d{2}/.test(inferredDate)) {
          // Has N10D date
          const dateObj = new Date(inferredDate);
          
          // If user manually edited this N10D date, show warning but keep their edit
          const displayDate = order.has_manual_override && order.manual_edit_date 
            ? order.manual_edit_date 
            : inferredDate;
          
          return {
            ...order,
            _category: status === 'Critical' ? 'C' : status === 'Urgent' ? 'B' : 'D',
            _sortDate: new Date(displayDate),
            _n10dStatus: status,
            _availDateDisplay: status,
            _dateSource: 'n10d',
            date_source: 'n10d',
            inferred_target_date: inferredDate,  // Store original N10D date
            inferred_target_label: status,  // Store original label
            target_avail_date: displayDate,  // Display date (may be manually overridden)
            last_n10d_update: new Date().toISOString(),  // Track when this was updated
            _hasUpdatedFromN10D: hasExistingN10DDate && inferredDate !== order.inferred_target_date,  // Flag if N10D changed
            _n10dUpdateWarning: order.has_manual_override ? 'User has manually edited this date. New N10D date available.' : null,
          };
        }
        
        // No N10D date, but has status
        if (status === 'Critical') {
          return {
            ...order,
            _category: 'C',
            _n10dStatus: status,
            _availDateDisplay: status,
            _dateSource: 'n10d',
            date_source: 'n10d',
            inferred_target_label: status,
          };
        }
        
        if (status === 'Urgent') {
          return {
            ...order,
            _category: 'B',
            _n10dStatus: status,
            _availDateDisplay: status,
            _dateSource: 'n10d',
            date_source: 'n10d',
            inferred_target_label: status,
          };
        }
        
        if (['Sufficient', 'Safe', 'OK'].includes(status)) {
          return {
            ...order,
            _category: 'D',
            _n10dStatus: status,
            _availDateDisplay: status,
            _dateSource: 'n10d',
            date_source: 'n10d',
            inferred_target_label: status,
          };
        }
      }
    }
    
    // Fallback
    return {
      ...order,
      _category: 'E',
      _dateSource: 'none',
      date_source: order.date_source || 'none',
    };
  });
}

function isSameDay(dateA, dateB) {
  if (!dateA || !dateB) return false;
  if (isNaN(dateA.getTime()) || isNaN(dateB.getTime())) return false;
  return dateA.getFullYear() === dateB.getFullYear() &&
         dateA.getMonth() === dateB.getMonth() &&
         dateA.getDate() === dateB.getDate();
}

## 4. HANDLE N10D UPLOAD — RECALCULATE AFFECTED ORDERS

When new N10D data is uploaded to Future Dispatches:

```javascript
async function handleN10DUpload(newN10DData) {
  try {
    // Get all ACTIVE orders
    const activeOrders = allOrders.filter(o =>
      o.status !== 'Done' && 
      o.status !== 'Cancel PO' && 
      o.status !== 'In Production' && 
      o.status !== 'On-going'
    );
    
    const ordersToUpdate = [];
    
    for (const order of activeOrders) {
      // Only process orders that got their date from N10D
      if (order.date_source !== 'n10d') continue;
      
      const materialCode = String(order.material_code_fg || '').trim();
      const n10dEntry = newN10DData.find(n =>
        String(n.material_code || n.material_code_fg || '').trim() === materialCode
      );
      
      if (!n10dEntry) continue;
      
      const newInferredDate = n10dEntry.inferred_target_date || n10dEntry.target_date;
      const oldInferredDate = order.inferred_target_date;
      const newStatus = String(n10dEntry.status || n10dEntry.stock_status || '').trim();
      const oldStatus = order.inferred_target_label;
      
      // Check if N10D data has changed
      const hasDateChange = newInferredDate && newInferredDate !== oldInferredDate;
      const hasStatusChange = newStatus !== oldStatus;
      
      if (!hasDateChange && !hasStatusChange) continue;
      
      // If user manually edited, flag it but DON'T auto-update
      if (order.has_manual_override) {
        ordersToUpdate.push({
          id: order.id,
          needsWarning: true,
          newDate: newInferredDate,
          newStatus: newStatus,
          oldDate: order.manual_edit_date,
        });
        continue;
      }
      
      // Auto-update the order
      const updateData = {
        inferred_target_date: newInferredDate,
        inferred_target_label: newStatus,
        target_avail_date: newInferredDate,  // Update display date
        last_n10d_update: new Date().toISOString(),
      };
      
      ordersToUpdate.push({
        id: order.id,
        data: updateData,
        wasUpdated: true,
        oldDate: oldInferredDate,
        newDate: newInferredDate,
      });
    }
    
    // Apply updates
    let updatedCount = 0;
    let warningCount = 0;
    
    for (const item of ordersToUpdate) {
      if (item.needsWarning) {
        // Just mark as needing attention — don't update
        warningCount++;
        // Optionally: save a flag to show warning icon
        await updateOrderMutation.mutateAsync({
          id: item.id,
          data: {
            _n10d_update_available: true,  // Show warning icon
            _n10d_update_new_date: item.newDate,
          }
        });
      } else if (item.wasUpdated) {
        await updateOrderMutation.mutateAsync({
          id: item.id,
          data: item.data
        });
        updatedCount++;
      }
    }
    
    await queryClient.invalidateQueries({ queryKey: ["orders"] });
    
    if (updatedCount > 0) {
      toast.success(`${updatedCount} order(s) updated from new N10D data`);
    }
    
    if (warningCount > 0) {
      toast.info(`${warningCount} order(s) have new N10D dates available (manually overridden)`);
    }
    
  } catch (error) {
    console.error('N10D upload error:', error);
    toast.error(`N10D upload failed: ${error.message}`);
  }
}

// In OrderTable.jsx — render avail date with N10D context

function renderAvailDate(order) {
  const isN10DDate = order.date_source === 'n10d';
  const hasManualOverride = order.has_manual_override;
  const hasN10DUpdate = order._n10d_update_available;
  
  return (
    <div className="avail-date-cell">
      {order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date)) ? (
        <>
          <span className="avail-date-value">
            {formatDate(order.target_avail_date)}
          </span>
          
          {isN10DDate && (
            <span className="avail-date-label">
              {order.inferred_target_label || 'n10d'}
            </span>
          )}
          
          {/* Warning icon if user manually edited an N10D date and new data arrived */}
          {hasManualOverride && hasN10DUpdate && (
            <span
              className="avail-date-warning"
              title={`New N10D date available: ${formatDate(order._n10d_update_new_date)}\n\nYou manually edited this date. Click to update or keep your edit.`}
            >
              ⚠️
            </span>
          )}
        </>
      ) : (
        <span className="avail-date-label">
          {order.inferred_target_label || order.target_avail_date || '—'}
        </span>
      )}
    </div>
  );
}

```javascript
async function handleManualDateEdit(orderId, newDate) {
  try {
    const order = allOrders.find(o => o.id === orderId);
    
    if (!order || order.date_source !== 'n10d') return;
    
    // Mark as manually edited
    await updateOrderMutation.mutateAsync({
      id: orderId,
      data: {
        target_avail_date: newDate,
        manual_edit_date: newDate,
        has_manual_override: true,
        _n10d_update_available: false,  // Clear the warning
      }
    });
    
    toast.success('Date updated. Will not auto-update from future N10D uploads.');
  } catch (error) {
    toast.error(`Update failed: ${error.message}`);
  }
}

// When user clicks the warning icon to accept new N10D date
async function handleAcceptN10DUpdate(orderId) {
  try {
    const order = allOrders.find(o => o.id === orderId);
    
    await updateOrderMutation.mutateAsync({
      id: orderId,
      data: {
        target_avail_date: order._n10d_update_new_date,
        has_manual_override: false,  // Reset override flag
        _n10d_update_available: false,
      }
    });
    
    toast.success('Order date updated from N10D data');
  } catch (error) {
    toast.error(`Update failed: ${error.message}`);
  }
}
