```javascript
function preSortOrders(orders, n10dData, masterData) {
  return orders.map(order => {
    const materialCode = String(order.material_code_fg || '').trim();
    
    // ALWAYS skip hard SAP dates (never override these)
    if (order.date_source === 'hard_date' && order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date))) {
      return {
        ...order,
        _category: 'A',
        _sortDate: new Date(order.target_avail_date),
        _dateSource: 'hard_date',
      };
    }
    
    // For N10D orders: RECALCULATE from current N10D data (don't use old stored dates)
    if (order.date_source === 'n10d' || order.date_source === 'none') {
      if (n10dData && n10dData.length > 0 && materialCode) {
        const n10dEntry = n10dData.find(n =>
          String(n.material_code || n.material_code_fg || '').trim() === materialCode
        );
        
        if (n10dEntry) {
          const inferredDate = n10dEntry.inferred_target_date || n10dEntry.target_date;
          const status = String(n10dEntry.status || n10dEntry.stock_status || '').trim();
          
          if (inferredDate && /^\d{4}-\d{2}-\d{2}/.test(inferredDate)) {
            // NEW DATE FROM N10D — use it
            return {
              ...order,
              _category: status === 'Critical' ? 'C' : status === 'Urgent' ? 'B' : 'D',
              _sortDate: new Date(inferredDate),
              _n10dStatus: status,
              _availDateDisplay: status,
              _dateSource: 'n10d',
              _resolvedDate: inferredDate,  // For display in preview
              inferred_target_date: inferredDate,
              inferred_target_label: status,
              target_avail_date: inferredDate,  // Will be saved only if user approves
              _freshFromN10D: true,  // Flag that this is fresh from latest N10D
            };
          }
          
          // Status-only (no date)
          if (status === 'Critical' || status === 'Urgent') {
            return {
              ...order,
              _category: status === 'Critical' ? 'C' : 'B',
              _n10dStatus: status,
              _availDateDisplay: status,
              _dateSource: 'n10d',
              inferred_target_label: status,
            };
          }
        }
      }
    }
    
    // Fallback
    return {
      ...order,
      _category: 'E',
      _dateSource: 'none',
    };
  });
}
```
