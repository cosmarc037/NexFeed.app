function preSortOrders(orders, n10dData, masterData) {
  return orders.map(order => {
    const materialCode = String(order.material_code_fg || '').trim();
    
    // Hard SAP date — never override
    if (order.date_source === 'hard_date' || 
        (order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date)) && order.date_source === 'hard_date')) {
      return {
        ...order,
        _category: 'A',
        _sortDate: new Date(order.target_avail_date),
        _dateSource: 'hard_date',
      };
    }
    
    // N10D date — ALWAYS recalculate from current N10D data
    if (n10dData && n10dData.length > 0 && materialCode) {
      const n10dEntry = n10dData.find(n =>
        String(n.material_code || n.material_code_fg || '').trim() === materialCode
      );
      
      if (n10dEntry) {
        const inferredDate = n10dEntry.inferred_target_date || n10dEntry.target_date;
        const status = String(n10dEntry.status || n10dEntry.stock_status || '').trim();
        
        if (inferredDate && /^\d{4}-\d{2}-\d{2}/.test(inferredDate)) {
          const dateObj = new Date(inferredDate);
          
          console.log(`[preSortOrders] ${order.item_description?.substring(0, 30)}: NEW N10D date = ${inferredDate}, Status = ${status}`);
          
          return {
            ...order,
            _category: status === 'Critical' ? 'C' : status === 'Urgent' ? 'B' : 'D',
            _sortDate: dateObj,
            _n10dStatus: status,
            _availDateDisplay: status,
            _dateSource: 'n10d',
            target_avail_date: inferredDate,  // UPDATE to new date
            inferred_target_date: inferredDate,
            inferred_target_label: status,
          };
        }
        
        // Status only, no date
        if (status === 'Critical') {
          return {
            ...order,
            _category: 'C',
            _n10dStatus: status,
            _availDateDisplay: status,
            _dateSource: 'n10d',
            inferred_target_label: status,
          };
        }
        
        if (status === 'Urgent') {
          return {
            ...order,
            _category: 'B',
            _n10dStatus: status,
            _availDateDisplay: status,
            _dateSource: 'n10d',
            inferred_target_label: status,
          };
        }
        
        if (['Sufficient', 'Safe', 'OK'].includes(status)) {
          return {
            ...order,
            _category: 'D',
            _n10dStatus: status,
            _availDateDisplay: status,
            _dateSource: 'n10d',
            inferred_target_label: status,
          };
        }
      }
    }
    
    // Fallback
    return {
      ...order,
      _category: 'E',
      _dateSource: 'none',
    };
  });
}
