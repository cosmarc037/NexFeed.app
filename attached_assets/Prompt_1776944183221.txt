function handlePlantLevelAutoSequence() {
  try {
    // ... existing code ...
    
    const allActiveOrders = enrichedOrders.filter(...);
    
    // Track which orders have date changes
    const dateChanges = new Map();
    
    const ordersWithFreshN10D = allActiveOrders.map(order => {
      const materialCode = String(order.material_code_fg || '').trim();
      const oldDate = order.target_avail_date;
      
      if (order.date_source === 'n10d' && n10dData && n10dData.length > 0 && materialCode) {
        const n10dEntry = n10dData.find(n =>
          String(n.material_code || n.material_code_fg || '').trim() === materialCode
        );
        
        if (n10dEntry) {
          const newInferredDate = n10dEntry.inferred_target_date || n10dEntry.target_date;
          const newStatus = String(n10dEntry.status || n10dEntry.stock_status || '').trim();
          
          if (newInferredDate && /^\d{4}-\d{2}-\d{2}/.test(newInferredDate)) {
            // Check if date actually changed
            if (oldDate !== newInferredDate) {
              dateChanges.set(String(order.id), {
                oldDate: oldDate,
                newDate: newInferredDate,
                oldLabel: order.inferred_target_label,
                newLabel: newStatus,
              });
            }
            
            return {
              ...order,
              target_avail_date: newInferredDate,
              inferred_target_date: newInferredDate,
              inferred_target_label: newStatus,
              _dateChanged: oldDate !== newInferredDate,  // Flag if changed
              _oldDate: oldDate,
            };
          }
        }
      }
      
      return order;
    });
    
    // Store date changes for display
    setDateChangesMap(dateChanges);
    
    // ... rest of code ...
  } catch (error) {
    // ...
  }
}
