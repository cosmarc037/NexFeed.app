async function cascadeSchedule(orders, startIndex = 0) {
  for (let i = startIndex; i < orders.length; i++) {
    const order = orders[i];
    
    // DONE orders: skip calculation, keep as-is
    if (order.status === 'Done' || order.status === 'Cancel PO' || order.status === 'In Production' || order.status === 'On-going') {
      console.log(`[Cascade] Skip ${order.item_description} (status: ${order.status})`);
      continue;
    }
    
    // Calculate changeover to NEXT ACTIVE order (not Done/Cancel)
    let changeover = 0;
    let nextActiveOrder = null;
    
    for (let j = i + 1; j < orders.length; j++) {
      if (orders[j].status !== 'Done' && orders[j].status !== 'Cancel PO' && orders[j].status !== 'In Production' && orders[j].status !== 'On-going') {
        nextActiveOrder = orders[j];
        break;
      }
    }
    
    if (nextActiveOrder) {
      changeover = calculateChangeoverBetween(order, nextActiveOrder, changeoverRules);
    }
    
    order.changeover = changeover;
    
    // Calculate production time
    const volume = parseFloat(order.volume_override || order.volume || order.total_volume_mt) || 0;
    const runRate = getLineRunRate(normalizeLine(order.feedmill_line));
    const productionTime = runRate > 0 ? volume / runRate : 0;
    
    order.production_time = productionTime;
    order._productionTimeHrs = productionTime.toFixed(2);
    
    // Estimated completion = start time + production time + changeover
    const startTime = order.start_datetime ? new Date(order.start_datetime) : new Date();
    const totalHours = productionTime + changeover;
    const completionTime = new Date(startTime.getTime() + totalHours * 60 * 60 * 1000);
    
    order.estimated_completion_date = completionTime.toISOString();
    
    console.log(
      `[Cascade] ${order.item_description?.substring(0, 30)} | ` +
      `Volume: ${volume} MT | Run Rate: ${runRate}/hr | ` +
      `Production: ${productionTime.toFixed(2)}h | Changeover: ${changeover.toFixed(2)}h | ` +
      `Total: ${totalHours.toFixed(2)}h`
    );
  }
}
