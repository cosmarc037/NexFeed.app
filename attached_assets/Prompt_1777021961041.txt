function calculatePreviewChangeovers(orders, changeoverRules) {
  for (let i = 0; i < orders.length; i++) {
    const currentOrder = orders[i];
    
    // FROZEN changeover (order is Done): keep as-is, don't recalculate
    if (currentOrder.frozen_changeover != null) {
      currentOrder.changeover = currentOrder.frozen_changeover;
      currentOrder._changeoverCalculated = false;
      currentOrder._isFrozen = true;
      continue;
    }
    
    // Done/Cancel orders that don't have frozen changeover: no changeover
    if (currentOrder.status === 'Done' || currentOrder.status === 'Cancel PO' || 
        currentOrder.status === 'In Production' || currentOrder.status === 'On-going') {
      currentOrder.changeover = 0;
      currentOrder._changeoverTotal = 0;
      currentOrder._changeoverDetails = 'Order is ' + currentOrder.status.toLowerCase();
      currentOrder._changeoverCalculated = false;
      continue;
    }
    
    // Last order on line: no changeover
    if (i === orders.length - 1) {
      currentOrder.changeover = 0;
      currentOrder._changeoverTotal = 0;
      currentOrder._changeoverDetails = 'Last order — no changeover';
      currentOrder._changeoverCalculated = true;
      continue;
    }
    
    // Find next ACTIVE order (skip Done/Cancel/In Production/On-going)
    let nextOrder = null;
    for (let j = i + 1; j < orders.length; j++) {
      if (orders[j].status !== 'Done' && orders[j].status !== 'Cancel PO' &&
          orders[j].status !== 'In Production' && orders[j].status !== 'On-going') {
        nextOrder = orders[j];
        break;
      }
    }
    
    // No active order after this one: no changeover
    if (!nextOrder) {
      currentOrder.changeover = 0;
      currentOrder._changeoverTotal = 0;
      currentOrder._changeoverDetails = 'No active orders following';
      currentOrder._changeoverCalculated = true;
      continue;
    }
    
    // Calculate changeover based on NEXT ACTIVE order
    const changeover = calculateChangeoverBetween(currentOrder, nextOrder, changeoverRules);
    currentOrder.changeover = changeover;
    currentOrder._changeoverTotal = changeover;
    currentOrder._changeoverDetails = `CO: ${changeover.toFixed(2)} hrs`;
    currentOrder._changeoverCalculated = true;
  }
}
