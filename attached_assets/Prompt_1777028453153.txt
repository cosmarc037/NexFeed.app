```javascript
function calculateEstimatedCompletionDates(lineOrdersMap, changeoverRules) {
  const results = {};
  
  for (const [line, orders] of Object.entries(lineOrdersMap)) {
    if (!orders || orders.length === 0) {
      results[line] = [];
      continue;
    }
    
    const enrichedOrders = [];
    let cumulativeTime = 0;
    let startDateTime = null;
    
    for (let i = 0; i < orders.length; i++) {
      const order = { ...orders[i] };
      
      // Get start date and time for first order
      if (i === 0) {
        if (order.start_datetime) {
          startDateTime = new Date(order.start_datetime);
        } else if (order.start_date) {
          // Parse start date and combine with default 08:00 time
          const dateStr = String(order.start_date).substring(0, 10);
          startDateTime = new Date(`${dateStr}T08:00:00`);
        } else {
          // Default: today at 08:00 AM
          const today = new Date();
          startDateTime = new Date(today.getFullYear(), today.getMonth(), today.getDate(), 8, 0, 0);
        }
        
        order.start_datetime = startDateTime.toISOString();
      }
      
      // Calculate production time
      const volume = parseFloat(order.volume_override || order.volume || order.total_volume_mt) || 0;
      const runRate = getLineRunRate(normalizeLine(order.feedmill_line));
      const productionTime = runRate > 0 ? volume / runRate : 0;
      
      order.production_time = productionTime;
      order._productionTimeHrs = productionTime.toFixed(2);
      
      // Get changeover (already calculated in preview)
      const changeover = order.changeover || 0;
      order._changeoverHrs = changeover.toFixed(2);
      
      // Calculate this order's completion time
      const orderStartTime = new Date(startDateTime.getTime() + cumulativeTime * 60 * 60 * 1000);
      const totalOrderTime = productionTime + changeover; // Production + changeover to next order
      const orderEndTime = new Date(orderStartTime.getTime() + totalOrderTime * 60 * 60 * 1000);
      
      order.estimated_completion_date = orderEndTime.toISOString();
      order._estimatedCompletionDisplay = formatDateTime(orderEndTime);
      
      // Add to cumulative for next order
      cumulativeTime += totalOrderTime;
      
      enrichedOrders.push(order);
    }
    
    results[line] = enrichedOrders;
  }
  
  return results;
}

function formatDateTime(date) {
  if (!date) return '—';
  const dateStr = new Date(date).toLocaleDateString('en-US', { month: '2-digit', day: '2-digit', year: 'numeric' });
  const timeStr = new Date(date).toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit', hour12: true });
  return `${dateStr} - ${timeStr}`;
}
```

## 5. CASCADE COMPLETION DATES WHEN START DATE/TIME CHANGES

```javascript
function cascadeCompletionDates(updatedOrder, lineOrders, changeoverRules) {
  const orderIndex = lineOrders.findIndex(o => o.id === updatedOrder.id);
  
  if (orderIndex === -1) return lineOrders;
  
  // Update the modified order
  lineOrders[orderIndex] = { ...lineOrders[orderIndex], ...updatedOrder };
  
  // Get start time for cascading
  let startDateTime = null;
  
  if (lineOrders[orderIndex].start_datetime) {
    startDateTime = new Date(lineOrders[orderIndex].start_datetime);
  } else if (lineOrders[orderIndex].start_date) {
    const dateStr = String(lineOrders[orderIndex].start_date).substring(0, 10);
    startDateTime = new Date(`${dateStr}T08:00:00`);
  } else {
    const today = new Date();
    startDateTime = new Date(today.getFullYear(), today.getMonth(), today.getDate(), 8, 0, 0);
  }
  
  let cumulativeTime = 0;
  
  // Recalculate from the modified order onwards
  for (let i = orderIndex; i < lineOrders.length; i++) {
    const order = lineOrders[i];
    
    const volume = parseFloat(order.volume_override || order.volume || order.total_volume_mt) || 0;
    const runRate = getLineRunRate(normalizeLine(order.feedmill_line));
    const productionTime = runRate > 0 ? volume / runRate : 0;
    
    const changeover = order.changeover || 0;
    
    const orderStartTime = new Date(startDateTime.getTime() + cumulativeTime * 60 * 60 * 1000);
    const totalOrderTime = productionTime + changeover;
    const orderEndTime = new Date(orderStartTime.getTime() + totalOrderTime * 60 * 60 * 1000);
    
    lineOrders[i].estimated_completion_date = orderEndTime.toISOString();
    lineOrders[i]._estimatedCompletionDisplay = formatDateTime(orderEndTime);
    
    cumulativeTime += totalOrderTime;
  }
  
  return lineOrders;
}
```

// In PlantAutoSequencePreview

function handleLineOrderUpdate(line, orderId, updates) {
  setSequencedResults(prev => {
    const updated = { ...prev };
    const lineOrders = [...(updated[line] || [])];
    
    const orderIndex = lineOrders.findIndex(o => o.id === orderId);
    if (orderIndex === -1) return prev;
    
    const updatedOrder = { ...lineOrders[orderIndex], ...updates };
    
    // Cascade completion dates if start date/time changed
    if (updates.start_date || updates.start_datetime) {
      lineOrders = cascadeCompletionDates(updatedOrder, lineOrders, changeoverRules);
    } else {
      lineOrders[orderIndex] = updatedOrder;
    }
    
    updated[line] = lineOrders;
    return updated;
  });
}
