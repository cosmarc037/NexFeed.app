## 1. PROPER START DATE/TIME INITIALIZATION

```javascript
function initializeStartDateTime(order, lineIndex, previousOrder) {
  // Only set start date/time for FIRST order in the line
  if (lineIndex === 0) {
    // Use provided start_datetime, or default to today 08:00 AM
    if (!order.start_datetime) {
      const today = new Date();
      order.start_datetime = new Date(today.getFullYear(), today.getMonth(), today.getDate(), 8, 0, 0).toISOString();
    }
    return order;
  }
  
  // For subsequent orders: inherit from previous order's completion time
  if (previousOrder && previousOrder.estimated_completion_date) {
    order.start_datetime = previousOrder.estimated_completion_date;
  } else {
    // Fallback: today 08:00 AM
    const today = new Date();
    order.start_datetime = new Date(today.getFullYear(), today.getMonth(), today.getDate(), 8, 0, 0).toISOString();
  }
  
  return order;
}

function calculateEstimatedCompletionDates(lineOrdersMap) {
  const results = {};
  
  for (const [line, orders] of Object.entries(lineOrdersMap)) {
    if (!orders || orders.length === 0) {
      results[line] = [];
      continue;
    }
    
    const enrichedOrders = [];
    
    for (let i = 0; i < orders.length; i++) {
      const order = { ...orders[i] };
      
      // Initialize start datetime
      const previousOrder = i > 0 ? enrichedOrders[i - 1] : null;
      order = initializeStartDateTime(order, i, previousOrder);
      
      // Calculate production time
      const volume = parseFloat(order.volume_override || order.volume || order.total_volume_mt) || 0;
      const runRate = getLineRunRate(normalizeLine(order.feedmill_line));
      const productionTime = runRate > 0 ? volume / runRate : 0;
      
      order.production_time = productionTime;
      order._productionTimeHrs = productionTime.toFixed(2);
      
      // Get changeover
      const changeover = order.changeover || 0;
      order._changeoverHrs = changeover.toFixed(2);
      
      // Calculate completion time: start + production + changeover
      const startTime = new Date(order.start_datetime);
      const totalHours = productionTime + changeover;
      const completionTime = new Date(startTime.getTime() + totalHours * 60 * 60 * 1000);
      
      order.estimated_completion_date = completionTime.toISOString();
      order._estimatedCompletionDisplay = formatDateTime(completionTime);
      
      enrichedOrders.push(order);
    }
    
    results[line] = enrichedOrders;
  }
  
  return results;
}
```

```jsx
      previousEnd.setHours(0, 0, 0, 0);
```

## 3. UPDATE HANDLER WITH CASCADING

```javascript
function handleLineOrderUpdate(line, orderId, updates, lineKey) {
  setSequencedResults(prev => {
    const updated = { ...prev };
    const lineOrders = [...(updated[lineKey] || [])];
    
    const orderIndex = lineOrders.findIndex(o => o.id === orderId);
    if (orderIndex === -1) return prev;
    
    // Update the order
    lineOrders[orderIndex] = { ...lineOrders[orderIndex], ...updates };
    
    // Cascade completion dates from this order onwards
    for (let i = orderIndex; i < lineOrders.length; i++) {
      const order = lineOrders[i];
      
      // Get start datetime
      let startDateTime = order.start_datetime ? new Date(order.start_datetime) : null;
      
      // For subsequent orders, use previous completion as start
      if (i > orderIndex && lineOrders[i - 1].estimated_completion_date) {
        startDateTime = new Date(lineOrders[i - 1].estimated_completion_date);
      }
      
      if (!startDateTime) {
        startDateTime = new Date();
        startDateTime.setHours(8, 0, 0, 0);
      }
      
      // Calculate completion
      const volume = parseFloat(order.volume_override || order.volume || order.total_volume_mt) || 0;
      const runRate = getLineRunRate(normalizeLine(order.feedmill_line));
      const productionTime = runRate > 0 ? volume / runRate : 0;
      const changeover = order.changeover || 0;
      
      const completionTime = new Date(startDateTime.getTime() + (productionTime + changeover) * 60 * 60 * 1000);
      
      lineOrders[i].estimated_completion_date = completionTime.toISOString();
      lineOrders[i]._estimatedCompletionDisplay = formatDateTime(completionTime);
    }
    
    updated[lineKey] = lineOrders;
    return updated;
  });
}
```
