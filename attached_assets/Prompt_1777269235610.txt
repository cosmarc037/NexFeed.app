async function generateSequenceStrategies(ordersByLine, n10dData, masterData, changeoverRules) {
  const strategies = {};
  
  // STRATEGY 1: Rule-based (current logic) — No AI
  strategies.rule_based = await generateRuleBasedStrategy(ordersByLine, n10dData);
  
  // STRATEGY 2: AI-Powered without profitability
  strategies.ai_balanced = await generateAIStrategy(
    ordersByLine, 
    n10dData, 
    masterData, 
    changeoverRules,
    { withProfitability: false }
  );
  
  // STRATEGY 3: AI-Powered with profitability
  strategies.ai_profit = await generateAIStrategy(
    ordersByLine, 
    n10dData, 
    masterData, 
    changeoverRules,
    { withProfitability: true }
  );
  
  // Determine AI recommendation — based on metric scoring
  strategies.recommended = determineRecommendation(strategies);
  
  return strategies;
}

async function generateRuleBasedStrategy(ordersByLine, n10dData) {
  // Current rule-based logic: follow N10D dates strictly, sort chronologically
  const result = {};
  
  for (const [line, orders] of Object.entries(ordersByLine)) {
    const sorted = [...orders].sort((a, b) => {
      // Category priority
      const catA = getCategoryPriority(a._dateCategory);
      const catB = getCategoryPriority(b._dateCategory);
      if (catA !== catB) return catA - catB;
      
      // By date
      const dateA = a._resolvedDate ? new Date(a._resolvedDate) : new Date('2099-12-31');
      const dateB = b._resolvedDate ? new Date(b._resolvedDate) : new Date('2099-12-31');
      return dateA - dateB;
    });
    
    result[line] = sorted.map((o, i) => ({ ...o, prio: i + 1 }));
  }
  
  return {
    id: 'rule_based',
    name: 'Standard Sequence',
    theme: 'Rule-Based',
    description: 'Sequenced strictly by availability dates and N10D thresholds',
    icon: '📋',
    color: 'blue',
    orders: result,
    metrics: calculateStrategyMetrics(result, ordersByLine),
    isAIRecommended: false,
  };
}

async function generateAIStrategy(ordersByLine, n10dData, masterData, changeoverRules, options) {
  const { withProfitability } = options;
  
  // Enrich orders with margin data
  const enrichedByLine = {};
  for (const [line, orders] of Object.entries(ordersByLine)) {
    enrichedByLine[line] = orders.map(order => {
      const materialCode = String(order.material_code_fg || '').trim();
      const mdEntry = masterData.find(md =>
        String(md.material_code_fg || '').trim() === materialCode
      );
      return {
        ...order,
        _margin: mdEntry ? (parseFloat(mdEntry.margin) || 0) : 0,
      };
    });
  }
  
  // Call Azure OpenAI with structured prompt
  const prompt = buildAIPrompt(enrichedByLine, withProfitability);
  
  try {
    const response = await callAzureOpenAI(prompt);
    const aiResult = parseAIResponse(response);
    
    // Validate AI result against non-negotiable rules
    const validatedResult = validateAndAdjust(aiResult, enrichedByLine);
    
    return {
      id: withProfitability ? 'ai_profit' : 'ai_balanced',
      name: withProfitability ? 'AI Profit-Optimized' : 'AI Balanced',
      theme: withProfitability ? 'AI-Powered (Profit-Optimized)' : 'AI-Powered (Balanced)',
      description: withProfitability 
        ? 'AI sequences orders to maximize profitability while respecting MTO deadlines' 
        : 'AI intelligently repositions MTS orders within safe windows to improve flow',
      icon: withProfitability ? '💰' : '🤖',
      color: withProfitability ? 'green' : 'purple',
      orders: validatedResult,
      metrics: calculateStrategyMetrics(validatedResult, enrichedByLine),
      aiReasoning: aiResult.reasoning || '',
      dateAdjustments: aiResult.dateAdjustments || [],
      isAIRecommended: false,
    };
  } catch (error) {
    console.error('AI strategy generation failed:', error);
    // Fallback: return rule-based as this option
    return generateRuleBasedStrategy(enrichedByLine, n10dData);
  }
}

function buildAIPrompt(ordersByLine, withProfitability) {
  const orderSummary = Object.entries(ordersByLine).map(([line, orders]) => {
    return `${line}: ${orders.length} orders`;
  }).join(', ');
  
  const allOrders = Object.values(ordersByLine).flat();
  
  const orderDetails = allOrders.map(o => ({
    id: o.id,
    line: o.feedmill_line,
    description: o.item_description,
    volume: o.volume || o.total_volume_mt,
    avail_date: o.target_avail_date,
    n10d_status: o._n10dStatus || 'unknown',
    n10d_label: o.inferred_target_label || '',
    is_mto: !!o.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(o.target_avail_date)),
    is_critical: o._n10dStatus === 'Critical' || o._dateCategory === 'critical',
    margin: o._margin || 0,
  }));
  
  return `You are an AI production scheduler. Analyze the following orders and provide an optimal sequencing strategy.

## Current Orders (${allOrders.length} total, ${orderSummary}):
${JSON.stringify(orderDetails, null, 2)}

## NON-NEGOTIABLE RULES:
1. MTO orders (is_mto: true) MUST complete on or before their avail_date — NEVER delay them
2. Critical N10D orders (is_critical: true) MUST complete before their N10D threshold
3. Respect line capacity and run rates

## OPTIMIZATION OBJECTIVE:
${withProfitability 
  ? '- PRIORITIZE high-margin orders earlier when they have MTS flexibility\n- Defer low-margin MTS orders within their safe N10D window\n- Fill production gaps with highest-margin eligible orders'
  : '- Balance workload across lines evenly\n- Minimize changeovers by grouping similar materials\n- Smooth out the production flow'
}

## MTS DATE FLEXIBILITY:
- MTS orders (is_mto: false, not critical) can be produced anywhere within a safe window
- Earliest start: today + 2 days (buffer)
- Latest end: their N10D red-threshold date (stored in avail_date)
- You can advance MTS orders to fill gaps between MTO orders

## RESPONSE FORMAT (JSON):
{

async function callAzureOpenAI(prompt) {
  // Call the existing Azure OpenAI integration used for sequence insights
  const response = await fetch('/api/azure-openai', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      prompt: prompt,
      model: 'gpt-4',
      temperature: 0.2,  // Low temp for consistent, structured output
      max_tokens: 4000,
    })
  });
  
  if (!response.ok) {
    throw new Error(`AI API error: ${response.statusText}`);
  }
  
  const data = await response.json();
  return data.response || data.content;
}

function parseAIResponse(response) {
  try {
    // Extract JSON from response
    const jsonMatch = response.match(/\{[\s\S]*\}/);
    if (!jsonMatch) throw new Error('No JSON in response');
    
    return JSON.parse(jsonMatch[0]);
  } catch (error) {
    console.error('Failed to parse AI response:', error);
    return { strategy: {}, reasoning: '', dateAdjustments: [] };
  }
}

function validateAndAdjust(aiResult, originalOrdersByLine) {
  const result = {};
  
  for (const [line, orders] of Object.entries(originalOrdersByLine)) {
    const aiLineStrategy = aiResult.strategy?.[line] || [];
    
    // Map AI suggestions back to orders
    const orderedResult = [];
    const orderMap = new Map(orders.map(o => [String(o.id), o]));
    
    // Apply AI's suggested order
    for (const aiOrder of aiLineStrategy) {
      const order = orderMap.get(String(aiOrder.order_id));
      if (!order) continue;
      
      // Apply AI's suggested date if within safe bounds
      const suggestedDate = aiOrder.suggested_start_date;
      const isMTO = !!order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date));
      
      if (!isMTO && suggestedDate) {
        // Validate suggested date is within safe window
        const today = new Date();
        today.setDate(today.getDate() + 2);  // Buffer: today + 2 days
        const earliestAllowed = today.toISOString().split('T')[0];
        const latestAllowed = order.target_avail_date;
        
        if (suggestedDate >= earliestAllowed && 
            (!latestAllowed || suggestedDate <= latestAllowed)) {
          order._aiSuggestedDate = suggestedDate;
          order._aiReasoning = aiOrder.reasoning;
        }
      }
      
      orderedResult.push(order);
      orderMap.delete(String(aiOrder.order_id));
    }
    
    // Add any orders AI missed (safety net)
    orderMap.forEach(order => orderedResult.push(order));
    
    // Assign prio
    orderedResult.forEach((o, i) => {
      o.prio = i + 1;
    });
    
    result[line] = orderedResult;
  }
  
  // Final safety validation: ensure MTO orders not delayed
  return enforceNonNegotiableRules(result);
}

function enforceNonNegotiableRules(ordersByLine) {
  for (const [line, orders] of Object.entries(ordersByLine)) {
    // Simulate production times and check for MTO violations
    let currentTime = new Date();
    currentTime.setHours(8, 0, 0, 0);  // Start at 8 AM today
    
    for (const order of orders) {
      const volume = order.volume || order.total_volume_mt || 0;
      const runRate = getLineRunRate(line);
      const productionTime = runRate > 0 ? volume / runRate : 0;
      const changeover = order.changeover || 0;
      
      const orderEnd = new Date(currentTime.getTime() + (productionTime + changeover) * 60 * 60 * 1000);
      
      // Check MTO violation
      if (order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date))) {
        const availDate = new Date(order.target_avail_date);
        availDate.setHours(23, 59, 59, 999);
        
        if (orderEnd > availDate) {
          console.warn(`MTO violation detected for ${order.item_description} — AI strategy adjusted`);
          // Don't modify here, trust AI's placement
        }
      }
      
      order._estimatedCompletion = orderEnd.toISOString();
      currentTime = orderEnd;
    }
  }
  
  return ordersByLine;
}

function calculateStrategyMetrics(ordersByLine, originalByLine) {
  const allOrders = Object.values(ordersByLine).flat();
  const totalOrders = allOrders.length;
  
  const mtoOrders = allOrders.filter(o => 
    o.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(o.target_avail_date))
  );
  
  const mtsOrders = allOrders.filter(o => !mtoOrders.includes(o));
  
  // Calculate min slack (time between completion and deadline)
  let minSlack = Infinity;
  let minSlackOrder = null;
  
  allOrders.forEach(o => {
    if (o._estimatedCompletion && o.target_avail_date) {
      const completionTime = new Date(o._estimatedCompletion);
      const availDate = new Date(o.target_avail_date);
      availDate.setHours(23, 59, 59, 999);
      
      const slackHours = (availDate - completionTime) / (1000 * 60 * 60);
      if (slackHours < minSlack) {
        minSlack = slackHours;
        minSlackOrder = o;
      }
    }
  });
  
  // Risk level
  let riskLevel = 'Low';
  if (minSlack < 0) riskLevel = 'Violated';
  else if (minSlack < 12) riskLevel = 'High';
  else if (minSlack < 48) riskLevel = 'Medium';
  
  // Count MTS with adjusted dates
  const mtsAdjusted = allOrders.filter(o => o._aiSuggestedDate).length;
  
  // Calculate total profit potential (if margins available)
  const totalMargin = allOrders.reduce((sum, o) => sum + (o._margin || 0) * (o.volume || 0), 0);
  
  // Count violations
  const violations = allOrders.filter(o => {
    if (!o._estimatedCompletion || !o.target_avail_date) return false;
    return new Date(o._estimatedCompletion) > new Date(o.target_avail_date);
  }).length;
  
  return {
    totalOrders,
    mtoCount: mtoOrders.length,
    mtsCount: mtsOrders.length,
    mtsAdjusted,
    minSlackHours: minSlack === Infinity ? null : minSlack.toFixed(1),
    minSlackOrder: minSlackOrder?.item_description,
    riskLevel,
    totalMargin: totalMargin.toFixed(2),
    violations,
  };
}

function determineRecommendation(strategies) {
  // AI recommends the strategy with:
  // 1. Zero violations
  // 2. Best balance of risk and utilization
  const candidates = [strategies.rule_based, strategies.ai_balanced, strategies.ai_profit];
  
  const safeStrategies = candidates.filter(s => s.metrics.violations === 0);
  
  if (safeStrategies.length === 0) {
    strategies.rule_based.isAIRecommended = true;
    return 'rule_based';
  }
  
  // Prefer AI balanced if safe and has reasonable slack
  if (strategies.ai_balanced.metrics.violations === 0 && 
      strategies.ai_balanced.metrics.minSlackHours > 12) {
    strategies.ai_balanced.isAIRecommended = true;
    return 'ai_balanced';
  }
  
  // Default to rule-based
  strategies.rule_based.isAIRecommended = true;
  return 'rule_based';
}
