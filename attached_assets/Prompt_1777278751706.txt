function validateAISuggestedDate(order, suggestedDate) {
  if (!suggestedDate) return { isValid: true };
  
  const isMTO = isMTOOrder(order);
  
  // MTO orders: never change
  if (isMTO) {
    return { isValid: false, reason: 'MTO orders cannot have dates changed by AI' };
  }
  
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  const buffer2Days = new Date(today);
  buffer2Days.setDate(buffer2Days.getDate() + 2);
  
  const suggestedDateObj = new Date(suggestedDate);
  
  // Can't suggest before today+2
  if (suggestedDateObj < buffer2Days) {
    return { isValid: false, reason: `Suggested date before buffer (today+2)` };
  }
  
  // Get effective ceiling (N10D threshold or hard avail date)
  const effectiveCeiling = getEffectiveAvailDateCeiling(order);
  
  if (effectiveCeiling) {
    const ceilingDateObj = new Date(effectiveCeiling);
    ceilingDateObj.setHours(23, 59, 59, 999);
    
    if (suggestedDateObj > ceilingDateObj) {
      return { isValid: false, reason: `Suggested date exceeds N10D threshold` };
    }
  }
  
  return { isValid: true };
}

function sortOrdersChronologically(orders) {
  return [...orders].sort((a, b) => {
    // Extract the effective avail date (either original or AI-suggested)
    const dateA = a.target_avail_date;
    const dateB = b.target_avail_date;
    
    // Both are real dates
    if (dateA && dateB && /^\d{4}-\d{2}-\d{2}/.test(String(dateA)) && /^\d{4}-\d{2}-\d{2}/.test(String(dateB))) {
      const timeA = new Date(dateA).getTime();
      const timeB = new Date(dateB).getTime();
      if (timeA !== timeB) return timeA - timeB;
    }
    
    // One is a real date, one is a string label — real dates come first
    const isADate = /^\d{4}-\d{2}-\d{2}/.test(String(dateA));
    const isBDate = /^\d{4}-\d{2}-\d{2}/.test(String(dateB));
    
    if (isADate && !isBDate) return -1;
    if (!isADate && isBDate) return 1;
    
    // Both are labels — sort by N10D urgency
    const categoryA = getCategoryPriority(a._dateCategory || categorizeOrder(a));
    const categoryB = getCategoryPriority(b._dateCategory || categorizeOrder(b));
    
    if (categoryA !== categoryB) return categoryA - categoryB;
    
    // Same category — maintain original order
    return 0;
  });
}

function categorizeOrder(order) {
  const n10dStatus = order._n10dStatus || order.inferred_target_label || '';
  
  if (n10dStatus === 'Critical') return 'critical';
  if (n10dStatus === 'Urgent' || n10dStatus === 'prio replenish') return 'urgent';
  if (n10dStatus === 'Monitor' || n10dStatus === 'monitor') return 'monitor';
  return 'sufficient';
}

function getCategoryPriority(category) {
  const priorities = {
    'critical': 0,
    'urgent': 1,
    'monitor': 2,
    'sufficient': 3,
  };
  return priorities[category] || 99;
}

function getEffectiveAvailDateCeiling(order) {
  // If order already has a real date, use it as ceiling
  if (order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date))) {
    return order.target_avail_date;
  }
  
  // Otherwise, infer from N10D
  if (order.inferred_target_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.inferred_target_date))) {
    return order.inferred_target_date;
  }
  
  // Fallback: today + 30 days
  const today = new Date();
  today.setDate(today.getDate() + 30);
  return today.toISOString().split('T')[0];
}

function isMTOOrder(order) {
  // MTO = has hard contractual avail date (not N10D inferred)
  if (!order.target_avail_date) return false;
  
  const dateStr = String(order.target_avail_date);
  
  // If it's a real date AND not sourced from N10D
  if (/^\d{4}-\d{2}-\d{2}/.test(dateStr) && !order._dateSourceIsN10D) {
    return true;
  }
  
  return false;
}

function calculateN10DDaysRemaining(order) {
  if (!order.target_avail_date || !/^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date))) {
    return 999;  // Unknown — very far away
  }
  
  const deadline = new Date(order.target_avail_date);
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  const daysRemaining = Math.ceil((deadline - today) / (1000 * 60 * 60 * 24));
  return Math.max(0, daysRemaining);
}
