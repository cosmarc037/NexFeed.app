function enrichOrdersWithMargin(ordersByLine, masterData) {
  const result = {};
  
  for (const [line, orders] of Object.entries(ordersByLine)) {
    result[line] = orders.map(order => {
      const materialCode = String(order.material_code_fg || '').trim();
      const mdEntry = masterData.find(md =>
        String(md.material_code_fg || '').trim() === materialCode
      );
      
      const margin = mdEntry ? (parseFloat(mdEntry.margin) || 0) : 0;
      const volume = order.volume_override || order.volume || order.total_volume_mt || 0;
      const productionTime = order.production_time || 0;
      const changeover = order.changeover || 0;
      const runtimeTotal = productionTime + changeover;
      
      return {
        ...order,
        _margin: margin,
        _marginTotal: margin * volume,  // Total margin contribution
        _profitRate: runtimeTotal > 0 ? margin / runtimeTotal : 0,  // Margin per hour
        _runtimeTotal: runtimeTotal,
      };
    });
  }
  
  return result;
}

function generateStandardSequence(enrichedByLine) {
  const result = {};
  
  for (const [line, orders] of Object.entries(enrichedByLine)) {
    // Sort chronologically by avail date (MTO dates first, then N10D dates)
    const sorted = sortOrdersChronologically(orders);
    
    sorted.forEach((o, i) => {
      o.prio = i + 1;
      o._noDateChange = true;  // Standard doesn't suggest new dates
    });
    
    result[line] = sorted;
  }
  
  return {
    id: 'standard',
    name: 'Standard Sequence',
    theme: 'Rule-Based',
    description: 'Orders sorted chronologically by availability dates and N10D thresholds',
    icon: '📋',
    color: 'blue',
    orders: result,
    metrics: calculateStrategyMetrics(result),
    isAIRecommended: false,
  };
}

async function generateAIBalancedSequence(enrichedByLine, n10dData, masterData, changeoverRules, options) {
  const { optimizeForProfit } = options;
  
  // Build AI prompt
  const prompt = buildAISequencePrompt(enrichedByLine, optimizeForProfit);
  
  let aiResult = { strategy: {}, reasoning: '', dateAdjustments: [] };
  
  try {
    const response = await callAzureOpenAI(prompt);
    aiResult = parseAIResponse(response);
  } catch (error) {
    console.error('AI sequence generation failed:', error);
    // Fallback to standard
    return generateStandardSequence(enrichedByLine);
  }
  
  // Apply AI reordering and date suggestions
  const result = {};
  
  for (const [line, orders] of Object.entries(enrichedByLine)) {
    const aiLineResult = aiResult.strategy?.[line] || [];
    
    if (aiLineResult.length === 0) {
      // No AI suggestion for this line
      const sorted = sortOrdersChronologically(orders);
      sorted.forEach((o, i) => o.prio = i + 1);
      result[line] = sorted;
      continue;
    }
    
    // Build AI suggestion map
    const aiMap = new Map();
    for (const aiOrder of aiLineResult) {
      aiMap.set(String(aiOrder.order_id), {
        suggestedDate: aiOrder.suggested_date,
        reasoning: aiOrder.reasoning,
        position: aiOrder.position,
      });
    }
    
    // Apply AI suggestions to orders
    const ordersWithAISuggestions = orders.map(order => {
      const aiSuggestion = aiMap.get(String(order.id));
      
      if (aiSuggestion?.suggestedDate) {
        // Validate suggested date
        const validation = validateAISuggestedDate(order, aiSuggestion.suggestedDate, n10dData);
        
        if (validation.isValid) {
          return {
            ...order,
            target_avail_date: aiSuggestion.suggestedDate,
            _aiSuggestedDate: aiSuggestion.suggestedDate,
            _originalAvailDate: order.target_avail_date,
            _aiReasoning: aiSuggestion.reasoning,
            _dateAdjustedByAI: true,
          };
        }
      }
      
      return order;
    });
    
    // Sort by AI's suggested position (which preserves avail date order but optimizes changeovers)
    const sorted = ordersWithAISuggestions
      .map((o, i) => ({ ...o, _aiPosition: i }))
      .sort((a, b) => {
        const posA = aiMap.get(String(a.id))?.position ?? a._aiPosition;
        const posB = aiMap.get(String(b.id))?.position ?? b._aiPosition;
        return posA - posB;
      });
    
    // Assign priorities
    sorted.forEach((o, i) => {
      o.prio = i + 1;
    });
    
    result[line] = sorted;
  }
  
  return {
    id: optimizeForProfit ? 'ai_profit' : 'ai_balanced',
    name: optimizeForProfit ? 'AI Profit-Optimized' : 'AI Balanced',
    theme: optimizeForProfit ? 'AI-Powered (Profit-Optimized)' : 'AI-Powered (Balanced)',
    description: optimizeForProfit
      ? 'AI prioritizes high-margin orders and optimizes changeovers while respecting MTO deadlines'
      : 'AI intelligently repositions orders to minimize changeovers and suggests flexible MTS dates within safe windows',
    icon: optimizeForProfit ? '💰' : '🤖',
    color: optimizeForProfit ? 'green' : 'purple',
    orders: result,
    metrics: calculateStrategyMetrics(result),
    aiReasoning: aiResult.reasoning || '',
    dateAdjustments: aiResult.dateAdjustments || [],
    isAIRecommended: false,
  };
}

function validateAISuggestedDate(order, suggestedDate, n10dData) {
  if (!suggestedDate) return { isValid: true };
  
  const isMTO = isMTOOrder(order);
  
  // MTO orders: never change
  if (isMTO) {
    return { isValid: false, reason: 'MTO cannot be changed' };
  }
  
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  const buffer2Days = new Date(today);
  buffer2Days.setDate(buffer2Days.getDate() + 2);
  
  const suggestedDateObj = new Date(suggestedDate);
  
  // Can't suggest before today+2
  if (suggestedDateObj < buffer2Days) {
    return { isValid: false, reason: 'Before today+2 buffer' };
  }
  
  // Get N10D threshold (red highlight date)
  const n10dThreshold = getN10DThreshold(order, n10dData);
  
  if (n10dThreshold) {
    const thresholdObj = new Date(n10dThreshold);
    thresholdObj.setHours(23, 59, 59, 999);
    
    if (suggestedDateObj > thresholdObj) {
      return { isValid: false, reason: 'Beyond N10D threshold' };
    }
  }
  
  // Check against hard avail date
  if (order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date))) {
    const availDateObj = new Date(order.target_avail_date);
    availDateObj.setHours(23, 59, 59, 999);
    
    if (suggestedDateObj > availDateObj) {
      return { isValid: false, reason: 'Beyond avail date' };
    }
  }
  
  return { isValid: true };
}

function getN10DThreshold(order, n10dData) {
  // N10D threshold = the red highlight date (day 10 from N10D)
  // For Stock Sufficient orders, this is typically 10 days from today
  const materialCode = String(order.material_code_fg || '').trim();
  const n10dEntry = n10dData?.find(n =>
    String(n.material_code || n.material_code_fg || '').trim() === materialCode
  );
  
  if (n10dEntry && n10dEntry.target_date && /^\d{4}-\d{2}-\d{2}/.test(String(n10dEntry.target_date))) {
    return n10dEntry.target_date;
  }
  
  // Fallback: 10 days from today
  const fallback = new Date();
  fallback.setDate(fallback.getDate() + 10);
  return fallback.toISOString().split('T')[0];
}

function calculateN10DDaysRemaining(order) {
  if (!order.target_avail_date || !/^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date))) {
    return 999;
  }
  
  const deadline = new Date(order.target_avail_date);
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  const daysRemaining = Math.ceil((deadline - today) / (1000 * 60 * 60 * 24));
  return Math.max(0, daysRemaining);
}

function sortOrdersChronologically(orders) {
  return [...orders].sort((a, b) => {
    // MTO orders first (hard dates)
    const isMTOA = isMTOOrder(a);
    const isMTOB = isMTOOrder(b);
    
    if (isMTOA && !isMTOB) return -1;
    if (!isMTOA && isMTOB) return 1;
    
    // Then by N10D category (critical → urgent → monitor → sufficient)
    const catA = getOrderCategory(a);
    const catB = getOrderCategory(b);
    const catPriority = { 'critical': 0, 'urgent': 1, 'monitor': 2, 'sufficient': 3 };
    
    if (catPriority[catA] !== catPriority[catB]) {
      return catPriority[catA] - catPriority[catB];
    }
    
    // Finally by avail date
    const dateA = a.target_avail_date ? new Date(a.target_avail_date).getTime() : Infinity;
    const dateB = b.target_avail_date ? new Date(b.target_avail_date).getTime() : Infinity;
    
    return dateA - dateB;
  });
}

function getOrderCategory(order) {
  if (order._n10dStatus === 'Critical') return 'critical';
  if (order._n10dStatus === 'Urgent' || order.inferred_target_label === 'prio replenish') return 'urgent';
  if (order._n10dStatus === 'Monitor') return 'monitor';
  return 'sufficient';
}

function isMTOOrder(order) {
  // MTO = has hard contractual avail date (not N10D inferred)
  if (!order.target_avail_date) return false;
  
  const dateStr = String(order.target_avail_date);
  
  // If it's a real date AND not sourced from N10D inference
  if (/^\d{4}-\d{2}-\d{2}/.test(dateStr) && !order._dateSourceIsN10D && order._n10dStatus !== 'Sufficient') {
    return true;
  }
  
  return false;
}

function calculateStrategyMetrics(ordersByLine) {
  const allOrders = Object.values(ordersByLine).flat();
  
  const mtoOrders = allOrders.filter(o => isMTOOrder(o));
  const totalOrders = allOrders.length;
  
  // Check for MTO violations
  let minSlack = Infinity;
  let violations = 0;
  
  mtoOrders.forEach(o => {
    if (o._estimatedCompletion && o.target_avail_date) {
      const completionTime = new Date(o._estimatedCompletion);
      const availDate = new Date(o.target_avail_date);
      availDate.setHours(23, 59, 59, 999);
      
      const slackHours = (availDate - completionTime) / (1000 * 60 * 60);
      
      if (slackHours < 1.5) violations++;
      minSlack = Math.min(minSlack, slackHours);
    }
  });
  
  const riskLevel = violations > 0 ? 'Violated' : minSlack < 12 ? 'High' : minSlack < 48 ? 'Medium' : 'Low';
  
  // Total profit
  const totalMargin = allOrders.reduce((sum, o) => sum + (o._marginTotal || 0), 0);
  
  return {
    totalOrders,
    mtoCount: mtoOrders.length,
    minSlackHours: minSlack === Infinity ? null : minSlack.toFixed(1),
    riskLevel,
    violations,
    totalMargin: totalMargin.toFixed(2),
  };
}

function determineRecommendation(strategies) {
  // Recommend AI Balanced if safe (no violations)
  // Prefer AI Profit if it's also safe and more profitable
  
  if (strategies.ai_balanced?.metrics.violations === 0) {
    strategies.ai_balanced.isAIRecommended = true;
    return 'ai_balanced';
  }
  
  if (strategies.ai_profit?.metrics.violations === 0 && 
      parseFloat(strategies.ai_profit.metrics.totalMargin) > parseFloat(strategies.ai_balanced?.metrics.totalMargin || 0)) {
    strategies.ai_profit.isAIRecommended = true;
    return 'ai_profit';
  }
  
  strategies.standard.isAIRecommended = true;
  return 'standard';
}
