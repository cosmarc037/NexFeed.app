```javascript
function calculateChangeoverBetween(fromOrder, toOrder, changeoverMatrix) {
  // Extract material attributes
  const fromForm = fromOrder.form || 'P';
  const toForm = toOrder.form || 'P';
  
  const fromColor = fromOrder.color || 'Unknown';
  const toColor = toOrder.color || 'Unknown';
  
  const fromDiameter = fromOrder.diameter || '3mm';
  const toDiameter = toOrder.diameter || '3mm';
  
  const fromCategory = fromOrder.category || 'Unknown';
  const toCategory = toOrder.category || 'Unknown';
  
  let totalChangeover = changeoverMatrix?.baseChangeover || 0.17;
  
  // 1. Diameter change
  if (fromDiameter !== toDiameter) {
    const diameterKey = `${fromDiameter}-${toDiameter}`;
    totalChangeover += changeoverMatrix?.diameterChanges?.[diameterKey] || 1.5; // Default FM1 cost
  }
  
  // 2. Color/SKU change
  if (fromColor !== toColor) {
    const fromColorLower = fromColor.toLowerCase();
    const toColorLower = toColor.toLowerCase();
    
    // Check for Red or Green (require flushing & cleaning)
    if (fromColorLower === 'red' || toColorLower === 'red' || 
        fromColorLower === 'green' || toColorLower === 'green') {
      totalChangeover += 1.0; // Flushing and Cleaning
    } else {
      totalChangeover += 0.33; // Regular cleaning
    }
  }
  
  // 3. Category change
  if (fromCategory !== toCategory) {
    totalChangeover += 0.33; // Different category cleaning
  }
  
  return totalChangeover;
}

function buildMaterialCompatibilityMap(orders, changeoverMatrix) {
  // Create a map of material "signatures" to identify groups
  const materialGroups = new Map();
  
  orders.forEach((order, index) => {
    const signature = `${order.category}|${order.color}|${order.diameter}|${order.form}`;
    if (!materialGroups.has(signature)) {
      materialGroups.set(signature, []);
    }
    materialGroups.get(signature).push(index);
  });
  
  return materialGroups;
}

function calculateTotalChangeoverCost(sequence, orders, changeoverMatrix) {
  let totalCO = 0;
  
  for (let i = 0; i < sequence.length - 1; i++) {
    const fromOrder = orders[sequence[i]];
    const toOrder = orders[sequence[i + 1]];
    
    totalCO += calculateChangeoverBetween(fromOrder, toOrder, changeoverMatrix);
  }
  
  return totalCO;
}
```

```javascript
function validateAISuggestedDate(order, suggestedDate, n10dData) {
  if (!suggestedDate) return { isValid: true };
  
  const isMTO = isMTOOrder(order);
  
  // MTO: never change
  if (isMTO) {
    return { isValid: false, reason: 'MTO cannot be changed' };
  }
  
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  const buffer2Days = new Date(today);
  buffer2Days.setDate(buffer2Days.getDate() + 2);
  
  const suggestedDateObj = new Date(suggestedDate);
  
  // Can't suggest before today+2
  if (suggestedDateObj < buffer2Days) {
    return { isValid: false, reason: 'Before today+2 buffer' };
  }
  
  // Check if order is Stock Sufficient
  const isSufficient = order._n10dStatus === 'Sufficient';
  
  if (isSufficient) {
    // Stock Sufficient: can go up to N10D threshold + 14 days
    const n10dThreshold = getN10DThreshold(order, n10dData);
    const maxDate = new Date(n10dThreshold);
    maxDate.setDate(maxDate.getDate() + 14);
    maxDate.setHours(23, 59, 59, 999);
    
    if (suggestedDateObj > maxDate) {
      return { isValid: false, reason: 'Beyond sufficient flexibility window' };
    }
  } else {
    // Urgent/Monitor/etc: can't go beyond avail date
    if (order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date))) {
      const availDateObj = new Date(order.target_avail_date);
      availDateObj.setHours(23, 59, 59, 999);
      
      if (suggestedDateObj > availDateObj) {
        return { isValid: false, reason: 'Beyond avail date' };
      }
    }
  }
  
  return { isValid: true };
}
```
