function sortOrdersChronologically(orders) {
  return [...orders].sort((a, b) => {
    // Both have real dates
    const dateAStr = String(a.target_avail_date || '');
    const dateBStr = String(b.target_avail_date || '');
    
    const isADate = /^\d{4}-\d{2}-\d{2}/.test(dateAStr);
    const isBDate = /^\d{4}-\d{2}-\d{2}/.test(dateBStr);
    
    // Both are real dates — sort by date
    if (isADate && isBDate) {
      const timeA = new Date(dateAStr).getTime();
      const timeB = new Date(dateBStr).getTime();
      if (timeA !== timeB) return timeA - timeB;  // Earlier date first
    }
    
    // One is real date, one is label — real dates come first
    if (isADate && !isBDate) return -1;
    if (!isADate && isBDate) return 1;
    
    // Both are labels (N10D status) — sort by priority
    const catA = getOrderCategory(a);
    const catB = getOrderCategory(b);
    const catPriority = { 'critical': 0, 'urgent': 1, 'monitor': 2, 'sufficient': 3 };
    
    return (catPriority[catA] || 99) - (catPriority[catB] || 99);
  });
}

```javascript
function validateAISuggestedDate(order, suggestedDate, n10dData) {
  if (!suggestedDate) return { isValid: true };
  
  const isMTO = isMTOOrder(order);
  
  // MTO: never change
  if (isMTO) {
    return { isValid: false, reason: 'MTO cannot be changed' };
  }
  
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  const buffer2Days = new Date(today);
  buffer2Days.setDate(buffer2Days.getDate() + 2);
  
  const suggestedDateObj = new Date(suggestedDate);
  suggestedDateObj.setHours(0, 0, 0, 0);
  
  // Can't suggest before today+2
  if (suggestedDateObj < buffer2Days) {
    console.warn(`Date ${suggestedDate} before today+2 buffer, rejecting`);
    return { isValid: false, reason: 'Before today+2 buffer' };
  }
  
  // Check if order is Stock Sufficient
  const isSufficient = order._n10dStatus === 'Sufficient';
  
  if (isSufficient) {
    // Stock Sufficient: can go up to N10D threshold + 14 days
    const n10dThreshold = getN10DThreshold(order, n10dData);
    const maxDate = new Date(n10dThreshold);
    maxDate.setDate(maxDate.getDate() + 14);
    maxDate.setHours(23, 59, 59, 999);
    
    if (suggestedDateObj > maxDate) {
      console.warn(`Date ${suggestedDate} beyond sufficient flexibility (${maxDate}), rejecting`);
      return { isValid: false, reason: 'Beyond sufficient flexibility window' };
    }
  } else {
    // Urgent/Monitor: can advance but not delay past avail date
    if (order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date))) {
      const availDateObj = new Date(order.target_avail_date);
      availDateObj.setHours(23, 59, 59, 999);
      
      if (suggestedDateObj > availDateObj) {
        console.warn(`Date ${suggestedDate} delays order past avail date ${order.target_avail_date}, rejecting`);
        return { isValid: false, reason: 'Delays order past avail date' };
      }
    }
  }
  
  return { isValid: true };
}

function getN10DThreshold(order, n10dData) {
  // N10D threshold = the day-10 date (red highlight)
  // For Stock Sufficient, this is typically N10D target date
  
  if (order.inferred_target_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.inferred_target_date))) {
    return order.inferred_target_date;
  }
  
  if (order.target_avail_date && /^\d{4}-\d{2}-\d{2}/.test(String(order.target_avail_date))) {
    return order.target_avail_date;
  }
  
  // Fallback: 10 days from today
  const fallback = new Date();
  fallback.setDate(fallback.getDate() + 10);
  return fallback.toISOString().split('T')[0];
}
