## 1. ADD RESIZABLE SPLIT PANE STATE

```jsx
// In PlantAutoSequencePreview.jsx

const DEFAULT_SPLIT = 50; // 50% / 50%
const MIN_SPLIT = 25;     // left panel minimum width %
const MAX_SPLIT = 75;     // left panel maximum width %

const [splitRatio, setSplitRatio] = useState(DEFAULT_SPLIT);
const [isDraggingDivider, setIsDraggingDivider] = useState(false);

const splitContainerRef = useRef(null);
const dividerRef = useRef(null);
```

---

## 2. RESET SPLIT WHEN MODAL OPENS

```jsx
// Reset to default every time modal opens
useEffect(() => {
  if (isOpen) {
    setSplitRatio(DEFAULT_SPLIT);
  }
}, [isOpen]);
```

If the component mounts/unmounts with modal open/close, this is enough:

```jsx
useEffect(() => {
  setSplitRatio(DEFAULT_SPLIT);
}, []);
```

## 3. ADD DRAG HANDLERS

```jsx
function clamp(value, min, max) {
  return Math.min(Math.max(value, min), max);
}

function handleDividerMouseDown(e) {
  e.preventDefault();
  setIsDraggingDivider(true);
  document.body.style.cursor = 'col-resize';
  document.body.style.userSelect = 'none';
}

function handleDividerDoubleClick() {
  setSplitRatio(DEFAULT_SPLIT);
}

useEffect(() => {
  function handleMouseMove(e) {
    if (!isDraggingDivider || !splitContainerRef.current) return;
    
    const rect = splitContainerRef.current.getBoundingClientRect();
    const relativeX = e.clientX - rect.left;
    const percent = (relativeX / rect.width) * 100;
    const clamped = clamp(percent, MIN_SPLIT, MAX_SPLIT);
    
    setSplitRatio(clamped);
  }
  
  function handleMouseUp() {
    if (!isDraggingDivider) return;
    setIsDraggingDivider(false);
    document.body.style.cursor = '';
    document.body.style.userSelect = '';
  }
  
  window.addEventListener('mousemove', handleMouseMove);
  window.addEventListener('mouseup', handleMouseUp);
  
  return () => {
    window.removeEventListener('mousemove', handleMouseMove);
    window.removeEventListener('mouseup', handleMouseUp);
  };
}, [isDraggingDivider]);
```

## 4. REPLACE THE CURRENT BEFORE/AFTER LAYOUT WITH A RESIZABLE SPLIT CONTAINER

```jsx
// BEFORE:
// <div className="as-before-after-wrap">
//   <div className="as-before-panel">...</div>
//   <div className="as-middle-divider">...</div>
//   <div className="as-after-panel">...</div>
// </div>

// AFTER:
<div
  ref={splitContainerRef}
  className={`as-before-after-split ${isDraggingDivider ? 'is-dragging' : ''}`}

  {/* LEFT: BEFORE */}
  <div
    className="as-split-panel as-split-panel-before"

      <span className="as-panel-header-icon">📋</span>

    </div>

      {/* existing Before table */}
      {renderBeforeTable()}
    </div>
  </div>
  
  {/* DRAGGABLE DIVIDER */}
  <div
    ref={dividerRef}
    className={`as-split-divider ${isDraggingDivider ? 'active' : ''}`}
    onMouseDown={handleDividerMouseDown}

    <div className="as-split-divider-handle">
      ↔
    </div>
  </div>
  
  {/* RIGHT: AFTER */}
  <div
    className="as-split-panel as-split-panel-after"
    style={{ width: `${100 - splitRatio}%` }}
  >
    <div className="as-panel-header as-panel-header-after">
      <span className="as-panel-header-icon">✨</span>
      <span>{afterHeaderLabel}</span>

    </div>
  </div>

```

## 5. KEEP EXISTING TABLES, JUST WRAP THEM

If your current code already has the two table render blocks, do not rewrite the table internals — just wrap them with the split layout above.

Example:

```jsx
function renderBeforeTable() {
  return (
    <table className="as-table as-table-before">
      {/* existing before table head/body */}
    </table>
  );
}

function renderAfterTable() {
  return (
    <table className="as-table as-table-after">
      {/* existing after table head/body */}
    </table>
  );
}
```

## 7. OPTIONAL: ADD KEYBOARD ACCESSIBILITY FOR DIVIDER

```jsx
function handleDividerKeyDown(e) {
  if (e.key === 'ArrowLeft') {
    setSplitRatio(prev => clamp(prev - 2, MIN_SPLIT, MAX_SPLIT));
  } else if (e.key === 'ArrowRight') {
    setSplitRatio(prev => clamp(prev + 2, MIN_SPLIT, MAX_SPLIT));
  } else if (e.key === 'Home') {
    setSplitRatio(MIN_SPLIT);
  } else if (e.key === 'End') {
    setSplitRatio(MAX_SPLIT);
  } else if (e.key === 'Enter' || e.key === ' ') {
    setSplitRatio(DEFAULT_SPLIT);
  }
}
```

Then apply it:

```jsx
<div
  ref={dividerRef}
  className={`as-split-divider ${isDraggingDivider ? 'active' : ''}`}
  onMouseDown={handleDividerMouseDown}
  onDoubleClick={handleDividerDoubleClick}
  onKeyDown={handleDividerKeyDown}
  tabIndex={0}
  title="Drag to resize • Double-click to reset"
  role="separator"
  aria-orientation="vertical"
  aria-label="Resize before and after tables"
>
```

## 9. DO NOT PERSIST WIDTH ACROSS MODAL REOPEN

Do NOT save `splitRatio` to localStorage or DB.

Each time the modal is opened:
- reset to 50 / 50

This matches the expected behavior:
- adjustable during current session
- reset when modal opens again
