## 1. RESTORE CHRONOLOGICAL SORT AS THE PRIMARY RULE

```javascript
function sortOrdersChronologicallyWithBalancedTieBreak(orders, changeoverRules) {
  return [...orders].sort((a, b) => {
    const dateA = normalizeDate(a._aiSuggestedDate || a.target_avail_date);
    const dateB = normalizeDate(b._aiSuggestedDate || b.target_avail_date);

    const isAReal = isISODate(dateA);
    const isBReal = isISODate(dateB);

    // Real dates first
    if (isAReal && isBReal) {
      const diff = new Date(dateA) - new Date(dateB);
      if (diff !== 0) return diff;
    } else if (isAReal && !isBReal) {
      return -1;
    } else if (!isAReal && isBReal) {
      return 1;
    }

    // Protected rank second
    const rankA = getProtectedRank(a);
    const rankB = getProtectedRank(b);
    if (rankA !== rankB) return rankA - rankB;

    // Same effective date / same protected rank:
    // prefer lower changeover opportunity by material similarity
    return compareMaterialCompatibility(a, b);
  });
}
```

## 2. ADD MATERIAL-COMPATIBILITY TIEBREAKER

This should only influence order when chronology is effectively equal.

```javascript
function compareMaterialCompatibility(a, b) {
  const sigA = getMaterialSignature(a);
  const sigB = getMaterialSignature(b);

  // identical signatures = same priority
  if (sigA === sigB) return 0;

  // sort by category, then color, then diameter, then form
  const catA = String(a.category || '').toLowerCase();
  const catB = String(b.category || '').toLowerCase();
  if (catA !== catB) return catA.localeCompare(catB);

  const colorA = String(a.color || '').toLowerCase();
  const colorB = String(b.color || '').toLowerCase();
  if (colorA !== colorB) return colorA.localeCompare(colorB);

  const diaA = String(a.diameter || '').toLowerCase();
  const diaB = String(b.diameter || '').toLowerCase();
  if (diaA !== diaB) return diaA.localeCompare(diaB);

  const formA = String(a.form || '').toLowerCase();
  const formB = String(b.form || '').toLowerCase();
  return formA.localeCompare(formB);
}
```

```javascript
function applyBalancedDateHeuristic(ordersByLine, changeoverRules, n10dData) {
  const result = {};

  Object.entries(ordersByLine).forEach(([line, orders]) => {
    const adjusted = orders.map(order => {
      if (isProtectedOrder(order)) {
        return {
          ...order,
          _aiSuggestedDate: normalizeDate(order.target_avail_date),
          _aiDateReason: 'Protected order — fixed by hard rule',
        };
      }

      const safeWindow = getBalancedSafeWindow(order, n10dData);
      const suggestedDate = chooseBestBalancedDate({
        order,
        line,
        orders,
        safeWindow,
        changeoverRules,
      });

      return {
        ...order,
        _originalAvailDate: normalizeDate(order.target_avail_date),
        target_avail_date: suggestedDate,
        _aiSuggestedDate: suggestedDate,
        _dateAdjustedByAI: normalizeDate(order.target_avail_date) !== suggestedDate,
        _aiDateReason: `Balanced heuristic selected ${suggestedDate} based on safe window, line load, and compatibility`,
      };
    });

    // IMPORTANT: final output is still chronological
    const sorted = sortOrdersChronologicallyWithBalancedTieBreak(adjusted, changeoverRules).map((o, i) => ({
      ...o,
      prio: i + 1,
    }));

    result[line] = sorted;
  });

  return result;
}
```

## 4. REFINE DATE SUGGESTION HEURISTIC — DO NOT LET IT BE RANDOM

The AI should suggest dates based on:
- earliest safe day

```javascript
function chooseBestBalancedDate({
  order,
  line,
  orders,
  safeWindow,
  changeoverRules,
}) {
  const candidateDates = enumerateDates(safeWindow.start, safeWindow.end);

  let bestDate = safeWindow.start;
  let bestScore = -Infinity;

  candidateDates.forEach(candidateDate => {
    const score = scoreBalancedCandidateDate({
      order,
      candidateDate,
      line,
      orders,
      changeoverRules,
    });

    if (score > bestScore) {
      bestScore = score;
      bestDate = candidateDate;
    }
  });

  return bestDate;
}
```

## 5. SCORE DATE CANDIDATES WITH CHRONOLOGY IN MIND

Balanced should prefer:
- earlier safe dates
- dates with low queue load
- dates that improve material compatibility

```javascript
function scoreBalancedCandidateDate({
  order,
  candidateDate,
  line,
  orders,
  changeoverRules,
}) {
  let score = 0;

  const loadMap = buildDailyLoadMap(orders, line);
  const load = loadMap.get(candidateDate) || { orderCount: 0, totalHours: 0 };

  // 1. Prefer lighter days
  score += Math.max(0, 12 - load.totalHours) * 4;
  score += Math.max(0, 4 - load.orderCount) * 2;

  // 2. Prefer compatibility with orders already on that day
  score += estimateCompatibilityGain(order, candidateDate, orders, changeoverRules) * 10;

  // 3. Prefer earlier safe dates (balanced planning)
  const today = startOfDay(new Date());
  const candidate = startOfDay(new Date(candidateDate));
  const daysFromToday = Math.round((candidate - today) / (1000 * 60 * 60 * 24));
  score += Math.max(0, 20 - daysFromToday) * 0.75;

  // 4. Penalize overloaded days
  if (load.totalHours > 16) score -= 20;
  if (load.totalHours > 20) score -= 30;

  return score;
}
```

```javascript
function getProtectedRank(order) {
  if (isMTOOrder(order)) return 0;

  const status = String(order._n10dStatus || order.inferred_target_label || '').toLowerCase();

  if (status === 'critical') return 1;
```

function collectDateAdjustments(ordersByLine) {
  const changes = [];

  Object.values(ordersByLine).flat().forEach(order => {
    const originalDate = normalizeDate(order._originalAvailDate);
    const newDate = normalizeDate(order._aiSuggestedDate || order.target_avail_date);

    if (originalDate && newDate && originalDate !== newDate) {
      changes.push({
        order_id: order.id,
        original: originalDate,
        new: newDate,
        reason: order._aiDateReason || order._aiReasoning || 'AI balanced adjustment',
      });
    }
  });

  return changes;
}
