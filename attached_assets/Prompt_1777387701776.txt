function calculateChangeoverBetween(fromOrder, toOrder, changeoverRules, feedmill) {
  const formKey = String(toOrder.form || fromOrder.form || '').trim();
  const baseChangeover =
    parseFloat(changeoverRules?.baseByForm?.[formKey]) ||
    parseFloat(changeoverRules?.baseChangeoverByForm?.[formKey]) ||
    parseFloat(changeoverRules?.baseDefault) ||
    0;

  let total = baseChangeover;

  const fromDiameter = normalizeDiameter(fromOrder.diameter);
  const toDiameter = normalizeDiameter(toOrder.diameter);
  const fromColor = normalizeColor(fromOrder.color);
  const toColor = normalizeColor(toOrder.color);
  const fromCategory = normalizeCategory(fromOrder.category);
  const toCategory = normalizeCategory(toOrder.category);

  const fmKey = normalizeFeedmillKey(feedmill || fromOrder.feedmill || toOrder.feedmill);

  // diameter change
  if (fromDiameter !== toDiameter) {
    total += parseFloat(changeoverRules?.additions?.diameterChange?.[fmKey]) || 0;
  }

  // SKU color change
  if (fromColor !== toColor) {
    total += resolveColorChangeCost(fromColor, toColor, changeoverRules, fmKey);
  }

  // category change
  if (fromCategory !== toCategory) {
    total += parseFloat(changeoverRules?.additions?.categoryChange?.[fmKey]) || 0;
  }

  return Number(total.toFixed(2));
}

function normalizeFeedmillKey(value) {
  const v = String(value || '').toLowerCase();
  if (v.includes('fm1') || v.includes('feedmill 1')) return 'FM1';
  if (v.includes('fm2') || v.includes('feedmill 2')) return 'FM2';
  if (v.includes('fm3') || v.includes('feedmill 3') || v.includes('pmx')) return 'FM3';
  return 'FM1';
}

function normalizeColor(value) {
  return String(value || '').trim().toLowerCase();
}

function normalizeCategory(value) {
  return String(value || '').trim().toLowerCase();
}

function normalizeDiameter(value) {
  return String(value || '').replace(/\s+/g, '').toLowerCase();
}

function resolveColorChangeCost(fromColor, toColor, changeoverRules, fmKey) {
  const rules = changeoverRules?.additions || {};

  const isFromRedGreen = fromColor === 'red' || fromColor === 'green';
  const isToRedGreen = toColor === 'red' || toColor === 'green';

  if ((fromColor === 'red' || fromColor === 'green') && toColor !== fromColor) {
    return parseFloat(rules?.redOrGreenToAny?.[fmKey]) || 0;
  }

  if (isToRedGreen && !isFromRedGreen) {
    return parseFloat(rules?.anyToRedOrGreen?.[fmKey]) || 0;
  }

  if (fromColor !== toColor) {
    return parseFloat(rules?.yellowBrownViceVersa?.[fmKey]) || 0;
  }

  return 0;
}

Adjust property names to match the actual structure from the Changeover Rules tab.

function estimateCompatibilityGain(order, candidateDate, orders, changeoverRules, line) {
  const sameDateOrders = orders.filter(o => {
    const dt = normalizeDate(o._aiSuggestedDate || o.target_avail_date);
    return dt === candidateDate && String(o.id) !== String(order.id);
  });

  if (sameDateOrders.length === 0) return 1;

  let bestGain = 0;

  sameDateOrders.forEach(other => {
    const co = calculateChangeoverBetween(order, other, changeoverRules, line);

    if (co <= 0.17) bestGain = Math.max(bestGain, 4);
    else if (co <= 0.50) bestGain = Math.max(bestGain, 3);
    else if (co <= 0.83) bestGain = Math.max(bestGain, 2);
    else bestGain = Math.max(bestGain, 1);
  });

  return bestGain;
}
