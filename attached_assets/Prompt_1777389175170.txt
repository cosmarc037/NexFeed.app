function sortWithinDateGroups(orders) {
  const groups = new Map();

  orders.forEach(order => {
    const key = normalizeDate(order._aiSuggestedDate || order.target_avail_date) || '__NO_DATE__';
    if (!groups.has(key)) groups.set(key, []);
    groups.get(key).push(order);
  });

  const sortedDates = Array.from(groups.keys()).sort((a, b) => {
    if (a === '__NO_DATE__') return 1;
    if (b === '__NO_DATE__') return -1;
    return new Date(a) - new Date(b);
  });

  const final = [];

  sortedDates.forEach(dateKey => {
    const bucket = groups.get(dateKey);

    bucket.sort((a, b) => {
      const rankA = getProtectedRank(a);
      const rankB = getProtectedRank(b);
      if (rankA !== rankB) return rankA - rankB;

      return compareMaterialCompatibility(a, b);
    });

    final.push(...bucket);
  });

  return final;
}
