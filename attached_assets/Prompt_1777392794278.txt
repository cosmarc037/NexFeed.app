```javascript
function calculateProfitRate(order, prevOrder, changeoverRules, line) {
  const margin = parseFloat(order._margin || 0) || 0;
  const productionHours = calculateProductionHours(order, line);

  const changeoverHours = prevOrder
    ? calculateChangeoverBetween(prevOrder, order, changeoverRules, line)
    : 0;

  const totalHours = productionHours + changeoverHours;

  if (totalHours <= 0) return 0;

  return margin / totalHours;
}
```

```javascript
function scoreProfitCandidateDate({
  order,
  candidateDate,
  line,
  orders,
  changeoverRules,
}) {
  let score = 0;

  const loadMap = buildDailyLoadMap(orders, line);
  const load = loadMap.get(candidateDate) || { orderCount: 0, totalHours: 0 };

  // 1. Profit rate weighting
  const prevOrder = findNearestPreviousOrderOnOrBeforeDate(order, candidateDate, orders);
  const profitRate = calculateProfitRate(order, prevOrder, changeoverRules, line);
  score += profitRate * 20;

  // 2. Prefer lighter-loaded days
  score += Math.max(0, 12 - load.totalHours) * 3;
  score += Math.max(0, 4 - load.orderCount) * 1.5;

  // 3. Reward compatibility / lower changeover
  score += estimateCompatibilityGain(order, candidateDate, orders, changeoverRules, line) * 8;

  // 4. Prefer earlier safe dates for higher-value orders
  const today = startOfDay(new Date());
  const candidate = startOfDay(new Date(candidateDate));
  const daysFromToday = Math.round((candidate - today) / (1000 * 60 * 60 * 24));
  score += Math.max(0, 20 - daysFromToday) * 0.75;

  // 5. Penalize overloaded days
  if (load.totalHours > 16) score -= 20;
  if (load.totalHours > 20) score -= 30;

  return score;
}
```

Adjust weights if needed after testing, but make sure profit rate is one of the strongest non-hard-constraint signals.

```javascript
function chooseBestProfitDate({
  order,
  line,
  orders,
  safeWindow,
  changeoverRules,
}) {
  const candidateDates = enumerateDates(safeWindow.start, safeWindow.end);

  let bestDate = safeWindow.start;
  let bestScore = -Infinity;

      line,
      orders,
      changeoverRules,
```

```javascript
function applyProfitDateHeuristic(ordersByLine, changeoverRules, n10dData) {
  const result = {};

  Object.entries(ordersByLine).forEach(([line, orders]) => {
    const adjusted = orders.map(order => {
      if (isMTOOrder(order)) {
        return {
          ...order,
          _aiSuggestedDate: normalizeDate(order.target_avail_date),
          _aiDateReason: 'MTO order — fixed date retained',
        };
      }

      const status = String(order._n10dStatus || order.inferred_target_label || '').toLowerCase();

      if (status === 'critical') {
        return {
          ...order,
          _aiSuggestedDate: normalizeDate(order.target_avail_date),
          _aiDateReason: 'Critical order — prioritized and fixed',
        };
      }

      const safeWindow = getProfitSafeWindow(order, n10dData);
      const suggestedDate = chooseBestProfitDate({
        order,
        line,
        orders,
        safeWindow,
        changeoverRules,
      });

      return {
        ...order,
        _originalAvailDate: normalizeDate(order.target_avail_date),
        target_avail_date: suggestedDate,
        _aiSuggestedDate: suggestedDate,
        _dateAdjustedByAI: normalizeDate(order.target_avail_date) !== suggestedDate,
        _aiDateReason: `Profit-optimized heuristic selected ${suggestedDate} based on profit rate, safe window, line load, and compatibility`,
      };
    });

    // Final display should still be chronologically correct
    const sorted = sortOrdersChronologicallyWithBalancedTieBreak(adjusted).map((o, i) => ({
      ...o,
      prio: i + 1,
    }));

    result[line] = sorted;
  });

  return result;
}
```

```javascript
function getProfitSafeWindow(order, n10dData) {
  const today = startOfDay(new Date());
  const earliest = addDays(today, 2);

  if (isMTOOrder(order)) {
    const fixed = normalizeDate(order.target_avail_date) || formatISODate(today);
    return { start: fixed, end: fixed };
  }

  const status = String(order._n10dStatus || order.inferred_target_label || '').toLowerCase();

  if (status === 'critical') {
    const fixed = normalizeDate(order.target_avail_date) || formatISODate(today);
    return { start: fixed, end: fixed };
  }

  if (status === 'urgent' || status === 'monitor' || status.includes('prio replenish')) {
    const latest = normalizeDate(order.target_avail_date) || formatISODate(addDays(today, 10));
    return {
      start: formatISODate(earliest),
      end: latest,
    };
  }

  // sufficient / flexible
  const n10dThreshold = resolveEffectiveN10DDate(order, n10dData);
  const thresholdDate = n10dThreshold ? startOfDay(new Date(n10dThreshold)) : addDays(today, 10);
  const latest = addDays(thresholdDate, 14);

  return {
    start: formatISODate(earliest),
    end: formatISODate(latest),
  };
}
```
