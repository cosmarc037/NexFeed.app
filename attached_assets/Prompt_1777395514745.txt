function buildOrderInsightContext({
  order,
  previousOrder,
  nextOrder,
  strategyId,
  line,
  lineOrders,
  index,
  changeoverRules,
}) {
  const effectiveAvailDate = normalizeDate(order._aiSuggestedDate || order.target_avail_date);
  const protectedReason = getProtectedReason(order);

  const previousChangeover = previousOrder
    ? calculateChangeoverBetween(previousOrder, order, changeoverRules, line)
    : 0;

  const nextChangeover = nextOrder
    ? calculateChangeoverBetween(order, nextOrder, changeoverRules, line)
    : 0;

  const productionHours = calculateProductionHours(order, line);
  const changeoverHours = previousOrder
    ? calculateChangeoverBetween(previousOrder, order, changeoverRules, line)
    : 0;

  const margin = parseFloat(order._margin || 0) || 0;
  const totalHours = productionHours + changeoverHours;
  const profitRate = totalHours > 0 ? margin / totalHours : 0;

  return {
    strategy: strategyId,
    line,
    position: index + 1,
    totalOrders: lineOrders.length,

    order_id: order.id,
    material_code_fg: order.material_code_fg,
    item_description: order.item_description,

    category: order.category,
    color: order.color,
    diameter: order.diameter,
    form: order.form,
    materialSignature: getMaterialSignature(order),

    effectiveAvailDate,
    originalAvailDate: normalizeDate(order._originalAvailDate || order.target_avail_date),
    estimatedCompletionDate: order.estimated_completion_date || order._estimatedCompletion || null,

    n10dStatus: order._n10dStatus || order.inferred_target_label || '',
    protectedReason,

    previousOrder: previousOrder
      ? {
          order_id: previousOrder.id,
          item_description: previousOrder.item_description,
          materialSignature: getMaterialSignature(previousOrder),
          effectiveAvailDate: normalizeDate(previousOrder._aiSuggestedDate || previousOrder.target_avail_date),
        }
      : null,

    nextOrder: nextOrder
      ? {
          order_id: nextOrder.id,
          item_description: nextOrder.item_description,
          materialSignature: getMaterialSignature(nextOrder),
          effectiveAvailDate: normalizeDate(nextOrder._aiSuggestedDate || nextOrder.target_avail_date),
        }
      : null,

    previousChangeover: Number(previousChangeover.toFixed(2)),
    nextChangeover: Number(nextChangeover.toFixed(2)),

    productionHours: Number(productionHours.toFixed(2)),
    changeoverHours: Number(changeoverHours.toFixed(2)),
    margin: Number(margin.toFixed(2)),
    profitRate: Number(profitRate.toFixed(2)),

    aiDateReason: order._aiDateReason || order._aiReasoning || '',
  };
}

## 5. ADD A PROTECTED-REASON HELPER

function getProtectedReason(order) {
  if (isMTOOrder(order)) return 'MTO';

  const status = String(order._n10dStatus || order.inferred_target_label || '').toLowerCase();

  if (status === 'critical') return 'Critical';
  if (status === 'urgent') return 'Urgent';
  if (status === 'monitor') return 'Monitor';
  if (status.includes('prio replenish')) return 'Prio Replenish';

  return null;
}

## 8. BATCH GENERATE PER LINE / PER STRATEGY

Use batch generation so you do NOT make one AI call per row.

Example approach:
- for each strategy
- for each line
- build row contexts for all orders on that line
- call AI once
- map back results

### Suggested implementation:
```javascript
async function generateStrategyInsights({
  strategyId,
  ordersByLine,
  changeoverRules,
  masterData,
  n10dData,
}) {
  const updated = structuredClone(ordersByLine);

  for (const [line, orders] of Object.entries(updated)) {
    if (!orders || orders.length === 0) continue;

    const lineContexts = orders.map((order, index) => {
      const previousOrder = index > 0 ? orders[index - 1] : null;
      const nextOrder = index < orders.length - 1 ? orders[index + 1] : null;

      return buildOrderInsightContext({
        order,
        previousOrder,
        nextOrder,
        strategyId,
        line,
        lineOrders: orders,
        index,
        changeoverRules,
      });
    });

    const prompt = buildStrategyInsightsPrompt({ strategyId, lineContexts });
    const response = await callAzureOpenAI(prompt);
    const parsed = parseInsightResponse(response);

    const resultMap = new Map(
      (parsed.results || []).map(r => [String(r.order_id), r])
    );

    updated[line] = orders.map(order => {
      const aiInsight = resultMap.get(String(order.id));

      return {
        ...order,
        _strategyInsights: {
          ...(order._strategyInsights || {}),
          [strategyId]: {
            short: aiInsight?.short || 'Position determined by current sequencing strategy.',
            long: aiInsight?.long || 'This order was placed based on the selected sequencing strategy and current schedule context.',
          },
        },
      };
    });
  }

  return updated;
}
```

## 9. ADD A RESPONSE PARSER

```javascript
function parseInsightResponse(responseText) {
  try {
    const match = responseText.match(/\{[\s\S]*\}/);
    if (!match) return { results: [] };
    return JSON.parse(match[0]);
  } catch (error) {
    console.error('Failed to parse AI insight response:', error);
    return { results: [] };
  }
}
```

    return {
      short: 'Placed according to chronology and priority rules.',
      long: 'This order was positioned using the standard sequencing logic based on date and scheduling priority.',
    };
  }

  if (strategyId === 'ai_balanced') {
    return {
      short: 'Placed to support smoother flow and lower changeover.',
      long: 'This order was positioned using AI balanced sequencing to improve flow while respecting scheduling constraints.',
    };
  }

  if (strategyId === 'ai_profit') {
    return {
      short: 'Placed to improve profitability while staying schedule-safe.',
      long: 'This order was positioned using AI profit-optimized sequencing to improve profitability while respecting scheduling constraints.',
    };
  }

  return {
    short: 'Placed by sequencing logic.',
    long: 'This order was placed by the current sequencing strategy.',
  };
}
