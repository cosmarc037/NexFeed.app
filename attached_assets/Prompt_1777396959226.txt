function getDisplayedProductionHours(order, line) {
  const direct =
    parseFloat(order.production_time ?? order._productionTimeHrs);

  if (!Number.isNaN(direct)) return direct;

  const volume = parseFloat(order.volume_override || order.volume || order.total_volume_mt || 0) || 0;
  const runRate = getLineRunRate(normalizeLine(line));

  return runRate > 0 ? volume / runRate : 0;
}

function getDisplayedChangeoverHours(order, previousOrder, changeoverRules, line) {
  const direct =
    parseFloat(order.changeover ?? order._changeoverHrs);

  if (!Number.isNaN(direct)) return direct;

  if (previousOrder) {
    return parseFloat(calculateChangeoverBetween(previousOrder, order, changeoverRules, line) || 0) || 0;
  }

  return 0;
}

function getDisplayedMargin(order) {
  const margin = parseFloat(order._margin ?? order.margin ?? 0);
  return Number.isNaN(margin) ? 0 : margin;
}

function calculateDisplayedProfitRate({ margin, productionHours, changeoverHours }) {
  const total = productionHours + changeoverHours;
  if (total <= 0) return 0;
  return margin / total;
}

function ensureMarginPresent(ordersByLine, masterData) {
  const result = {};

  Object.entries(ordersByLine).forEach(([line, orders]) => {
    result[line] = orders.map(order => {
      if (order._margin != null && !Number.isNaN(parseFloat(order._margin))) {
        return order;
      }

      const materialCode = String(order.material_code_fg || '').trim();
      const mdEntry = masterData.find(md =>
        String(md.material_code_fg || md.material_code || '').trim() === materialCode
      );

      const margin = mdEntry ? parseFloat(mdEntry.Margin ?? mdEntry.margin ?? 0) || 0 : 0;

      return {
        ...order,
        _margin: margin,
      };
    });
  });

  return result;
}

## 5. ADD CONTEXT VALIDATION BEFORE SENDING TO AI

This prevents mismatched row values from entering the prompt.

```javascript
function validateProfitInsightContext(ctx) {
  const expectedProfitRate =
    (ctx.productionHours + ctx.changeoverHours) > 0
      ? ctx.margin / (ctx.productionHours + ctx.changeoverHours)
      : 0;

  const roundedExpected = Number(expectedProfitRate.toFixed(2));

  if (Math.abs(roundedExpected - ctx.profitRate) > 0.01) {
    console.warn('[Profit Insight] Profit rate mismatch', {
      order_id: ctx.order_id,
      margin: ctx.margin,
      productionHours: ctx.productionHours,
      changeoverHours: ctx.changeoverHours,
      ctxProfitRate: ctx.profitRate,
      expectedProfitRate: roundedExpected,
    });

    ctx.profitRate = roundedExpected;
  }

  return ctx;
}
```

This ensures:
- profit rate is always mathematically consistent with the production and changeover values sent to AI

function getMarginStatus(order) {
  const raw = order._margin ?? order.margin;

  if (raw == null || raw === '') {
    return { margin: 0, marginFound: false };

  const parsed = parseFloat(raw);
