```javascript
function getMarginFromMasterData(order, masterData) {
  const orderCode = normalizeMaterialCode(order.material_code_fg || order.material_code);

  if (!orderCode) {
    return { margin: 0, marginFound: false, matchedField: null };
  }

  const match = masterData.find(md => {
    const codesToCheck = [
      md.material_code_fg,
      md.material_code,
      md.MaterialCode,
      md['Material Code (FG)'],
      md['Material Code'],
    ].map(normalizeMaterialCode);

    return codesToCheck.includes(orderCode);
  });

  if (!match) {
    return { margin: 0, marginFound: false, matchedField: null };
  }

  const rawMargin =
    match.Margin ??
    match.margin ??
    match['Margin'] ??
    match['MARGIN'] ??
    null;

  const parsedMargin = parseFloat(rawMargin);

  if (Number.isNaN(parsedMargin)) {
    return { margin: 0, marginFound: false, matchedField: 'margin-unparseable' };
  }

  return {
    margin: parsedMargin,
    marginFound: true,
    matchedField: 'Margin',
  };
}
```

Important:
- Match master data by normalized material code
- Check multiple possible field names if needed
- Do not assume only one exact property spelling
- If not found, explicitly mark `marginFound: false`

function ensureMarginPresent(ordersByLine, masterData) {
  const result = {};

  Object.entries(ordersByLine).forEach(([line, orders]) => {
    result[line] = orders.map(order => {
      const existing = parseFloat(order._margin);
      if (!Number.isNaN(existing) && existing !== 0) {
        return order;
      }

      const { margin, marginFound, matchedField } = getMarginFromMasterData(order, masterData);

      return {
        ...order,
        _margin: margin,
        _marginFound: marginFound,
        _marginMatchedField: matchedField,
      };
    });
  });

  return result;
}

function validateProfitInsightContext(ctx) {
  const expectedProfitRate =
    (ctx.productionHours + ctx.changeoverHours) > 0
      ? ctx.margin / (ctx.productionHours + ctx.changeoverHours)
      : 0;

  const roundedExpected = Number(expectedProfitRate.toFixed(2));

  if (Math.abs(roundedExpected - ctx.profitRate) > 0.01) {
    console.warn('[Profit Insight] Profit rate mismatch', {
      order_id: ctx.order_id,
      margin: ctx.margin,
      productionHours: ctx.productionHours,
      changeoverHours: ctx.changeoverHours,
      ctxProfitRate: ctx.profitRate,
      expectedProfitRate: roundedExpected,
    });

    ctx.profitRate = roundedExpected;
  }

  return ctx;
}
