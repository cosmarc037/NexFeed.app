### Calculation rules:

### Helpers:
```javascript
function calculateTotalChangeoverTime(lineOrders) {
  return Number(
    lineOrders.reduce((sum, order) => {
      const co = parseFloat(order.changeover ?? order._changeoverHrs ?? 0) || 0;
      return sum + co;
    }, 0).toFixed(2)
  );
}

function calculateTimeSavedVsStandard(standardOrders, strategyOrders) {
  const standardCO = calculateTotalChangeoverTime(standardOrders);
  const strategyCO = calculateTotalChangeoverTime(strategyOrders);

  const rawSaved = standardCO - strategyCO;

  return {
    rawSaved: Number(rawSaved.toFixed(2)),
    displaySaved: Number(Math.max(0, rawSaved).toFixed(2)),
  };
}
```
