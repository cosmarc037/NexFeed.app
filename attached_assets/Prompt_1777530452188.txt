### Helper:
```javascript
function calculateTotalChangeoverTime(lineOrders) {
  return Number(
    lineOrders.reduce((sum, order) => {
      const co = parseFloat(
        order.changeover ??
        order._changeoverHrs ??
        order._changeover ??
        order.change_over ??
        0
      ) || 0;

      return sum + co;
    }, 0).toFixed(2)
  );
}
```

Important:
- `lineOrders` must be the final executed order rows for that specific strategy
- compute after final sequencing/date application and final changeover recalculation

### Helper:
```javascript
function calculateTimeSavedVsStandard(standardOrders, strategyOrders) {
  const standardCO = calculateTotalChangeoverTime(standardOrders);
  const strategyCO = calculateTotalChangeoverTime(strategyOrders);

  const rawSaved = standardCO - strategyCO;

  return {
    standardCO: Number(standardCO.toFixed(2)),
    strategyCO: Number(strategyCO.toFixed(2)),
    rawSaved: Number(rawSaved.toFixed(2)),
    displaySaved: Number(Math.max(0, rawSaved).toFixed(2)),

}
```
