### Helper:
```javascript
function getRowDisplayedChangeover(order) {
  const candidates = [
    order.changeover,
    order._changeoverHrs,
    order._changeover,
    order.change_over,
  ];

  for (const val of candidates) {
    const parsed = parseFloat(val);
    if (!Number.isNaN(parsed)) return parsed;
  }

  return 0;
}

function calculateAfterTableTotalChangeover(lineOrders) {
  return Number(
    (lineOrders || []).reduce((sum, order) => {
      return sum + getRowDisplayedChangeover(order);

  );
}
```

```javascript
function formatSignedHours(value) {
  const num = parseFloat(value || 0) || 0;
  const sign = num > 0 ? '+' : '';

}
```
