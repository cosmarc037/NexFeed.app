## 2. CALCULATE PRE-MOVE AND POST-MOVE LOADS SEPARATELY

### Helpers
```javascript
function getOrderVolumeMT(order) {
  const candidates = [
    order.volume_override,
    order.volume,
    order.total_volume_mt,
    order.volume_mt,
  ];

  for (const val of candidates) {
    const parsed = parseFloat(val);
    if (!Number.isNaN(parsed)) return parsed;
  }

  return 0;
}

function calculateLineTotalMT(orders) {
  return Number(
    (orders || []).reduce((sum, order) => {
      return sum + getOrderVolumeMT(order);
    }, 0).toFixed(2)
  );
}

function calculateQueueTimeHours(totalMT, runRate) {
  const mt = parseFloat(totalMT) || 0;
  const rr = parseFloat(runRate) || 0;

  if (rr <= 0) return 0;

  return Number((mt / rr).toFixed(2));
}
```

### Required logic
For a move from `sourceLine` to `targetLine`:

```javascript
const movedMT = getOrderVolumeMT(orderToMove);

const sourceBeforeOrders = preMoveOrdersByLine[sourceLine] || [];
const targetBeforeOrders = preMoveOrdersByLine[targetLine] || [];

const sourceBeforeMT = calculateLineTotalMT(sourceBeforeOrders);
const targetBeforeMT = calculateLineTotalMT(targetBeforeOrders);

const sourceAfterMT = Number((sourceBeforeMT - movedMT).toFixed(2));
const targetAfterMT = Number((targetBeforeMT + movedMT).toFixed(2));

const sourceRunRate = getLineRunRate(sourceLine);
const targetRunRate = getLineRunRate(targetLine);

const sourceBeforeQueue = calculateQueueTimeHours(sourceBeforeMT, sourceRunRate);
const sourceAfterQueue = calculateQueueTimeHours(sourceAfterMT, sourceRunRate);
const targetBeforeQueue = calculateQueueTimeHours(targetBeforeMT, targetRunRate);
const targetAfterQueue = calculateQueueTimeHours(targetAfterMT, targetRunRate);
```

Important:
- `sourceBeforeMT` must include the moved order if it was still on the source line at that time
- do NOT compute it after the order has already been removed
