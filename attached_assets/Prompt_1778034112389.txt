## 1. DEFINE A SHARED HELPER FOR EFFECTIVE DISPLAYED ORDER VOLUME

Create or update a helper that returns the actual effective volume that the app is displaying / using operationally.

### Suggested helper:
```javascript
function adjustVolumeToBatchCeiling(volume, batchSize) {
  const v = parseFloat(volume || 0) || 0;
  const b = parseFloat(batchSize || 0) || 0;

  if (b <= 0) return Number(v.toFixed(2));

  return Number((Math.ceil(v / b) * b).toFixed(2));
}

function getEffectiveOrderVolume(order) {
  const overrideVolume = parseFloat(order.volume_override);
  if (!Number.isNaN(overrideVolume)) {
    return Number(overrideVolume.toFixed(2));
  }

  const combinedEffective = parseFloat(order._combined_effective_volume);
  if (!Number.isNaN(combinedEffective)) {
    return Number(combinedEffective.toFixed(2));
  }

  const explicitEffective = parseFloat(order._effectiveVolumeMT);
  if (!Number.isNaN(explicitEffective)) {
    return Number(explicitEffective.toFixed(2));
  }

  const rawVolume = parseFloat(
    order.total_volume_mt ??
    order.volume ??
    order.volume_mt ??
    0
  ) || 0;

  const batchSize = parseFloat(order.batch_size ?? order.batchSize ?? 0) || 0;

  return adjustVolumeToBatchCeiling(rawVolume, batchSize);
}
```

Important:
- this helper is for **display/summary/operational totaling**
- not for pre-combine raw-basis summation
