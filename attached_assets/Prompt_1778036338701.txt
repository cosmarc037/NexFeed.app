function getFinalDisplayedOrderVolume(order) {
  const candidates = [
    order.display_volume_mt,
    order._displayVolumeMT,
    order.final_volume_mt,
    order._finalVolumeMT,
    order.volume_override,
    order.total_volume_mt,
    order.volume_mt,
    order.volume,
