## 1. DEFINE APP-ADJUSTED MT FOR DISPLAY

An order is app-adjusted if:
- it has no user override
- and its raw/current volume is not divisible by its batch size
- and the app rounds it up to the nearest valid batch multiple

### Suggested helper:
```javascript
function adjustVolumeToBatchCeiling(volume, batchSize) {
  const v = parseFloat(volume || 0) || 0;
  const b = parseFloat(batchSize || 0) || 0;

  if (b <= 0) return Number(v.toFixed(2));

  return Number((Math.ceil(v / b) * b).toFixed(2));
}

function getRawOrderVolume(order) {
  return Number(
    parseFloat(
      order.total_volume_mt ??
      order.volume_mt ??
      order.volume ??
      0
    ) || 0
  );
}

function getBatchSize(order) {
  return Number(
    parseFloat(order.batch_size ?? order.batchSize ?? 0) || 0
  );
}

function getVolumeOverride(order) {
  const parsed = parseFloat(order.volume_override);
  return Number.isNaN(parsed) ? null : Number(parsed.toFixed(2));
}
```

### Determine app-adjusted state:
```javascript
function getOrderVolumeDisplayState(order) {
  const override = getVolumeOverride(order);
  const rawVolume = getRawOrderVolume(order);
  const batchSize = getBatchSize(order);

  if (override != null) {
    return {
      displayVolume: override,
      originalVolume: rawVolume,
      isUserOverride: true,
      isAppAdjusted: false,
      batchSize,
    };
  }

  const adjusted = adjustVolumeToBatchCeiling(rawVolume, batchSize);
  const isAppAdjusted = batchSize > 0 && adjusted !== rawVolume;

  return {
    displayVolume: adjusted,
    originalVolume: rawVolume,
    isUserOverride: false,
    isAppAdjusted,
    batchSize,
  };
}
```

Important:
- for app-adjusted rows, the displayed volume should now be the adjusted value
- not the original raw volume
